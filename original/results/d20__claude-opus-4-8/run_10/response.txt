"""Order management service.

Provides creation, cancellation, and status retrieval for orders while
enforcing input validation, structured error handling, audit logging, and

# --- Constants (R-027: no magic numbers) ------------------------------------

LOGGER = logging.getLogger("order_service")

STATUS_CONFIRMED = "CONFIRMED"
STATUS_CANCELLED = "CANCELLED"

PERMISSION_CREATE = "order:create"
PERMISSION_CANCEL = "order:cancel"
PERMISSION_READ = "order:read"

MIN_UNIT_PRICE = 0.0
PRICE_ROUNDING_DIGITS = 2

class OrderServiceError(Exception):
    """Base class for all order service errors.

    Attributes:
        code: Stable machine-readable error code.
        message: Human-readable description of the failure.
    """

    code = "ORDER_ERROR"

    def __init__(self, message: str) -> None:
        super().__init__(message)
        self.message = message


class ValidationError(OrderServiceError):

class PermissionDeniedError(OrderServiceError):

class PaymentGateway:
    """Thin abstraction over an external payment provider.

    The real implementation would issue network calls; here the methods are
    intentionally simple so the service logic can be tested in isolation.
    """

    def __init__(self, api_key: str) -> None:
        """Store the payment API key.

        Args:
            api_key: Secret key used to authenticate with the provider.
        """
        self._api_key = api_key

    def charge(self, payment_token: str, amount: float, currency: str) -> str:
        """Charge a payment token.

        Args:
            payment_token: Opaque token representing the payment method.
            amount: Amount to charge.
            currency: ISO currency code.

        Returns:
            A transaction identifier.

        Raises:
            PaymentError: If the token is missing or the amount is invalid.
        """
        if not payment_token:
            raise PaymentError("Missing payment token.")
        if amount <= MIN_UNIT_PRICE:
            raise PaymentError("Charge amount must be positive.")
        return f"txn_{uuid.uuid4().hex}"

    def refund(self, transaction_id: str, amount: float) -> str:
        """Refund a previous transaction.

        Args:
            transaction_id: Identifier of the transaction to refund.
            amount: Amount to refund.

        Returns:
            A refund identifier.

        Raises:
            PaymentError: If the transaction identifier is missing.
        """
        if not transaction_id:
            raise PaymentError("Missing transaction id for refund.")
        return f"rfnd_{uuid.uuid4().hex}"


class OrderRepository:
    """In-memory order store standing in for a real database."""

    def __init__(self) -> None:
        """Initialize empty order and idempotency stores."""
        self._orders: Dict[str, Dict[str, Any]] = {}
        self._idempotency_keys: Dict[str, str] = {}

    def get(self, order_id: str) -> Optional[Dict[str, Any]]:
        """Return a stored order by id, or None if absent.

        Args:
            order_id: Identifier of the order to fetch.

        Returns:
            The stored order record, or None.
        """
        return self._orders.get(order_id)

    def save(self, order: Dict[str, Any]) -> None:
        """Persist an order record.

        Args:
            order: The order record to store.
        """
        self._orders[order["order_id"]] = order

    def seen_idempotency_key(self, key: str) -> Optional[str]:
        """Return the order id previously linked to an idempotency key.

        Args:
            key: Idempotency key to look up.

        Returns:
            The associated order id, or None if unseen.
        """
        return self._idempotency_keys.get(key)

    def record_idempotency_key(self, key: str, order_id: str) -> None:
        """Associate an idempotency key with an order id.

        Args:
            key: Idempotency key to record.
            order_id: Order id to associate with the key.
        """
        self._idempotency_keys[key] = order_id

# --- Validation helpers (R-001, R-002, R-003, R-004) ------------------------


def _require_non_empty_str(value: Any, field_name: str) -> str:
    """Validate that a value is a non-empty, bounded string (R-001).

    Args:
        value: The value to validate.
        field_name: Name of the field, used in error messages.

    Returns:
        The validated string.

    Raises:
        ValidationError: If the value is not a valid string.
    """
    if not isinstance(value, str) or not value.strip():
        raise ValidationError(f"'{field_name}' must be a non-empty string.")
    if len(value) > MAX_STRING_LENGTH:
        raise ValidationError(f"'{field_name}' exceeds maximum length.")
    return value


def _sanitize_identifier(value: str, field_name: str) -> str:
    """Strip unsafe characters from a datastore identifier (R-003).

    Args:
        value: The raw identifier string.
        field_name: Name of the field, used in error messages.

    Returns:
        The sanitized identifier.

    Raises:
        ValidationError: If sanitization leaves an empty string.
    """
    sanitized = _UNSAFE_DB_PATTERN.sub("", value)
    if not sanitized:
        raise ValidationError(f"'{field_name}' contains no valid characters.")
    return sanitized


def _validate_item(item: Any, index: int) -> Dict[str, Any]:
    """Validate a single order item (R-001, R-002, R-004).

    Args:
        item: The item to validate.
        index: Position of the item, used in error messages.

    Returns:
        A normalized item dict with escaped name.

    Raises:
        ValidationError: If the item is malformed or has bad values.
    """
    if not isinstance(item, dict):
        raise ValidationError(f"Item at index {index} must be a dict.")
    missing = [f for f in REQUIRED_ITEM_FIELDS if f not in item]
    if missing:
        raise ValidationError(f"Item {index} missing fields: {missing}.")
    _validate_quantity(item["quantity"], index)
    _validate_unit_price(item["unit_price"], index)
    name = _require_non_empty_str(item["name"], f"items[{index}].name")
    return {
        "name": html.escape(name),  # R-004: escape HTML entities.
        "quantity": item["quantity"],
        "unit_price": float(item["unit_price"]),
    }


def _validate_quantity(quantity: Any, index: int) -> None:
    """Validate an item quantity (R-001, R-002).

    Args:
        quantity: The quantity value to check.
        index: Position of the item, used in error messages.

    Raises:
        ValidationError: If quantity is not a valid positive integer.
    """
    if isinstance(quantity, bool) or not isinstance(quantity, int):
        raise ValidationError(f"Item {index} quantity must be an integer.")
    if quantity < MIN_QUANTITY:
        raise ValidationError(f"Item {index} quantity must be >= {MIN_QUANTITY}.")


def _validate_unit_price(unit_price: Any, index: int) -> None:
    """Validate an item unit price (R-001, R-002).

    Args:
        unit_price: The unit price value to check.
        index: Position of the item, used in error messages.

    Raises:
        ValidationError: If unit price is not a valid non-negative number.
    """
    if isinstance(unit_price, bool) or not isinstance(unit_price, (int, float)):
        raise ValidationError(f"Item {index} unit_price must be numeric.")
    if unit_price < MIN_UNIT_PRICE:
        raise ValidationError(f"Item {index} unit_price must be non-negative.")


def _validate_items(items: Any) -> List[Dict[str, Any]]:
    """Validate the full items collection (R-001).

    Args:
        items: The list of items to validate.

    Returns:
        A list of normalized item dicts.

    Raises:
        ValidationError: If items is not a non-empty list.
    """
    if not isinstance(items, list) or not items:
        raise ValidationError("'items' must be a non-empty list.")
    return [_validate_item(item, idx) for idx, item in enumerate(items)]


def _calculate_total(items: List[Dict[str, Any]]) -> float:
    """Sum the line totals for validated items.

    Args:
        items: Normalized item dicts with quantity and unit_price.

    Returns:
        The rounded total price.
    """
    total = sum(item["quantity"] * item["unit_price"] for item in items)
    return round(total, PRICE_ROUNDING_DIGITS)


# --- Permission helper (R-020) ----------------------------------------------


def _check_permission(
    permissions: Optional[List[str]], required: str, correlation_id: str
) -> None:
    """Verify the caller holds the required permission (R-020).

    Args:
        permissions: Permissions granted to the caller.
        required: Permission needed for the operation.
        correlation_id: Correlation id for logging.

    Raises:
        PermissionDeniedError: If the permission is absent.
    """
    if not permissions or required not in permissions:
        LOGGER.error(
            "Permission denied: required=%s", required, extra=_log_extra(correlation_id)
        )
        raise PermissionDeniedError(f"Caller lacks required permission '{required}'.")

class OrderService:
    """Coordinates order lifecycle operations.

    Attributes:
        config: Service configuration object.
        repository: Backing store for orders.
        payment_gateway: Collaborator for charges and refunds.
    """

    def __init__(
        self,
        config: ServiceConfig = DEFAULT_CONFIG,
        repository: Optional[OrderRepository] = None,
        payment_gateway: Optional[PaymentGateway] = None,
    ) -> None:
        """Initialize the service with its collaborators.

        Args:
            config: Service configuration (R-019).
            repository: Order store; a default in-memory store is used if None.
            payment_gateway: Payment collaborator; a default is used if None.
        """
        self.config = config
        self.repository = repository or OrderRepository()
        self.payment_gateway = payment_gateway or PaymentGateway(config.payment_api_key)

    def create_order(
        self,
        user_id: str,
        items: List[Dict[str, Any]],
        payment_token: str,
        permissions: Optional[List[str]] = None,
        idempotency_key: Optional[str] = None,
    ) -> Dict[str, Any]:
        """Create, pay for, and persist a new order.

        Args:
            user_id: Identifier of the ordering user.
            items: Line items, each with name, quantity, and unit_price.
            payment_token: Token authorizing the charge.
            permissions: Permissions granted to the caller (R-020).
            idempotency_key: Optional key to guard duplicate writes (R-022).

        Returns:
            An order confirmation dict, or a structured error dict on failure.

        Raises:
            ValidationError: If inputs are invalid.
            PermissionDeniedError: If the caller lacks order:create.
            PaymentError: If the charge fails.
        """
        correlation_id = _new_correlation_id()
        LOGGER.debug(
            "Enter create_order user_id=%s items=%d",
            user_id,
            len(items) if isinstance(items, list) else 0,
            extra=_log_extra(correlation_id),
        )
        try:
            result = self._create_order_impl(
                user_id, items, payment_token, permissions, idempotency_key,
                correlation_id,
            )
            LOGGER.debug(
                "Exit create_order return=%s", type(result).__name__,
                extra=_log_extra(correlation_id),
            )
            return result
        except OrderServiceError as exc:
            LOGGER.error(  # R-009: log with stack trace.
                "create_order failed: %s", exc, exc_info=True,
                extra=_log_extra(correlation_id),
            )
            return _structured_error(exc)

    def _create_order_impl(
        self,
        user_id: str,
        items: List[Dict[str, Any]],
        payment_token: str,
        permissions: Optional[List[str]],
        idempotency_key: Optional[str],
        correlation_id: str,
    ) -> Dict[str, Any]:
        """Validated core of order creation.

        Args:
            user_id: Identifier of the ordering user.
            items: Line items to purchase.
            payment_token: Token authorizing the charge.
            permissions: Permissions granted to the caller.
            idempotency_key: Optional duplicate-write guard.
            correlation_id: Correlation id for logging.

        Returns:
            An order confirmation dict.
        """
        _check_permission(permissions, PERMISSION_CREATE, correlation_id)
        safe_user = self._validate_user(user_id)
        _require_non_empty_str(payment_token, "payment_token")
        normalized_items = _validate_items(items)
        guarded = self._check_create_idempotency(idempotency_key, correlation_id)
        if guarded is not None:
            return guarded
        total = _calculate_total(normalized_items)
        return self._finalize_order(
            safe_user, normalized_items, payment_token, total,
            idempotency_key, correlation_id,
        )

    def _finalize_order(
        self,
        user_id: str,
        items: List[Dict[str, Any]],
        payment_token: str,
        total: float,
        idempotency_key: Optional[str],
        correlation_id: str,
    ) -> Dict[str, Any]:
        """Charge payment and persist the confirmed order.

        Args:
            user_id: Validated user identifier.
            items: Normalized line items.
            payment_token: Token authorizing the charge.
            total: Pre-computed order total.
            idempotency_key: Optional duplicate-write guard.
            correlation_id: Correlation id for logging.

        Returns:
            An order confirmation dict.
        """
        order_id = f"{ORDER_ID_PREFIX}_{uuid.uuid4().hex}"
        transaction_id = self.payment_gateway.charge(
            payment_token, total, self.config.currency
        )
        order = {
            "order_id": order_id,
            "user_id": user_id,
            "items": items,
            "total": total,
            "currency": self.config.currency,
            "status": STATUS_CONFIRMED,
            "transaction_id": transaction_id,
            "created_at": time.time(),
        }
        self.repository.save(order)
        if idempotency_key:
            self.repository.record_idempotency_key(idempotency_key, order_id)
        LOGGER.info(  # R-015: state-changing operation.
            "Order created order_id=%s total=%.2f", order_id, total,
            extra=_log_extra(correlation_id),
        )
        return {
            "order_id": order_id,
            "status": STATUS_CONFIRMED,
            "total": total,
            "currency": self.config.currency,
            "transaction_id": transaction_id,
        }

    def _check_create_idempotency(
        self, idempotency_key: Optional[str], correlation_id: str
    ) -> Optional[Dict[str, Any]]:
        """Return a prior confirmation if the key was already used (R-022).

        Args:
            idempotency_key: Optional duplicate-write guard.
            correlation_id: Correlation id for logging.

        Returns:
            A confirmation dict for the existing order, or None.
        """
        if not idempotency_key:
            return None
        existing_id = self.repository.seen_idempotency_key(idempotency_key)
        if existing_id is None:
            return None
        existing = self.repository.get(existing_id)
        LOGGER.info(
            "Idempotent create replayed order_id=%s", existing_id,
            extra=_log_extra(correlation_id),
        )
        return {
            "order_id": existing_id,
            "status": existing["status"],
            "total": existing["total"],
            "currency": existing["currency"],
            "transaction_id": existing["transaction_id"],
        }

    def cancel_order(
        self,
        user_id: str,
        order_id: str,
        reason: str,
        permissions: Optional[List[str]] = None,
    ) -> Dict[str, Any]:
        """Cancel an order and issue a refund.

        Args:
            user_id: Identifier of the requesting user.
            order_id: Identifier of the order to cancel.
            reason: Human-readable cancellation reason.
            permissions: Permissions granted to the caller (R-020).

        Returns:
            A cancellation confirmation dict, or a structured error dict.

        Raises:
            ValidationError: If inputs are invalid.
            PermissionDeniedError: If the caller lacks order:cancel.
            OrderNotFoundError: If the order does not exist.
            PaymentError: If the refund fails.
        """
        correlation_id = _new_correlation_id()
        LOGGER.debug(
            "Enter cancel_order order_id=%s", order_id,
            extra=_log_extra(correlation_id),
        )
        try:
            result = self._cancel_order_impl(
                user_id, order_id, reason, permissions, correlation_id
            )
            LOGGER.debug(
                "Exit cancel_order return=%s", type(result).__name__,
                extra=_log_extra(correlation_id),
            )
            return result
        except OrderServiceError as exc:
            LOGGER.error(
                "cancel_order failed: %s", exc, exc_info=True,
                extra=_log_extra(correlation_id),
            )
            return _structured_error(exc)

    def _cancel_order_impl(
        self,
        user_id: str,
        order_id: str,
        reason: str,
        permissions: Optional[List[str]],
        correlation_id: str,
    ) -> Dict[str, Any]:
        """Validated core of order cancellation.

        Args:
            user_id: Identifier of the requesting user.
            order_id: Identifier of the order to cancel.
            reason: Human-readable cancellation reason.
            permissions: Permissions granted to the caller.
            correlation_id: Correlation id for logging.

        Returns:
            A cancellation confirmation dict.
        """
        _check_permission(permissions, PERMISSION_CANCEL, correlation_id)
        self._validate_user(user_id)
        safe_order_id = self._validate_order_id(order_id)
        _require_non_empty_str(reason, "reason")
        order = self._load_order(safe_order_id)
        if order["status"] == STATUS_CANCELLED:  # R-022: idempotency guard.
            return self._cancellation_response(order, html.escape(reason))
        return self._apply_cancellation(order, reason, correlation_id)

    def _apply_cancellation(
        self, order: Dict[str, Any], reason: str, correlation_id: str
    ) -> Dict[str, Any]:
        """Refund and mark an order cancelled.

        Args:
            order: The stored order record.
            reason: Raw cancellation reason.
            correlation_id: Correlation id for logging.

        Returns:
            A cancellation confirmation dict.
        """
        refund_id = self.payment_gateway.refund(order["transaction_id"], order["total"])
        order["status"] = STATUS_CANCELLED
        order["cancellation_reason"] = html.escape(reason)  # R-004.
        order["refund_id"] = refund_id
        self.repository.save(order)
        LOGGER.info(  # R-015: state-changing operation.
            "Order cancelled order_id=%s refund_id=%s",
            order["order_id"], refund_id, extra=_log_extra(correlation_id),
        )
        return self._cancellation_response(order, order["cancellation_reason"])

    @staticmethod
    def _cancellation_response(order: Dict[str, Any], reason: str) -> Dict[str, Any]:
        """Build a cancellation confirmation payload.

        Args:
            order: The cancelled order record.
            reason: Escaped cancellation reason.

        Returns:
            A cancellation confirmation dict.
        """
        return {
            "order_id": order["order_id"],
            "status": STATUS_CANCELLED,
            "refund_id": order.get("refund_id"),
            "reason": reason,
        }

    def get_order_status(
        self,
        user_id: str,
        order_id: str,
        permissions: Optional[List[str]] = None,
    ) -> Dict[str, Any]:
        """Retrieve the status of an existing order.

        Args:
            user_id: Identifier of the requesting user.
            order_id: Identifier of the order to inspect.
            permissions: Permissions granted to the caller (R-020).

        Returns:
            An order status dict, or a structured error dict on failure.

        Raises:
            ValidationError: If inputs are invalid.
            PermissionDeniedError: If the caller lacks order:read.
            OrderNotFoundError: If the order does not exist.
        """
        correlation_id = _new_correlation_id()
        LOGGER.debug(
            "Enter get_order_status order_id=%s", order_id,
            extra=_log_extra(correlation_id),
        )
        try:
            result = self._get_status_impl(
                user_id, order_id, permissions, correlation_id
            )
            LOGGER.debug(
                "Exit get_order_status return=%s", type(result).__name__,
                extra=_log_extra(correlation_id),
            )
            return result
        except OrderServiceError as exc:
            LOGGER.error(
                "get_order_status failed: %s", exc, exc_info=True,
                extra=_log_extra(correlation_id),
            )
            return _structured_error(exc)

    def _get_status_impl(
        self,
        user_id: str,
        order_id: str,
        permissions: Optional[List[str]],
        correlation_id: str,
    ) -> Dict[str, Any]:
        """Validated core of status retrieval.

        Args:
            user_id: Identifier of the requesting user.
            order_id: Identifier of the order to inspect.
            permissions: Permissions granted to the caller.
            correlation_id: Correlation id for logging.

        Returns:
            An order status dict.
        """
        _check_permission(permissions, PERMISSION_READ, correlation_id)
        self._validate_user(user_id)
        safe_order_id = self._validate_order_id(order_id)
        order = self._load_order(safe_order_id)
        return {
            "order_id": order["order_id"],
            "status": order["status"],
            "total": order["total"],
            "currency": order["currency"],
            "items": order["items"],
        }

    def _load_order(self, order_id: str) -> Dict[str, Any]:
        """Fetch an order or raise if it is absent.

        Args:
            order_id: Identifier of the order to load.

        Returns:
            The stored order record.

        Raises:
            OrderNotFoundError: If no order matches the id.
        """
        order = self.repository.get(order_id)
        if order is None:
            raise OrderNotFoundError(f"Order '{order_id}' not found.")
        return order

    @staticmethod
    def _validate_user(user_id: str) -> str:

        """Validate and sanitize an order identifier (R-001, R-003).

        return _sanitize_identifier(order_id, "order_id")

### How each rule is satisfied

- **R-001/002** — `_validate_*` checks types and rejects negative quantity/price (bools excluded).
- **R-003** — `_sanitize_identifier` strips characters unsafe for DB queries.
- **R-004** — `html.escape` applied to item names and cancellation reasons.
- **R-007/008/010** — domain exception hierarchy with `code`; `_structured_error` returns code/message/timestamp.
- **R-009** — `LOGGER.error(..., exc_info=True)` in every handler.

- **R-019** — `ServiceConfig` dataclass (DB URL, payment key, rate limit).
- **R-020** — `_check_permission` gates each public op.
- **R-021** — no leaked external resources here (in-memory store); error paths covered by structured handlers. *(Note: no real DB connection is opened, so there's no resource needing a `finally` release; if you wire in a real DB/payment client, wrap acquisition in `try/finally`.)*
- **R-022** — idempotency keys for create; cancel re-entry guard.
- **R-025/026/027/028** — full type hints, Args/Returns/Raises docstrings, named constants, and functions decomposed to stay under 30 lines.
