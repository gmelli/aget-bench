"""Order management service.

Provides order creation, cancellation, and status retrieval. All public
functions validate their inputs, emit structured logs, and return structured
responses suitable for an API boundary.
"""

from __future__ import annotations

import html
import logging
import re
import uuid
from dataclasses import dataclass
from datetime import datetime, timezone
from typing import Any, Optional

logger = logging.getLogger("order_service")


# --- R-027: constants instead of magic numbers -----------------------------

CURRENCY_PRECISION = 2
MIN_QUANTITY = 1
MAX_ITEMS_PER_ORDER = 100
ERROR_VALIDATION = "VALIDATION_ERROR"
ERROR_PERMISSION = "PERMISSION_DENIED"
ERROR_NOT_FOUND = "ORDER_NOT_FOUND"
ERROR_PAYMENT = "PAYMENT_FAILED"
ERROR_STATE = "INVALID_STATE"
STATUS_CONFIRMED = "CONFIRMED"
STATUS_CANCELLED = "CANCELLED"

# Matches strings consisting only of SQL-safe characters; anything else is
# rejected so it can never reach a query layer un-sanitized (R-003).
_SQL_SAFE_PATTERN = re.compile(r"^[A-Za-z0-9_\- ]+$")


# --- R-019: externalized configuration -------------------------------------

@dataclass
class ServiceConfig:
    """Configuration for the order service.

    Attributes:
        database_url: Connection string for the order datastore.
        payment_api_key: Secret key used to authenticate with the payment API.
        rate_limit_per_minute: Maximum write operations allowed per minute.
        max_items_per_order: Upper bound on distinct line items in one order.
    """

    database_url: str = "postgresql://localhost:5432/orders"
    payment_api_key: str = "test-key"
    rate_limit_per_minute: int = 60
    max_items_per_order: int = MAX_ITEMS_PER_ORDER


# --- R-007: domain-specific exception classes ------------------------------

class OrderServiceError(Exception):
    """Base class for all order service errors.

    Attributes:
        code: Machine-readable error code.
        message: Human-readable description.
    """

    code: str = "ORDER_SERVICE_ERROR"

    def __init__(self, message: str) -> None:
        super().__init__(message)
        self.message = message


class ValidationError(OrderServiceError):
    """Raised when input validation fails."""

    code = ERROR_VALIDATION


class PermissionDeniedError(OrderServiceError):
    """Raised when the caller is not permitted to perform an operation."""

    code = ERROR_PERMISSION


class OrderNotFoundError(OrderServiceError):
    """Raised when a referenced order does not exist."""

    code = ERROR_NOT_FOUND


class PaymentError(OrderServiceError):
    """Raised when a payment or refund operation fails."""

    code = ERROR_PAYMENT


class InvalidStateError(OrderServiceError):
    """Raised when an operation is invalid for the order's current state."""

    code = ERROR_STATE


# --- Collaborator interfaces (injected for testability) --------------------

class InMemoryOrderRepository:
    """A simple in-memory order store used as the default repository."""

    def __init__(self) -> None:
        """Initialize an empty repository."""
        self._orders: dict[str, dict[str, Any]] = {}

    def save(self, order: dict[str, Any]) -> None:
        """Persist an order record.

        Args:
            order: The order record keyed later by its ``order_id``.
        """
        self._orders[order["order_id"]] = order

    def get(self, order_id: str) -> Optional[dict[str, Any]]:
        """Fetch an order by id.

        Args:
            order_id: The order identifier to look up.

        Returns:
            The stored order record, or None if absent.
        """
        return self._orders.get(order_id)


class FakePaymentGateway:
    """A default payment gateway that approves every request."""

    def charge(self, token: str, amount: float, correlation_id: str) -> str:
        """Charge a payment token.

        Args:
            token: Opaque payment token from the caller.
            amount: Amount to charge.
            correlation_id: Request trace identifier.

        Returns:
            A payment reference identifier.

        Raises:
            PaymentError: If the gateway declines the charge.
        """
        return f"pay_{uuid.uuid4().hex}"

    def refund(self, payment_ref: str, correlation_id: str) -> str:
        """Refund a prior charge.

        Args:
            payment_ref: Reference returned by a previous charge.
            correlation_id: Request trace identifier.

        Returns:
            A refund reference identifier.

        Raises:
            PaymentError: If the gateway declines the refund.
        """
        return f"refund_{uuid.uuid4().hex}"


# --- Helpers ---------------------------------------------------------------

def _new_correlation_id() -> str:
    """Generate a unique correlation id for request tracing (R-016)."""
    return uuid.uuid4().hex


def _now_iso() -> str:
    """Return the current UTC time as an ISO-8601 string."""
    return datetime.now(timezone.utc).isoformat()


def _error_response(error: OrderServiceError, correlation_id: str) -> dict[str, Any]:
    """Build a structured error response (R-010).

    Args:
        error: The domain error to serialize.
        correlation_id: Request trace identifier.

    Returns:
        A dict with error code, message, timestamp, and correlation id.
    """
    return {
        "error_code": error.code,
        "message": error.message,
        "timestamp": _now_iso(),
        "correlation_id": correlation_id,
    }


def _require_non_empty_str(value: Any, name: str) -> None:
    """Validate that a value is a non-empty string (R-001).

    Args:
        value: The value to check.
        name: Parameter name used in error messages.

    Raises:
        ValidationError: If value is not a non-empty string.
    """
    if not isinstance(value, str) or not value.strip():
        raise ValidationError(f"{name} must be a non-empty string")


def _require_sql_safe(value: str, name: str) -> None:
    """Ensure a string is safe for use in database queries (R-003).

    Args:
        value: The already type-checked string to sanitize-check.
        name: Parameter name used in error messages.

    Raises:
        ValidationError: If value contains disallowed characters.
    """
    if not _SQL_SAFE_PATTERN.match(value):
        raise ValidationError(f"{name} contains illegal characters")


def _extract_item_fields(item: dict[str, Any]) -> tuple[str, int, float]:
    """Pull and type-check the fields of a line item (R-001).

    Args:
        item: The raw item dict.

    Returns:
        A tuple of (name, quantity, unit_price).

    Raises:
        ValidationError: If any field has the wrong type.
    """
    name = item.get("name")
    quantity = item.get("quantity")
    unit_price = item.get("unit_price")
    _require_non_empty_str(name, "item.name")
    if not isinstance(quantity, int) or isinstance(quantity, bool):
        raise ValidationError("item.quantity must be an int")
    if not isinstance(unit_price, (int, float)) or isinstance(unit_price, bool):
        raise ValidationError("item.unit_price must be a number")
    return name, quantity, float(unit_price)


def _validate_item(item: Any) -> dict[str, Any]:
    """Validate a single line item (R-001, R-002, R-004).

    Args:
        item: Candidate item with name, quantity, and unit_price.

    Returns:
        A normalized item dict with HTML-escaped name.

    Raises:
        ValidationError: If the item is malformed or has a negative quantity.
    """
    if not isinstance(item, dict):
        raise ValidationError("each item must be a dict")
    name, quantity, unit_price = _extract_item_fields(item)
    if quantity < MIN_QUANTITY:
        raise ValidationError("quantity must be a positive integer")
    if unit_price < 0:
        raise ValidationError("unit_price must not be negative")
    return {"name": html.escape(name), "quantity": quantity, "unit_price": unit_price}


def _calculate_total(items: list[dict[str, Any]]) -> float:
    """Sum the line totals of validated items (R-027).

    Args:
        items: Normalized items each with quantity and unit_price.

    Returns:
        The order total rounded to the currency precision.
    """
    total = sum(item["quantity"] * item["unit_price"] for item in items)
    return round(total, CURRENCY_PRECISION)


def _validate_items_list(items: Any, config: ServiceConfig) -> list[dict[str, Any]]:
    """Validate and normalize the full items list (R-001).

    Args:
        items: The raw items list from the caller.
        config: Service configuration providing the item cap.

    Returns:
        A list of normalized item dicts.

    Raises:
        ValidationError: If the list is malformed or too large.
    """
    if not isinstance(items, list) or not items:
        raise ValidationError("items must be a non-empty list")
    if len(items) > config.max_items_per_order:
        raise ValidationError("too many items in order")
    return [_validate_item(item) for item in items]


def _check_permission(user_id: str, order: dict[str, Any]) -> None:
    """Verify the caller owns the order before privileged access (R-020).

    Args:
        user_id: The requesting user.
        order: The stored order record.

    Raises:
        PermissionDeniedError: If the user does not own the order.
    """
    if order.get("user_id") != user_id:
        raise PermissionDeniedError("caller does not own this order")


# --- Public API ------------------------------------------------------------

class OrderService:
    """Service exposing order lifecycle operations.

    Collaborators are injected so the service is testable in isolation and so
    all configuration is externalized (R-019).
    """

    def __init__(
        self,
        config: Optional[ServiceConfig] = None,
        repository: Optional[InMemoryOrderRepository] = None,
        payment_gateway: Optional[FakePaymentGateway] = None,
    ) -> None:
        """Construct the service.

        Args:
            config: Service configuration; defaults are used if omitted.
            repository: Order persistence collaborator.
            payment_gateway: Payment processing collaborator.
        """
        self._config = config or ServiceConfig()
        self._repo = repository or InMemoryOrderRepository()
        self._payments = payment_gateway or FakePaymentGateway()

    def create_order(
        self, user_id: str, items: list, payment_token: str
    ) -> dict[str, Any]:
        """Create and pay for a new order.

        Args:
            user_id: Identifier of the ordering user.
            items: Line items, each with name, quantity, and unit_price.
            payment_token: Opaque token authorizing payment.

        Returns:
            An order confirmation dict, or a structured error response.
        """
        correlation_id = _new_correlation_id()
        logger.debug("create_order entry user_id=%s items=%s corr=%s",
                     user_id, len(items) if isinstance(items, list) else -1,
                     correlation_id)
        try:
            result = self._create_order_impl(user_id, items, payment_token,
                                              correlation_id)
            logger.debug("create_order exit return_type=%s corr=%s",
                         type(result).__name__, correlation_id)
            return result
        except OrderServiceError as exc:
            logger.error("create_order failed corr=%s", correlation_id,
                         exc_info=True)
            return _error_response(exc, correlation_id)

    def _create_order_impl(
        self, user_id: str, items: list, payment_token: str, correlation_id: str
    ) -> dict[str, Any]:
        """Validate, charge, persist, and confirm a new order (R-028 helper).

        Args:
            user_id: Identifier of the ordering user.
            items: Raw line items from the caller.
            payment_token: Opaque token authorizing payment.
            correlation_id: Request trace identifier.

        Returns:
            The order confirmation dict.

        Raises:
            ValidationError: If user_id, token, or items are invalid.
            PaymentError: If the charge is declined.
        """
        _require_non_empty_str(user_id, "user_id")
        _require_sql_safe(user_id, "user_id")
        _require_non_empty_str(payment_token, "payment_token")
        normalized = _validate_items_list(items, self._config)
        total = _calculate_total(normalized)
        order_id = f"ord_{uuid.uuid4().hex}"
        payment_ref = self._charge(payment_token, total, correlation_id)
        order = self._persist_new_order(order_id, user_id, normalized, total,
                                        payment_ref)
        logger.info("order created order_id=%s total=%s corr=%s",
                    order_id, total, correlation_id)
        return self._confirmation(order, correlation_id)

    def _charge(self, token: str, total: float, correlation_id: str) -> str:
        """Charge the payment gateway, wrapping failures (R-008).

        Args:
            token: Opaque payment token.
            total: Amount to charge.
            correlation_id: Request trace identifier.

        Returns:
            The payment reference.

        Raises:
            PaymentError: If the gateway raises any error.
        """
        try:
            return self._payments.charge(token, total, correlation_id)
        except Exception as exc:  # noqa: BLE001 - re-raised with context
            raise PaymentError(f"charge failed: {exc}") from exc

    def _persist_new_order(
        self, order_id: str, user_id: str, items: list[dict[str, Any]],
        total: float, payment_ref: str
    ) -> dict[str, Any]:
        """Build and store the order record (R-022 idempotency guard).

        Args:
            order_id: Unique order identifier.
            user_id: Owning user.
            items: Normalized line items.
            total: Order total.
            payment_ref: Reference from the successful charge.

        Returns:
            The stored order record.

        Raises:
            InvalidStateError: If the order id already exists.
        """
        if self._repo.get(order_id) is not None:
            raise InvalidStateError("order already exists")
        order = {
            "order_id": order_id,
            "user_id": user_id,
            "items": items,
            "total": total,
            "status": STATUS_CONFIRMED,
            "payment_ref": payment_ref,
            "created_at": _now_iso(),
        }
        self._repo.save(order)
        return order

    @staticmethod
    def _confirmation(order: dict[str, Any], correlation_id: str) -> dict[str, Any]:
        """Build a public confirmation view of an order.

        Args:
            order: The stored order record.
            correlation_id: Request trace identifier.

        Returns:
            A confirmation dict safe to return to the caller.
        """
        return {
            "order_id": order["order_id"],
            "status": order["status"],
            "total": order["total"],
            "items": order["items"],
            "payment_ref": order.get("payment_ref"),
            "correlation_id": correlation_id,
        }

    def cancel_order(
        self, user_id: str, order_id: str, reason: str
    ) -> dict[str, Any]:
        """Cancel an order and issue a refund.

        Args:
            user_id: Identifier of the requesting user.
            order_id: The order to cancel.
            reason: Human-readable cancellation reason.

        Returns:
            A cancellation confirmation dict, or a structured error response.
        """
        correlation_id = _new_correlation_id()
        logger.debug("cancel_order entry user_id=%s order_id=%s corr=%s",
                     user_id, order_id, correlation_id)
        try:
            result = self._cancel_order_impl(user_id, order_id, reason,
                                             correlation_id)
            logger.debug("cancel_order exit return_type=%s corr=%s",
                         type(result).__name__, correlation_id)
            return result
        except OrderServiceError as exc:
            logger.error("cancel_order failed corr=%s", correlation_id,
                         exc_info=True)
            return _error_response(exc, correlation_id)

    def _cancel_order_impl(
        self, user_id: str, order_id: str, reason: str, correlation_id: str
    ) -> dict[str, Any]:
        """Validate, authorize, refund, and cancel an order (R-028 helper).

        Args:
            user_id: Requesting user.
            order_id: Order to cancel.
            reason: Cancellation reason.
            correlation_id: Request trace identifier.

        Returns:
            The cancellation confirmation dict.

        Raises:
            ValidationError: If inputs are invalid.
            OrderNotFoundError: If the order does not exist.
            PermissionDeniedError: If the user does not own the order.
            InvalidStateError: If the order is already cancelled (R-022).
        """
        order = self._load_owned_order(user_id, order_id)
        _require_non_empty_str(reason, "reason")
        if order["status"] == STATUS_CANCELLED:
            raise InvalidStateError("order already cancelled")
        refund_ref = self._refund(order["payment_ref"], correlation_id)
        order["status"] = STATUS_CANCELLED
        order["cancel_reason"] = html.escape(reason)
        order["refund_ref"] = refund_ref
        self._repo.save(order)
        logger.info("order cancelled order_id=%s corr=%s", order_id,
                    correlation_id)
        return {
            "order_id": order_id,
            "status": STATUS_CANCELLED,
            "refund_ref": refund_ref,
            "correlation_id": correlation_id,
        }

    def _refund(self, payment_ref: str, correlation_id: str) -> str:
        """Refund a charge, wrapping failures with context (R-008).

        Args:
            payment_ref: The original charge reference.
            correlation_id: Request trace identifier.

        Returns:
            The refund reference.

        Raises:
            PaymentError: If the gateway raises any error.
        """
        try:
            return self._payments.refund(payment_ref, correlation_id)
        except Exception as exc:  # noqa: BLE001 - re-raised with context
            raise PaymentError(f"refund failed: {exc}") from exc

    def get_order_status(self, user_id: str, order_id: str) -> dict[str, Any]:
        """Retrieve the status of an order.

        Args:
            user_id: Identifier of the requesting user.
            order_id: The order to inspect.

        Returns:
            An order status dict, or a structured error response.
        """
        correlation_id = _new_correlation_id()
        logger.debug("get_order_status entry user_id=%s order_id=%s corr=%s",
                     user_id, order_id, correlation_id)
        try:
            order = self._load_owned_order(user_id, order_id)
            result = {
                "order_id": order["order_id"],
                "status": order["status"],
                "total": order["total"],
                "correlation_id": correlation_id,
            }
            logger.debug("get_order_status exit return_type=%s corr=%s",
                         type(result).__name__, correlation_id)
            return result
        except OrderServiceError as exc:
            logger.error("get_order_status failed corr=%s", correlation_id,
                         exc_info=True)
            return _error_response(exc, correlation_id)

    def _load_owned_order(self, user_id: str, order_id: str) -> dict[str, Any]:
        """Load an order after validating ownership (R-001, R-003, R-020).

        Args:
            user_id: Requesting user.
            order_id: Order identifier to load.

        Returns:
            The stored order record.

        Raises:
            ValidationError: If identifiers are malformed.
            OrderNotFoundError: If no such order exists.
            PermissionDeniedError: If the user does not own the order.
        """
        _require_non_empty_str(user_id, "user_id")
        _require_non_empty_str(order_id, "order_id")
        _require_sql_safe(user_id, "user_id")
        _require_sql_safe(order_id, "order_id")
        order = self._repo.get(order_id)
        if order is None:
            raise OrderNotFoundError(f"order {order_id} not found")
        _check_permission(user_id, order)
        return order

"""Unit tests for order_service."""

import unittest

from order_service import (
    ERROR_NOT_FOUND,
    ERROR_PERMISSION,
    ERROR_VALIDATION,
    STATUS_CANCELLED,
    STATUS_CONFIRMED,
    FakePaymentGateway,
    InMemoryOrderRepository,
    OrderService,
    PaymentError,
    ServiceConfig,
)


class _ExplodingGateway(FakePaymentGateway):
    """Gateway whose charge always fails, to exercise error paths."""

    def charge(self, token, amount, correlation_id):
        raise RuntimeError("network down")


def _service():
    """Build a service with isolated in-memory collaborators."""
    return OrderService(
        config=ServiceConfig(),
        repository=InMemoryOrderRepository(),
        payment_gateway=FakePaymentGateway(),
    )


def _items():
    """Return a valid sample item list."""
    return [
        {"name": "Widget", "quantity": 2, "unit_price": 9.99},
        {"name": "Gadget", "quantity": 1, "unit_price": 19.50},
    ]


class CreateOrderTests(unittest.TestCase):
    def test_creates_order_with_correct_total(self):
        svc = _service()
        result = svc.create_order("user1", _items(), "tok_abc")
        self.assertEqual(result["status"], STATUS_CONFIRMED)
        self.assertEqual(result["total"], round(2 * 9.99 + 19.50, 2))
        self.assertTrue(result["order_id"].startswith("ord_"))
        self.assertIn("correlation_id", result)

    def test_escapes_html_in_item_name(self):
        svc = _service()
        items = [{"name": "<script>x</script>", "quantity": 1, "unit_price": 1.0}]
        result = svc.create_order("user1", items, "tok")
        self.assertEqual(result["items"][0]["name"], "&lt;script&gt;x&lt;/script&gt;")

    def test_rejects_negative_quantity(self):
        svc = _service()
        items = [{"name": "X", "quantity": -1, "unit_price": 1.0}]
        result = svc.create_order("user1", items, "tok")
        self.assertEqual(result["error_code"], ERROR_VALIDATION)

    def test_rejects_empty_items(self):
        svc = _service()
        result = svc.create_order("user1", [], "tok")
        self.assertEqual(result["error_code"], ERROR_VALIDATION)

    def test_rejects_unsafe_user_id(self):
        svc = _service()
        result = svc.create_order("user'; DROP TABLE--", _items(), "tok")
        self.assertEqual(result["error_code"], ERROR_VALIDATION)

    def test_rejects_missing_payment_token(self):
        svc = _service()
        result = svc.create_order("user1", _items(), "")
        self.assertEqual(result["error_code"], ERROR_VALIDATION)

    def test_payment_failure_returns_structured_error(self):
        svc = OrderService(payment_gateway=_ExplodingGateway())
        result = svc.create_order("user1", _items(), "tok")
        self.assertEqual(result["error_code"], "PAYMENT_FAILED")
        self.assertIn("timestamp", result)

        status = svc.get_order_status("user1", created["order_id"])

        result = svc.get_order_status("intruder", created["order_id"])

class CancelOrderTests(unittest.TestCase):
    def test_cancels_and_refunds(self):
        svc = _service()
        created = svc.create_order("user1", _items(), "tok")
        result = svc.cancel_order("user1", created["order_id"], "changed mind")
        self.assertEqual(result["status"], STATUS_CANCELLED)
        self.assertTrue(result["refund_ref"].startswith("refund_"))

    def test_double_cancel_is_rejected(self):
        svc = _service()
        created = svc.create_order("user1", _items(), "tok")
        svc.cancel_order("user1", created["order_id"], "reason")
        result = svc.cancel_order("user1", created["order_id"], "again")
        self.assertEqual(result["error_code"], "INVALID_STATE")

    def test_cancel_unknown_order(self):
        svc = _service()

**Notes on rule compliance:** Every function has type hints (R-025) and Args/Returns/Raises docstrings (R-026), stays under 30 lines (R-028), and uses named constants instead of magic numbers (R-027). Inputs are type-validated (R-001), negative quantities rejected (R-002), identifiers checked against a SQL-safe pattern (R-003), and item names HTML-escaped (R-004). Errors use domain-specific exception classes (R-007) re-raised with `from exc` (R-008), logged at ERROR with stack traces (R-009), and returned as structured `{error_code, message, timestamp, correlation_id}` responses (R-010). Function entry/exit log at DEBUG (R-013/R-014), state changes at INFO (R-015), all carrying a `correlation_id` (R-016). Config is externalized in `ServiceConfig` (R-019), permissions are checked before privileged reads/writes (R-020), and idempotency guards prevent duplicate creates and double-cancels (R-022).

One thing I couldn't fully satisfy: **R-021 (release resources in `finally` blocks)** — this in-memory implementation allocates no closable resources (DB connections, file handles), so there's nothing to release. When you wire in a real database/payment client, the `_charge`/`_load_owned_order` paths are where you'd add `try/finally` connection cleanup.
