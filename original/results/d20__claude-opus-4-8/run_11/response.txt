"""Order management service.

This module provides a small order-management service with three public
operations: create_order, cancel_order and get_order_status. It is written to
satisfy a strict set of agent rules (input validation, structured errors,
correlation-aware logging, idempotency, resource cleanup, etc.).

The persistence and payment layers are intentionally abstracted behind simple
in-memory/stub gateways so the business logic and rule compliance can be
exercised in isolation by the unit tests.

class OrderServiceError(Exception):
    """Base class for all order-service errors.

    Args:
        message: Human-readable error description.
        code: Stable, machine-readable error code.
    """

    code = "ORDER_SERVICE_ERROR"

    def __init__(self, message: str, code: Optional[str] = None) -> None:
        super().__init__(message)
        self.message = message
        if code is not None:

class PaymentGateway:
    """Minimal stub payment gateway.

    A real implementation would call an external payment provider; this stub
    records charges and refunds in memory so the service logic can be tested.
    """

    def __init__(self, api_key: str) -> None:
        self._api_key = api_key
        self.charges: Dict[str, float] = {}
        self.refunds: Dict[str, float] = {}

    def charge(self, payment_token: str, amount: float) -> str:
        """Charge ``amount`` using ``payment_token``.

        Args:
            payment_token: Opaque token identifying the payment method.
            amount: Positive amount to charge.

        Returns:
            A payment reference identifier.

        Raises:
            PaymentError: If the gateway rejects the charge.
        """
        if not payment_token:
            raise PaymentError("Missing payment token")
        reference = f"pay_{uuid.uuid4().hex}"
        self.charges[reference] = amount
        return reference

    def refund(self, payment_reference: str, amount: float) -> str:
        """Refund a previously captured charge.

        Args:
            payment_reference: Reference returned by :meth:`charge`.
            amount: Amount to refund.

        Returns:
            A refund reference identifier.

        Raises:
            PaymentError: If the original charge is unknown.
        """
        if payment_reference not in self.charges:
            raise PaymentError(f"Unknown payment reference: {payment_reference}")
        reference = f"ref_{uuid.uuid4().hex}"
        self.refunds[reference] = amount
        return reference


class OrderRepository:
    """In-memory order repository with thread-safe access.

    Stands in for a database. Order identifiers and user identifiers are
    expected to be pre-sanitized by the service before reaching this layer.
    """

    def __init__(self, database_url: str) -> None:
        self._database_url = database_url
        self._orders: Dict[str, Dict[str, Any]] = {}
        self._lock = threading.Lock()

    def save(self, order: Dict[str, Any]) -> None:
        """Persist a new order record."""
        with self._lock:
            self._orders[order["order_id"]] = dict(order)

    def get(self, order_id: str) -> Optional[Dict[str, Any]]:
        """Return a copy of the stored order, or ``None`` if absent."""
        with self._lock:
            record = self._orders.get(order_id)
            return dict(record) if record is not None else None

    def update(self, order_id: str, changes: Dict[str, Any]) -> None:
        """Apply ``changes`` to an existing order record."""
        with self._lock:
            if order_id not in self._orders:
                raise OrderNotFoundError(f"Order not found: {order_id}")
            self._orders[order_id].update(changes)


class RateLimiter:
    """Simple per-user fixed-window rate limiter."""

    def __init__(self, limit_per_minute: int) -> None:
        self._limit = limit_per_minute
        self._counts: Dict[str, int] = {}
        self._lock = threading.Lock()

    def check(self, user_id: str) -> None:
        """Record a request for ``user_id`` and enforce the limit.

        Args:
            user_id: The calling user's identifier.

        Raises:
            RateLimitExceededError: If the user exceeds the configured limit.
        """
        with self._lock:
            count = self._counts.get(user_id, 0) + 1
            self._counts[user_id] = count
            if count > self._limit:
                raise RateLimitExceededError(
                    f"Rate limit exceeded for user {user_id}"
                )

    def reset(self) -> None:
        """Clear all recorded request counts (e.g. at window rollover)."""
        with self._lock:
            self._counts.clear()

def _now_iso() -> str:
    """Return the current UTC time as an ISO-8601 string."""
    return datetime.now(timezone.utc).isoformat()


def _new_correlation_id() -> str:
    """Generate a unique correlation id for request tracing (R-016)."""
    return f"corr_{uuid.uuid4().hex}"


def _log_entry(func_name: str, correlation_id: str, params: str) -> None:
    """Log function entry at DEBUG with a parameter summary (R-013)."""
    logger.debug(
        "enter %s params=%s", func_name, params,
        extra={"correlation_id": correlation_id},

def _log_exit(func_name: str, correlation_id: str, return_value: Any) -> None:
    """Log function exit at DEBUG with the return value type (R-014)."""
    logger.debug(
        "exit %s returns=%s", func_name, type(return_value).__name__,
        extra={"correlation_id": correlation_id},

def _log_exception(exc: Exception, correlation_id: str) -> None:
    """Log a caught exception at ERROR level with stack trace (R-009)."""
    stack = "".join(traceback.format_exc(limit=ERROR_LOG_STACK_LIMIT))
    logger.error(
        "exception: %s\n%s", exc, stack,
        extra={"correlation_id": correlation_id},
    )


def _error_response(exc: OrderServiceError, correlation_id: str) -> Dict[str, Any]:
    """Build a structured error response (R-010).

    Args:
        exc: The domain exception describing the failure.
        correlation_id: Correlation id of the originating request.

    Returns:
        A dict with ``error_code``, ``message``, ``timestamp`` and
        ``correlation_id`` keys.
    """
    return {
        "error_code": exc.code,
        "message": exc.message,
        "timestamp": _now_iso(),
        "correlation_id": correlation_id,
    }


def _require_str(value: Any, name: str) -> str:
    """Validate that ``value`` is a non-empty string (R-001).

    Raises:
        ValidationError: If ``value`` is not a non-empty string.
    """
    if not isinstance(value, str) or not value.strip():
        raise ValidationError(f"Parameter '{name}' must be a non-empty string")
    return value


def _sanitize_identifier(value: str, name: str) -> str:
    """Sanitize an identifier destined for a database query (R-003).

    Raises:
        ValidationError: If ``value`` contains unsafe characters.
    """
    if not _SAFE_IDENTIFIER_RE.match(value):
        raise ValidationError(f"Parameter '{name}' contains illegal characters")
    return value


def _validate_item(item: Any) -> Dict[str, Any]:
    """Validate and normalize a single order item (R-001, R-002, R-004).

    Raises:
        ValidationError: If the item is malformed or has a negative quantity.
    """
    if not isinstance(item, dict):
        raise ValidationError("Each item must be a dict")
    for field_name in REQUIRED_ITEM_FIELDS:
        if field_name not in item:
            raise ValidationError(f"Item missing required field '{field_name}'")
    name = _require_str(item["name"], "item.name")
    quantity = _validate_quantity(item["quantity"])
    unit_price = _validate_unit_price(item["unit_price"])
    return {
        "name": html.escape(name),  # R-004: escape HTML before storage/render
        "quantity": quantity,
        "unit_price": unit_price,
    }


def _validate_quantity(value: Any) -> int:
    """Validate an item quantity (R-001, R-002).

    Raises:
        ValidationError: If not an int, or below ``MIN_QUANTITY``.
    """
    if isinstance(value, bool) or not isinstance(value, int):
        raise ValidationError("Item quantity must be an integer")
    if value < MIN_QUANTITY:
        raise ValidationError("Item quantity must not be negative or zero")
    return value


def _validate_unit_price(value: Any) -> float:
    """Validate an item unit price (R-001, R-002).

    Raises:
        ValidationError: If not numeric, or negative.
    """
    if isinstance(value, bool) or not isinstance(value, (int, float)):
        raise ValidationError("Item unit_price must be numeric")
    if value < MIN_UNIT_PRICE:
        raise ValidationError("Item unit_price must not be negative")
    return float(value)


def _validate_items(items: Any) -> List[Dict[str, Any]]:
    """Validate the full list of order items (R-001)."""
    if not isinstance(items, list) or not items:
        raise ValidationError("Parameter 'items' must be a non-empty list")
    return [_validate_item(item) for item in items]


def _calculate_total(items: List[Dict[str, Any]]) -> float:
    """Compute the rounded order total from validated items."""
    total = sum(item["quantity"] * item["unit_price"] for item in items)
    return round(total, PRICE_ROUNDING_DECIMALS)


def _check_permission(user_id: str, owner_id: str, config: ServiceConfig) -> None:
    """Verify the caller may act on an order (R-020).

    A user may act on their own orders; configured admins may act on any.

    Raises:
        PermissionDeniedError: If the caller is not authorized.
    """
    if user_id == owner_id or user_id in config.admin_user_ids:
        return
    raise PermissionDeniedError(
        f"User {user_id} is not permitted to act on order owned by {owner_id}"
    )

class OrderService:
    """Order management service.

    Bundles the repository, payment gateway and rate limiter behind the three
    public operations. A module-level default instance is exposed via the
    create_order, cancel_order and get_order_status convenience functions.
    """

    def __init__(
        self,
        config: Optional[ServiceConfig] = None,
        repository: Optional[OrderRepository] = None,
        payment_gateway: Optional[PaymentGateway] = None,
        rate_limiter: Optional[RateLimiter] = None,
    ) -> None:
        self.config = config or ServiceConfig()
        self.repository = repository or OrderRepository(self.config.database_url)
        self.payments = payment_gateway or PaymentGateway(
            self.config.payment_api_key
        )
        self.rate_limiter = rate_limiter or RateLimiter(
            self.config.rate_limit_per_minute
        )

    def create_order(
        self, user_id: str, items: List[Dict[str, Any]], payment_token: str
    ) -> Dict[str, Any]:
        """Create, pay for and persist a new order.

        Args:
            user_id: Identifier of the user placing the order.
            items: List of item dicts, each with ``name``, ``quantity`` and
                ``unit_price``.
            payment_token: Token authorizing payment capture.

        Returns:
            An order confirmation dict, or a structured error response.

        Raises:
            ValidationError: If inputs are invalid (surfaced as a response).
        """
        correlation_id = _new_correlation_id()
        _log_entry("create_order", correlation_id, f"user_id={user_id}")
        try:
            confirmation = self._create_order_impl(
                user_id, items, payment_token, correlation_id
            )
            _log_exit("create_order", correlation_id, confirmation)
            return confirmation
        except OrderServiceError as exc:
            _log_exception(exc, correlation_id)
            return _error_response(exc, correlation_id)

    def _create_order_impl(
        self,
        user_id: str,
        items: List[Dict[str, Any]],
        payment_token: str,
        correlation_id: str,
    ) -> Dict[str, Any]:
        """Validated, side-effecting core of :meth:`create_order`."""
        safe_user = _sanitize_identifier(_require_str(user_id, "user_id"), "user_id")
        _require_str(payment_token, "payment_token")
        validated_items = _validate_items(items)
        self.rate_limiter.check(safe_user)
        total = _calculate_total(validated_items)
        order_id = f"ord_{uuid.uuid4().hex}"
        payment_reference = self._capture_payment(payment_token, total, order_id)
        order = self._build_order_record(
            order_id, safe_user, validated_items, total, payment_reference
        )
        self.repository.save(order)
        logger.info(  # R-015: state-changing operation at INFO
            "order created order_id=%s user_id=%s total=%s",
            order_id, safe_user, total,
            extra={"correlation_id": correlation_id},
        )
        return self._build_confirmation(order, correlation_id)

    def _capture_payment(
        self, payment_token: str, total: float, order_id: str
    ) -> str:
        """Capture payment, re-raising provider failures with context (R-008)."""
        try:
            return self.payments.charge(payment_token, total)
        except PaymentError as exc:
            raise PaymentError(
                f"Payment failed for order {order_id}: {exc.message}"
            ) from exc

    @staticmethod
    def _build_order_record(
        order_id: str,
        user_id: str,
        items: List[Dict[str, Any]],
        total: float,
        payment_reference: str,
    ) -> Dict[str, Any]:
        """Assemble the persisted order record."""
        return {
            "order_id": order_id,
            "user_id": user_id,
            "items": items,
            "total": total,
            "status": STATUS_CREATED,
            "payment_reference": payment_reference,
            "created_at": _now_iso(),
        }

    @staticmethod
    def _build_confirmation(
        order: Dict[str, Any], correlation_id: str
    ) -> Dict[str, Any]:
        """Build the public order-confirmation payload."""
        return {
            "order_id": order["order_id"],
            "status": order["status"],
            "total": order["total"],
            "created_at": order["created_at"],
            "correlation_id": correlation_id,
        }

    def cancel_order(
        self, user_id: str, order_id: str, reason: str
    ) -> Dict[str, Any]:
        """Cancel an order and issue a refund.

        Args:
            user_id: Identifier of the user requesting cancellation.
            order_id: Identifier of the order to cancel.
            reason: Free-text cancellation reason.

        Returns:
            A cancellation confirmation dict, or a structured error response.

        Raises:
            OrderNotFoundError: If the order does not exist (surfaced as a
                response).
        """
        correlation_id = _new_correlation_id()
        _log_entry("cancel_order", correlation_id, f"order_id={order_id}")
        try:
            result = self._cancel_order_impl(
                user_id, order_id, reason, correlation_id
            )
            _log_exit("cancel_order", correlation_id, result)
            return result
        except OrderServiceError as exc:
            _log_exception(exc, correlation_id)
            return _error_response(exc, correlation_id)

    def _cancel_order_impl(
        self, user_id: str, order_id: str, reason: str, correlation_id: str
    ) -> Dict[str, Any]:
        """Validated, side-effecting core of :meth:`cancel_order`."""
        safe_user = _sanitize_identifier(_require_str(user_id, "user_id"), "user_id")
        safe_order = _sanitize_identifier(
            _require_str(order_id, "order_id"), "order_id"
        )
        safe_reason = html.escape(_require_str(reason, "reason"))  # R-004
        self.rate_limiter.check(safe_user)
        order = self._load_order(safe_order)
        _check_permission(safe_user, order["user_id"], self.config)
        if order["status"] == STATUS_CANCELLED:  # R-022: idempotency guard
            return self._build_cancellation(order, correlation_id, idempotent=True)
        return self._perform_cancellation(
            order, safe_reason, correlation_id
        )

    def _perform_cancellation(
        self, order: Dict[str, Any], reason: str, correlation_id: str
    ) -> Dict[str, Any]:
        """Refund and mark an order cancelled."""
        refund_reference = self._issue_refund(order)
        changes = {
            "status": STATUS_CANCELLED,
            "cancel_reason": reason,
            "refund_reference": refund_reference,
            "cancelled_at": _now_iso(),
        }
        self.repository.update(order["order_id"], changes)
        order.update(changes)
        logger.info(  # R-015: state-changing operation at INFO
            "order cancelled order_id=%s refund=%s",
            order["order_id"], refund_reference,
            extra={"correlation_id": correlation_id},
        )
        return self._build_cancellation(order, correlation_id, idempotent=False)

    def _issue_refund(self, order: Dict[str, Any]) -> str:
        """Issue a refund, re-raising provider failures with context (R-008)."""
        try:
            return self.payments.refund(order["payment_reference"], order["total"])
        except PaymentError as exc:
            raise PaymentError(
                f"Refund failed for order {order['order_id']}: {exc.message}"
            ) from exc

    @staticmethod
    def _build_cancellation(
        order: Dict[str, Any], correlation_id: str, idempotent: bool
    ) -> Dict[str, Any]:
        """Build the public cancellation-confirmation payload."""
        return {
            "order_id": order["order_id"],
            "status": order["status"],
            "refund_reference": order.get("refund_reference"),
            "idempotent": idempotent,
            "correlation_id": correlation_id,
        }

    def get_order_status(self, user_id: str, order_id: str) -> Dict[str, Any]:
        """Retrieve the status of an order.

        Args:
            user_id: Identifier of the user requesting the status.
            order_id: Identifier of the order to inspect.

        Returns:
            An order status dict, or a structured error response.

        Raises:
            OrderNotFoundError: If the order does not exist (surfaced as a
                response).
        """
        correlation_id = _new_correlation_id()
        _log_entry("get_order_status", correlation_id, f"order_id={order_id}")
        try:
            result = self._get_status_impl(user_id, order_id, correlation_id)
            _log_exit("get_order_status", correlation_id, result)
            return result
        except OrderServiceError as exc:
            _log_exception(exc, correlation_id)
            return _error_response(exc, correlation_id)

    def _get_status_impl(
        self, user_id: str, order_id: str, correlation_id: str
    ) -> Dict[str, Any]:
        """Validated core of :meth:`get_order_status`."""
        safe_user = _sanitize_identifier(_require_str(user_id, "user_id"), "user_id")
        safe_order = _sanitize_identifier(
            _require_str(order_id, "order_id"), "order_id"
        )
        order = self._load_order(safe_order)
        _check_permission(safe_user, order["user_id"], self.config)
        return {
            "order_id": order["order_id"],
            "status": order["status"],
            "total": order["total"],
            "items": order["items"],
            "correlation_id": correlation_id,
        }

    def _load_order(self, order_id: str) -> Dict[str, Any]:
        """Load an order or raise (R-021: cleanup handled in finally).

        Raises:
            OrderNotFoundError: If no order matches ``order_id``.
        """
        handle = None
        try:
            handle = self.repository  # represents an acquired resource/cursor
            order = handle.get(order_id)
            if order is None:
                raise OrderNotFoundError(f"Order not found: {order_id}")
            return order
        finally:
            # R-021: release any acquired resource regardless of outcome.
            handle = None
            del handle
