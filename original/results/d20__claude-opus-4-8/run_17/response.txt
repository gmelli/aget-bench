Provides order creation, cancellation, and status retrieval backed by a
pluggable persistence layer and payment gateway. Every public operation
validates its inputs, logs entry/exit/state-changes with a correlation id,
raises domain-specific exceptions, externalizes configuration, and guards
writes for idempotency.

MIN_QUANTITY: int = 1

class _CorrelationIdFilter(logging.Filter):
    """Injects the current correlation id into every log record."""

    def filter(self, record: logging.LogRecord) -> bool:
        record.correlation_id = _correlation_id.get()
        return True


def _build_logger() -> logging.Logger:
    """Construct the module logger with a correlation-id filter.

    Returns:
        A configured ``logging.Logger`` instance.
    """
    logger = logging.getLogger("order_service")
    if not any(isinstance(f, _CorrelationIdFilter) for f in logger.filters):
        logger.addFilter(_CorrelationIdFilter())
    if not logger.handlers:
        handler = logging.StreamHandler()
        fmt = "%(asctime)s %(levelname)s [%(correlation_id)s] %(message)s"
        handler.setFormatter(logging.Formatter(fmt))
        logger.addHandler(handler)
        logger.setLevel(logging.DEBUG)
    return logger


LOGGER = _build_logger()


def _set_correlation_id(correlation_id: Optional[str]) -> str:
    """Set the active correlation id, generating one if absent.

    Args:
        correlation_id: Caller-supplied id, or ``None`` to auto-generate.

    Returns:
        The correlation id that was set.

def _require_non_empty_str(value: Any, field_name: str) -> str:
    """Validate that ``value`` is a non-empty string.

    Args:
        value: The value to check.
        field_name: Name used in error messages.

    Returns:
        The validated string.

    Raises:
        ValidationError: If the value is not a non-empty string.
    """
    if not isinstance(value, str) or not value.strip():
        raise ValidationError(f"{field_name} must be a non-empty string")
    return value


def sanitize_for_query(value: str) -> str:
    """Sanitize a string for safe use in a database query (R-003).

    Args:
        value: Raw string to sanitize.

    Returns:
        The string with characters used in SQL injection removed.
    """
    return _UNSAFE_SQL_PATTERN.sub("", value).strip()


def escape_html_content(value: str) -> str:
    """Escape HTML entities in user-provided content (R-004).

    Args:
        value: Raw user content.

    Returns:
        The HTML-escaped string.
    """
    return html.escape(value, quote=True)


def _validate_item(item: Any, index: int) -> Dict[str, Any]:
    """Validate and normalize a single order item.

    Args:
        item: The candidate item mapping.
        index: Position of the item, used in error messages.

    Returns:
        A normalized item dict with sanitized name and numeric fields.

    Raises:
        ValidationError: If the item is malformed or has invalid values.
    """
    if not isinstance(item, dict):
        raise ValidationError(f"item[{index}] must be a dict")
    for key in ITEM_REQUIRED_FIELDS:
        if key not in item:
            raise ValidationError(f"item[{index}] missing field '{key}'")
    return _normalize_item(item, index)


def _normalize_item(item: Dict[str, Any], index: int) -> Dict[str, Any]:
    """Coerce and range-check an item's fields (R-001, R-002).

    Args:
        item: The validated item mapping.
        index: Position of the item, used in error messages.

    Returns:
        A normalized item dict.

    Raises:
        ValidationError: If quantity/unit_price are wrong typed or negative.
    """
    name = escape_html_content(_require_non_empty_str(item["name"], "name"))
    quantity, unit_price = item["quantity"], item["unit_price"]
    if not isinstance(quantity, int) or isinstance(quantity, bool):
        raise ValidationError(f"item[{index}] quantity must be an int")
    if not isinstance(unit_price, (int, float)) or isinstance(unit_price, bool):
        raise ValidationError(f"item[{index}] unit_price must be numeric")
    if quantity < MIN_QUANTITY:
        raise ValidationError(f"item[{index}] quantity must be >= {MIN_QUANTITY}")
    if unit_price < MIN_UNIT_PRICE:
        raise ValidationError(f"item[{index}] unit_price must be non-negative")
    return {"name": name, "quantity": quantity, "unit_price": float(unit_price)}


def _validate_items(items: Any) -> List[Dict[str, Any]]:
    """Validate the full items collection.

    Args:
        items: Candidate list of item mappings.

    Returns:
        A list of normalized item dicts.

    Raises:
        ValidationError: If ``items`` is not a non-empty list.
    """
    if not isinstance(items, list) or not items:
        raise ValidationError("items must be a non-empty list")
    return [_validate_item(item, i) for i, item in enumerate(items)]


def _calculate_total(items: List[Dict[str, Any]]) -> float:
    """Calculate the order total from normalized items.

    Args:
        items: Normalized item dicts.

    Returns:
        The total price rounded to ``PRICE_ROUNDING_DP`` decimals.
    """
    total = sum(item["quantity"] * item["unit_price"] for item in items)
    return round(total, PRICE_ROUNDING_DP)

# Permission check (R-020)
# --------------------------------------------------------------------------- #
def _check_permission(user_id: str, config: ServiceConfig) -> None:
    """Verify the caller may perform a privileged operation (R-020).

    Args:
        user_id: Identifier of the calling user.
        config: Active service configuration.

    Raises:
        PermissionDeniedError: If the caller is not authorized.
    """
    role = "admin" if user_id.startswith("admin:") else "user"
    if role not in config.authorized_roles:
        raise PermissionDeniedError(f"user '{user_id}' lacks permission")

class InMemoryOrderStore:
    """Thread-safe in-memory order store used as the default backend."""

    def __init__(self) -> None:
        self._orders: Dict[str, Dict[str, Any]] = {}
        self._lock = threading.Lock()

    def save(self, order: Dict[str, Any]) -> None:
        """Persist an order, acquiring and releasing the lock safely.

        Args:
            order: The order record to store.
        """
        self._lock.acquire()
        try:
            self._orders[order["order_id"]] = order
        finally:
            self._lock.release()

    def get(self, order_id: str) -> Optional[Dict[str, Any]]:
        """Retrieve an order by id.

        Args:
            order_id: Identifier of the order.

        Returns:
            The order record, or ``None`` if not present.
        """
        with self._lock:
            return self._orders.get(order_id)


class FakePaymentGateway:
    """Minimal payment gateway stand-in for charges and refunds."""

    def charge(self, payment_token: str, amount: float) -> str:
        """Charge the given amount against a payment token.

        Args:
            payment_token: Tokenized payment instrument.
            amount: Amount to charge.

        Returns:
            A payment reference id.

        Raises:
            PaymentError: If the gateway rejects the charge.
        """
        if payment_token.startswith("decline"):
            raise PaymentError("payment declined by gateway")
        return f"pay_{uuid.uuid4().hex}"

    def refund(self, payment_ref: str, amount: float) -> str:
        """Refund a previously captured payment.

        Args:
            payment_ref: Reference returned by :meth:`charge`.
            amount: Amount to refund.

        Returns:
            A refund reference id.
        """
        return f"ref_{uuid.uuid4().hex}"

def create_order(
    user_id: str,
    items: List[Dict[str, Any]],
    payment_token: str,
    config: Optional[ServiceConfig] = None,
    correlation_id: Optional[str] = None,
    idempotency_key: Optional[str] = None,
) -> Dict[str, Any]:
    """Create a new order, charge payment, and persist it.

    Args:
        user_id: Identifier of the ordering user.
        items: List of items, each with name, quantity, and unit_price.
        payment_token: Tokenized payment instrument.
        config: Optional service configuration; defaults are used if omitted.
        correlation_id: Optional id for request tracing.
        idempotency_key: Optional key to guard against duplicate creation.

    Returns:
        An order confirmation dict, or a structured error response.

    Raises:
        ValidationError: Propagated only when inputs cannot be validated.
    """
    cfg = config or ServiceConfig()
    _set_correlation_id(correlation_id)
    LOGGER.debug("enter create_order(user_id=%s, items=%d)", user_id, len(items or []))
    try:
        confirmation = _do_create_order(user_id, items, payment_token, cfg,
                                        idempotency_key)
        LOGGER.debug("exit create_order -> %s", type(confirmation).__name__)
        return confirmation
    except OrderServiceError as exc:
        LOGGER.error("create_order failed: %s", exc, exc_info=True)
        return _error_response(exc)


def _do_create_order(
    user_id: str,
    items: List[Dict[str, Any]],
    payment_token: str,
    config: ServiceConfig,
    idempotency_key: Optional[str],
) -> Dict[str, Any]:
    """Validate, charge, and persist a new order (core logic).

    Args:
        user_id: Identifier of the ordering user.
        items: Raw items to validate.
        payment_token: Tokenized payment instrument.
        config: Active service configuration.
        idempotency_key: Optional duplicate-suppression key.

    Returns:
        The order confirmation dict.

    Raises:
        OrderServiceError: On validation, permission, or payment failure.
    """
    _require_non_empty_str(user_id, "user_id")
    _require_non_empty_str(payment_token, "payment_token")
    _check_permission(user_id, config)
    existing = _STORE.get(idempotency_key) if idempotency_key else None
    if existing is not None:  # R-022: idempotency guard
        LOGGER.info("idempotent hit for key=%s", idempotency_key)
        return _confirmation_view(existing)
    normalized = _validate_items(items)
    return _persist_new_order(user_id, normalized, payment_token, idempotency_key)


def _persist_new_order(
    user_id: str,
    items: List[Dict[str, Any]],
    payment_token: str,
    idempotency_key: Optional[str],
) -> Dict[str, Any]:
    """Charge payment and store the resulting order record.

    Args:
        user_id: Identifier of the ordering user.
        items: Normalized item dicts.
        payment_token: Tokenized payment instrument.
        idempotency_key: Optional key stored alongside the order id.

    Returns:
        The order confirmation view.

    Raises:
        PaymentError: If the charge is rejected.
    """
    total = _calculate_total(items)
    payment_ref = _GATEWAY.charge(payment_token, total)
    order = _build_order_record(user_id, items, total, payment_ref)
    _STORE.save(order)
    if idempotency_key:
        _STORE.save({**order, "order_id": idempotency_key})
    LOGGER.info("order %s created total=%.2f", order["order_id"], total)
    return _confirmation_view(order)


def _build_order_record(
    user_id: str,
    items: List[Dict[str, Any]],
    total: float,
    payment_ref: str,
) -> Dict[str, Any]:
    """Assemble the canonical persisted order record.

    Args:
        user_id: Identifier of the ordering user.
        items: Normalized item dicts.
        total: Pre-computed order total.
        payment_ref: Reference from the payment charge.

    Returns:
        A fully-populated order record dict.
    """
    return {
        "order_id": f"ord_{uuid.uuid4().hex}",
        "user_id": user_id,
        "items": items,
        "total": total,
        "status": STATUS_CREATED,
        "payment_ref": payment_ref,
        "created_at": _now_iso(),
    }


def _confirmation_view(order: Dict[str, Any]) -> Dict[str, Any]:
    """Project an order record into a client confirmation dict.

    Args:
        order: The stored order record.

    Returns:
        A confirmation dict safe to return to the caller.
    """
    return {
        "order_id": order["order_id"],
        "status": order["status"],
        "total": order["total"],
        "created_at": order["created_at"],
        "correlation_id": _correlation_id.get(),
    }

def _do_cancel_order(
    user_id: str,
    order_id: str,
    reason: str,
    config: ServiceConfig,
) -> Dict[str, Any]:
    """Validate and perform an order cancellation with refund (core logic).

    Args:
        user_id: Identifier of the requesting user.
        order_id: Identifier of the order to cancel.
        reason: Free-text cancellation reason.
        config: Active service configuration.

    Returns:
        The cancellation confirmation dict.

    Raises:
        OrderServiceError: On validation, permission, or lookup failure.
    """
    _require_non_empty_str(user_id, "user_id")
    _require_non_empty_str(order_id, "order_id")
    _check_permission(user_id, config)
    order = _load_owned_order(user_id, order_id)
    if order["status"] == STATUS_CANCELLED:  # R-022: idempotent cancel
        LOGGER.info("order %s already cancelled", order_id)
        return _cancellation_view(order, escape_html_content(reason or ""))
    return _apply_cancellation(order, reason)


def _apply_cancellation(order: Dict[str, Any], reason: str) -> Dict[str, Any]:
    """Refund, mark cancelled, and persist the updated order.

    Args:
        order: The order record to cancel.
        reason: Free-text cancellation reason.

    Returns:
        The cancellation confirmation dict.
    """
    refund_ref = _GATEWAY.refund(order["payment_ref"], order["total"])
    order["status"] = STATUS_CANCELLED
    order["refund_ref"] = refund_ref
    order["cancel_reason"] = escape_html_content(reason or "")
    order["cancelled_at"] = _now_iso()
    _STORE.save(order)
    LOGGER.info("order %s cancelled refund=%s", order["order_id"], refund_ref)
    return _cancellation_view(order, order["cancel_reason"])


def _cancellation_view(order: Dict[str, Any], reason: str) -> Dict[str, Any]:
    """Project a cancelled order into a confirmation dict.

    Args:
        order: The cancelled order record.
        reason: Sanitized cancellation reason.

    Returns:
        A cancellation confirmation dict.
    """
    return {
        "order_id": order["order_id"],
        "status": order["status"],
        "refund_ref": order.get("refund_ref"),
        "reason": reason,
        "correlation_id": _correlation_id.get(),
    }

**How it maps to the 20 rules:**

| Rule | Where |
|------|-------|
| R-001 type validation | `_require_non_empty_str`, `_normalize_item` |
| R-002 reject negatives | `_normalize_item` quantity/price checks |
| R-003 sanitize for queries | `sanitize_for_query`, used in `_load_owned_order` |
| R-004 escape HTML | `escape_html_content`, applied to item names & reasons |
| R-007 domain exceptions | `ValidationError`, `PaymentError`, etc. |
| R-008 error context | re-raises preserve original via the typed hierarchy |
| R-009 log at ERROR + trace | `LOGGER.error(..., exc_info=True)` in each public fn |
| R-010 structured errors | `_error_response` (code, message, timestamp) |
| R-013/R-014 entry/exit DEBUG | `LOGGER.debug` at start/end of public fns |
| R-015 state-change INFO | `LOGGER.info` on create/cancel |
| R-016 correlation_id | `_CorrelationIdFilter` injects into every record |
| R-019 externalized config | `ServiceConfig` |
| R-020 permission check | `_check_permission` |
| R-021 finally release | `InMemoryOrderStore.save` lock in `finally` |
| R-022 idempotency | idempotency_key guard + idempotent cancel |
| R-025/R-026 hints & docstrings | all signatures and public fns |
| R-027 constants | `MIN_QUANTITY`, `STATUS_*`, etc. |
| R-028 <30 lines | public fns delegate to small `_do_*`/helper fns |

A note on a tradeoff I made deliberately: to keep every function under 30 lines (R-028) while still doing validation, logging, permission checks, and idempotency, I split each public function into a thin wrapper plus several private helpers. This satisfies the rule but raises the function count — if you'd prefer fewer, larger functions over the strict line limit, let me know and I'll consolidate.
