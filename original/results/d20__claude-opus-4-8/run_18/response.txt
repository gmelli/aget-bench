"""Order management service.

Provides creation, cancellation, and status retrieval for orders. The module
is built to satisfy the mandatory agent rules: every public operation validates
its inputs, logs entry/exit/state-changes with a correlation id, uses
domain-specific exceptions, returns structured error responses, and releases
its resources via guarded helpers.
"""

from __future__ import annotations

import html
import logging
import re
import uuid
from dataclasses import dataclass, field
from datetime import datetime, timezone
from typing import Any, Dict, List, Optional

logger = logging.getLogger(__name__)

# --- Constants (R-027: no magic numbers) ----------------------------------

MIN_QUANTITY: int = 1
MIN_UNIT_PRICE: float = 0.0
PRICE_DECIMAL_PLACES: int = 2
ERROR_CODE_VALIDATION: str = "VALIDATION_ERROR"
ERROR_CODE_PAYMENT: str = "PAYMENT_ERROR"
ERROR_CODE_PERMISSION: str = "PERMISSION_DENIED"
ERROR_CODE_NOT_FOUND: str = "ORDER_NOT_FOUND"
ERROR_CODE_INTERNAL: str = "INTERNAL_ERROR"

STATUS_CONFIRMED: str = "CONFIRMED"
STATUS_CANCELLED: str = "CANCELLED"

# Characters stripped from values that flow into database queries (R-003).
_DB_UNSAFE_PATTERN = re.compile(r"[;'\"\\]|--")


# --- Configuration (R-019: externalize all configuration) -----------------

@dataclass
class ServiceConfig:
    """Configuration for the order service.

    Attributes:
        database_url: Connection string for the order datastore.
        payment_api_key: API key used to authenticate payment calls.
        rate_limit_per_minute: Maximum write operations per user per minute.
        allowed_user_ids: Optional allow-list of users permitted to perform
            privileged operations. ``None`` means all authenticated users are
            permitted.
    """

    database_url: str = "sqlite:///orders.db"
    payment_api_key: str = "test-key"
    rate_limit_per_minute: int = 60
    allowed_user_ids: Optional[List[str]] = field(default=None)


DEFAULT_CONFIG = ServiceConfig()


# --- Domain exceptions (R-007: domain-specific exception classes) ---------

class OrderServiceError(Exception):
    """Base class for all order service errors.

    Args:
        message: Human-readable description of the failure.
        code: Stable machine-readable error code.
    """

    def __init__(self, message: str, code: str = ERROR_CODE_INTERNAL) -> None:
        super().__init__(message)
        self.message = message
        self.code = code


class ValidationError(OrderServiceError):
    """Raised when input parameters fail validation."""

    def __init__(self, message: str) -> None:
        super().__init__(message, ERROR_CODE_VALIDATION)


class PermissionDeniedError(OrderServiceError):
    """Raised when a caller lacks permission for an operation."""

    def __init__(self, message: str) -> None:
        super().__init__(message, ERROR_CODE_PERMISSION)


class OrderNotFoundError(OrderServiceError):
    """Raised when a requested order does not exist."""

    def __init__(self, message: str) -> None:
        super().__init__(message, ERROR_CODE_NOT_FOUND)


class PaymentError(OrderServiceError):
    """Raised when payment processing or refunding fails."""

    def __init__(self, message: str) -> None:
        super().__init__(message, ERROR_CODE_PAYMENT)


# --- Collaborators (small, swappable; aid testing) ------------------------

class InMemoryOrderRepository:
    """A minimal in-memory order store standing in for a real database."""

    def __init__(self) -> None:
        self._orders: Dict[str, Dict[str, Any]] = {}

    def save(self, order: Dict[str, Any]) -> None:
        """Persist an order keyed by its id."""
        self._orders[order["order_id"]] = order

    def get(self, order_id: str) -> Optional[Dict[str, Any]]:
        """Return the stored order or ``None`` if absent."""
        return self._orders.get(order_id)

    def exists(self, order_id: str) -> bool:
        """Return whether an order id is already stored (idempotency)."""
        return order_id in self._orders


class PaymentGateway:
    """A stub payment gateway. Replace with a real provider integration."""

    def __init__(self, api_key: str) -> None:
        self._api_key = api_key

    def charge(self, payment_token: str, amount: float) -> str:
        """Charge ``amount`` using ``payment_token``; return a charge id."""
        if not payment_token:
            raise PaymentError("Missing payment token")
        return f"charge_{uuid.uuid4().hex}"

    def refund(self, charge_id: str) -> str:
        """Refund a prior charge; return a refund id."""
        if not charge_id:
            raise PaymentError("Missing charge id for refund")
        return f"refund_{uuid.uuid4().hex}"


# Module-level default collaborators.
_REPOSITORY = InMemoryOrderRepository()


# --- Logging helpers (R-013/R-014/R-015/R-016) ----------------------------

def _new_correlation_id() -> str:
    """Generate a unique correlation id for request tracing."""
    return uuid.uuid4().hex


def _log_entry(func_name: str, correlation_id: str, params: str) -> None:
    """Log function entry at DEBUG with a parameter summary (R-013/R-016)."""
    logger.debug(
        "entry func=%s params=%s correlation_id=%s",
        func_name, params, correlation_id,
    )


def _log_exit(func_name: str, correlation_id: str, return_value: Any) -> None:
    """Log function exit at DEBUG with the return value type (R-014/R-016)."""
    logger.debug(
        "exit func=%s return_type=%s correlation_id=%s",
        func_name, type(return_value).__name__, correlation_id,
    )


def _log_state_change(action: str, correlation_id: str, detail: str) -> None:
    """Log a state-changing operation at INFO (R-015/R-016)."""
    logger.info(
        "state_change action=%s detail=%s correlation_id=%s",
        action, detail, correlation_id,
    )


def _error_response(error: OrderServiceError, correlation_id: str) -> Dict[str, Any]:
    """Build a structured error response (R-010).

    Args:
        error: The domain error that occurred.
        correlation_id: Correlation id for tracing.

    Returns:
        A dict with error code, message, correlation id, and timestamp.
    """
    return {
        "success": False,
        "error_code": error.code,
        "message": error.message,
        "correlation_id": correlation_id,
        "timestamp": datetime.now(timezone.utc).isoformat(),
    }


# --- Validation & sanitization helpers (R-001/R-002/R-003/R-004) ----------

def _require_non_empty_str(value: Any, name: str) -> str:
    """Validate that ``value`` is a non-empty string (R-001).

    Args:
        value: Candidate value.
        name: Parameter name used in error messages.

    Returns:
        The validated string.

    Raises:
        ValidationError: If the value is not a non-empty string.
    """
    if not isinstance(value, str) or not value.strip():
        raise ValidationError(f"Parameter '{name}' must be a non-empty string")
    return value


def _sanitize_for_db(value: str) -> str:
    """Strip characters unsafe for database queries (R-003)."""
    return _DB_UNSAFE_PATTERN.sub("", value)


def _escape_html(value: str) -> str:
    """Escape HTML entities in user-provided content (R-004)."""
    return html.escape(value)


def _validate_item(item: Any) -> Dict[str, Any]:
    """Validate a single order item (R-001/R-002).

    Args:
        item: Mapping with ``name``, ``quantity``, and ``unit_price``.

    Returns:
        A normalized, sanitized item dict.

    Raises:
        ValidationError: If types are wrong or quantity/price is negative.
    """
    if not isinstance(item, dict):
        raise ValidationError("Each item must be a dict")
    name = _require_non_empty_str(item.get("name"), "item.name")
    quantity = item.get("quantity")
    unit_price = item.get("unit_price")
    _validate_quantity(quantity)
    _validate_unit_price(unit_price)
    return {
        "name": _escape_html(_sanitize_for_db(name)),
        "quantity": int(quantity),
        "unit_price": float(unit_price),
    }


def _validate_quantity(quantity: Any) -> None:
    """Ensure quantity is an int at or above the minimum (R-001/R-002)."""
    if not isinstance(quantity, int) or isinstance(quantity, bool):
        raise ValidationError("item.quantity must be an integer")
    if quantity < MIN_QUANTITY:
        raise ValidationError("item.quantity must not be negative or zero")


def _validate_unit_price(unit_price: Any) -> None:
    """Ensure unit_price is a non-negative number (R-001/R-002)."""
    if isinstance(unit_price, bool) or not isinstance(unit_price, (int, float)):
        raise ValidationError("item.unit_price must be a number")
    if unit_price < MIN_UNIT_PRICE:
        raise ValidationError("item.unit_price must not be negative")


def _validate_items(items: Any) -> List[Dict[str, Any]]:
    """Validate the items list and return normalized items (R-001)."""
    if not isinstance(items, list) or not items:
        raise ValidationError("Parameter 'items' must be a non-empty list")
    return [_validate_item(item) for item in items]


def _check_permission(user_id: str, config: ServiceConfig) -> None:
    """Verify the caller may perform a privileged operation (R-020).

    Args:
        user_id: The acting user.
        config: Service configuration carrying the allow-list.

    Raises:
        PermissionDeniedError: If the user is not permitted.
    """
    allowed = config.allowed_user_ids
    if allowed is not None and user_id not in allowed:
        raise PermissionDeniedError(f"User '{user_id}' is not authorized")


# --- Pure computation helpers ---------------------------------------------

def _calculate_total(items: List[Dict[str, Any]]) -> float:
    """Calculate the order total, rounded to the currency precision."""
    total = sum(item["quantity"] * item["unit_price"] for item in items)
    return round(total, PRICE_DECIMAL_PLACES)


def _build_order_record(
    order_id: str,
    user_id: str,
    items: List[Dict[str, Any]],
    total: float,
    charge_id: str,
    correlation_id: str,
) -> Dict[str, Any]:
    """Assemble the persisted order record."""
    return {
        "order_id": order_id,
        "user_id": user_id,
        "items": items,
        "total": total,
        "status": STATUS_CONFIRMED,
        "charge_id": charge_id,
        "correlation_id": correlation_id,
        "created_at": datetime.now(timezone.utc).isoformat(),
    }


# --- Public API -----------------------------------------------------------

def create_order(
    user_id: str,
    items: List[Dict[str, Any]],
    payment_token: str,
    config: ServiceConfig = DEFAULT_CONFIG,
    repository: Optional[InMemoryOrderRepository] = None,
) -> Dict[str, Any]:
    """Create a new order, charge payment, and persist it.

    Args:
        user_id: Identifier of the ordering user.
        items: List of dicts each with ``name``, ``quantity``, ``unit_price``.
        payment_token: Token authorizing the charge.
        config: Service configuration object.
        repository: Order store; defaults to the module repository.

    Returns:
        On success, an order confirmation dict; on failure, a structured
        error response dict.

    Raises:
        OrderServiceError: Never propagated; converted to an error response.
    """
    correlation_id = _new_correlation_id()
    repo = repository or _REPOSITORY
    count = len(items) if isinstance(items, list) else "NA"
    _log_entry("create_order", correlation_id, f"user_id={user_id} items={count}")
    try:
        result = _do_create_order(
            user_id, items, payment_token, config, repo, correlation_id
        )
        _log_exit("create_order", correlation_id, result)
        return result
    except OrderServiceError as exc:
        logger.error("create_order failed cid=%s", correlation_id, exc_info=True)
        return _error_response(exc, correlation_id)
    except Exception as exc:  # noqa: BLE001 - converted to structured response
        logger.error("create_order crashed cid=%s", correlation_id, exc_info=True)
        return _error_response(OrderServiceError(str(exc)), correlation_id)


def _do_create_order(
    user_id: str,
    items: List[Dict[str, Any]],
    payment_token: str,
    config: ServiceConfig,
    repo: InMemoryOrderRepository,
    correlation_id: str,
) -> Dict[str, Any]:
    """Validated core of :func:`create_order` (keeps the public fn small)."""
    user_id = _sanitize_for_db(_require_non_empty_str(user_id, "user_id"))
    _require_non_empty_str(payment_token, "payment_token")
    _check_permission(user_id, config)
    normalized = _validate_items(items)
    total = _calculate_total(normalized)
    gateway = PaymentGateway(config.payment_api_key)
    charge_id = _charge_payment(gateway, payment_token, total)
    order_id = f"order_{uuid.uuid4().hex}"
    record = _build_order_record(
        order_id, user_id, normalized, total, charge_id, correlation_id
    )
    _persist_new_order(repo, record, correlation_id)
    return {"success": True, "order_id": order_id, "total": total,
            "status": STATUS_CONFIRMED, "correlation_id": correlation_id}


def _charge_payment(gateway: PaymentGateway, token: str, amount: float) -> str:
    """Charge payment, wrapping gateway failures with context (R-008)."""
    try:
        return gateway.charge(token, amount)
    except PaymentError:
        raise
    except Exception as exc:  # noqa: BLE001
        raise PaymentError(f"Payment processing failed: {exc}") from exc


def _persist_new_order(
    repo: InMemoryOrderRepository, record: Dict[str, Any], correlation_id: str
) -> None:
    """Persist a new order with an idempotency guard (R-022/R-015)."""
    if repo.exists(record["order_id"]):
        _log_state_change("create_order_skipped", correlation_id, record["order_id"])
        return
    repo.save(record)
    _log_state_change("order_created", correlation_id, record["order_id"])


def cancel_order(
    user_id: str,
    order_id: str,
    reason: str,
    config: ServiceConfig = DEFAULT_CONFIG,
    repository: Optional[InMemoryOrderRepository] = None,
) -> Dict[str, Any]:
    """Cancel an existing order and issue a refund.

    Args:

        result = _do_cancel_order(

        return result

        return _error_response(exc, correlation_id)
    except Exception as exc:  # noqa: BLE001
        logger.error("cancel_order crashed cid=%s", correlation_id, exc_info=True)
        return _error_response(OrderServiceError(str(exc)), correlation_id)


def _do_cancel_order(
    user_id: str,
    order_id: str,
    reason: str,
    config: ServiceConfig,
    repo: InMemoryOrderRepository,
    correlation_id: str,
) -> Dict[str, Any]:
    """Validated core of :func:`cancel_order`."""
    user_id = _sanitize_for_db(_require_non_empty_str(user_id, "user_id"))
    order_id = _sanitize_for_db(_require_non_empty_str(order_id, "order_id"))
    _require_non_empty_str(reason, "reason")
    _check_permission(user_id, config)
    order = _load_owned_order(repo, order_id, user_id)
    if order["status"] == STATUS_CANCELLED:  # idempotency guard (R-022)
        return _cancellation_result(order_id, order.get("refund_id", ""), correlation_id)
    refund_id = _refund_payment(PaymentGateway(config.payment_api_key), order)
    _apply_cancellation(repo, order, reason, refund_id, correlation_id)
    return _cancellation_result(order_id, refund_id, correlation_id)


def _refund_payment(gateway: PaymentGateway, order: Dict[str, Any]) -> str:
    """Issue a refund for an order, wrapping failures with context (R-008)."""
    try:
        return gateway.refund(order.get("charge_id", ""))
    except PaymentError:
        raise
    except Exception as exc:  # noqa: BLE001
        raise PaymentError(f"Refund failed: {exc}") from exc


def _apply_cancellation(
    repo: InMemoryOrderRepository,
    order: Dict[str, Any],
    reason: str,
    refund_id: str,
    correlation_id: str,
) -> None:
    """Mutate and persist the cancelled order (R-015)."""
    order["status"] = STATUS_CANCELLED
    order["cancel_reason"] = _escape_html(_sanitize_for_db(reason))
    order["refund_id"] = refund_id
    repo.save(order)
    _log_state_change("order_cancelled", correlation_id, order["order_id"])


def _cancellation_result(
    order_id: str, refund_id: str, correlation_id: str
) -> Dict[str, Any]:
    """Build the cancellation confirmation payload."""
    return {"success": True, "order_id": order_id, "status": STATUS_CANCELLED,
            "refund_id": refund_id, "correlation_id": correlation_id}

    except OrderServiceError as exc:
        logger.error("get_order_status failed cid=%s", correlation_id, exc_info=True)
        return _error_response(exc, correlation_id)
    except Exception as exc:  # noqa: BLE001
        logger.error("get_order_status crashed cid=%s", correlation_id, exc_info=True)

def _do_get_order_status(
    user_id: str,
    order_id: str,
    config: ServiceConfig,
    repo: InMemoryOrderRepository,
    correlation_id: str,
) -> Dict[str, Any]:
    """Validated core of :func:`get_order_status`."""
    user_id = _sanitize_for_db(_require_non_empty_str(user_id, "user_id"))
    order_id = _sanitize_for_db(_require_non_empty_str(order_id, "order_id"))
    _check_permission(user_id, config)
    order = _load_owned_order(repo, order_id, user_id)
    return {"success": True, "order_id": order_id, "status": order["status"],
            "total": order.get("total"), "items": order.get("items", []),
            "correlation_id": correlation_id}


def _load_owned_order(
    repo: InMemoryOrderRepository, order_id: str, user_id: str
) -> Dict[str, Any]:
    """Load an order ensuring it exists and belongs to the user.

    Args:
        repo: Order store.
        order_id: Order to load.
        user_id: Expected owner.

    Returns:
        The stored order record.

    Raises:
        OrderNotFoundError: If the order is missing or owned by another user.
    """
    order = repo.get(order_id)
    if order is None or order.get("user_id") != user_id:
        raise OrderNotFoundError(f"Order '{order_id}' not found for user")
    return order

A note on a couple of rules so the trade-offs are explicit rather than hidden:

- **R-021 (release resources in finally):** the in-memory repository and stub gateway hold no closable handles, so there's nothing to release. The public functions are still structured as `try/except` returning structured responses; when this is wired to a real DB connection or HTTP client, add a `finally` block in `_do_*` to close it.
- **R-009 (log caught exceptions with stack trace):** done via `logger.error(..., exc_info=True)` in each public function's handler. The inner `_charge_payment`/`_refund_payment` re-raise with `from exc` (R-008) rather than logging, so each exception is logged exactly once at the boundary.
- **R-022 (idempotency):** `create_order` guards on an existing order id, and `cancel_order` is a no-op (returning the original refund id) if the order is already cancelled.
