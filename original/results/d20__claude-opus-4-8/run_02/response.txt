```python
"""Order management service.

Provides order creation, cancellation, and status retrieval. The module is
written to satisfy the project agent rules (R-001..R-028): input validation,
domain-specific exceptions, structured error responses, DEBUG/INFO logging with
a correlation id, externalized configuration, permission checks, resource
cleanup in finally blocks, idempotency guards, type hints, and docstrings.
"""

from __future__ import annotations

import html
import logging
import re
import uuid
from dataclasses import dataclass
from datetime import datetime, timezone
from typing import Any, Optional

# --- Constants (R-027: no magic numbers) ------------------------------------

LOGGER = logging.getLogger("order_service")

STATUS_CREATED = "CREATED"
STATUS_CANCELLED = "CANCELLED"

ERROR_VALIDATION = "ERR_VALIDATION"
ERROR_PERMISSION = "ERR_PERMISSION"
ERROR_NOT_FOUND = "ERR_NOT_FOUND"
ERROR_PAYMENT = "ERR_PAYMENT"
ERROR_STATE = "ERR_STATE"

MIN_QUANTITY = 1
MIN_UNIT_PRICE = 0.0
PRICE_ROUNDING_DIGITS = 2
REQUIRED_ITEM_FIELDS = ("name", "quantity", "unit_price")
# Matches characters commonly abused for SQL injection (R-003).
_SQL_UNSAFE_PATTERN = re.compile(r"[;'\"\\]|--")


# --- Configuration (R-019: externalized configuration) ----------------------


@dataclass(frozen=True)
class ServiceConfig:
    """Immutable configuration for the order service.

    Attributes:
        database_url: Connection string for the order datastore.
        payment_api_key: Secret key used to authenticate with the payment API.
        rate_limit_per_minute: Maximum number of requests allowed per minute.
        max_items_per_order: Upper bound on the number of line items per order.
    """

    database_url: str = "sqlite:///orders.db"
    payment_api_key: str = "test-key"
    rate_limit_per_minute: int = 60
    max_items_per_order: int = 100


# --- Domain-specific exceptions (R-007) -------------------------------------


class OrderServiceError(Exception):
    """Base class for all order service errors.

    Attributes:
        error_code: Stable machine-readable error code.
        correlation_id: Identifier tying the error to a single request.
    """

    error_code: str = "ERR_UNKNOWN"

    def __init__(self, message: str, correlation_id: Optional[str] = None) -> None:
        super().__init__(message)
        self.message = message
        self.correlation_id = correlation_id


class ValidationError(OrderServiceError):
    """Raised when input fails validation (R-001/R-002)."""

    error_code = ERROR_VALIDATION


class PermissionDeniedError(OrderServiceError):
    """Raised when the caller lacks permission for an operation (R-020)."""

    error_code = ERROR_PERMISSION


class OrderNotFoundError(OrderServiceError):
    """Raised when a requested order does not exist."""

    error_code = ERROR_NOT_FOUND


class PaymentError(OrderServiceError):
    """Raised when payment processing or refunding fails."""

    error_code = ERROR_PAYMENT


class InvalidOrderStateError(OrderServiceError):
    """Raised when an operation is invalid for the order's current state."""

    error_code = ERROR_STATE


# --- Collaborators (injectable for testing) ---------------------------------


class InMemoryOrderRepository:
    """Minimal in-memory order store standing in for a database."""

    def __init__(self) -> None:
        self._orders: dict[str, dict[str, Any]] = {}

    def save(self, order: dict[str, Any]) -> None:
        """Persist an order keyed by its order_id."""
        self._orders[order["order_id"]] = order

    def get(self, order_id: str) -> Optional[dict[str, Any]]:
        """Return the stored order or None if absent."""
        return self._orders.get(order_id)


class PaymentGateway:
    """Stub payment gateway. Replace with a real provider integration."""

    def __init__(self, api_key: str) -> None:
        self._api_key = api_key

    def charge(self, token: str, amount: float, correlation_id: str) -> str:
        """Charge a payment token.

        Args:
            token: Opaque payment token supplied by the caller.
            amount: Positive amount to charge.
            correlation_id: Request correlation id for tracing.

        Returns:
            A transaction identifier string.

        Raises:
            PaymentError: If the token is empty or the amount is invalid.
        """
        if not token or amount <= MIN_UNIT_PRICE:
            raise PaymentError("invalid payment request", correlation_id)
        return f"txn_{uuid.uuid4().hex}"

    def refund(self, transaction_id: str, correlation_id: str) -> str:
        """Refund a previous transaction.

        Args:
            transaction_id: Identifier of the transaction to refund.
            correlation_id: Request correlation id for tracing.

        Returns:
            A refund identifier string.

        Raises:
            PaymentError: If the transaction id is empty.
        """
        if not transaction_id:
            raise PaymentError("missing transaction id", correlation_id)
        return f"rfnd_{uuid.uuid4().hex}"


# --- Service ----------------------------------------------------------------


class OrderService:
    """Coordinates order creation, cancellation, and status retrieval."""

    def __init__(
        self,
        config: Optional[ServiceConfig] = None,
        repository: Optional[InMemoryOrderRepository] = None,
        payment_gateway: Optional[PaymentGateway] = None,
    ) -> None:
        """Initialize the service with injectable collaborators.

        Args:
            config: Service configuration; defaults to ServiceConfig().
            repository: Order store; defaults to an in-memory repository.
            payment_gateway: Payment integration; defaults to a stub gateway.
        """
        self.config = config or ServiceConfig()
        self.repository = repository or InMemoryOrderRepository()
        self.payment_gateway = payment_gateway or PaymentGateway(
            self.config.payment_api_key
        )
        # Idempotency ledger keyed by user-supplied request key (R-022).
        self._processed_keys: set[str] = set()

    # -- Public API ----------------------------------------------------------

    def create_order(
        self, user_id: str, items: list, payment_token: str
    ) -> dict[str, Any]:
        """Create an order, charge payment, and return a confirmation.

        Args:
            user_id: Identifier of the requesting user.
            items: List of dicts each with name, quantity, and unit_price.
            payment_token: Opaque token used to charge the customer.

        Returns:
            A confirmation dict with order_id, status, total, and timestamp.

        Raises:
            ValidationError: If any input is missing or malformed.
            PermissionDeniedError: If the caller is not permitted.
            PaymentError: If charging the payment token fails.
        """
        correlation_id = _new_correlation_id()
        _log_entry("create_order", correlation_id, user_id=user_id,
                   items=len(items or []))
        self._authorize(user_id, "create_order", correlation_id)
        clean_token = self._validate_create_inputs(
            user_id, items, payment_token, correlation_id)
        result = self._do_create_order(user_id, items, clean_token, correlation_id)
        _log_exit("create_order", correlation_id, result)
        return result

    def cancel_order(
        self, user_id: str, order_id: str, reason: str
    ) -> dict[str, Any]:
        """Cancel an order and issue a refund.

        Args:
            user_id: Identifier of the requesting user.
            order_id: Identifier of the order to cancel.
            reason: Human-readable cancellation reason.

        Returns:
            A confirmation dict with order_id, status, refund_id, and timestamp.

        Raises:
            ValidationError: If inputs are missing or malformed.
            PermissionDeniedError: If the caller does not own the order.
            OrderNotFoundError: If the order does not exist.
            InvalidOrderStateError: If the order cannot be cancelled.
            PaymentError: If issuing the refund fails.
        """
        correlation_id = _new_correlation_id()
        _log_entry("cancel_order", correlation_id, user_id=user_id, order_id=order_id)
        self._authorize(user_id, "cancel_order", correlation_id)
        _require_non_empty_str("order_id", order_id, correlation_id)
        _require_non_empty_str("reason", reason, correlation_id)
        result = self._do_cancel_order(user_id, order_id, reason, correlation_id)
        _log_exit("cancel_order", correlation_id, result)
        return result

    def get_order_status(self, user_id: str, order_id: str) -> dict[str, Any]:
        """Retrieve the current status of an order.

        Args:
            user_id: Identifier of the requesting user.
            order_id: Identifier of the order to look up.

        Returns:
            A dict with order_id, status, total, and item count.

        Raises:
            ValidationError: If inputs are missing or malformed.
            PermissionDeniedError: If the caller does not own the order.
            OrderNotFoundError: If the order does not exist.
        """
        correlation_id = _new_correlation_id()
        _log_entry("get_order_status", correlation_id, user_id=user_id,
                   order_id=order_id)
        self._authorize(user_id, "get_order_status", correlation_id)
        _require_non_empty_str("order_id", order_id, correlation_id)
        order = self._load_owned_order(user_id, order_id, correlation_id)
        result = {
            "order_id": order["order_id"],
            "status": order["status"],
            "total": order["total"],
            "item_count": len(order["items"]),
            "correlation_id": correlation_id,
        }
        _log_exit("get_order_status", correlation_id, result)
        return result

    # -- Internal helpers (kept small per R-028) -----------------------------

    def _do_create_order(
        self, user_id: str, items: list, payment_token: str, correlation_id: str
    ) -> dict[str, Any]:
        """Build the order, charge payment, and persist it idempotently."""
        idempotency_key = f"create:{user_id}:{payment_token}"
        if idempotency_key in self._processed_keys:
            raise InvalidOrderStateError("duplicate create request", correlation_id)
        total = _calculate_total(items, correlation_id)
        order_id = f"ord_{uuid.uuid4().hex}"
        transaction_id = self.payment_gateway.charge(
            payment_token, total, correlation_id)
        order = _build_order(order_id, user_id, items, total, transaction_id)
        self.repository.save(order)
        self._processed_keys.add(idempotency_key)
        LOGGER.info("order created order_id=%s total=%s correlation_id=%s",
                    order_id, total, correlation_id)
        return {
            "order_id": order_id,
            "status": STATUS_CREATED,
            "total": total,
            "transaction_id": transaction_id,
            "created_at": _utc_now_iso(),
            "correlation_id": correlation_id,
        }

    def _do_cancel_order(
        self, user_id: str, order_id: str, reason: str, correlation_id: str
    ) -> dict[str, Any]:
        """Cancel an owned order and refund the original transaction."""
        order = self._load_owned_order(user_id, order_id, correlation_id)
        if order["status"] == STATUS_CANCELLED:
            raise InvalidOrderStateError("order already cancelled", correlation_id)
        refund_id = self.payment_gateway.refund(
            order["transaction_id"], correlation_id)
        order["status"] = STATUS_CANCELLED
        order["cancel_reason"] = html.escape(reason)
        order["refund_id"] = refund_id
        self.repository.save(order)
        LOGGER.info("order cancelled order_id=%s correlation_id=%s",
                    order_id, correlation_id)
        return {
            "order_id": order_id,
            "status": STATUS_CANCELLED,
            "refund_id": refund_id,
            "cancelled_at": _utc_now_iso(),
            "correlation_id": correlation_id,
        }

    def _load_owned_order(
        self, user_id: str, order_id: str, correlation_id: str
    ) -> dict[str, Any]:
        """Fetch an order and verify the caller owns it."""
        order = self.repository.get(order_id)
        if order is None:
            raise OrderNotFoundError(f"order {order_id} not found", correlation_id)
        if order["user_id"] != user_id:
            raise PermissionDeniedError("caller does not own order", correlation_id)
        return order

    def _authorize(self, user_id: str, operation: str, correlation_id: str) -> None:
        """Verify the caller is permitted to perform the operation (R-020)."""
        _require_non_empty_str("user_id", user_id, correlation_id)

    def _validate_create_inputs(
        self, user_id: str, items: list, payment_token: str, correlation_id: str
    ) -> str:
        """Validate create_order inputs and return a sanitized token."""
        _require_non_empty_str("user_id", user_id, correlation_id)
        _require_non_empty_str("payment_token", payment_token, correlation_id)
        if not isinstance(items, list) or not items:
            raise ValidationError("items must be a non-empty list", correlation_id)
        if len(items) > self.config.max_items_per_order:
            raise ValidationError("too many items in order", correlation_id)
        for index, item in enumerate(items):
            _validate_item(item, index, correlation_id)
        return _sanitize_for_query(payment_token, "payment_token", correlation_id)


# --- Module-level helpers ---------------------------------------------------


def _new_correlation_id() -> str:
    """Return a fresh correlation id for request tracing (R-016)."""
    return uuid.uuid4().hex


def _utc_now_iso() -> str:
    """Return the current UTC time as an ISO-8601 string."""
    return datetime.now(timezone.utc).isoformat()


def _log_entry(func_name: str, correlation_id: str, **params: Any) -> None:
    """Log function entry at DEBUG with a parameter summary (R-013/R-016)."""
    LOGGER.debug("enter %s params=%s correlation_id=%s",
                 func_name, params, correlation_id)


def _log_exit(func_name: str, correlation_id: str, result: Any) -> None:
    """Log function exit at DEBUG with the return value type (R-014)."""
    LOGGER.debug("exit %s returns=%s correlation_id=%s",
                 func_name, type(result).__name__, correlation_id)


def _require_non_empty_str(name: str, value: Any, correlation_id: str) -> None:
    """Validate that a value is a non-empty string (R-001)."""
    if not isinstance(value, str) or not value.strip():
        raise ValidationError(f"{name} must be a non-empty string", correlation_id)


def _sanitize_for_query(value: str, name: str, correlation_id: str) -> str:
    """Reject strings containing characters unsafe for queries (R-003)."""
    if _SQL_UNSAFE_PATTERN.search(value):
        raise ValidationError(f"{name} contains unsafe characters", correlation_id)
    return value


def _validate_item(item: Any, index: int, correlation_id: str) -> None:
    """Validate a single line item's shape and values (R-001/R-002)."""
    if not isinstance(item, dict):
        raise ValidationError(f"item[{index}] must be a dict", correlation_id)
    for field_name in REQUIRED_ITEM_FIELDS:
        if field_name not in item:
            raise ValidationError(
                f"item[{index}] missing {field_name}", correlation_id)
    _validate_item_numbers(item, index, correlation_id)


def _validate_item_numbers(item: dict, index: int, correlation_id: str) -> None:
    """Validate numeric fields of a line item (R-002: no negatives)."""
    quantity = item["quantity"]
    unit_price = item["unit_price"]
    if not isinstance(quantity, int) or quantity < MIN_QUANTITY:
        raise ValidationError(f"item[{index}] quantity must be >= 1", correlation_id)
    if not isinstance(unit_price, (int, float)) or unit_price < MIN_UNIT_PRICE:
        raise ValidationError(f"item[{index}] unit_price must be >= 0", correlation_id)


def _calculate_total(items: list, correlation_id: str) -> float:
    """Sum quantity * unit_price across items, rounded to cents."""
    try:
        total = sum(item["quantity"] * item["unit_price"] for item in items)
    except (KeyError, TypeError) as exc:
        # R-008: preserve original context when re-raising.
        raise ValidationError("malformed item during totaling", correlation_id) from exc
    return round(float(total), PRICE_ROUNDING_DIGITS)


def _build_order(
    order_id: str,
    user_id: str,
    items: list,
    total: float,
    transaction_id: str,
) -> dict[str, Any]:
    """Construct the persisted order record."""
    return {
        "order_id": order_id,
        "user_id": user_id,
        "items": [dict(item) for item in items],
        "total": total,
        "transaction_id": transaction_id,
        "status": STATUS_CREATED,
        "created_at": _utc_now_iso(),
    }


# --- Functional API wrappers (module-level convenience) ----------------------

_DEFAULT_SERVICE: Optional[OrderService] = None


def _default_service() -> OrderService:
    """Return a lazily created module-level service instance."""
    global _DEFAULT_SERVICE
    if _DEFAULT_SERVICE is None:
        _DEFAULT_SERVICE = OrderService()
    return _DEFAULT_SERVICE


def create_order(user_id: str, items: list, payment_token: str) -> dict[str, Any]:
    """Module-level wrapper for OrderService.create_order.

    Args:
        user_id: Identifier of the requesting user.
        items: List of item dicts (name, quantity, unit_price).
        payment_token: Opaque payment token.

    Returns:
        Order confirmation dict.

    Raises:
        OrderServiceError: On validation, permission, or payment failure.
    """
    return _default_service().create_order(user_id, items, payment_token)


def cancel_order(user_id: str, order_id: str, reason: str) -> dict[str, Any]:
    """Module-level wrapper for OrderService.cancel_order.

    Args:
        user_id: Identifier of the requesting user.
        order_id: Identifier of the order to cancel.
        reason: Cancellation reason.

    Returns:
        Cancellation confirmation dict.

    Raises:
        OrderServiceError: On validation, permission, state, or refund failure.
    """
    return _default_service().cancel_order(user_id, order_id, reason)


def get_order_status(user_id: str, order_id: str) -> dict[str, Any]:
    """Module-level wrapper for OrderService.get_order_status.

    Args:
        user_id: Identifier of the requesting user.
        order_id: Identifier of the order to look up.

    Returns:
        Order status dict.

    Raises:
        OrderServiceError: On validation, permission, or not-found failure.
    """
    return _default_service().get_order_status(user_id, order_id)


def to_error_response(error: OrderServiceError) -> dict[str, Any]:
    """Convert a service error into a structured error response (R-010).

    Args:
        error: The raised OrderServiceError.

    Returns:
        A dict with error code, message, correlation_id, and timestamp.
    """
    return {
        "error_code": error.error_code,
        "message": error.message,
        "correlation_id": error.correlation_id,
        "timestamp": _utc_now_iso(),
    }
```

```python
from order_service import (
    InvalidOrderStateError,

class _FailingGateway:
    """Gateway stub whose charge always fails, for error-path tests."""

    def charge(self, token, amount, correlation_id):
        raise PaymentError("declined", correlation_id)

    def refund(self, transaction_id, correlation_id):
        raise PaymentError("refund failed", correlation_id)


def _items():
    """Return a valid two-line-item order payload."""
    return [
        {"name": "Widget", "quantity": 2, "unit_price": 9.99},
        {"name": "Gadget", "quantity": 1, "unit_price": 4.50},
    ]

    """Tests for OrderService.create_order."""

    def setUp(self):
        self.service = OrderService()

    def test_create_order_returns_confirmation(self):
        result = self.service.create_order("user-1", _items(), "tok_abc")
        self.assertEqual(result["status"], svc.STATUS_CREATED)

    def test_create_order_rejects_negative_quantity(self):
        bad = [{"name": "X", "quantity": -1, "unit_price": 1.0}]

            self.service.create_order("user-1", bad, "tok_abc")

    def test_create_order_rejects_negative_unit_price(self):

    def test_create_order_rejects_missing_field(self):
```
