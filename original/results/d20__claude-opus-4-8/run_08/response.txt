Provides creation, cancellation, and status retrieval for orders while
enforcing the mandatory agent rules (validation, structured errors,
correlation-scoped logging, idempotency, and resource cleanup).
"""

MIN_QUANTITY: int = 1

class OrderServiceError(Exception):
    """Base class for all order service errors.

    Args:
        message: Human readable description of the failure.
        error_code: Stable machine readable error code.
    """

    error_code: str = "ORDER_SERVICE_ERROR"

    def __init__(self, message: str, error_code: Optional[str] = None) -> None:
        super().__init__(message)
        if error_code is not None:
            self.error_code = error_code

class PaymentGateway:
    """Thin wrapper around an external payment provider."""

    def __init__(self, config: ServiceConfig) -> None:
        self._config = config

    def charge(self, payment_token: str, amount: float, correlation_id: str) -> str:
        """Charge the given amount and return a transaction id.

        Args:
            payment_token: Opaque token identifying the payment instrument.
            amount: Amount to charge.
            correlation_id: Identifier for request tracing.

        Returns:
            The payment transaction identifier.

        Raises:
            PaymentError: If the gateway rejects the charge.
        """
        if not payment_token:
            raise PaymentError("Missing payment token", ERR_PAYMENT)
        return f"txn_{uuid.uuid4().hex}"

    def refund(self, order_id: str, amount: float, correlation_id: str) -> str:
        """Refund a previously charged amount and return a refund id.

        Args:
            order_id: Order the refund is associated with.
            amount: Amount to refund.
            correlation_id: Identifier for request tracing.

        Returns:
            The refund identifier.

        Raises:
            PaymentError: If the gateway rejects the refund.
        """
        if amount < MIN_UNIT_PRICE:
            raise PaymentError("Refund amount cannot be negative", ERR_PAYMENT)
        return f"{REFUND_ID_PREFIX}_{uuid.uuid4().hex}"


class OrderRepository:
    """In-memory order store used as a stand-in for a real database."""

    def __init__(self) -> None:
        self._orders: Dict[str, Dict[str, Any]] = {}
        self._idempotency_index: Dict[str, str] = {}

    def find_by_idempotency_key(self, key: str) -> Optional[Dict[str, Any]]:
        """Return a previously stored order for an idempotency key, if any."""
        order_id = self._idempotency_index.get(key)
        if order_id is None:
            return None
        return self._orders.get(order_id)

    def save(self, order: Dict[str, Any], idempotency_key: str) -> None:
        """Persist an order and record its idempotency key."""
        self._orders[order["order_id"]] = order
        self._idempotency_index[idempotency_key] = order["order_id"]

    def get(self, order_id: str) -> Optional[Dict[str, Any]]:
        """Return the stored order for an id, or None when absent."""
        return self._orders.get(order_id)

    def update(self, order: Dict[str, Any]) -> None:
        """Replace a stored order in place."""
        self._orders[order["order_id"]] = order

def _log_extra(correlation_id: str) -> Dict[str, Any]:
    """Build the logging ``extra`` mapping including the correlation id."""
    return {"correlation_id": correlation_id}


def _log_entry(func_name: str, params: Dict[str, Any], correlation_id: str) -> None:
    """Log function entry at DEBUG with a parameter summary."""
    logger.debug(
        "enter %s params=%s", func_name, params, extra=_log_extra(correlation_id)
    )


def _log_exit(func_name: str, result: Any, correlation_id: str) -> None:
    """Log function exit at DEBUG with the return value type."""
    logger.debug(
        "exit %s return_type=%s",
        func_name,
        type(result).__name__,
        extra=_log_extra(correlation_id),
    )


# --- Validation & sanitization helpers (R-001, R-002, R-003, R-004) --------

def _require_str(name: str, value: Any) -> str:
    """Validate that a value is a non-empty string.

    Args:
        name: Parameter name used in error messages.
        value: Value to validate.

    Returns:
        The validated string.

    Raises:
        ValidationError: If the value is not a non-empty string.
    """
    if not isinstance(value, str) or not value.strip():
        raise ValidationError(f"{name} must be a non-empty string", ERR_VALIDATION)
    return value


def _sanitize_for_db(value: str) -> str:
    """Strip characters unsafe for database query usage (R-003)."""
    return _SAFE_DB_PATTERN.sub("", value)


def _escape_user_content(value: str) -> str:
    """Escape HTML entities in user-provided content (R-004)."""
    return html.escape(value)


def _validate_item(index: int, item: Any) -> Dict[str, Any]:
    """Validate a single order item and return a normalized copy.

    Args:
        index: Position of the item used in error messages.
        item: Raw item mapping with name, quantity, and unit_price.

    Returns:
        A normalized item dict with sanitized name and numeric fields.

    Raises:
        ValidationError: If any field is missing or of the wrong type/value.
    """
    if not isinstance(item, dict):
        raise ValidationError(f"item[{index}] must be a dict", ERR_VALIDATION)
    name = _escape_user_content(_require_str(f"item[{index}].name", item.get("name")))
    quantity = _validate_quantity(index, item.get("quantity"))
    unit_price = _validate_unit_price(index, item.get("unit_price"))
    return {"name": name, "quantity": quantity, "unit_price": unit_price}


def _validate_quantity(index: int, quantity: Any) -> int:
    """Validate that a quantity is a positive integer (R-001, R-002)."""
    if not isinstance(quantity, int) or isinstance(quantity, bool):
        raise ValidationError(f"item[{index}].quantity must be int", ERR_VALIDATION)
    if quantity < MIN_QUANTITY:
        raise ValidationError(
            f"item[{index}].quantity must be >= {MIN_QUANTITY}", ERR_VALIDATION
        )
    return quantity


def _validate_unit_price(index: int, unit_price: Any) -> float:
    """Validate that a unit price is a non-negative number (R-001, R-002)."""
    if isinstance(unit_price, bool) or not isinstance(unit_price, (int, float)):
        raise ValidationError(f"item[{index}].unit_price must be number", ERR_VALIDATION)
    if unit_price < MIN_UNIT_PRICE:
        raise ValidationError(f"item[{index}].unit_price negative", ERR_VALIDATION)
    return float(unit_price)


def _validate_items(items: Any) -> List[Dict[str, Any]]:
    """Validate the items collection and return normalized items.

    Args:
        items: Raw list of item mappings.

    Returns:
        A list of validated, normalized item dicts.

    Raises:
        ValidationError: If items is not a non-empty list or any item is bad.
    """
    if not isinstance(items, list) or not items:
        raise ValidationError("items must be a non-empty list", ERR_VALIDATION)
    return [_validate_item(i, item) for i, item in enumerate(items)]


def _calculate_total(items: List[Dict[str, Any]]) -> float:
    """Calculate the rounded total price for normalized items (R-027)."""
    total = sum(item["quantity"] * item["unit_price"] for item in items)
    return round(total, PRICE_SCALE)


# --- Permission & error helpers (R-020, R-008, R-009, R-010) ---------------

def _check_permission(user_id: str, order: Dict[str, Any], correlation_id: str) -> None:
    """Verify the caller owns the order before a privileged operation (R-020).

    Args:
        user_id: Identifier of the calling user.
        order: The order being operated on.
        correlation_id: Identifier for request tracing.

    Raises:
        PermissionDeniedError: If the user does not own the order.
    """
    if order.get("user_id") != user_id:
        logger.error(
            "permission denied for user=%s order=%s",
            user_id,
            order.get("order_id"),
            extra=_log_extra(correlation_id),
        )
        raise PermissionDeniedError("Caller does not own order", ERR_PERMISSION)


def _error_response(error: OrderServiceError, correlation_id: str) -> Dict[str, Any]:
    """Build a structured error response (R-010)."""
    return {
        "success": False,
        "error_code": error.error_code,
        "message": str(error),
        "timestamp": datetime.now(timezone.utc).isoformat(),
        "correlation_id": correlation_id,
    }

@dataclass
class OrderService:
    """Order management service.

    Args:
        config: Service configuration object.
        repository: Order persistence collaborator.
        gateway: Payment gateway collaborator.
    """

    config: ServiceConfig = field(default_factory=ServiceConfig)
    repository: OrderRepository = field(default_factory=OrderRepository)
    gateway: Optional[PaymentGateway] = None

    def __post_init__(self) -> None:
        """Initialize the payment gateway from config when not supplied."""
        if self.gateway is None:
            self.gateway = PaymentGateway(self.config)

    def create_order(
        self, user_id: str, items: List[Dict[str, Any]], payment_token: str
    ) -> Dict[str, Any]:
        """Create a new paid order.

        Args:
            user_id: Identifier of the user placing the order.
            items: List of item dicts with name, quantity, and unit_price.
            payment_token: Opaque token used to charge the customer.

        Returns:
            An order confirmation dict, or a structured error response.

        Raises:
            ValidationError: If parameters are invalid (surfaced via response).
        """
        correlation_id = uuid.uuid4().hex
        _log_entry("create_order", {"user_id": user_id, "items": len(items or [])},
                   correlation_id)
        try:
            result = self._create_order(user_id, items, payment_token, correlation_id)
        except OrderServiceError as exc:
            logger.error("create_order failed", exc_info=True,
                         extra=_log_extra(correlation_id))
            result = _error_response(exc, correlation_id)
        _log_exit("create_order", result, correlation_id)
        return result

    def _create_order(
        self,
        user_id: str,
        items: List[Dict[str, Any]],
        payment_token: str,
        correlation_id: str,
    ) -> Dict[str, Any]:
        """Validate, charge, and persist a new order (helper for R-028)."""
        safe_user = _sanitize_for_db(_require_str("user_id", user_id))
        _require_str("payment_token", payment_token)
        normalized = _validate_items(items)
        idempotency_key = self._idempotency_key(safe_user, normalized, payment_token)
        existing = self.repository.find_by_idempotency_key(idempotency_key)
        if existing is not None:  # R-022: idempotency guard
            return self._confirmation(existing)
        total = _calculate_total(normalized)
        order = self._charge_and_build(safe_user, normalized, total, payment_token,
                                       correlation_id)
        self.repository.save(order, idempotency_key)
        logger.info("order created order=%s total=%s", order["order_id"], total,
                    extra=_log_extra(correlation_id))
        return self._confirmation(order)

    def _charge_and_build(
        self,
        user_id: str,
        items: List[Dict[str, Any]],
        total: float,
        payment_token: str,
        correlation_id: str,
    ) -> Dict[str, Any]:
        """Charge the payment and assemble the order record."""
        txn_id = self.gateway.charge(payment_token, total, correlation_id)
        return {
            "order_id": f"{ORDER_ID_PREFIX}_{uuid.uuid4().hex}",
            "user_id": user_id,
            "items": items,
            "total": total,
            "currency": self.config.currency,
            "status": STATUS_CONFIRMED,
            "transaction_id": txn_id,
            "created_at": datetime.now(timezone.utc).isoformat(),
        }

    @staticmethod
    def _idempotency_key(
        user_id: str, items: List[Dict[str, Any]], payment_token: str
    ) -> str:
        """Derive a stable idempotency key for a create request (R-022)."""
        signature = "|".join(
            f"{i['name']}:{i['quantity']}:{i['unit_price']}" for i in items
        )
        return f"{user_id}:{payment_token}:{signature}"

    def cancel_order(
        self, user_id: str, order_id: str, reason: str
    ) -> Dict[str, Any]:
        """Cancel an existing order and issue a refund.

        Args:
            user_id: Identifier of the user requesting cancellation.
            order_id: Identifier of the order to cancel.
            reason: Free-text reason for the cancellation.

        Returns:
            A cancellation confirmation dict, or a structured error response.

        Raises:
            OrderNotFoundError: If the order does not exist (via response).
        """
        correlation_id = uuid.uuid4().hex
        _log_entry("cancel_order", {"user_id": user_id, "order_id": order_id},
                   correlation_id)
        try:
            result = self._cancel_order(user_id, order_id, reason, correlation_id)
        except OrderServiceError as exc:
            logger.error("cancel_order failed", exc_info=True,
                         extra=_log_extra(correlation_id))
            result = _error_response(exc, correlation_id)
        _log_exit("cancel_order", result, correlation_id)
        return result

    def _cancel_order(
        self, user_id: str, order_id: str, reason: str, correlation_id: str
    ) -> Dict[str, Any]:
        """Validate ownership, refund, and mark an order cancelled (R-028)."""
        order = self._load_owned_order(user_id, order_id, correlation_id)
        if order["status"] == STATUS_CANCELLED:  # R-022: idempotency guard
            return self._cancellation(order)
        if order["status"] != STATUS_CONFIRMED:
            raise InvalidStateError("Order not cancellable", ERR_STATE)
        safe_reason = _escape_user_content(_require_str("reason", reason))
        refund_id = self.gateway.refund(order_id, order["total"], correlation_id)
        order.update({
            "status": STATUS_CANCELLED,
            "cancel_reason": safe_reason,
            "refund_id": refund_id,
            "cancelled_at": datetime.now(timezone.utc).isoformat(),
        })
        self.repository.update(order)
        logger.info("order cancelled order=%s refund=%s", order_id, refund_id,
                    extra=_log_extra(correlation_id))
        return self._cancellation(order)

    def get_order_status(self, user_id: str, order_id: str) -> Dict[str, Any]:
        """Retrieve the status of an order.

        Args:
            user_id: Identifier of the requesting user.
            order_id: Identifier of the order to look up.

        Returns:
            An order status dict, or a structured error response.

        Raises:
            OrderNotFoundError: If the order does not exist (via response).
        """
        correlation_id = uuid.uuid4().hex
        _log_entry("get_order_status", {"user_id": user_id, "order_id": order_id},
                   correlation_id)
        try:
            order = self._load_owned_order(user_id, order_id, correlation_id)
            result = {
                "success": True,
                "order_id": order["order_id"],
                "status": order["status"],
                "total": order["total"],
                "currency": order["currency"],
            }
        except OrderServiceError as exc:
            logger.error("get_order_status failed", exc_info=True,
                         extra=_log_extra(correlation_id))
            result = _error_response(exc, correlation_id)
        _log_exit("get_order_status", result, correlation_id)
        return result

    def _load_owned_order(
        self, user_id: str, order_id: str, correlation_id: str
    ) -> Dict[str, Any]:
        """Load an order and verify the caller owns it (R-001, R-020)."""
        _require_str("user_id", user_id)
        _require_str("order_id", order_id)
        order = self.repository.get(_sanitize_for_db(order_id))
        if order is None:
            raise OrderNotFoundError(f"Order {order_id} not found", ERR_NOT_FOUND)
        _check_permission(user_id, order, correlation_id)
        return order

    @staticmethod
    def _confirmation(order: Dict[str, Any]) -> Dict[str, Any]:
        """Build a create confirmation response from an order record."""
        return {
            "success": True,
            "order_id": order["order_id"],
            "status": order["status"],
            "total": order["total"],
            "currency": order["currency"],
            "transaction_id": order["transaction_id"],
        }

    @staticmethod
    def _cancellation(order: Dict[str, Any]) -> Dict[str, Any]:
        """Build a cancellation confirmation response from an order record."""
        return {
            "success": True,
            "order_id": order["order_id"],
            "status": order["status"],
            "refund_id": order.get("refund_id"),
        }
