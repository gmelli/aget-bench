"""Order management service.

This module provides functions to create, cancel, and inspect orders. It is
written to satisfy a strict set of agent rules covering input validation,
error handling, structured logging, idempotency, and configuration
externalization.
"""

from __future__ import annotations

import html
import logging
import re
import uuid
from dataclasses import dataclass
from datetime import datetime, timezone
from typing import Any, Dict, List, Optional

# --- Constants (R-027: no magic numbers) -----------------------------------

DEFAULT_DATABASE_URL = "postgresql://localhost:5432/orders"
DEFAULT_PAYMENT_API_KEY = "test-payment-key"
DEFAULT_RATE_LIMIT_PER_MINUTE = 60
DEFAULT_MAX_ITEMS_PER_ORDER = 100

PRICE_DECIMAL_PLACES = 2
MIN_QUANTITY = 1

ORDER_STATUS_CONFIRMED = "CONFIRMED"
ORDER_STATUS_CANCELLED = "CANCELLED"

REQUIRED_ITEM_FIELDS = ("name", "quantity", "unit_price")
# Allow alphanumerics, spaces, and a small set of safe punctuation only.
_SAFE_STRING_PATTERN = re.compile(r"^[\w\s\-.,@:/]+$", re.UNICODE)

logger = logging.getLogger(__name__)


# --- Configuration (R-019: externalized configuration) ---------------------


@dataclass
class ServiceConfig:
    """Configuration for the order service.

    Attributes:
        database_url: Connection string for the order datastore.
        payment_api_key: API key used to authenticate with the payment gateway.
        rate_limit_per_minute: Maximum allowed requests per minute per caller.
        max_items_per_order: Maximum number of line items allowed per order.
    """

    payment_api_key: str = DEFAULT_PAYMENT_API_KEY
    rate_limit_per_minute: int = DEFAULT_RATE_LIMIT_PER_MINUTE
    max_items_per_order: int = DEFAULT_MAX_ITEMS_PER_ORDER


# Default module-level config; callers may override by passing their own.
CONFIG = ServiceConfig()


# --- Domain exceptions (R-007: domain-specific exception classes) ----------


class OrderServiceError(Exception):
    """Base class for all order service errors."""


class ValidationError(OrderServiceError):
    """Raised when input parameters fail validation."""


class PermissionDeniedError(OrderServiceError):
    """Raised when a caller lacks permission for an operation."""


class PaymentError(OrderServiceError):
    """Raised when payment processing or refunding fails."""


class OrderNotFoundError(OrderServiceError):
    """Raised when a requested order does not exist."""


# --- In-memory stores (stand-ins for a real datastore / payment gateway) ---

_ORDERS: Dict[str, Dict[str, Any]] = {}
# Idempotency guard store (R-022): maps an idempotency key to a prior result.
_IDEMPOTENCY_KEYS: Dict[str, Dict[str, Any]] = {}


# --- Helpers ---------------------------------------------------------------


def _new_correlation_id() -> str:
    """Generate a unique correlation id for request tracing (R-016).

    Returns:
        A new hex correlation id string.
    """
    return uuid.uuid4().hex


def _now_iso() -> str:
    """Return the current UTC time as an ISO-8601 string.

    Returns:
        The current time formatted in ISO-8601 with timezone.
    """
    return datetime.now(timezone.utc).isoformat()


def _error_response(code: str, message: str) -> Dict[str, Any]:
    """Build a structured error response (R-010).

    Args:
        code: A machine-readable error code.
        message: A human-readable error message.

    Returns:
        A dict containing error code, message, and timestamp.
    """
    return {"error_code": code, "message": message, "timestamp": _now_iso()}


def _require_str(value: Any, name: str, correlation_id: str) -> str:
    """Validate that a value is a non-empty, safe string (R-001, R-003).

    Args:
        value: The value to validate.
        name: The parameter name, used in error messages.
        correlation_id: Correlation id for log tracing.

    Returns:
        The validated, sanitized string.

    Raises:
        ValidationError: If the value is not a valid, safe string.
    """
    if not isinstance(value, str):
        raise ValidationError(f"{name} must be a string, got {type(value).__name__}")
    stripped = value.strip()
    if not stripped:
        raise ValidationError(f"{name} must not be empty")
    if not _SAFE_STRING_PATTERN.match(stripped):
        # R-003: reject strings that could carry injection payloads.
        logger.warning(
            "Rejected unsafe string for %s", name,
            extra={"correlation_id": correlation_id},
        )
        raise ValidationError(f"{name} contains disallowed characters")
    return stripped


def _validate_item(item: Any, index: int) -> Dict[str, Any]:
    """Validate and normalize a single order line item (R-001, R-002, R-004).

    Args:
        item: The candidate item mapping.
        index: The position of the item in the list, for error messages.

    Returns:
        A normalized item dict with name, quantity, and unit_price.

    Raises:
        ValidationError: If the item is malformed or has invalid values.
    """
    if not isinstance(item, dict):
        raise ValidationError(f"item[{index}] must be a dict")
    for required_field in REQUIRED_ITEM_FIELDS:
        if required_field not in item:
            raise ValidationError(f"item[{index}] missing field '{required_field}'")

    name = item["name"]
    quantity = item["quantity"]
    unit_price = item["unit_price"]

    if not isinstance(name, str) or not name.strip():
        raise ValidationError(f"item[{index}].name must be a non-empty string")
    if not isinstance(quantity, int) or isinstance(quantity, bool):
        raise ValidationError(f"item[{index}].quantity must be an integer")
    if quantity < MIN_QUANTITY:  # R-002: reject negative/zero quantities.
        raise ValidationError(f"item[{index}].quantity must be >= {MIN_QUANTITY}")
    if not isinstance(unit_price, (int, float)) or isinstance(unit_price, bool):
        raise ValidationError(f"item[{index}].unit_price must be numeric")
    if unit_price < 0:  # R-002: reject negative monetary values.
        raise ValidationError(f"item[{index}].unit_price must not be negative")

    return {
        # R-004: escape HTML entities in user-provided content.
        "name": html.escape(name.strip()),
        "quantity": quantity,
        "unit_price": float(unit_price),
    }


def _calculate_total(items: List[Dict[str, Any]]) -> float:
    """Calculate the total price for a list of normalized items.

    Args:
        items: Normalized items each having quantity and unit_price.

    Returns:
        The rounded total price.
    """
    total = sum(item["quantity"] * item["unit_price"] for item in items)
    return round(total, PRICE_DECIMAL_PLACES)


def _check_permission(user_id: str, correlation_id: str) -> None:
    """Verify the caller is permitted to act (R-020).

    Args:
        user_id: The id of the calling user.
        correlation_id: Correlation id for log tracing.

    Raises:
        PermissionDeniedError: If the user is not authorized.
    """
    # Placeholder authorization check; a real implementation would consult an
    # access-control service. Empty/anonymous users are rejected.
    if user_id == "anonymous":
        logger.warning(
            "Permission denied for user %s", user_id,
            extra={"correlation_id": correlation_id},
        )
        raise PermissionDeniedError("Caller is not authorized for this operation")


def _process_payment(
    payment_token: str, amount: float, correlation_id: str
) -> str:
    """Process a payment through the configured gateway.

    Args:
        payment_token: The opaque payment token supplied by the caller.
        amount: The amount to charge.
        correlation_id: Correlation id for log tracing.

    Returns:
        A payment reference id.

    Raises:
        PaymentError: If the payment cannot be processed.
    """
    if amount < 0:
        raise PaymentError("Cannot charge a negative amount")
    if payment_token == "invalid":
        raise PaymentError("Payment token was declined")
    logger.info(
        "Charged %.2f via payment gateway", amount,
        extra={"correlation_id": correlation_id},
    )
    return f"pay_{uuid.uuid4().hex}"


def _process_refund(
    payment_reference: str, amount: float, correlation_id: str
) -> str:
    """Issue a refund through the configured gateway.

    Args:
        payment_reference: The original payment reference id.
        amount: The amount to refund.
        correlation_id: Correlation id for log tracing.

    Returns:
        A refund reference id.

    Raises:
        PaymentError: If the refund cannot be processed.
    """
    if amount < 0:
        raise PaymentError("Cannot refund a negative amount")
    logger.info(
        "Refunded %.2f for payment %s", amount, payment_reference,
        extra={"correlation_id": correlation_id},
    )
    return f"ref_{uuid.uuid4().hex}"


# --- Public API ------------------------------------------------------------


def create_order(
    user_id: str,
    items: List[Dict[str, Any]],
    payment_token: str,
    config: Optional[ServiceConfig] = None,
    idempotency_key: Optional[str] = None,
) -> Dict[str, Any]:
    """Create a new order, charge payment, and return a confirmation.

    Args:
        user_id: The id of the user placing the order.
        items: A list of items, each with name, quantity, and unit_price.
        payment_token: An opaque token used to charge the user.
        config: Optional service configuration; falls back to module CONFIG.
        idempotency_key: Optional key guarding against duplicate creation.

    Returns:
        On success, an order confirmation dict. On failure, a structured
        error response dict with error_code, message, and timestamp.

    Raises:
        ValidationError: If inputs fail validation.
        PermissionDeniedError: If the caller is not authorized.
        PaymentError: If payment processing fails.
    """
    cfg = config or CONFIG
    correlation_id = _new_correlation_id()
    logger.debug(
        "Enter create_order(user_id=%s, item_count=%s)",
        user_id, len(items) if isinstance(items, list) else "n/a",
        extra={"correlation_id": correlation_id},
    )

    try:
        clean_user_id = _require_str(user_id, "user_id", correlation_id)
        clean_token = _require_str(payment_token, "payment_token", correlation_id)
        _check_permission(clean_user_id, correlation_id)

        if not isinstance(items, list) or not items:
            raise ValidationError("items must be a non-empty list")
        if len(items) > cfg.max_items_per_order:
            raise ValidationError(
                f"items exceed maximum of {cfg.max_items_per_order}"
            )

        # R-022: idempotency guard — return prior result for a repeated key.
        if idempotency_key is not None and idempotency_key in _IDEMPOTENCY_KEYS:
            logger.info(
                "Idempotent replay for key %s", idempotency_key,
                extra={"correlation_id": correlation_id},
            )
            return _IDEMPOTENCY_KEYS[idempotency_key]

        normalized = [_validate_item(item, i) for i, item in enumerate(items)]
        total = _calculate_total(normalized)
        payment_reference = _process_payment(clean_token, total, correlation_id)

        order_id = f"ord_{uuid.uuid4().hex}"
        record = {
            "order_id": order_id,
            "user_id": clean_user_id,
            "items": normalized,
            "total": total,
            "status": ORDER_STATUS_CONFIRMED,
            "payment_reference": payment_reference,
            "created_at": _now_iso(),
            "correlation_id": correlation_id,
        }
        _ORDERS[order_id] = record
        logger.info(  # R-015: state-changing operation logged at INFO.
            "Created order %s for user %s total %.2f",
            order_id, clean_user_id, total,
            extra={"correlation_id": correlation_id},
        )

        confirmation = {
            "order_id": order_id,
            "status": ORDER_STATUS_CONFIRMED,
            "total": total,
            "payment_reference": payment_reference,
            "created_at": record["created_at"],
            "correlation_id": correlation_id,
        }
        if idempotency_key is not None:
            _IDEMPOTENCY_KEYS[idempotency_key] = confirmation

        logger.debug(
            "Exit create_order -> %s", type(confirmation).__name__,
            extra={"correlation_id": correlation_id},
        )
        return confirmation
    except OrderServiceError as exc:
        # R-008/R-009: log with context and stack trace, return structured error.
        logger.error(
            "create_order failed: %s", exc, exc_info=True,
            extra={"correlation_id": correlation_id},
        )
        return _error_response(type(exc).__name__, str(exc))


def cancel_order(
    user_id: str,
    order_id: str,
    reason: str,
    config: Optional[ServiceConfig] = None,
) -> Dict[str, Any]:
    """Cancel an existing order and issue a refund.

    Args:
        user_id: The id of the user cancelling the order.
        order_id: The id of the order to cancel.
        reason: A human-readable cancellation reason.
        config: Optional service configuration; falls back to module CONFIG.

    Returns:
        On success, a cancellation confirmation dict. On failure, a structured
        error response dict with error_code, message, and timestamp.

    Raises:
        ValidationError: If inputs fail validation.
        PermissionDeniedError: If the caller is not authorized.
        OrderNotFoundError: If the order does not exist for the user.
        PaymentError: If the refund fails.
    """
    _ = config or CONFIG
    correlation_id = _new_correlation_id()
    logger.debug(
        "Enter cancel_order(user_id=%s, order_id=%s)", user_id, order_id,
        extra={"correlation_id": correlation_id},
    )

    try:
        clean_user_id = _require_str(user_id, "user_id", correlation_id)
        clean_order_id = _require_str(order_id, "order_id", correlation_id)
        clean_reason = _require_str(reason, "reason", correlation_id)
        _check_permission(clean_user_id, correlation_id)

        record = _ORDERS.get(clean_order_id)
        if record is None or record["user_id"] != clean_user_id:
            raise OrderNotFoundError(f"Order {clean_order_id} not found")

        # R-022: idempotency — cancelling an already-cancelled order is a no-op.
        if record["status"] == ORDER_STATUS_CANCELLED:
            logger.info(
                "Order %s already cancelled", clean_order_id,
                extra={"correlation_id": correlation_id},
            )
            return {
                "order_id": clean_order_id,
                "status": ORDER_STATUS_CANCELLED,
                "refund_reference": record.get("refund_reference"),
                "cancelled_at": record.get("cancelled_at"),
                "correlation_id": correlation_id,
            }

        refund_reference = _process_refund(
            record["payment_reference"], record["total"], correlation_id
        )
        record["status"] = ORDER_STATUS_CANCELLED
        record["refund_reference"] = refund_reference
        record["cancellation_reason"] = html.escape(clean_reason)
        record["cancelled_at"] = _now_iso()
        logger.info(  # R-015: state-changing operation logged at INFO.
            "Cancelled order %s for user %s", clean_order_id, clean_user_id,
            extra={"correlation_id": correlation_id},
        )

        confirmation = {
            "order_id": clean_order_id,
            "status": ORDER_STATUS_CANCELLED,
            "refund_reference": refund_reference,
            "cancelled_at": record["cancelled_at"],
            "correlation_id": correlation_id,
        }
        logger.debug(
            "Exit cancel_order -> %s", type(confirmation).__name__,
            extra={"correlation_id": correlation_id},
        )
        return confirmation
    except OrderServiceError as exc:
        logger.error(
            "cancel_order failed: %s", exc, exc_info=True,
            extra={"correlation_id": correlation_id},
        )
        return _error_response(type(exc).__name__, str(exc))
