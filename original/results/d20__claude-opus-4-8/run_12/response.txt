class PaymentGateway:
    """Minimal payment gateway abstraction."""

    def __init__(self, api_key: str) -> None:
        """Initialize the gateway.

        Args:
            api_key: API key used to authenticate calls.
        """
        self._api_key = api_key

    def charge(self, payment_token: str, amount: float) -> str:
        """Charge a payment token.

        Args:
            payment_token: Tokenized payment instrument.
            amount: Amount to charge.

        Returns:
            A transaction identifier.

        Raises:
            PaymentError: If the charge cannot be completed.
        """
        if not payment_token:
            raise PaymentError("Empty payment token")
        return f"txn_{uuid.uuid4().hex}"

    def refund(self, transaction_id: str, amount: float) -> str:
        """Refund a previous charge.

        Args:
            transaction_id: Identifier of the original charge.
            amount: Amount to refund.

        Returns:
            A refund identifier.

        Raises:
            PaymentError: If the refund cannot be completed.
        """
        if not transaction_id:
            raise PaymentError("Missing transaction id for refund")
        return f"rfnd_{uuid.uuid4().hex}"


# --- In-memory store (resource lifecycle illustrated in finally blocks) ----


class OrderStore:
    """Thread-safe in-memory order store standing in for a database."""

    def __init__(self) -> None:
        """Initialize the store."""
        self._orders: Dict[str, Dict[str, Any]] = {}
        self._lock = threading.Lock()

    def acquire(self) -> "OrderStore":
        """Acquire the store lock (resource to be released in finally)."""
        self._lock.acquire()
        return self

    def release(self) -> None:
        """Release the store lock if held."""
        if self._lock.locked():
            self._lock.release()

    def get(self, order_id: str) -> Optional[Dict[str, Any]]:
        """Return a copy of the stored order, or ``None``."""
        record = self._orders.get(order_id)
        return dict(record) if record is not None else None

    def put(self, order_id: str, record: Dict[str, Any]) -> None:
        """Insert or replace an order record."""
        self._orders[order_id] = dict(record)

def _require_non_empty_str(value: Any, name: str) -> str:

def _validate_item(item: Any, index: int) -> Dict[str, Any]:
    """Validate and normalize a single order item (R-001, R-002, R-004).

    Args:

        raise ValidationError(f"item[{index}] must be a dict")

    _validate_quantity(quantity, index)
    _validate_unit_price(unit_price, index)

        "quantity": int(quantity),

    Raises:

def _validate_unit_price(unit_price: Any, index: int) -> None:
    """Validate an item unit price (R-001, R-002).

def _validate_items(items: Any) -> List[Dict[str, Any]]:

    Raises:

        raise ValidationError("items must be a non-empty list")

def _calculate_total(items: List[Dict[str, Any]]) -> float:
    """Sum item quantity * unit_price into a rounded total (R-027)."""
    total = sum(item["quantity"] * item["unit_price"] for item in items)

def create_order(
    user_id: str, items: List[Dict[str, Any]], payment_token: str
) -> Dict[str, Any]:
    """Create a new order, charge payment, and return a confirmation.

    Args:
        user_id: Identifier of the user placing the order.
        items: List of item dicts, each with ``name``, ``quantity`` and
            ``unit_price``.
        payment_token: Tokenized payment instrument used for the charge.

    Returns:
        On success, an order confirmation dict containing the order id, status,
        total, items, transaction id, timestamp, and correlation id. On failure,
        a structured error response dict (R-010).

    Raises:
        OrderServiceError: Domain errors are caught and converted to structured
            error responses rather than propagated.
    """
    correlation_id = _new_correlation_id()
    item_count = len(items) if isinstance(items, list) else "n/a"
    _log_entry("create_order", correlation_id, {"user_id": user_id, "items": item_count})
    store = _store.acquire()
    try:
        result = _create_order_locked(user_id, items, payment_token, correlation_id)
    except OrderServiceError as exc:
        _log_exception(correlation_id, exc)  # R-009
        result = _error_response(exc, correlation_id)  # R-010
    finally:
        store.release()  # R-021: release resources in finally.
    _log_exit("create_order", correlation_id, result)
    return result


def _create_order_locked(
    user_id: str,
    items: List[Dict[str, Any]],
    payment_token: str,
    correlation_id: str,
) -> Dict[str, Any]:
    """Perform validated order creation while holding the store lock.

    Args:
        user_id: Identifier of the user placing the order.
        items: Raw list of item dicts.
        payment_token: Tokenized payment instrument.
        correlation_id: Correlation id for tracing.

    Returns:
        An order confirmation dict.

    Raises:
        ValidationError: If user, token, or items are invalid.
        PermissionDeniedError: If the caller lacks create permission.
        PaymentError: If the charge fails.
    """
    safe_user = _sanitize_for_query(_require_non_empty_str(user_id, "user_id"))
    _require_non_empty_str(payment_token, "payment_token")
    _check_permission(safe_user, PERMISSION_CREATE)  # R-020
    key = _idempotency_key("create", safe_user, payment_token)
    if key in _idempotency_ledger:  # R-022: idempotency guard.
        return dict(_idempotency_ledger[key])
    normalized = _validate_items(items)
    confirmation = _persist_new_order(safe_user, normalized, payment_token, correlation_id)
    _idempotency_ledger[key] = dict(confirmation)
    return confirmation


def _persist_new_order(
    user_id: str,
    items: List[Dict[str, Any]],
    payment_token: str,
    correlation_id: str,
) -> Dict[str, Any]:
    """Charge payment and write the new order record.

    Args:
        user_id: Sanitized user identifier.
        items: Normalized item list.
        payment_token: Tokenized payment instrument.
        correlation_id: Correlation id for tracing.

    Returns:
        An order confirmation dict.

    Raises:
        PaymentError: If the charge fails.
    """
    total = _calculate_total(items)
    order_id = uuid.uuid4().hex
    transaction_id = _payment_gateway.charge(payment_token, total)
    record = {
        "order_id": order_id,
        "user_id": user_id,
        "status": ORDER_STATUS_CONFIRMED,
        "items": items,
        "total": total,
        "transaction_id": transaction_id,
        "created_at": _now_iso(),
        "correlation_id": correlation_id,
    }
    _store.put(order_id, record)
    _log_state_change(f"order created order_id={order_id}", correlation_id)  # R-015
    return dict(record)

def _cancel_order_locked(
    user_id: str, order_id: str, reason: str, correlation_id: str
) -> Dict[str, Any]:
    """Perform validated cancellation while holding the store lock.

    Args:
        user_id: Identifier of the user cancelling the order.
        order_id: Identifier of the order to cancel.
        reason: Free-text cancellation reason.
        correlation_id: Correlation id for tracing.

    Returns:
        A cancellation confirmation dict.

    Raises:
        ValidationError: If user, order, or reason are invalid.
        PermissionDeniedError: If the caller lacks cancel permission.
        OrderNotFoundError: If the order is missing or not owned by the user.
        PaymentError: If the refund fails.
    """
    safe_user = _sanitize_for_query(_require_non_empty_str(user_id, "user_id"))
    safe_order = _sanitize_for_query(_require_non_empty_str(order_id, "order_id"))
    _require_non_empty_str(reason, "reason")
    _check_permission(safe_user, PERMISSION_CANCEL)  # R-020
    record = _load_owned_order(safe_user, safe_order)
    if record["status"] == ORDER_STATUS_CANCELLED:  # R-022: idempotent cancel.
        return _cancellation_confirmation(record, reason, correlation_id)
    return _apply_cancellation(record, reason, correlation_id)


def _apply_cancellation(
    record: Dict[str, Any], reason: str, correlation_id: str
) -> Dict[str, Any]:
    """Refund and mark an order cancelled.

    Args:
        record: The existing order record.
        reason: Cancellation reason (HTML-escaped before storage).
        correlation_id: Correlation id for tracing.

    Returns:
        A cancellation confirmation dict.

    Raises:
        PaymentError: If the refund fails.
    """
    refund_id = _payment_gateway.refund(record["transaction_id"], record["total"])
    record["status"] = ORDER_STATUS_CANCELLED
    record["refund_id"] = refund_id
    record["cancellation_reason"] = html.escape(reason)  # R-004
    record["cancelled_at"] = _now_iso()
    _store.put(record["order_id"], record)
    _log_state_change(f"order cancelled order_id={record['order_id']}", correlation_id)
    return _cancellation_confirmation(record, reason, correlation_id)


def _cancellation_confirmation(
    record: Dict[str, Any], reason: str, correlation_id: str

    """Build a cancellation confirmation dict from a record."""
    return {
        "order_id": record["order_id"],
        "status": ORDER_STATUS_CANCELLED,
        "refund_id": record.get("refund_id"),
        "reason": html.escape(reason),
        "timestamp": _now_iso(),
        "correlation_id": correlation_id,
    }

def _get_status_locked(
    user_id: str, order_id: str, correlation_id: str
) -> Dict[str, Any]:
    """Validate input and build an order status dict (read-only path).

    Args:
        user_id: Identifier of the requesting user.
        order_id: Identifier of the order to look up.
        correlation_id: Correlation id for tracing.

    Returns:
        An order status dict.

    Raises:
        ValidationError: If user or order id are invalid.
        PermissionDeniedError: If the caller lacks read permission.
        OrderNotFoundError: If the order is missing or not owned by the user.
    """
    safe_user = _sanitize_for_query(_require_non_empty_str(user_id, "user_id"))
    safe_order = _sanitize_for_query(_require_non_empty_str(order_id, "order_id"))
    _check_permission(safe_user, PERMISSION_READ)  # R-020
    record = _load_owned_order(safe_user, safe_order)
    return {
        "order_id": record["order_id"],
        "status": record["status"],
        "total": record["total"],
        "items": record["items"],

        "correlation_id": correlation_id,

def _load_owned_order(user_id: str, order_id: str) -> Dict[str, Any]:
    """Load an order and verify ownership.

    Args:
        user_id: Sanitized identifier of the requesting user.
        order_id: Sanitized identifier of the order.

    Returns:
        The order record.

    Raises:
        OrderNotFoundError: If the order is missing or owned by another user.
    """
    record = _store.get(order_id)
    if record is None or record.get("user_id") != user_id:
        raise OrderNotFoundError(f"order {order_id} not found for user {user_id}")
    return record

import unittest


def _reset_state() -> None:
    """Reset module state for isolated tests."""
    global _store, _idempotency_ledger
    _store = OrderStore()
    _idempotency_ledger = {}
    configure(ServiceConfig(), permission_resolver=lambda u, p: True)


def _sample_items() -> List[Dict[str, Any]]:
    """Return a valid item list for tests."""
    return [
        {"name": "Widget", "quantity": 2, "unit_price": 9.99},
        {"name": "Gadget", "quantity": 1, "unit_price": 4.50},
    ]


class CreateOrderTests(unittest.TestCase):
    """Tests for :func:`create_order`."""

    def setUp(self) -> None:
        _reset_state()

    def test_create_order_success(self) -> None:
        result = create_order("user-1", _sample_items(), "tok_abc")
        self.assertEqual(result["status"], ORDER_STATUS_CONFIRMED)
        self.assertEqual(result["total"], round(2 * 9.99 + 4.50, 2))
        self.assertIn("order_id", result)
        self.assertIn("correlation_id", result)

    def test_create_order_rejects_negative_quantity(self) -> None:
        items = [{"name": "X", "quantity": -1, "unit_price": 1.0}]
        result = create_order("user-1", items, "tok_abc")
        self.assertEqual(result["error_code"], ERR_VALIDATION)

    def test_create_order_rejects_negative_price(self) -> None:
        items = [{"name": "X", "quantity": 1, "unit_price": -5.0}]
        result = create_order("user-1", items, "tok_abc")
        self.assertEqual(result["error_code"], ERR_VALIDATION)

    def test_create_order_rejects_empty_items(self) -> None:
        result = create_order("user-1", [], "tok_abc")
        self.assertEqual(result["error_code"], ERR_VALIDATION)

    def test_create_order_rejects_bad_user(self) -> None:
        result = create_order("", _sample_items(), "tok_abc")
        self.assertEqual(result["error_code"], ERR_VALIDATION)

    def test_create_order_requires_payment_token(self) -> None:
        result = create_order("user-1", _sample_items(), "")
        self.assertEqual(result["error_code"], ERR_VALIDATION)

    def test_create_order_escapes_html_in_name(self) -> None:
        items = [{"name": "<script>x</script>", "quantity": 1, "unit_price": 1.0}]
        result = create_order("user-1", items, "tok_abc")
        self.assertNotIn("<script>", result["items"][0]["name"])

    def test_create_order_permission_denied(self) -> None:
        configure(permission_resolver=lambda u, p: False)
        result = create_order("user-1", _sample_items(), "tok_abc")
        self.assertEqual(result["error_code"], ERR_PERMISSION)

    def test_create_order_idempotent(self) -> None:
        first = create_order("user-1", _sample_items(), "tok_dupe")
        second = create_order("user-1", _sample_items(), "tok_dupe")
        self.assertEqual(first["order_id"], second["order_id"])
