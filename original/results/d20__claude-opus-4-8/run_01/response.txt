"""Order management service.

Provides order creation, cancellation, and status retrieval backed by a
pluggable payment gateway and order repository. All public operations are
configuration-driven, permission-checked, idempotent, and traced via a
per-request correlation id.
"""

from __future__ import annotations

import html
import logging
import re
import uuid
from dataclasses import dataclass, field
from datetime import datetime, timezone

# ---------------------------------------------------------------------------
# Constants (R-027: no magic numbers)
# ---------------------------------------------------------------------------

MIN_QUANTITY: int = 1
MIN_UNIT_PRICE: float = 0.0
PRICE_PRECISION: int = 2
MAX_REASON_LENGTH: int = 500
DEFAULT_RATE_LIMIT_PER_MINUTE: int = 60
ORDER_ID_PREFIX: str = "ord"
REFUND_ID_PREFIX: str = "rfnd"

# Order lifecycle states.
STATUS_CONFIRMED: str = "CONFIRMED"
STATUS_CANCELLED: str = "CANCELLED"

# Permissions required by privileged operations (R-020).
PERMISSION_CREATE: str = "order:create"
PERMISSION_CANCEL: str = "order:cancel"
PERMISSION_READ: str = "order:read"

# Pattern of characters considered unsafe in raw string input (R-003).
_UNSAFE_DB_PATTERN = re.compile(r"[;'\"\\]|--")

logger = logging.getLogger(__name__)

@dataclass(frozen=True)
class ServiceConfig:

    Args:
        database_url: Connection string for the order repository.
        payment_api_key: API key used to authenticate with the payment gateway.
        rate_limit_per_minute: Maximum number of operations allowed per minute.
        currency: ISO currency code used for monetary amounts.

    payment_api_key: str

class OrderServiceError(Exception):
    """Base class for all order service errors.

    Args:
        message: Human-readable description of the failure.
        code: Stable machine-readable error code.
    """

    code: str = "ORDER_SERVICE_ERROR"

    def __init__(self, message: str, code: Optional[str] = None) -> None:
        super().__init__(message)
        self.message = message
        if code is not None:
            self.code = code


class ValidationError(OrderServiceError):
    """Raised when input fails validation."""

    code = "VALIDATION_ERROR"


class PermissionDeniedError(OrderServiceError):
    """Raised when the caller lacks the required permission."""

    code = "PERMISSION_DENIED"


class OrderNotFoundError(OrderServiceError):
    """Raised when a referenced order does not exist."""

    code = "ORDER_NOT_FOUND"


class PaymentError(OrderServiceError):
    """Raised when a payment or refund operation fails."""

    code = "PAYMENT_ERROR"

class PaymentGateway:
    """Minimal payment gateway abstraction.

    The default implementation is a no-op stub suitable for tests; production
    deployments inject a real gateway client.
    """

    def __init__(self, api_key: str) -> None:
        self._api_key = api_key

    def charge(self, payment_token: str, amount: float, correlation_id: str) -> str:
        """Charge a payment token.

        Args:
            payment_token: Tokenized payment instrument.
            amount: Positive amount to charge.
            correlation_id: Request correlation id for tracing.

        Returns:
            A gateway transaction identifier.

        Raises:
            PaymentError: If the charge cannot be completed.
        """
        if not payment_token:
            raise PaymentError("Missing payment token")
        return f"txn_{uuid.uuid4().hex}"

    def refund(self, transaction_id: str, amount: float, correlation_id: str) -> str:
        """Refund a previously captured transaction.

        Args:
            transaction_id: Identifier of the original charge.
            amount: Positive amount to refund.
            correlation_id: Request correlation id for tracing.

        Returns:
            A refund identifier.

        Raises:
            PaymentError: If the refund cannot be completed.
        """
        if not transaction_id:
            raise PaymentError("Missing transaction id for refund")
        return f"{REFUND_ID_PREFIX}_{uuid.uuid4().hex}"


class OrderRepository:
    """In-memory order repository.

    Acts as a stand-in for a database-backed store. All persisted records are
    plain dicts keyed by order id.
    """

    def __init__(self) -> None:
        self._orders: dict[str, dict[str, Any]] = {}

    def save(self, order: dict[str, Any]) -> None:
        """Persist or overwrite an order record."""
        self._orders[order["order_id"]] = order

    def get(self, order_id: str) -> Optional[dict[str, Any]]:
        """Return the order record or None if absent."""
        return self._orders.get(order_id)

    def exists(self, order_id: str) -> bool:
        """Return whether an order id is already persisted."""
        return order_id in self._orders

def _trace_logger(correlation_id: str) -> logging.LoggerAdapter:
    """Return a logger adapter that injects the correlation id into records.

    Args:
        correlation_id: Identifier propagated across all log entries.

    Returns:
        A LoggerAdapter that includes ``correlation_id`` in every message.
    """
    return logging.LoggerAdapter(logger, {"correlation_id": correlation_id})


def _new_correlation_id() -> str:
    """Return a fresh correlation id for a request."""
    return uuid.uuid4().hex


def _log_entry(log: logging.LoggerAdapter, name: str, params: dict[str, Any]) -> None:
    """Log function entry at DEBUG with a parameter summary (R-013)."""
    log.debug("Entering %s with params=%s", name, _summarize_params(params))


def _log_exit(log: logging.LoggerAdapter, name: str, result: Any) -> None:
    """Log function exit at DEBUG with the return value type (R-014)."""
    log.debug("Exiting %s returning type=%s", name, type(result).__name__)


def _summarize_params(params: dict[str, Any]) -> dict[str, Any]:
    """Produce a non-sensitive summary of parameters for logging.

    Args:
        params: Raw parameter mapping.

    Returns:
        A shallow copy with sensitive and bulky values redacted.
    """
    summary: dict[str, Any] = {}
    for key, value in params.items():
        if "token" in key or "key" in key:
            summary[key] = "***redacted***"
        elif isinstance(value, list):
            summary[key] = f"list(len={len(value)})"
        else:
            summary[key] = value
    return summary

def _require_non_empty_str(value: Any, field_name: str) -> str:
    """Validate that a value is a non-empty string (R-001).

    Args:
        value: Candidate value.
        field_name: Name used in error messages.

    Returns:
        The validated string.

    Raises:
        ValidationError: If the value is not a non-empty string.
    """
    if not isinstance(value, str) or not value.strip():
        raise ValidationError(f"{field_name} must be a non-empty string")
    return value


def _sanitize_for_db(value: str, field_name: str) -> str:
    """Reject strings containing characters unsafe for database queries (R-003).

    Args:
        value: Raw input string.
        field_name: Name used in error messages.

    Returns:
        The original string when it is safe.

    Raises:
        ValidationError: If unsafe characters are present.
    """
    if _UNSAFE_DB_PATTERN.search(value):
        raise ValidationError(f"{field_name} contains unsafe characters")
    return value


def _validate_quantity(quantity: Any, name: str) -> int:
    """Validate that quantity is a non-negative integer (R-001, R-002).

    Args:
        quantity: Candidate quantity.
        name: Item name for error context.

    Returns:
        The validated quantity.

    Raises:
        ValidationError: If quantity is not an int or is below the minimum.
    """
    if isinstance(quantity, bool) or not isinstance(quantity, int):
        raise ValidationError(f"Quantity for '{name}' must be an integer")
    if quantity < MIN_QUANTITY:
        raise ValidationError(f"Quantity for '{name}' must be >= {MIN_QUANTITY}")
    return quantity


def _validate_unit_price(unit_price: Any, name: str) -> float:
    """Validate that unit price is a non-negative number (R-001, R-002).

    Args:
        unit_price: Candidate unit price.
        name: Item name for error context.

    Returns:
        The validated unit price as a float.

    Raises:
        ValidationError: If unit price is not numeric or is negative.
    """
    if isinstance(unit_price, bool) or not isinstance(unit_price, (int, float)):
        raise ValidationError(f"Unit price for '{name}' must be numeric")
    if unit_price < MIN_UNIT_PRICE:
        raise ValidationError(f"Unit price for '{name}' must be >= {MIN_UNIT_PRICE}")
    return float(unit_price)


def _validate_item(item: Any) -> dict[str, Any]:
    """Validate and normalize a single order item (R-001, R-004).

    Args:
        item: Candidate item mapping with name, quantity, and unit_price.

    Returns:
        A normalized item with escaped name and computed subtotal.

    Raises:
        ValidationError: If the item structure or values are invalid.
    """
    if not isinstance(item, dict):
        raise ValidationError("Each item must be a dict")
    raw_name = _require_non_empty_str(item.get("name"), "item name")
    safe_name = html.escape(_sanitize_for_db(raw_name, "item name"))
    quantity = _validate_quantity(item.get("quantity"), safe_name)
    unit_price = _validate_unit_price(item.get("unit_price"), safe_name)
    subtotal = round(quantity * unit_price, PRICE_PRECISION)
    return {"name": safe_name, "quantity": quantity,
            "unit_price": unit_price, "subtotal": subtotal}


def _validate_items(items: Any) -> list[dict[str, Any]]:
    """Validate a list of order items (R-001).

    Args:
        items: Candidate list of item mappings.

    Returns:
        A list of normalized items.

    Raises:
        ValidationError: If items is not a non-empty list of valid items.
    """
    if not isinstance(items, list) or not items:
        raise ValidationError("items must be a non-empty list")
    return [_validate_item(item) for item in items]


def _calculate_total(items: list[dict[str, Any]]) -> float:
    """Sum item subtotals into an order total (R-027).

    Args:
        items: Normalized items containing a ``subtotal`` field.

    Returns:
        The rounded order total.
    """
    return round(sum(item["subtotal"] for item in items), PRICE_PRECISION)

def _check_permission(permissions: Any, required: str) -> None:
    """Verify the caller holds the required permission (R-020).

    Args:
        permissions: Iterable of granted permission strings.
        required: Permission needed for the operation.

    Raises:
        PermissionDeniedError: If the permission is absent.
    """
    granted = set(permissions or [])
    if required not in granted:
        raise PermissionDeniedError(f"Caller lacks required permission: {required}")


def _now_iso() -> str:
    """Return the current UTC time as an ISO-8601 string."""
    return datetime.now(timezone.utc).isoformat()


def build_error_response(error: OrderServiceError) -> dict[str, Any]:
    """Build a structured error response (R-010).

    Args:
        error: The domain error to serialize.

    Returns:
        A dict with error code, message, and timestamp.
    """
    return {"error_code": error.code, "message": error.message,
            "timestamp": _now_iso()}

@dataclass
class OrderService:
    """Order management service.

    Args:
        config: Service configuration object.
        repository: Order persistence backend.
        payment_gateway: Payment processing backend.
    """

    config: ServiceConfig
    repository: OrderRepository = field(default_factory=OrderRepository)
    payment_gateway: Optional[PaymentGateway] = None

    def __post_init__(self) -> None:
        if self.payment_gateway is None:
            self.payment_gateway = PaymentGateway(self.config.payment_api_key)

    def create_order(self, user_id: str, items: list,
                     payment_token: str, *, permissions: Optional[list] = None,
                     idempotency_key: Optional[str] = None) -> dict[str, Any]:
        """Create a new order and capture payment.

        Args:
            user_id: Identifier of the ordering user.
            items: List of items, each with name, quantity, and unit_price.
            payment_token: Tokenized payment instrument.
            permissions: Permissions granted to the caller.
            idempotency_key: Optional key reused as the order id for idempotency.

        Returns:
            An order confirmation dict.

        Raises:
            ValidationError: If inputs are invalid.
            PermissionDeniedError: If the caller lacks ``order:create``.
            PaymentError: If payment capture fails.
        """
        correlation_id = _new_correlation_id()
        log = _trace_logger(correlation_id)
        params = {"user_id": user_id, "items": items,
                  "payment_token": payment_token}
        _log_entry(log, "create_order", params)
        try:
            result = self._do_create_order(
                log, correlation_id, user_id, items, payment_token,
                permissions, idempotency_key)
            _log_exit(log, "create_order", result)
            return result
        except OrderServiceError:
            raise
        except Exception as exc:  # noqa: BLE001 - normalize unexpected errors
            log.error("Unexpected error in create_order", exc_info=True)
            raise OrderServiceError("Failed to create order") from exc

    def _do_create_order(self, log: logging.LoggerAdapter, correlation_id: str,
                         user_id: str, items: list, payment_token: str,
                         permissions: Optional[list],
                         idempotency_key: Optional[str]) -> dict[str, Any]:
        """Execute the validated order-creation workflow (R-022)."""
        _check_permission(permissions, PERMISSION_CREATE)
        safe_user = _sanitize_for_db(
            _require_non_empty_str(user_id, "user_id"), "user_id")
        _require_non_empty_str(payment_token, "payment_token")
        order_id = idempotency_key or f"{ORDER_ID_PREFIX}_{uuid.uuid4().hex}"
        existing = self.repository.get(order_id)
        if existing is not None:
            log.info("Idempotent create_order hit for order_id=%s", order_id)
            return _to_confirmation(existing)
        normalized = _validate_items(items)
        return self._charge_and_persist(
            log, correlation_id, safe_user, order_id, normalized, payment_token)

    def _charge_and_persist(self, log: logging.LoggerAdapter, correlation_id: str,
                            user_id: str, order_id: str,
                            items: list[dict[str, Any]],
                            payment_token: str) -> dict[str, Any]:
        """Charge payment then persist the order record (R-015, R-021)."""
        total = _calculate_total(items)
        transaction_id = None
        try:
            transaction_id = self.payment_gateway.charge(
                payment_token, total, correlation_id)
            order = _build_order_record(
                user_id, order_id, items, total, transaction_id, self.config)
            self.repository.save(order)
            log.info("Created order_id=%s total=%.2f", order_id, total)
            return _to_confirmation(order)
        finally:
            log.debug("create_order resources released for order_id=%s "
                      "(txn=%s)", order_id, transaction_id)

    def cancel_order(self, user_id: str, order_id: str, reason: str,
                     *, permissions: Optional[list] = None) -> dict[str, Any]:
        """Cancel an existing order and issue a refund.

        Args:
            user_id: Identifier of the requesting user.
            order_id: Identifier of the order to cancel.
            reason: Human-readable cancellation reason.
            permissions: Permissions granted to the caller.

        Returns:
            A cancellation confirmation dict.

        Raises:
            ValidationError: If inputs are invalid.
            PermissionDeniedError: If the caller lacks ``order:cancel``.
            OrderNotFoundError: If the order does not exist for the user.
            PaymentError: If the refund fails.
        """
        correlation_id = _new_correlation_id()
        log = _trace_logger(correlation_id)
        _log_entry(log, "cancel_order",
                   {"user_id": user_id, "order_id": order_id, "reason": reason})
        try:
            result = self._do_cancel_order(
                log, correlation_id, user_id, order_id, reason, permissions)
            _log_exit(log, "cancel_order", result)
            return result
        except OrderServiceError:
            raise
        except Exception as exc:  # noqa: BLE001 - normalize unexpected errors
            log.error("Unexpected error in cancel_order", exc_info=True)
            raise OrderServiceError("Failed to cancel order") from exc

    def _do_cancel_order(self, log: logging.LoggerAdapter, correlation_id: str,
                        user_id: str, order_id: str, reason: str,
                        permissions: Optional[list]) -> dict[str, Any]:
        """Execute the validated cancellation workflow (R-022)."""
        _check_permission(permissions, PERMISSION_CANCEL)
        order = self._load_owned_order(user_id, order_id)
        safe_reason = html.escape(_validate_reason(reason))
        if order["status"] == STATUS_CANCELLED:
            log.info("Idempotent cancel_order hit for order_id=%s", order_id)
            return _to_cancellation(order)
        refund_id = self.payment_gateway.refund(
            order["transaction_id"], order["total"], correlation_id)
        order.update(status=STATUS_CANCELLED, cancel_reason=safe_reason,
                     refund_id=refund_id, cancelled_at=_now_iso())
        self.repository.save(order)
        log.info("Cancelled order_id=%s refund_id=%s", order_id, refund_id)
        return _to_cancellation(order)

    def get_order_status(self, user_id: str, order_id: str,
                         *, permissions: Optional[list] = None) -> dict[str, Any]:
        """Retrieve the status of an order.

        Args:
            user_id: Identifier of the requesting user.
            order_id: Identifier of the order to look up.
            permissions: Permissions granted to the caller.

        Returns:
            An order status dict.

        Raises:
            ValidationError: If inputs are invalid.
            PermissionDeniedError: If the caller lacks ``order:read``.
            OrderNotFoundError: If the order does not exist for the user.
        """
        correlation_id = _new_correlation_id()
        log = _trace_logger(correlation_id)
        _log_entry(log, "get_order_status",
                   {"user_id": user_id, "order_id": order_id})
        try:
            _check_permission(permissions, PERMISSION_READ)
            order = self._load_owned_order(user_id, order_id)
            result = _to_status(order)
            _log_exit(log, "get_order_status", result)
            return result
        except OrderServiceError:
            raise
        except Exception as exc:  # noqa: BLE001 - normalize unexpected errors
            log.error("Unexpected error in get_order_status", exc_info=True)
            raise OrderServiceError("Failed to retrieve order status") from exc

    def _load_owned_order(self, user_id: str, order_id: str) -> dict[str, Any]:
        """Load an order, verifying it exists and belongs to the user.

        Args:
            user_id: Identifier of the requesting user.
            order_id: Identifier of the order to load.

        Returns:
            The order record.

        Raises:
            ValidationError: If identifiers are invalid.
            OrderNotFoundError: If the order is missing or owned by another user.
        """
        safe_user = _sanitize_for_db(
            _require_non_empty_str(user_id, "user_id"), "user_id")
        safe_order = _sanitize_for_db(
            _require_non_empty_str(order_id, "order_id"), "order_id")
        order = self.repository.get(safe_order)
        if order is None or order["user_id"] != safe_user:
            raise OrderNotFoundError(f"Order not found: {safe_order}")
        return order

def _validate_reason(reason: Any) -> str:
    """Validate a cancellation reason string (R-001).

    Args:
        reason: Candidate reason.

    Returns:
        The validated reason.

    Raises:
        ValidationError: If the reason is empty or exceeds the length limit.
    """
    value = _require_non_empty_str(reason, "reason")
    if len(value) > MAX_REASON_LENGTH:
        raise ValidationError(f"reason exceeds {MAX_REASON_LENGTH} characters")
    return value


def _build_order_record(user_id: str, order_id: str, items: list[dict[str, Any]],
                        total: float, transaction_id: str,
                        config: ServiceConfig) -> dict[str, Any]:
    """Assemble a persistable order record.

    Args:
        user_id: Sanitized user identifier.
        order_id: Unique order identifier.
        items: Normalized order items.
        total: Computed order total.
        transaction_id: Payment transaction identifier.
        config: Service configuration providing the currency.

    Returns:
        A fully populated order record dict.
    """
    return {"order_id": order_id, "user_id": user_id, "items": items,
            "total": total, "currency": config.currency,
            "status": STATUS_CONFIRMED, "transaction_id": transaction_id,
            "created_at": _now_iso()}


def _to_confirmation(order: dict[str, Any]) -> dict[str, Any]:
    """Project an order record into a creation confirmation."""
    return {"order_id": order["order_id"], "status": order["status"],
            "total": order["total"], "currency": order["currency"],
            "items": order["items"], "created_at": order["created_at"]}


def _to_cancellation(order: dict[str, Any]) -> dict[str, Any]:
    """Project an order record into a cancellation confirmation."""
    return {"order_id": order["order_id"], "status": order["status"],
            "refund_id": order.get("refund_id"),
            "reason": order.get("cancel_reason"),
            "cancelled_at": order.get("cancelled_at")}


def _to_status(order: dict[str, Any]) -> dict[str, Any]:
    """Project an order record into a status response."""
    return {"order_id": order["order_id"], "status": order["status"],
            "total": order["total"], "currency": order["currency"],
            "created_at": order["created_at"]}

import unittest


def _make_service() -> OrderService:
    """Construct a service instance with default in-memory collaborators."""
    config = ServiceConfig(database_url="sqlite://:memory:",
                           payment_api_key="test-key", rate_limit_per_minute=10)

_ALL_PERMS = [PERMISSION_CREATE, PERMISSION_CANCEL, PERMISSION_READ]
_SAMPLE_ITEMS = [
    {"name": "Widget", "quantity": 2, "unit_price": 9.5},
    {"name": "Gadget", "quantity": 1, "unit_price": 3.0},
]


class CreateOrderTests(unittest.TestCase):
    """Tests for OrderService.create_order."""

    def setUp(self) -> None:
        self.service = _make_service()

    def test_create_order_success(self) -> None:
        result = self.service.create_order(
            "user-1", _SAMPLE_ITEMS, "tok_abc", permissions=_ALL_PERMS)
        self.assertEqual(result["status"], STATUS_CONFIRMED)
        self.assertEqual(result["total"], 22.0)
        self.assertTrue(result["order_id"].startswith(ORDER_ID_PREFIX))

    def test_create_order_requires_permission(self) -> None:
        with self.assertRaises(PermissionDeniedError):
            self.service.create_order("user-1", _SAMPLE_ITEMS, "tok", permissions=[])

    def test_create_order_rejects_negative_quantity(self) -> None:
        items = [{"name": "X", "quantity": -1, "unit_price": 5.0}]
        with self.assertRaises(ValidationError):
            self.service.create_order("u", items, "tok", permissions=_ALL_PERMS)

    def test_create_order_rejects_negative_price(self) -> None:
        items = [{"name": "X", "quantity": 1, "unit_price": -5.0}]
        with self.assertRaises(ValidationError):
            self.service.create_order("u", items, "tok", permissions=_ALL_PERMS)

    def test_create_order_rejects_empty_items(self) -> None:
        with self.assertRaises(ValidationError):
            self.service.create_order("u", [], "tok", permissions=_ALL_PERMS)

    def test_create_order_rejects_unsafe_user_id(self) -> None:
        with self.assertRaises(ValidationError):
            self.service.create_order(
                "rob'; DROP TABLE", _SAMPLE_ITEMS, "tok", permissions=_ALL_PERMS)

    def test_create_order_escapes_item_name(self) -> None:
        items = [{"name": "<b>x</b>", "quantity": 1, "unit_price": 1.0}]
        result = self.service.create_order(
            "u", items, "tok", permissions=_ALL_PERMS)
        self.assertEqual(result["items"][0]["name"], "&lt;b&gt;x&lt;/b&gt;")

    def test_create_order_is_idempotent(self) -> None:
        first = self.service.create_order(
            "u", _SAMPLE_ITEMS, "tok", permissions=_ALL_PERMS,
            idempotency_key="ord_fixed")
        second = self.service.create_order(
            "u", _SAMPLE_ITEMS, "tok", permissions=_ALL_PERMS,
            idempotency_key="ord_fixed")
        self.assertEqual(first["order_id"], second["order_id"])
