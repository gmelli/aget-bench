class InMemoryOrderRepository:
    """Thread-safe in-memory order store.

    This stands in for a real database. All methods are safe to call
    concurrently because access is guarded by a lock.
    """

    def __init__(self) -> None:
        self._orders: Dict[str, Dict[str, Any]] = {}
        self._lock = Lock()

    def save(self, order: Dict[str, Any]) -> None:
        """Persist an order, overwriting any existing record with the same id.

        Args:
            order: The order record to store. Must contain an ``order_id``.

        Returns:
            None.

        Raises:
            ValidationError: If the order has no ``order_id``.
        """
        order_id = order.get("order_id")
        if not order_id:
            raise ValidationError("Order record is missing 'order_id'.")
        with self._lock:
            self._orders[order_id] = dict(order)

    def get(self, order_id: str) -> Optional[Dict[str, Any]]:
        """Return a copy of the order with the given id, or None.

        Args:
            order_id: Identifier of the order to fetch.

        Returns:
            A copy of the stored order, or None if not present.

        Raises:
            None.
        """
        with self._lock:
            record = self._orders.get(order_id)
            return dict(record) if record is not None else None


class PaymentGateway:
    """Minimal payment gateway abstraction.

    The default implementation simulates success. Real deployments inject a
    concrete client. Methods are intentionally side-effect free here.
    """

    def __init__(self, api_key: str) -> None:
        self._api_key = api_key

    def charge(self, payment_token: str, amount: float, currency: str) -> str:
        """Charge the given amount and return a payment reference.

        Args:
            payment_token: Tokenized payment instrument.
            amount: Amount to charge; must be non-negative.
            currency: ISO currency code.

        Returns:
            A payment reference identifier.

        Raises:
            PaymentError: If the charge cannot be completed.
        """
        if not payment_token:
            raise PaymentError("Payment token is required to charge.")
        return f"pay_{uuid.uuid4().hex}"

    def refund(self, payment_reference: str, amount: float) -> str:
        """Refund a previous charge and return a refund reference.

        Args:
            payment_reference: Reference of the original charge.
            amount: Amount to refund; must be non-negative.

        Returns:
            A refund reference identifier.

        Raises:
            PaymentError: If the refund cannot be completed.
        """
        if not payment_reference:
            raise PaymentError("Payment reference is required to refund.")
        return f"{REFUND_ID_PREFIX}_{uuid.uuid4().hex}"


class RateLimiter:
    """Sliding-window rate limiter keyed by user id."""

    def __init__(self, limit_per_minute: int) -> None:
        self._limit = limit_per_minute
        self._events: Dict[str, List[float]] = {}
        self._lock = Lock()

    def check(self, user_id: str) -> None:
        """Record a request and raise if the user exceeds the limit.

        Args:
            user_id: Identifier of the calling user.

        Returns:
            None.

        Raises:
            RateLimitError: If the user exceeds the configured limit.
        """
        window_start = time.monotonic() - 60.0
        with self._lock:
            recent = [t for t in self._events.get(user_id, []) if t >= window_start]
            if len(recent) >= self._limit:
                raise RateLimitError(
                    f"Rate limit of {self._limit}/min exceeded for user."
                )
            recent.append(time.monotonic())
            self._events[user_id] = recent


class PermissionService:
    """Authorization check abstraction (R-020).

    The default implementation grants all permissions; production code injects
    a real implementation backed by an identity provider.
    """

    def has_permission(self, user_id: str, permission: str) -> bool:
        """Return whether the user holds the given permission.

        Args:
            user_id: Identifier of the calling user.
            permission: Permission identifier being checked.

        Returns:
            True if the user is authorized, otherwise False.

        Raises:
            None.
        """
        return bool(user_id) and bool(permission)

def _sanitize_for_db(value: str) -> str:
    """Strip characters unsafe for database queries (R-003).

    Args:
        value: Raw string value.

    Returns:
        The sanitized string.

    Raises:
        None.
    """
    return _DB_UNSAFE_PATTERN.sub("", value)


def _require_non_empty_str(value: Any, name: str) -> str:
    """Validate that a value is a non-empty string (R-001).

    Args:
        value: Value to validate.
        name: Parameter name used in error messages.

    Returns:
        The validated string.

    Raises:
        ValidationError: If the value is not a non-empty string.
    """
    if not isinstance(value, str) or not value.strip():
        raise ValidationError(f"Parameter '{name}' must be a non-empty string.")
    return value


def _validate_item(item: Any, index: int) -> Dict[str, Any]:
    """Validate and normalize a single order item (R-001, R-002, R-004).

    Args:
        item: Raw item; expected to have name, quantity, unit_price.
        index: Position of the item, used in error messages.

    Returns:
        A normalized item dict with sanitized, escaped name.

    Raises:
        ValidationError: If the item fails type or value checks.
    """
    if not isinstance(item, dict):
        raise ValidationError(f"Item at index {index} must be a dict.")
    name = _require_non_empty_str(item.get("name"), f"items[{index}].name")
    quantity = item.get("quantity")
    unit_price = item.get("unit_price")
    if not isinstance(quantity, int) or isinstance(quantity, bool):
        raise ValidationError(f"items[{index}].quantity must be an int.")
    if quantity < MIN_QUANTITY:
        raise ValidationError(f"items[{index}].quantity must be >= {MIN_QUANTITY}.")
    if not isinstance(unit_price, (int, float)) or isinstance(unit_price, bool):
        raise ValidationError(f"items[{index}].unit_price must be numeric.")
    if unit_price < MIN_UNIT_PRICE:
        raise ValidationError(f"items[{index}].unit_price must be non-negative.")
    safe_name = html.escape(_sanitize_for_db(name))
    return {"name": safe_name, "quantity": quantity, "unit_price": float(unit_price)}


def _validate_items(items: Any) -> List[Dict[str, Any]]:
    """Validate the full items collection (R-001, R-002).

    Args:
        items: Raw items list.

    Returns:
        A list of normalized item dicts.

    Raises:
        ValidationError: If the collection or any item is invalid.
    """
    if not isinstance(items, list) or not items:
        raise ValidationError("Parameter 'items' must be a non-empty list.")
    if len(items) > MAX_ITEMS_PER_ORDER:
        raise ValidationError(f"Cannot exceed {MAX_ITEMS_PER_ORDER} items per order.")
    return [_validate_item(item, i) for i, item in enumerate(items)]


def _calculate_total(items: List[Dict[str, Any]]) -> float:
    """Calculate the rounded order total (R-027).

    Args:
        items: Normalized item dicts.

    Returns:
        The total price rounded to ``PRICE_DECIMAL_PLACES``.

    Raises:
        None.
    """
    total = sum(item["quantity"] * item["unit_price"] for item in items)
    return round(total, PRICE_DECIMAL_PLACES)


def _authorize(user_id: str, permission: str, correlation_id: str) -> None:
    """Verify the caller holds the required permission (R-020).

    Args:
        user_id: Identifier of the calling user.
        permission: Permission required for the operation.
        correlation_id: Correlation id for tracing.

    Returns:
        None.

    Raises:
        AuthorizationError: If the caller is not authorized.
    """
    if not _SERVICE.permissions.has_permission(user_id, permission):
        raise AuthorizationError(f"User is not authorized for '{permission}'.")

class _ServiceContext:
    """Container for the service's configuration and collaborators."""

    config: ServiceConfig = field(default_factory=ServiceConfig)
    repository: InMemoryOrderRepository = field(default_factory=InMemoryOrderRepository)
    permissions: PermissionService = field(default_factory=PermissionService)
    payments: Optional[PaymentGateway] = None
    limiter: Optional[RateLimiter] = None

    def __post_init__(self) -> None:
        if self.payments is None:
            self.payments = PaymentGateway(self.config.payment_api_key)
        if self.limiter is None:
            self.limiter = RateLimiter(self.config.rate_limit_per_minute)

def create_order(
    user_id: str, items: List[Dict[str, Any]], payment_token: str
) -> Dict[str, Any]:
    """Create a new order and process its payment.

    Args:
        user_id: Identifier of the user placing the order.
        items: List of item dicts, each with ``name``, ``quantity``, and
            ``unit_price``.
        payment_token: Tokenized payment instrument used to charge the order.

    Returns:
        An order confirmation dict on success, or a structured error response
        dict containing ``error_code``, ``message``, and ``timestamp``.

    Raises:
        None: All domain errors are converted to structured error responses.
    """
    correlation_id = _new_correlation_id()
    extra = _log_extra(correlation_id)
    item_count = len(items) if isinstance(items, list) else 0
    logger.debug("Entering create_order user=%s items=%d", user_id, item_count, extra=extra)
    try:
        user_id = _require_non_empty_str(user_id, "user_id")
        payment_token = _require_non_empty_str(payment_token, "payment_token")
        _SERVICE.limiter.check(user_id)
        _authorize(user_id, PERMISSION_CREATE, correlation_id)
        confirmation = _do_create_order(user_id, items, payment_token, correlation_id)
        logger.debug("Exiting create_order return=%s", type(confirmation).__name__, extra=extra)
        return confirmation
    except OrderServiceError as exc:
        logger.error("create_order failed: %s", exc, exc_info=True, extra=extra)
        return _error_response(exc)


def _do_create_order(
    user_id: str, items: Any, payment_token: str, correlation_id: str
) -> Dict[str, Any]:
    """Validate, charge, and persist a new order (helper for ``create_order``).

    Args:
        user_id: Validated user identifier.
        items: Raw items to validate and price.
        payment_token: Validated payment token.
        correlation_id: Correlation id for tracing.

    Returns:
        The order confirmation dict.

    Raises:
        OrderServiceError: For any validation, payment, or state failure.
    """
    extra = _log_extra(correlation_id)
    normalized = _validate_items(items)
    total = _calculate_total(normalized)
    order_id = f"{ORDER_ID_PREFIX}_{uuid.uuid4().hex}"
    try:
        payment_ref = _SERVICE.payments.charge(payment_token, total, _SERVICE.config.currency)
        order = _build_order_record(order_id, user_id, normalized, total, payment_ref, correlation_id)
        _SERVICE.repository.save(order)
        logger.info("Order created order_id=%s total=%.2f", order_id, total, extra=extra)
    except PaymentError:
        raise
    except OrderServiceError as exc:
        raise OrderServiceError(f"Failed to create order {order_id}: {exc.message}") from exc
    finally:
        logger.debug("create_order cleanup order_id=%s", order_id, extra=extra)
    return _confirmation_from_order(order)


def _build_order_record(
    order_id: str,
    user_id: str,
    items: List[Dict[str, Any]],
    total: float,
    payment_ref: str,
    correlation_id: str,
) -> Dict[str, Any]:
    """Assemble the persisted order record.

    Args:
        order_id: Unique order identifier.
        user_id: Owning user identifier.
        items: Normalized item dicts.
        total: Calculated order total.
        payment_ref: Payment reference from the gateway.
        correlation_id: Correlation id for tracing.

    Returns:
        The order record dict.

    Raises:
        None.
    """
    return {
        "order_id": order_id,
        "user_id": user_id,
        "items": items,
        "total": total,
        "currency": _SERVICE.config.currency,
        "status": OrderStatus.CONFIRMED.value,
        "payment_reference": payment_ref,
        "refund_reference": None,
        "correlation_id": correlation_id,
        "created_at": _now_iso(),
    }


def _confirmation_from_order(order: Dict[str, Any]) -> Dict[str, Any]:
    """Project an order record into a public confirmation dict.

    Args:
        order: The persisted order record.

    Returns:
        A confirmation dict safe to return to callers.

    Raises:
        None.
    """
    return {
        "order_id": order["order_id"],
        "status": order["status"],
        "total": order["total"],
        "currency": order["currency"],
        "items": order["items"],
        "created_at": order["created_at"],
    }


def cancel_order(user_id: str, order_id: str, reason: str) -> Dict[str, Any]:
    """Cancel an existing order and issue a refund.

    Args:
        user_id: Identifier of the user cancelling the order.
        order_id: Identifier of the order to cancel.
        reason: Free-text cancellation reason.

    Returns:
        A cancellation confirmation dict on success, or a structured error
        response dict.

    Raises:
        None: All domain errors are converted to structured error responses.
    """
    correlation_id = _new_correlation_id()
    extra = _log_extra(correlation_id)
    logger.debug("Entering cancel_order user=%s order_id=%s", user_id, order_id, extra=extra)
    try:
        user_id = _require_non_empty_str(user_id, "user_id")
        order_id = _require_non_empty_str(order_id, "order_id")
        reason = html.escape(_sanitize_for_db(_require_non_empty_str(reason, "reason")))
        _SERVICE.limiter.check(user_id)
        _authorize(user_id, PERMISSION_CANCEL, correlation_id)
        result = _do_cancel_order(user_id, order_id, reason, correlation_id)
        logger.debug("Exiting cancel_order return=%s", type(result).__name__, extra=extra)
        return result
    except OrderServiceError as exc:
        logger.error("cancel_order failed: %s", exc, exc_info=True, extra=extra)
        return _error_response(exc)


def _do_cancel_order(
    user_id: str, order_id: str, reason: str, correlation_id: str
) -> Dict[str, Any]:
    """Load, refund, and mark an order cancelled (helper for ``cancel_order``).

    Implements an idempotency guard (R-022): cancelling an already-cancelled
    order returns the existing cancellation confirmation without re-refunding.

    Args:
        user_id: Validated user identifier.
        order_id: Validated order identifier.
        reason: Sanitized cancellation reason.
        correlation_id: Correlation id for tracing.

    Returns:
        The cancellation confirmation dict.

    Raises:
        OrderNotFoundError: If the order does not exist or is not the user's.
        OrderServiceError: For refund or persistence failures.
    """
    extra = _log_extra(correlation_id)
    order = _load_owned_order(user_id, order_id)
    if order["status"] == OrderStatus.CANCELLED.value:
        logger.info("cancel_order idempotent hit order_id=%s", order_id, extra=extra)
        return _cancellation_from_order(order)
    refund_ref = _SERVICE.payments.refund(order["payment_reference"], order["total"])
    order["status"] = OrderStatus.CANCELLED.value
    order["refund_reference"] = refund_ref
    order["cancel_reason"] = reason
    order["cancelled_at"] = _now_iso()
    _SERVICE.repository.save(order)
    logger.info("Order cancelled order_id=%s refund=%s", order_id, refund_ref, extra=extra)
    return _cancellation_from_order(order)


def _cancellation_from_order(order: Dict[str, Any]) -> Dict[str, Any]:
    """Project an order record into a public cancellation dict.

    Args:
        order: The persisted, cancelled order record.

    Returns:
        A cancellation confirmation dict.

    Raises:
        None.
    """
    return {
        "order_id": order["order_id"],
        "status": order["status"],
        "refund_reference": order.get("refund_reference"),
        "reason": order.get("cancel_reason"),
        "cancelled_at": order.get("cancelled_at"),
    }


def get_order_status(user_id: str, order_id: str) -> Dict[str, Any]:
    """Retrieve the current status of an order.

    Args:
        user_id: Identifier of the requesting user.
        order_id: Identifier of the order to inspect.

    Returns:
        An order status dict on success, or a structured error response dict.

    Raises:
        None: All domain errors are converted to structured error responses.
    """
    correlation_id = _new_correlation_id()
    extra = _log_extra(correlation_id)
    logger.debug("Entering get_order_status user=%s order_id=%s", user_id, order_id, extra=extra)
    try:
        user_id = _require_non_empty_str(user_id, "user_id")
        order_id = _require_non_empty_str(order_id, "order_id")
        _authorize(user_id, PERMISSION_READ, correlation_id)
        order = _load_owned_order(user_id, order_id)
        status = {
            "order_id": order["order_id"],
            "status": order["status"],
            "total": order["total"],
            "currency": order["currency"],
            "created_at": order.get("created_at"),
        }
        logger.debug("Exiting get_order_status return=%s", type(status).__name__, extra=extra)
        return status
    except OrderServiceError as exc:
        logger.error("get_order_status failed: %s", exc, exc_info=True, extra=extra)
        return _error_response(exc)


def _load_owned_order(user_id: str, order_id: str) -> Dict[str, Any]:
    """Load an order and verify ownership.

    Args:
        user_id: Identifier expected to own the order.
        order_id: Identifier of the order to load.

    Returns:
        The order record.

    Raises:
        OrderNotFoundError: If the order is missing or owned by another user.
    """
    safe_id = _sanitize_for_db(order_id)
    order = _SERVICE.repository.get(safe_id)
    if order is None or order.get("user_id") != user_id:
        raise OrderNotFoundError(f"Order '{order_id}' was not found.")
    return order
