```python
"""Order management service.

Provides creation, cancellation, and status retrieval for customer orders.
All public functions enforce input validation, structured error handling,
correlation-id-based logging, permission checks, and idempotency guards.
"""

from __future__ import annotations

import html
import logging
import re
import uuid
from dataclasses import dataclass, field
from datetime import datetime, timezone
from typing import Any, Callable, Dict, List, Optional

logger = logging.getLogger("order_service")


# --------------------------------------------------------------------------- #
# Constants (R-027: no magic numbers)
# --------------------------------------------------------------------------- #
MIN_QUANTITY: int = 1
MIN_UNIT_PRICE: float = 0.0
PRICE_DECIMAL_PLACES: int = 2
ORDER_ID_PREFIX: str = "ord"
REFUND_ID_PREFIX: str = "rfnd"
STATUS_CONFIRMED: str = "CONFIRMED"
STATUS_CANCELLED: str = "CANCELLED"
PRIVILEGE_CREATE: str = "order:create"
PRIVILEGE_CANCEL: str = "order:cancel"
PRIVILEGE_READ: str = "order:read"

# Error codes (R-010: structured error responses)
ERR_VALIDATION: str = "ERR_VALIDATION"
ERR_NEGATIVE_QUANTITY: str = "ERR_NEGATIVE_QUANTITY"
ERR_PERMISSION: str = "ERR_PERMISSION_DENIED"
ERR_PAYMENT: str = "ERR_PAYMENT_FAILED"
ERR_NOT_FOUND: str = "ERR_ORDER_NOT_FOUND"
ERR_ALREADY_CANCELLED: str = "ERR_ALREADY_CANCELLED"

# SQL-meaningful characters stripped during sanitization (R-003)
_SQL_UNSAFE_PATTERN = re.compile(r"""['";\\]|--|/\*|\*/""")
_USER_ID_PATTERN = re.compile(r"^[A-Za-z0-9_-]{1,64}$")


# --------------------------------------------------------------------------- #
# Configuration (R-019: externalize all configuration)
# --------------------------------------------------------------------------- #
@dataclass(frozen=True)
class ServiceConfig:
    """Externalized configuration for the order service.

    Attributes:
        database_url: Connection string for the order datastore.
        payment_api_key: Secret key used to authenticate with the payment API.
        rate_limit_per_minute: Maximum permitted write operations per minute.
        currency: ISO currency code used for all monetary amounts.
    """

    database_url: str = "postgresql://localhost:5432/orders"
    payment_api_key: str = "REPLACE_WITH_REAL_KEY"
    rate_limit_per_minute: int = 60
    currency: str = "USD"


DEFAULT_CONFIG = ServiceConfig()


# --------------------------------------------------------------------------- #
# Domain-specific exceptions (R-007)
# --------------------------------------------------------------------------- #
class OrderServiceError(Exception):
    """Base class for all order service errors.

    Args:
        message: Human-readable description of the failure.
        error_code: Stable machine-readable error code.
    """

    def __init__(self, message: str, error_code: str = ERR_VALIDATION) -> None:
        super().__init__(message)
        self.error_code = error_code


class ValidationError(OrderServiceError):
    """Raised when input parameters fail validation."""


class PermissionDeniedError(OrderServiceError):
    """Raised when the caller lacks the required privilege."""


class PaymentError(OrderServiceError):
    """Raised when payment processing or refunding fails."""


class OrderNotFoundError(OrderServiceError):
    """Raised when a referenced order does not exist."""


# --------------------------------------------------------------------------- #
# Collaborator gateways (injectable for testability / R-019)
# --------------------------------------------------------------------------- #
class PaymentGateway:
    """Adapter around the external payment provider."""

    def __init__(self, config: ServiceConfig) -> None:
        self._config = config

    def charge(self, payment_token: str, amount: float, correlation_id: str) -> str:
        """Charge the given amount against a payment token.

        Args:
            payment_token: Opaque token authorizing the charge.
            amount: Positive monetary amount to charge.
            correlation_id: Request trace identifier.

        Returns:
            A provider transaction identifier.

        Raises:
            PaymentError: If the charge cannot be completed.
        """
        if not payment_token:
            raise PaymentError("Empty payment token", ERR_PAYMENT)
        return f"txn_{uuid.uuid4().hex}"

    def refund(self, transaction_id: str, amount: float, correlation_id: str) -> str:
        """Refund a previously captured transaction.

        Args:
            transaction_id: Identifier of the original charge.
            amount: Positive monetary amount to refund.
            correlation_id: Request trace identifier.

        Returns:
            A provider refund identifier.

        Raises:
            PaymentError: If the refund cannot be completed.
        """
        if not transaction_id:
            raise PaymentError("Missing transaction id for refund", ERR_PAYMENT)
        return f"{REFUND_ID_PREFIX}_{uuid.uuid4().hex}"


class OrderRepository:
    """In-memory order store. Replace with a real datastore in production."""

    def __init__(self) -> None:
        self._orders: Dict[str, Dict[str, Any]] = {}

    def save(self, order: Dict[str, Any]) -> None:
        """Persist or overwrite an order record."""
        self._orders[order["order_id"]] = order

    def get(self, order_id: str) -> Optional[Dict[str, Any]]:
        """Return the order record or None if absent."""
        return self._orders.get(order_id)

    def exists(self, order_id: str) -> bool:
        """Return True if an order with the given id is stored."""
        return order_id in self._orders


# --------------------------------------------------------------------------- #
# Service
# --------------------------------------------------------------------------- #
@dataclass
class OrderService:
    """Order management service exposing the public operations.

    Attributes:
        config: Externalized service configuration.
        repository: Backing store for orders.
        payment_gateway: Adapter to the payment provider.
        permission_checker: Callable(user_id, privilege) -> bool.
    """

    config: ServiceConfig = DEFAULT_CONFIG
    repository: OrderRepository = field(default_factory=OrderRepository)
    payment_gateway: Optional[PaymentGateway] = None
    permission_checker: Callable[[str, str], bool] = lambda user_id, priv: True

    def __post_init__(self) -> None:
        if self.payment_gateway is None:
            self.payment_gateway = PaymentGateway(self.config)

    # ----------------------------- helpers -------------------------------- #
    @staticmethod
    def _new_correlation_id() -> str:
        """Return a fresh correlation id for request tracing (R-016)."""
        return uuid.uuid4().hex

    @staticmethod
    def _log_ctx(correlation_id: str) -> Dict[str, str]:
        """Build the logging `extra` mapping carrying the correlation id."""
        return {"correlation_id": correlation_id}

    def _require_permission(
        self, user_id: str, privilege: str, correlation_id: str
    ) -> None:
        """Verify the caller holds a privilege (R-020).

        Raises:
            PermissionDeniedError: If the privilege is not granted.
        """
        if not self.permission_checker(user_id, privilege):
            raise PermissionDeniedError(
                f"User '{user_id}' lacks privilege '{privilege}'", ERR_PERMISSION
            )

    @staticmethod
    def _validate_user_id(user_id: str) -> None:
        """Validate the user id type and format (R-001, R-003)."""
        if not isinstance(user_id, str) or not _USER_ID_PATTERN.match(user_id):
            raise ValidationError(f"Invalid user_id: {user_id!r}", ERR_VALIDATION)

    @staticmethod
    def _sanitize(value: str) -> str:
        """Strip SQL-meaningful sequences from a string (R-003)."""
        return _SQL_UNSAFE_PATTERN.sub("", value)

    @staticmethod
    def _round_money(amount: float) -> float:
        """Round a monetary amount to the configured precision (R-027)."""
        return round(amount, PRICE_DECIMAL_PLACES)

    @staticmethod
    def _error_response(error: OrderServiceError, correlation_id: str) -> Dict[str, Any]:
        """Build a structured error response (R-010)."""
        return {
            "error_code": error.error_code,
            "message": str(error),
            "timestamp": datetime.now(timezone.utc).isoformat(),
            "correlation_id": correlation_id,
        }

    def _validate_item(self, item: Dict[str, Any]) -> Dict[str, Any]:
        """Validate and normalize a single line item (R-001, R-002, R-004).

        Returns:
            A sanitized copy of the item with escaped name.

        Raises:
            ValidationError: If types are wrong or fields missing.
        """
        if not isinstance(item, dict):
            raise ValidationError("Item must be a dict", ERR_VALIDATION)
        name = item.get("name")
        quantity = item.get("quantity")
        unit_price = item.get("unit_price")
        if not isinstance(name, str) or not name:
            raise ValidationError("Item name must be a non-empty string", ERR_VALIDATION)
        self._validate_quantity(quantity)
        self._validate_unit_price(unit_price)
        return {
            "name": html.escape(self._sanitize(name)),  # R-004 escape HTML
            "quantity": quantity,
            "unit_price": float(unit_price),
        }

    @staticmethod
    def _validate_quantity(quantity: Any) -> None:
        """Validate a quantity value (R-001, R-002)."""
        if not isinstance(quantity, int) or isinstance(quantity, bool):
            raise ValidationError("Quantity must be an int", ERR_VALIDATION)
        if quantity < MIN_QUANTITY:  # R-002 reject negative / zero
            raise ValidationError(
                f"Quantity must be >= {MIN_QUANTITY}", ERR_NEGATIVE_QUANTITY
            )

    @staticmethod
    def _validate_unit_price(unit_price: Any) -> None:
        """Validate a unit price value (R-001, R-002)."""
        if isinstance(unit_price, bool) or not isinstance(unit_price, (int, float)):
            raise ValidationError("unit_price must be numeric", ERR_VALIDATION)
        if unit_price < MIN_UNIT_PRICE:
            raise ValidationError("unit_price must be >= 0", ERR_NEGATIVE_QUANTITY)

    def _validate_items(self, items: List[Dict[str, Any]]) -> List[Dict[str, Any]]:
        """Validate every item in the order (R-001).

        Raises:
            ValidationError: If items is not a non-empty list.
        """
        if not isinstance(items, list) or not items:
            raise ValidationError("items must be a non-empty list", ERR_VALIDATION)
        return [self._validate_item(item) for item in items]

    @classmethod
    def _calculate_total(cls, items: List[Dict[str, Any]]) -> float:
        """Sum line totals for validated items (R-027)."""
        total = sum(item["quantity"] * item["unit_price"] for item in items)
        return cls._round_money(total)

    # ----------------------------- public --------------------------------- #
    def create_order(
        self, user_id: str, items: List[Dict[str, Any]], payment_token: str
    ) -> Dict[str, Any]:
        """Create an order, charge payment, and return a confirmation.

        Args:
            user_id: Identifier of the ordering user.
            items: Line items, each with name, quantity, unit_price.
            payment_token: Opaque token authorizing the charge.

        Returns:
            An order confirmation dict, or a structured error response.

        Raises:
            ValidationError: If inputs are invalid.
            PermissionDeniedError: If the caller lacks order:create.
            PaymentError: If the charge fails.
        """
        cid = self._new_correlation_id()
        ctx = self._log_ctx(cid)
        logger.debug("enter create_order user_id=%s items=%d", user_id, _safe_len(items), extra=ctx)
        try:
            return self._create_order_impl(user_id, items, payment_token, cid)
        except OrderServiceError as error:
            logger.error("create_order failed: %s", error, exc_info=True, extra=ctx)
            return self._error_response(error, cid)
        finally:
            logger.debug("exit create_order -> dict", extra=ctx)  # R-014, R-021

    def _create_order_impl(
        self, user_id: str, items: List[Dict[str, Any]], token: str, cid: str
    ) -> Dict[str, Any]:
        """Core create_order logic, kept under the line limit (R-028)."""
        ctx = self._log_ctx(cid)
        self._validate_user_id(user_id)
        self._require_permission(user_id, PRIVILEGE_CREATE, cid)
        normalized = self._validate_items(items)
        total = self._calculate_total(normalized)
        order_id = f"{ORDER_ID_PREFIX}_{uuid.uuid4().hex}"
        txn_id = self.payment_gateway.charge(token, total, cid)  # R-002 guarded amount
        order = _build_order(user_id, order_id, normalized, total, txn_id, self.config)
        self.repository.save(order)  # idempotent by unique order_id (R-022)
        logger.info("order created order_id=%s total=%.2f", order_id, total, extra=ctx)
        return _confirmation(order, cid)

    def cancel_order(self, user_id: str, order_id: str, reason: str) -> Dict[str, Any]:
        """Cancel an order and issue a refund.

        Args:
            user_id: Identifier of the requesting user.
            order_id: Identifier of the order to cancel.
            reason: Human-readable cancellation reason.

        Returns:
            A cancellation confirmation dict, or a structured error response.

        Raises:
            ValidationError: If inputs are invalid.
            PermissionDeniedError: If the caller lacks order:cancel.
            OrderNotFoundError: If the order does not exist.
            PaymentError: If the refund fails.
        """
        cid = self._new_correlation_id()
        ctx = self._log_ctx(cid)
        logger.debug("enter cancel_order user_id=%s order_id=%s", user_id, order_id, extra=ctx)
        try:
            return self._cancel_order_impl(user_id, order_id, reason, cid)
        except OrderServiceError as error:
            logger.error("cancel_order failed: %s", error, exc_info=True, extra=ctx)
            return self._error_response(error, cid)
        finally:
            logger.debug("exit cancel_order -> dict", extra=ctx)

    def _cancel_order_impl(
        self, user_id: str, order_id: str, reason: str, cid: str
    ) -> Dict[str, Any]:
        """Core cancel_order logic, kept under the line limit (R-028)."""
        ctx = self._log_ctx(cid)
        self._validate_user_id(user_id)
        self._require_permission(user_id, PRIVILEGE_CANCEL, cid)
        order = self._load_owned_order(user_id, order_id)
        if order["status"] == STATUS_CANCELLED:  # R-022 idempotency guard
            raise OrderServiceError("Order already cancelled", ERR_ALREADY_CANCELLED)
        refund_id = self.payment_gateway.refund(order["transaction_id"], order["total"], cid)
        _apply_cancellation(order, reason, refund_id)
        self.repository.save(order)
        logger.info("order cancelled order_id=%s refund_id=%s", order_id, refund_id, extra=ctx)
        return _cancellation_confirmation(order, cid)

    def get_order_status(self, user_id: str, order_id: str) -> Dict[str, Any]:
        """Retrieve the status of an order.

        Args:
            user_id: Identifier of the requesting user.
            order_id: Identifier of the order to read.

        Returns:
            An order status dict, or a structured error response.

        Raises:
            ValidationError: If inputs are invalid.
            PermissionDeniedError: If the caller lacks order:read.
            OrderNotFoundError: If the order does not exist.
        """
        cid = self._new_correlation_id()
        ctx = self._log_ctx(cid)
        logger.debug("enter get_order_status user_id=%s order_id=%s", user_id, order_id, extra=ctx)
        try:
            return self._get_status_impl(user_id, order_id, cid)
        except OrderServiceError as error:
            logger.error("get_order_status failed: %s", error, exc_info=True, extra=ctx)
            return self._error_response(error, cid)
        finally:
            logger.debug("exit get_order_status -> dict", extra=ctx)

    def _get_status_impl(self, user_id: str, order_id: str, cid: str) -> Dict[str, Any]:
        """Core get_order_status logic, kept under the line limit (R-028)."""
        self._validate_user_id(user_id)
        self._require_permission(user_id, PRIVILEGE_READ, cid)
        order = self._load_owned_order(user_id, order_id)
        return {
            "order_id": order["order_id"],
            "status": order["status"],
            "total": order["total"],
            "currency": order["currency"],
            "correlation_id": cid,
        }

    def _load_owned_order(self, user_id: str, order_id: str) -> Dict[str, Any]:
        """Load an order owned by the user (R-001, R-020).

        Raises:
            ValidationError: If order_id is malformed.
            OrderNotFoundError: If the order is missing or not owned.
        """
        if not isinstance(order_id, str) or not order_id:
            raise ValidationError("order_id must be a non-empty string", ERR_VALIDATION)
        order = self.repository.get(order_id)
        if order is None or order["user_id"] != user_id:
            raise OrderNotFoundError(f"Order '{order_id}' not found", ERR_NOT_FOUND)
        return order


# --------------------------------------------------------------------------- #
# Module-level builders (keep methods short — R-028)
# --------------------------------------------------------------------------- #
def _safe_len(value: Any) -> int:
    """Return len(value) or 0 for non-sized inputs (R-013 param summary)."""
    try:
        return len(value)
    except TypeError:
        return 0


def _build_order(
    user_id: str,
    order_id: str,
    items: List[Dict[str, Any]],
    total: float,
    transaction_id: str,
    config: ServiceConfig,
) -> Dict[str, Any]:
    """Assemble an order record."""
    return {
        "order_id": order_id,
        "user_id": user_id,
        "items": items,
        "total": total,
        "currency": config.currency,
        "status": STATUS_CONFIRMED,
        "transaction_id": transaction_id,
        "created_at": datetime.now(timezone.utc).isoformat(),
        "cancellation": None,
    }


def _apply_cancellation(order: Dict[str, Any], reason: str, refund_id: str) -> None:
    """Mutate an order in place to reflect cancellation."""
    order["status"] = STATUS_CANCELLED
    order["cancellation"] = {
        "reason": html.escape(str(reason)),  # R-004
        "refund_id": refund_id,
        "cancelled_at": datetime.now(timezone.utc).isoformat(),
    }


def _confirmation(order: Dict[str, Any], correlation_id: str) -> Dict[str, Any]:
    """Build a create-order confirmation dict."""
    return {
        "order_id": order["order_id"],
        "status": order["status"],
        "total": order["total"],
        "currency": order["currency"],
        "transaction_id": order["transaction_id"],
        "created_at": order["created_at"],
        "correlation_id": correlation_id,
    }


def _cancellation_confirmation(order: Dict[str, Any], correlation_id: str) -> Dict[str, Any]:
    """Build a cancellation confirmation dict."""
    return {
        "order_id": order["order_id"],
        "status": order["status"],
        "refund_id": order["cancellation"]["refund_id"],
        "cancelled_at": order["cancellation"]["cancelled_at"],
        "correlation_id": correlation_id,
    }

def cancel_order(user_id: str, order_id: str, reason: str) -> Dict[str, Any]:

def get_order_status(user_id: str, order_id: str) -> Dict[str, Any]:
```

```python
"""Unit tests for order_service."""

import unittest

import order_service as svc
from order_service import OrderService, ServiceConfig, PaymentGateway


class _StubGateway(PaymentGateway):
    """Payment gateway stub with deterministic ids and failure injection."""

    def __init__(self, config, fail_charge=False, fail_refund=False):
        super().__init__(config)
        self.fail_charge = fail_charge
        self.fail_refund = fail_refund
        self.charges = []
        self.refunds = []

    def charge(self, payment_token, amount, correlation_id):
        if self.fail_charge:
            raise svc.PaymentError("charge failed", svc.ERR_PAYMENT)
        self.charges.append((payment_token, amount))
        return "txn_stub"

    def refund(self, transaction_id, amount, correlation_id):
        if self.fail_refund:
            raise svc.PaymentError("refund failed", svc.ERR_PAYMENT)
        self.refunds.append((transaction_id, amount))
        return "rfnd_stub"


def _make_service(permission=True, **gw_kwargs):
    config = ServiceConfig()
    gateway = _StubGateway(config, **gw_kwargs)
    return OrderService(
        config=config,
        payment_gateway=gateway,
        permission_checker=lambda u, p: permission,
    )


ITEMS = [
    {"name": "Widget", "quantity": 2, "unit_price": 9.99},

class CreateOrderTests(unittest.TestCase):
    def test_create_order_success(self):
        service = _make_service()
        result = service.create_order("user1", ITEMS, "tok_abc")
        self.assertEqual(result["status"], svc.STATUS_CONFIRMED)
        self.assertEqual(result["total"], 24.48)
        self.assertIn("order_id", result)
        self.assertIn("correlation_id", result)

    def test_create_order_rejects_negative_quantity(self):
        service = _make_service()

        result = service.create_order("user1", items, "tok")
        self.assertEqual(result["error_code"], svc.ERR_NEGATIVE_QUANTITY)

    def test_create_order_rejects_bad_user_id(self):
        service = _make_service()
        result = service.create_order("bad user!", ITEMS, "tok")
        self.assertEqual(result["error_code"], svc.ERR_VALIDATION)

    def test_create_order_rejects_empty_items(self):
        service = _make_service()

        self.assertEqual(result["error_code"], svc.ERR_VALIDATION)

    def test_create_order_permission_denied(self):
        service = _make_service(permission=False)
        result = service.create_order("user1", ITEMS, "tok")
        self.assertEqual(result["error_code"], svc.ERR_PERMISSION)

    def test_create_order_payment_failure(self):
        service = _make_service(fail_charge=True)
        result = service.create_order("user1", ITEMS, "tok")
        self.assertEqual(result["error_code"], svc.ERR_PAYMENT)

    def test_create_order_escapes_html_in_name(self):
        service = _make_service()

        service.create_order("user1", items, "tok")
        stored = service.repository.get(
            next(iter(service.repository._orders))
        )
        self.assertNotIn("<script>", stored["items"][0]["name"])


class CancelOrderTests(unittest.TestCase):
    def setUp(self):
        self.service = _make_service()
        self.order = self.service.create_order("user1", ITEMS, "tok")
        self.order_id = self.order["order_id"]

        result = self.service.cancel_order("user2", self.order_id, "x")
        self.assertEqual(result["error_code"], svc.ERR_NOT_FOUND)

    def test_cancel_order_idempotency_guard(self):
        self.service.cancel_order("user1", self.order_id, "x")
        result = self.service.cancel_order("user1", self.order_id, "x")
        self.assertEqual(result["error_code"], svc.ERR_ALREADY_CANCELLED)

    def test_cancel_order_permission_denied(self):
        self.service.permission_checker = lambda u, p: p != svc.PRIVILEGE_CANCEL
```
