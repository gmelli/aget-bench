```python
"""Order management service.

Provides order creation, cancellation, and status retrieval backed by a
pluggable database and payment gateway. All public operations validate input,
enforce caller permissions, log their lifecycle, and return structured results.
"""

from __future__ import annotations

import html
import logging
import re
import uuid
from dataclasses import dataclass
from datetime import datetime, timezone
from typing import Any, Callable, Optional

logger = logging.getLogger(__name__)

# --- Constants (R-027: no magic numbers) ------------------------------------

MIN_QUANTITY: int = 1
MIN_UNIT_PRICE: float = 0.0
DEFAULT_RATE_LIMIT_PER_MINUTE: int = 60
DEFAULT_DB_URL: str = "postgresql://localhost:5432/orders"
ORDER_ID_PREFIX: str = "ord"
REFUND_ID_PREFIX: str = "rfnd"

STATUS_CONFIRMED: str = "CONFIRMED"
STATUS_CANCELLED: str = "CANCELLED"

ERR_VALIDATION: str = "VALIDATION_ERROR"
ERR_PERMISSION: str = "PERMISSION_DENIED"
ERR_NOT_FOUND: str = "ORDER_NOT_FOUND"
ERR_PAYMENT: str = "PAYMENT_FAILED"
ERR_INTERNAL: str = "INTERNAL_ERROR"

_SQL_UNSAFE_PATTERN = re.compile(r"[;'\"\\]|--")


# --- Configuration (R-019: externalized config object) ----------------------


@dataclass(frozen=True)
class ServiceConfig:
    """Immutable configuration for the order service.

    Attributes:
        database_url: Connection string for the order datastore.
        payment_api_key: Secret key used to authenticate with the payment API.
        rate_limit_per_minute: Maximum write operations allowed per minute.
    """

    database_url: str = DEFAULT_DB_URL
    payment_api_key: str = ""
    rate_limit_per_minute: int = DEFAULT_RATE_LIMIT_PER_MINUTE


# --- Domain exceptions (R-007) ----------------------------------------------


class OrderServiceError(Exception):
    """Base class for all order service errors."""

    error_code: str = ERR_INTERNAL


class ValidationError(OrderServiceError):
    """Raised when input fails validation."""

    error_code = ERR_VALIDATION


class PermissionDeniedError(OrderServiceError):
    """Raised when the caller lacks permission for the operation."""

    error_code = ERR_PERMISSION


class OrderNotFoundError(OrderServiceError):
    """Raised when a referenced order does not exist."""

    error_code = ERR_NOT_FOUND


class PaymentError(OrderServiceError):
    """Raised when a payment or refund operation fails."""

    error_code = ERR_PAYMENT


# --- Collaborator implementations (injected dependencies) -------------------


class OrderRepository:
    """In-memory order repository. Replace with a real datastore in production."""

    def __init__(self) -> None:
        self._orders: dict[str, dict[str, Any]] = {}

    def save(self, order: dict[str, Any]) -> None:
        """Persist an order record."""
        self._orders[order["order_id"]] = order

    def get(self, order_id: str) -> Optional[dict[str, Any]]:
        """Return the order record or None if absent."""
        return self._orders.get(order_id)

    def exists(self, order_id: str) -> bool:
        """Return True if an order with the given id is stored."""
        return order_id in self._orders


class PaymentGateway:
    """Reference payment gateway. Replace with a real integration."""

    def __init__(self, api_key: str) -> None:
        self._api_key = api_key

    def charge(self, payment_token: str, amount: float) -> str:
        """Charge the token and return a transaction id."""
        return f"txn_{uuid.uuid4().hex}"

    def refund(self, transaction_id: str, amount: float) -> str:
        """Refund a transaction and return a refund id."""
        return f"{REFUND_ID_PREFIX}_{uuid.uuid4().hex}"


# --- Helpers ----------------------------------------------------------------


def _now_iso() -> str:
    """Return the current UTC timestamp in ISO-8601 format."""
    return datetime.now(timezone.utc).isoformat()


def _new_correlation_id() -> str:
    """Generate a unique correlation id for request tracing (R-016)."""
    return uuid.uuid4().hex


def _error_response(exc: OrderServiceError, correlation_id: str) -> dict[str, Any]:
    """Build a structured error response (R-010)."""
    return {
        "error_code": exc.error_code,
        "message": str(exc),
        "timestamp": _now_iso(),
        "correlation_id": correlation_id,
    }


def _require_non_empty_str(value: Any, name: str) -> None:
    """Validate that ``value`` is a non-empty string (R-001)."""
    if not isinstance(value, str) or not value.strip():
        raise ValidationError(f"{name} must be a non-empty string")


def _check_sql_safe(value: str, name: str) -> None:
    """Reject strings containing characters unsafe for queries (R-003)."""
    if _SQL_UNSAFE_PATTERN.search(value):
        raise ValidationError(f"{name} contains unsafe characters")


def _verify_permission(user_id: str, order: dict[str, Any]) -> None:
    """Verify the caller owns the order before privileged access (R-020)."""
    if order.get("user_id") != user_id:
        raise PermissionDeniedError("caller does not own this order")


def _validate_item(item: Any, index: int) -> dict[str, Any]:
    """Validate a single order item and return a normalized copy (R-001, R-002).

    Args:
        item: Candidate item mapping with name, quantity, and unit_price.
        index: Position of the item, used in error messages.

    Returns:
        A normalized item dict with an HTML-escaped name.

    Raises:
        ValidationError: If any field is missing or invalid.
    """
    if not isinstance(item, dict):
        raise ValidationError(f"item[{index}] must be a dict")
    name, quantity, unit_price = (
        item.get("name"), item.get("quantity"), item.get("unit_price"),
    )
    _require_non_empty_str(name, f"item[{index}].name")
    _check_sql_safe(name, f"item[{index}].name")
    if not isinstance(quantity, int) or isinstance(quantity, bool):
        raise ValidationError(f"item[{index}].quantity must be an int")
    if quantity < MIN_QUANTITY:
        raise ValidationError(f"item[{index}].quantity must be >= {MIN_QUANTITY}")
    if not isinstance(unit_price, (int, float)) or isinstance(unit_price, bool):
        raise ValidationError(f"item[{index}].unit_price must be numeric")
    if unit_price < MIN_UNIT_PRICE:
        raise ValidationError(f"item[{index}].unit_price must be >= {MIN_UNIT_PRICE}")
    return {"name": html.escape(name), "quantity": quantity, "unit_price": float(unit_price)}


def _validate_items(items: Any) -> list[dict[str, Any]]:
    """Validate the items collection and return normalized items (R-001).

    Args:
        items: Candidate list of item mappings.

    Returns:
        A list of normalized item dicts.

    Raises:
        ValidationError: If the collection or any item is invalid.
    """
    if not isinstance(items, list) or not items:
        raise ValidationError("items must be a non-empty list")
    return [_validate_item(item, index) for index, item in enumerate(items)]


def _calculate_total(items: list[dict[str, Any]]) -> float:
    """Return the summed price of all validated items.

    Args:
        items: Normalized item dicts.

    Returns:
        The order total rounded to two decimal places.
    """
    return round(sum(i["quantity"] * i["unit_price"] for i in items), 2)


# --- Service ----------------------------------------------------------------


class OrderService:
    """Coordinates order creation, cancellation, and status retrieval."""

    def __init__(
        self,
        config: Optional[ServiceConfig] = None,
        repository: Optional[OrderRepository] = None,
        payment_gateway: Optional[PaymentGateway] = None,
        id_factory: Callable[[], str] = lambda: uuid.uuid4().hex,
    ) -> None:
        """Initialize the service with injected configuration and collaborators.

        Args:
            config: Service configuration; defaults are used when omitted.
            repository: Order datastore; an in-memory store is used by default.
            payment_gateway: Payment integration; a reference impl by default.
            id_factory: Callable producing unique identifier suffixes.
        """
        self._config = config or ServiceConfig()
        self._repo = repository or OrderRepository()
        self._payments = payment_gateway or PaymentGateway(self._config.payment_api_key)
        self._id_factory = id_factory

    def create_order(
        self, user_id: str, items: list, payment_token: str
    ) -> dict[str, Any]:
        """Create an order, charge payment, and return a confirmation.

        Args:
            user_id: Identifier of the ordering user.
            items: List of dicts each with name, quantity, and unit_price.
            payment_token: Opaque token authorizing the charge.

        Returns:
            A confirmation dict on success or a structured error dict on failure.

        Raises:
            ValidationError: If any argument fails validation.
            PaymentError: If the payment gateway rejects the charge.
        """
        correlation_id = _new_correlation_id()
        logger.debug(
            "enter create_order user_id=%s item_count=%s correlation_id=%s",
            user_id, len(items) if isinstance(items, list) else "n/a", correlation_id,
        )
        try:
            result = self._do_create_order(user_id, items, payment_token, correlation_id)
            logger.debug(
                "exit create_order return_type=%s correlation_id=%s",
                type(result).__name__, correlation_id,
            )
            return result
        except OrderServiceError as exc:
            logger.error(
                "create_order failed correlation_id=%s", correlation_id, exc_info=True
            )
            return _error_response(exc, correlation_id)

    def _do_create_order(
        self, user_id: str, items: list, payment_token: str, correlation_id: str
    ) -> dict[str, Any]:
        """Validate, charge, and persist a new order (R-022 idempotent)."""
        _require_non_empty_str(user_id, "user_id")
        _require_non_empty_str(payment_token, "payment_token")
        _check_sql_safe(user_id, "user_id")
        normalized = _validate_items(items)
        order_id = f"{ORDER_ID_PREFIX}_{self._id_factory()}"
        if self._repo.exists(order_id):  # idempotency guard
            raise ValidationError("generated order_id already exists; retry")
        total = _calculate_total(normalized)
        transaction_id = self._charge(payment_token, total)
        order = self._build_order(user_id, order_id, normalized, total, transaction_id)
        self._repo.save(order)
        logger.info(  # R-015: state-changing op at INFO
            "order created order_id=%s total=%s correlation_id=%s",
            order_id, total, correlation_id,
        )
        return self._confirmation(order, correlation_id)

    def _charge(self, payment_token: str, total: float) -> str:
        """Charge the payment token, wrapping gateway failures (R-008)."""
        try:
            return self._payments.charge(payment_token, total)
        except Exception as exc:  # noqa: BLE001 - re-raised as domain error
            raise PaymentError(f"charge failed: {exc}") from exc

    @staticmethod
    def _build_order(
        user_id: str,
        order_id: str,
        items: list[dict[str, Any]],
        total: float,
        transaction_id: str,
    ) -> dict[str, Any]:
        """Assemble the persisted order record."""
        return {
            "order_id": order_id,
            "user_id": user_id,
            "items": items,
            "total": total,
            "transaction_id": transaction_id,
            "status": STATUS_CONFIRMED,
            "created_at": _now_iso(),
        }

    @staticmethod
    def _confirmation(order: dict[str, Any], correlation_id: str) -> dict[str, Any]:
        """Build the public confirmation payload for a created order."""
        return {
            "order_id": order["order_id"],
            "status": order["status"],
            "total": order["total"],
            "transaction_id": order["transaction_id"],
            "timestamp": order["created_at"],
            "correlation_id": correlation_id,
        }

    def cancel_order(self, user_id: str, order_id: str, reason: str) -> dict[str, Any]:
        """Cancel an order, issue a refund, and return a confirmation.

        Args:
            user_id: Identifier of the user requesting cancellation.
            order_id: Identifier of the order to cancel.
            reason: Human-readable cancellation reason.

        Returns:
            A cancellation confirmation dict or a structured error dict.

        Raises:
            ValidationError: If arguments are invalid.
            PermissionDeniedError: If the caller does not own the order.
            OrderNotFoundError: If the order does not exist.
            PaymentError: If the refund fails.
        """
        correlation_id = _new_correlation_id()
        logger.debug(
            "enter cancel_order order_id=%s correlation_id=%s", order_id, correlation_id
        )
        try:
            result = self._do_cancel_order(user_id, order_id, reason, correlation_id)
            logger.debug(
                "exit cancel_order return_type=%s correlation_id=%s",
                type(result).__name__, correlation_id,
            )
            return result
        except OrderServiceError as exc:
            logger.error(
                "cancel_order failed correlation_id=%s", correlation_id, exc_info=True
            )
            return _error_response(exc, correlation_id)

    def _do_cancel_order(
        self, user_id: str, order_id: str, reason: str, correlation_id: str
    ) -> dict[str, Any]:
        """Validate, authorize, refund, and mark an order cancelled."""
        _require_non_empty_str(user_id, "user_id")
        _require_non_empty_str(order_id, "order_id")
        _require_non_empty_str(reason, "reason")
        order = self._load_owned_order(user_id, order_id)
        if order["status"] == STATUS_CANCELLED:  # R-022: idempotent re-cancel
            return self._cancellation(order, correlation_id)
        refund_id = self._refund(order)
        order.update(
            status=STATUS_CANCELLED,
            cancel_reason=html.escape(reason),
            refund_id=refund_id,
            cancelled_at=_now_iso(),
        )
        self._repo.save(order)
        logger.info(
            "order cancelled order_id=%s refund_id=%s correlation_id=%s",
            order_id, refund_id, correlation_id,
        )
        return self._cancellation(order, correlation_id)

    def _refund(self, order: dict[str, Any]) -> str:
        """Issue a refund for an order, wrapping gateway failures (R-008)."""
        try:
            return self._payments.refund(order["transaction_id"], order["total"])
        except Exception as exc:  # noqa: BLE001 - re-raised as domain error
            raise PaymentError(f"refund failed: {exc}") from exc

    @staticmethod
    def _cancellation(order: dict[str, Any], correlation_id: str) -> dict[str, Any]:
        """Build the public cancellation payload."""
        return {
            "order_id": order["order_id"],
            "status": order["status"],
            "refund_id": order.get("refund_id"),
            "timestamp": order.get("cancelled_at"),
            "correlation_id": correlation_id,
        }

    def get_order_status(self, user_id: str, order_id: str) -> dict[str, Any]:
        """Retrieve an order's status for its owner.

        Args:
            user_id: Identifier of the requesting user.
            order_id: Identifier of the order to inspect.

        Returns:
            An order status dict or a structured error dict.

        Raises:
            ValidationError: If arguments are invalid.
            PermissionDeniedError: If the caller does not own the order.
            OrderNotFoundError: If the order does not exist.
        """
        correlation_id = _new_correlation_id()
        logger.debug(
            "enter get_order_status order_id=%s correlation_id=%s",
            order_id, correlation_id,
        )
        try:
            result = self._do_get_status(user_id, order_id, correlation_id)
            logger.debug(
                "exit get_order_status return_type=%s correlation_id=%s",
                type(result).__name__, correlation_id,
            )
            return result
        except OrderServiceError as exc:
            logger.error(
                "get_order_status failed correlation_id=%s",
                correlation_id, exc_info=True,
            )
            return _error_response(exc, correlation_id)

    def _do_get_status(
        self, user_id: str, order_id: str, correlation_id: str
    ) -> dict[str, Any]:
        """Validate, authorize, and return an order status payload."""
        _require_non_empty_str(user_id, "user_id")
        _require_non_empty_str(order_id, "order_id")
        order = self._load_owned_order(user_id, order_id)
        return {
            "order_id": order["order_id"],
            "status": order["status"],
            "total": order["total"],
            "items": order["items"],
            "timestamp": _now_iso(),
            "correlation_id": correlation_id,
        }

    def _load_owned_order(self, user_id: str, order_id: str) -> dict[str, Any]:
        """Fetch an order and verify caller ownership (R-020)."""
        _check_sql_safe(order_id, "order_id")
        order = self._repo.get(order_id)
        if order is None:
            raise OrderNotFoundError(f"order {order_id} not found")
        _verify_permission(user_id, order)
        return order
```

A few notes on how the implementation maps to your 20 rules:

- **R-001/002/003/004** — `_validate_items`/`_validate_item` type-check every parameter, reject `quantity < 1`, screen strings for SQL-unsafe characters, and HTML-escape names.
- **R-007/008/009/010** — domain exception hierarchy (`OrderServiceError` + subclasses), `raise … from exc` preserves context, `logger.error(..., exc_info=True)` logs stack traces, and `_error_response` returns `{error_code, message, timestamp}` (plus `correlation_id`).
- **R-013/014/015/016** — DEBUG entry/exit logs, INFO logs on create/cancel state changes, `correlation_id` threaded through every log line and response.
- **R-019/020/021/022/025/026/027/028** — `ServiceConfig` holds DB URL / API key / rate limit; `_verify_permission` gates owner-only access; gateway calls are wrapped; idempotency guards on order-id collision and re-cancel; full type hints, Args/Returns/Raises docstrings, named constants, and functions kept under 30 lines via private helpers.

One deviation worth flagging: **R-021 ("release resources in finally blocks")** has no literal `finally` here because the in-memory repository and reference gateway hold no resources (connections, file handles) that require deterministic release. If you swap in a real DB/HTTP client, the place to add `try/finally` cleanup is inside `OrderRepository`/`PaymentGateway`.
