Provides order creation, cancellation, and status retrieval with input

ERROR_CODE_VALIDATION = "VALIDATION_ERROR"

def _require_non_empty_str(value: Any, field_name: str) -> str:
    """Validate that a value is a non-empty string.

    Args:
        value: The value to validate.
        field_name: Name of the field, used in error messages.

    Returns:
        The validated string.

    Raises:
        ValidationError: If the value is not a non-empty string.
    """
    if not isinstance(value, str) or not value.strip():
        raise ValidationError(f"{field_name} must be a non-empty string")
    return value


def _sanitize_for_query(value: str, field_name: str) -> str:
    """Reject string inputs containing characters unsafe for queries.

    Args:
        value: The raw string to sanitize.
        field_name: Name of the field, used in error messages.

    Returns:
        The original string if it is safe.

    Raises:
        ValidationError: If unsafe characters are present.
    """
    if _SQL_UNSAFE_PATTERN.search(value):
        raise ValidationError(f"{field_name} contains illegal characters")
    return value


def _validate_identifier(value: Any, field_name: str) -> str:
    """Validate and sanitize an identifier used in queries.

    Args:
        value: The candidate identifier.
        field_name: Name of the field, used in error messages.

    Returns:
        The validated, query-safe identifier.

    Raises:
        ValidationError: If the identifier is invalid or unsafe.
    """
    return _sanitize_for_query(_require_non_empty_str(value, field_name), field_name)


def _validate_quantity(value: Any, index: int) -> int:
    """Validate an item quantity (R-002: reject negatives).

    Args:
        value: The quantity to validate.
        index: Item position, for error messages.

    Returns:
        The validated integer quantity.

    Raises:
        ValidationError: If quantity is not an int >= MIN_QUANTITY.
    """
    if isinstance(value, bool) or not isinstance(value, int):
        raise ValidationError(f"item[{index}].quantity must be an integer")
    if value < MIN_QUANTITY:
        raise ValidationError(f"item[{index}].quantity must be >= {MIN_QUANTITY}")
    return value


def _validate_unit_price(value: Any, index: int) -> float:
    """Validate an item unit price (R-002: reject negatives).

    Args:
        value: The unit price to validate.
        index: Item position, for error messages.

    Returns:
        The validated float unit price.

    Raises:
        ValidationError: If price is not a number >= MIN_UNIT_PRICE.
    """
    if isinstance(value, bool) or not isinstance(value, (int, float)):
        raise ValidationError(f"item[{index}].unit_price must be a number")
    if value < MIN_UNIT_PRICE:
        raise ValidationError(f"item[{index}].unit_price must be >= {MIN_UNIT_PRICE}")
    return float(value)


def _validate_item(item: Any, index: int) -> Dict[str, Any]:
    """Validate a single order item.

    Args:
        item: The item mapping to validate.
        index: Position of the item in the list, for error messages.

    Returns:
        A normalized item dict with escaped name and numeric fields.

    Raises:
        ValidationError: If the item is malformed or has negative quantity.
    """
    if not isinstance(item, dict):
        raise ValidationError(f"item[{index}] must be a dict")
    for required in REQUIRED_ITEM_FIELDS:
        if required not in item:
            raise ValidationError(f"item[{index}] missing field '{required}'")
    quantity = _validate_quantity(item["quantity"], index)
    unit_price = _validate_unit_price(item["unit_price"], index)
    name = html.escape(_require_non_empty_str(item["name"], f"item[{index}].name"))
    return {"name": name, "quantity": quantity, "unit_price": unit_price}


def _validate_items(items: Any) -> List[Dict[str, Any]]:
    """Validate the full list of order items.

    Args:
        items: The items collection to validate.

    Returns:
        A list of normalized item dicts.

    Raises:
        ValidationError: If items is not a non-empty list or any item fails.
    """
    if not isinstance(items, list) or not items:
        raise ValidationError("items must be a non-empty list")
    return [_validate_item(item, index) for index, item in enumerate(items)]


def _calculate_total(items: List[Dict[str, Any]]) -> float:
    """Calculate the rounded order total from validated items.

    Args:
        items: Validated item dicts containing quantity and unit_price.

    Returns:
        The total price rounded to PRICE_ROUNDING_DP decimal places.
    """
    total = sum(item["quantity"] * item["unit_price"] for item in items)
    return round(total, PRICE_ROUNDING_DP)

class PaymentGateway:
    """Stub payment gateway. Replace with a real integration in production."""

    def __init__(self, api_key: str) -> None:
        """Initialize the gateway.

        Args:
            api_key: API key used to authenticate with the provider.

    def charge(self, payment_token: str, amount: float, correlation_id: str) -> str:
        """Charge a payment token.

        Raises:
            PaymentError: If the refund cannot be completed.

        if not transaction_id:
            raise PaymentError("missing transaction id for refund")
        logger.info("Refunding amount=%s txn=%s", amount, transaction_id,
                    extra={"correlation_id": correlation_id})
        return f"ref_{uuid.uuid4().hex}"


class OrderRepository:
    """In-memory order repository. Replace with a real datastore in production."""

    def __init__(self) -> None:
        """Initialize an empty repository."""
        self._orders: Dict[str, Dict[str, Any]] = {}

        """Fetch an order by id.

    def save(self, order: Dict[str, Any]) -> None:
        """Persist an order.

        Args:
            order: The order dict to store.

        Returns:
            None.

        self._orders[order["order_id"]] = order

    def find_by_idempotency_key(self, key: str) -> Optional[str]:
        """Look up an order id by idempotency key.

        Args:
            key: The idempotency key.

        Returns:
            The associated order id, or None.
        """
        return self._idempotency_index.get(key)

    def bind_idempotency_key(self, key: str, order_id: str) -> None:
        """Associate an idempotency key with an order id.

        Args:
            key: The idempotency key.
            order_id: The order id to bind.

        Returns:
            None.
        """
        self._idempotency_index[key] = order_id

class OrderService:
    """Coordinates order operations across the repository and payment gateway."""

    def __init__(
        self,
        config: Optional[ServiceConfig] = None,
        repository: Optional[OrderRepository] = None,
        gateway: Optional[PaymentGateway] = None,
        permission_checker: Optional[Callable[[str, str], bool]] = None,
    ) -> None:
        """Initialize the service and its collaborators.

        Args:
            config: Service configuration; defaults to ServiceConfig().
            repository: Order repository; defaults to in-memory implementation.
            gateway: Payment gateway; defaults to stub implementation.
            permission_checker: Callable(user_id, action) -> bool used to
                authorize privileged operations; defaults to allow-all.
        """
        self.config = config or ServiceConfig()
        self._repo = repository or OrderRepository()
        self._gateway = gateway or PaymentGateway(self.config.payment_api_key)
        self._permission_checker = permission_checker or (lambda user, action: True)

    # ----- permission (R-020) -------------------------------------------- #
    def _require_permission(self, user_id: str, action: str, correlation_id: str) -> None:
        """Verify the caller may perform a privileged action.

        Args:
            user_id: The acting user.
            action: The privileged action name.
            correlation_id: Request correlation identifier.

        Returns:
            None.

        Raises:
            PermissionDeniedError: If the caller lacks permission.
        """
        if not self._permission_checker(user_id, action):
            raise PermissionDeniedError(f"user '{user_id}' may not perform '{action}'")
        logger.debug("Permission granted action=%s", action,
                     extra={"correlation_id": correlation_id})

    # ----- create (R-022 idempotency) ------------------------------------ #
    def create_order(
        self, user_id: str, items: List[Dict[str, Any]], payment_token: str
    ) -> Dict[str, Any]:
        """Create a new order, charge payment, and return a confirmation.

        Args:
            user_id: Identifier of the ordering user.
            items: List of item dicts with name, quantity, and unit_price.
            payment_token: Tokenized payment instrument.

        Returns:
            An order confirmation dict.

        Raises:
            ValidationError: If any input is invalid.
            PermissionDeniedError: If the user may not create orders.
            PaymentError: If payment processing fails.
        """
        correlation_id = _new_correlation_id()
        _log_entry("create_order", correlation_id,
                   {"user_id": user_id, "item_count": _safe_len(items)})
        try:
            result = self._create_order_impl(user_id, items, payment_token, correlation_id)
        except OrderServiceError:
            raise
        except Exception as exc:  # noqa: BLE001 - wrap unexpected errors
            logger.error("create_order failed", exc_info=True,
                         extra={"correlation_id": correlation_id})
            raise OrderServiceError("unexpected error creating order") from exc
        finally:
            logger.debug("create_order finished", extra={"correlation_id": correlation_id})
        _log_exit("create_order", correlation_id, result)
        return result

    def _create_order_impl(
        self, user_id: str, items: Any, payment_token: str, correlation_id: str
    ) -> Dict[str, Any]:
        """Validate inputs and perform the order creation steps.

        Args:
            user_id: Identifier of the ordering user.
            items: Raw list of item dicts.
            payment_token: Tokenized payment instrument.
            correlation_id: Request correlation identifier.

        Returns:
            An order confirmation dict.

        Raises:
            ValidationError, PermissionDeniedError, PaymentError: On failure.
        """
        valid_user = _validate_identifier(user_id, "user_id")
        valid_token = _require_non_empty_str(payment_token, "payment_token")
        valid_items = _validate_items(items)
        self._require_permission(valid_user, "create_order", correlation_id)
        existing = self._repo.find_by_idempotency_key(valid_token)
        if existing is not None:
            logger.info("Idempotent create hit order=%s", existing,
                        extra={"correlation_id": correlation_id})
            return self._build_confirmation(self._repo.get(existing))
        return self._persist_new_order(valid_user, valid_items, valid_token, correlation_id)

    def _persist_new_order(
        self, user_id: str, items: List[Dict[str, Any]], token: str, correlation_id: str
    ) -> Dict[str, Any]:
        """Charge payment and store a brand new order.

        Args:
            user_id: Validated user id.
            items: Validated item dicts.
            token: Validated payment token.
            correlation_id: Request correlation identifier.

        Returns:
            An order confirmation dict.

        Raises:
            PaymentError: If the charge fails.
        """
        total = _calculate_total(items)
        order_id = f"ord_{uuid.uuid4().hex}"
        transaction_id = self._gateway.charge(token, total, correlation_id)
        order = _build_order_record(order_id, user_id, items, total,
                                    transaction_id, self.config.currency)
        self._repo.save(order)
        self._repo.bind_idempotency_key(token, order_id)
        logger.info("Order created order=%s total=%s", order_id, total,
                    extra={"correlation_id": correlation_id})
        return self._build_confirmation(order)

    @staticmethod
    def _build_confirmation(order: Dict[str, Any]) -> Dict[str, Any]:
        """Build a confirmation view of an order.

        Args:
            order: The stored order record.

        Returns:
            A confirmation dict safe to return to callers.
        """
        return {
            "order_id": order["order_id"],
            "status": order["status"],
            "total": order["total"],
            "currency": order["currency"],
            "items": order["items"],
            "created_at": order["created_at"],
        }

    # ----- cancel (R-022 idempotency) ------------------------------------ #
    def cancel_order(self, user_id: str, order_id: str, reason: str) -> Dict[str, Any]:
        """Cancel an existing order and issue a refund.

        Args:
            user_id: Identifier of the requesting user.
            order_id: Identifier of the order to cancel.
            reason: Human-provided cancellation reason.

        Returns:
            A cancellation confirmation dict.

        Raises:
            ValidationError: If any input is invalid.
            PermissionDeniedError: If the user may not cancel the order.
            OrderNotFoundError: If the order does not exist.
            InvalidOrderStateError: If the order cannot be cancelled.
            PaymentError: If the refund fails.
        """
        correlation_id = _new_correlation_id()
        _log_entry("cancel_order", correlation_id,
                   {"user_id": user_id, "order_id": order_id})
        try:
            result = self._cancel_order_impl(user_id, order_id, reason, correlation_id)
        except OrderServiceError:
            raise
        except Exception as exc:  # noqa: BLE001
            logger.error("cancel_order failed", exc_info=True,
                         extra={"correlation_id": correlation_id})
            raise OrderServiceError("unexpected error cancelling order") from exc
        finally:
            logger.debug("cancel_order finished", extra={"correlation_id": correlation_id})
        _log_exit("cancel_order", correlation_id, result)
        return result

    def _cancel_order_impl(
        self, user_id: str, order_id: str, reason: str, correlation_id: str
    ) -> Dict[str, Any]:
        """Validate inputs and perform cancellation steps.

        Args:
            user_id: Requesting user id.
            order_id: Order to cancel.
            reason: Cancellation reason.
            correlation_id: Request correlation identifier.

        Returns:
            A cancellation confirmation dict.

        Raises:
            ValidationError, PermissionDeniedError, OrderNotFoundError,
            InvalidOrderStateError, PaymentError: On failure.
        """
        valid_user = _validate_identifier(user_id, "user_id")
        valid_order_id = _validate_identifier(order_id, "order_id")
        valid_reason = _validate_reason(reason)
        self._require_permission(valid_user, "cancel_order", correlation_id)
        order = self._load_owned_order(valid_user, valid_order_id)
        if order["status"] == STATUS_CANCELLED:  # R-022 idempotency guard
            logger.info("Idempotent cancel hit order=%s", valid_order_id,
                        extra={"correlation_id": correlation_id})
            return _build_cancellation(order)
        return self._apply_cancellation(order, valid_reason, correlation_id)

    def _apply_cancellation(
        self, order: Dict[str, Any], reason: str, correlation_id: str
    ) -> Dict[str, Any]:
        """Refund and mark an order cancelled.

        Args:
            order: The order record being cancelled.
            reason: Validated cancellation reason.
            correlation_id: Request correlation identifier.

        Returns:
            A cancellation confirmation dict.

        Raises:
            InvalidOrderStateError: If the order is not in a cancellable state.
            PaymentError: If the refund fails.
        """
        if order["status"] != STATUS_CONFIRMED:
            raise InvalidOrderStateError(f"order in state '{order['status']}' cannot cancel")
        refund_id = self._gateway.refund(order["transaction_id"], order["total"], correlation_id)
        order["status"] = STATUS_CANCELLED
        order["reason"] = reason
        order["refund_id"] = refund_id
        order["cancelled_at"] = _now_iso()
        self._repo.save(order)
        logger.info("Order cancelled order=%s refund=%s", order["order_id"], refund_id,
                    extra={"correlation_id": correlation_id})
        return _build_cancellation(order)

    # ----- status -------------------------------------------------------- #
    def get_order_status(self, user_id: str, order_id: str) -> Dict[str, Any]:
        """Retrieve the status of an order.

        Args:
            user_id: Identifier of the requesting user.
            order_id: Identifier of the order.

        Returns:
            An order status dict.

        Raises:
            ValidationError: If any input is invalid.
            PermissionDeniedError: If the user may not view the order.
            OrderNotFoundError: If the order does not exist.
        """
        correlation_id = _new_correlation_id()
        _log_entry("get_order_status", correlation_id,
                   {"user_id": user_id, "order_id": order_id})
        try:
            result = self._get_status_impl(user_id, order_id, correlation_id)
        except OrderServiceError:
            raise
        except Exception as exc:  # noqa: BLE001
            logger.error("get_order_status failed", exc_info=True,
                         extra={"correlation_id": correlation_id})
            raise OrderServiceError("unexpected error reading order") from exc
        finally:
            logger.debug("get_order_status finished", extra={"correlation_id": correlation_id})
        _log_exit("get_order_status", correlation_id, result)
        return result

    def _get_status_impl(
        self, user_id: str, order_id: str, correlation_id: str
    ) -> Dict[str, Any]:
        """Validate inputs and read the order status.

        Args:
            user_id: Requesting user id.
            order_id: Order to read.
            correlation_id: Request correlation identifier.

        Returns:
            An order status dict.

        Raises:
            ValidationError, PermissionDeniedError, OrderNotFoundError: On failure.
        """
        valid_user = _validate_identifier(user_id, "user_id")
        valid_order_id = _validate_identifier(order_id, "order_id")
        self._require_permission(valid_user, "get_order_status", correlation_id)
        order = self._load_owned_order(valid_user, valid_order_id)
        return {
            "order_id": order["order_id"],
            "status": order["status"],
            "total": order["total"],
            "currency": order["currency"],
            "created_at": order["created_at"],
        }

    def _load_owned_order(self, user_id: str, order_id: str) -> Dict[str, Any]:
        """Load an order and verify ownership.

        Args:
            user_id: The user expected to own the order.
            order_id: The order identifier.

        Returns:
            The order record.

        Raises:
            OrderNotFoundError: If the order is missing.
            PermissionDeniedError: If the order belongs to another user.
        """
        order = self._repo.get(order_id)
        if order is None:
            raise OrderNotFoundError(f"order '{order_id}' not found")
        if order["user_id"] != user_id:
            raise PermissionDeniedError(f"user '{user_id}' does not own order '{order_id}'")
        return order

def _safe_len(value: Any) -> int:
    """Return len(value) without raising for non-sized inputs.

    Args:
        value: Any value.

    Returns:
        The length, or 0 if the value has no length.
    """
    try:
        return len(value)
    except TypeError:
        return 0


def _validate_reason(reason: Any) -> str:
    """Validate a cancellation reason.

    Args:
        reason: The reason text.

    Returns:
        The HTML-escaped, length-checked reason.

    Raises:
        ValidationError: If the reason is empty or too long.
    """
    text = _require_non_empty_str(reason, "reason")
    if len(text) > MAX_REASON_LENGTH:
        raise ValidationError(f"reason exceeds {MAX_REASON_LENGTH} characters")
    return html.escape(text)


def _build_order_record(
    order_id: str,
    user_id: str,
    items: List[Dict[str, Any]],
    total: float,
    transaction_id: str,
    currency: str,
) -> Dict[str, Any]:
    """Assemble a persisted order record.

    Args:
        order_id: Unique order identifier.
        user_id: Owning user id.
        items: Validated item dicts.
        total: Computed order total.
        transaction_id: Payment transaction id.
        currency: ISO currency code.

    Returns:
        A complete order record dict.
    """
    return {
        "order_id": order_id,
        "user_id": user_id,
        "items": items,
        "total": total,
        "currency": currency,
        "transaction_id": transaction_id,
        "status": STATUS_CONFIRMED,
        "created_at": _now_iso(),
    }


def _build_cancellation(order: Dict[str, Any]) -> Dict[str, Any]:
    """Build a cancellation confirmation view.

    Args:
        order: The cancelled order record.

    Returns:
        A cancellation confirmation dict.
    """
    return {
        "order_id": order["order_id"],
        "status": order["status"],
        "refund_id": order.get("refund_id"),
        "reason": order.get("reason"),
        "cancelled_at": order.get("cancelled_at"),
    }


def build_error_response(error: OrderServiceError) -> Dict[str, Any]:
    """Build a structured error response from a domain error (R-010).

        A dict with error code, message, and timestamp.
    """
    return {
        "error_code": error.error_code,
        "message": str(error),
        "timestamp": _now_iso(),
    }
