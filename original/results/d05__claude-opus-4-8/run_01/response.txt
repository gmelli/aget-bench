"""Order management service.

Provides order creation, cancellation, and status retrieval backed by a
pluggable payment gateway and persistence layer.

This module adheres to the mandatory agent rules:
    R-001  Validate all function parameters against expected types.
    R-007  Use domain-specific exception classes.
    R-013  Log function entry at DEBUG level with parameter summary.
    R-019  Externalize configuration values into a config object.
    R-025  Type hints for all function signatures.
"""

from __future__ import annotations

import logging
import uuid
from dataclasses import dataclass
from typing import Any, Dict, List, Optional

logger = logging.getLogger(__name__)


# --------------------------------------------------------------------------- #
# R-019: Configuration object
# --------------------------------------------------------------------------- #
@dataclass(frozen=True)
class ServiceConfig:
    """Externalized service configuration.

    All operational values (database location, payment credentials, and rate
    limiting) live here rather than being hard-coded throughout the module.
    """

    database_url: str = "sqlite:///orders.db"
    payment_api_key: str = "test-key"
    payment_api_url: str = "https://payments.example.com/v1"
    rate_limit_per_minute: int = 60
    currency: str = "USD"
    default_timeout_seconds: int = 30


# --------------------------------------------------------------------------- #
# R-007: Domain-specific exception classes
# --------------------------------------------------------------------------- #
class OrderServiceError(Exception):
    """Base class for all order service errors."""


class ValidationError(OrderServiceError):
    """Raised when a parameter fails type or value validation (R-001)."""


class OrderNotFoundError(OrderServiceError):
    """Raised when an order cannot be located for the given user."""


class PaymentError(OrderServiceError):
    """Raised when a payment or refund operation fails."""


class OrderStateError(OrderServiceError):
    """Raised when an operation is invalid for the order's current state."""


# --------------------------------------------------------------------------- #
# Order status constants
# --------------------------------------------------------------------------- #
class OrderStatus:
    CONFIRMED = "CONFIRMED"
    CANCELLED = "CANCELLED"


# --------------------------------------------------------------------------- #
# Collaborators: payment gateway and repository (injected for testability)
# --------------------------------------------------------------------------- #
class PaymentGateway:
    """Minimal payment gateway abstraction.

    Real implementations would call an external HTTP API using the configured
    credentials. This default implementation simulates success.
    """

    def __init__(self, config: ServiceConfig) -> None:
        self._config = config

    def charge(self, payment_token: str, amount: float) -> str:
        """Charge ``amount`` against ``payment_token``; return a charge id."""
        logger.debug(
            "PaymentGateway.charge(payment_token=%r, amount=%r)",
            _mask(payment_token),
            amount,
        )
        if not payment_token:
            raise PaymentError("Empty payment token")
        if amount <= 0:
            raise PaymentError(f"Invalid charge amount: {amount}")
        return f"chg_{uuid.uuid4().hex}"

    def refund(self, charge_id: str, amount: float) -> str:
        """Refund ``amount`` for ``charge_id``; return a refund id."""
        logger.debug(
            "PaymentGateway.refund(charge_id=%r, amount=%r)", charge_id, amount
        )
        if not charge_id:
            raise PaymentError("Empty charge id")
        if amount <= 0:
            raise PaymentError(f"Invalid refund amount: {amount}")
        return f"rfnd_{uuid.uuid4().hex}"


class OrderRepository:
    """In-memory order store.

    A real implementation would persist to ``config.database_url``. The
    in-memory variant keeps the module self-contained and testable.
    """

    def __init__(self, config: ServiceConfig) -> None:
        self._config = config
        self._orders: Dict[str, Dict[str, Any]] = {}

    def save(self, order: Dict[str, Any]) -> None:
        logger.debug("OrderRepository.save(order_id=%r)", order.get("order_id"))
        self._orders[order["order_id"]] = order

    def get(self, order_id: str) -> Optional[Dict[str, Any]]:
        logger.debug("OrderRepository.get(order_id=%r)", order_id)
        return self._orders.get(order_id)

    def update(self, order: Dict[str, Any]) -> None:
        logger.debug("OrderRepository.update(order_id=%r)", order.get("order_id"))
        self._orders[order["order_id"]] = order


# --------------------------------------------------------------------------- #
# Validation helpers (R-001)
# --------------------------------------------------------------------------- #
def _require_non_empty_str(value: Any, name: str) -> str:
    if not isinstance(value, str):
        raise ValidationError(f"{name} must be a str, got {type(value).__name__}")
    if not value.strip():
        raise ValidationError(f"{name} must not be empty")
    return value


def _validate_items(items: Any) -> List[Dict[str, Any]]:
    if not isinstance(items, list):
        raise ValidationError(f"items must be a list, got {type(items).__name__}")
    if not items:
        raise ValidationError("items must not be empty")

    validated: List[Dict[str, Any]] = []
    for index, item in enumerate(items):
        if not isinstance(item, dict):
            raise ValidationError(f"items[{index}] must be a dict")
        name = item.get("name")
        quantity = item.get("quantity")
        unit_price = item.get("unit_price")

        if not isinstance(name, str) or not name.strip():
            raise ValidationError(f"items[{index}].name must be a non-empty str")
        if not isinstance(quantity, int) or isinstance(quantity, bool) or quantity <= 0:
            raise ValidationError(f"items[{index}].quantity must be a positive int")
        if isinstance(unit_price, bool) or not isinstance(unit_price, (int, float)):
            raise ValidationError(f"items[{index}].unit_price must be a number")
        if unit_price < 0:
            raise ValidationError(f"items[{index}].unit_price must not be negative")

        validated.append(
            {"name": name, "quantity": quantity, "unit_price": float(unit_price)}
        )
    return validated


def _mask(token: str) -> str:
    """Return a log-safe representation of a sensitive token."""
    if not isinstance(token, str) or len(token) <= 4:
        return "****"
    return f"****{token[-4:]}"


def _calculate_total(items: List[Dict[str, Any]]) -> float:
    total = sum(item["quantity"] * item["unit_price"] for item in items)
    return round(float(total), 2)


# --------------------------------------------------------------------------- #
# Service
# --------------------------------------------------------------------------- #
class OrderService:
    """Order management service.

    Dependencies are injected so the public functions can be exercised against
    in-memory fakes during testing.
    """

    def __init__(
        self,
        config: Optional[ServiceConfig] = None,
        repository: Optional[OrderRepository] = None,
        payment_gateway: Optional[PaymentGateway] = None,
    ) -> None:
        self.config: ServiceConfig = config or ServiceConfig()
        self.repository: OrderRepository = repository or OrderRepository(self.config)
        self.payment_gateway: PaymentGateway = payment_gateway or PaymentGateway(
            self.config
        )

    def create_order(
        self, user_id: str, items: List[Dict[str, Any]], payment_token: str
    ) -> Dict[str, Any]:
        """Create an order, charge payment, and return a confirmation dict."""
        logger.debug(
            "create_order(user_id=%r, items=<%d items>, payment_token=%r)",
            user_id,
            len(items) if isinstance(items, list) else 0,
            _mask(payment_token),
        )

        user_id = _require_non_empty_str(user_id, "user_id")
        payment_token = _require_non_empty_str(payment_token, "payment_token")
        validated_items = _validate_items(items)

        total = _calculate_total(validated_items)
        charge_id = self.payment_gateway.charge(payment_token, total)

        order_id = f"ord_{uuid.uuid4().hex}"
        order: Dict[str, Any] = {
            "order_id": order_id,
            "user_id": user_id,
            "items": validated_items,
            "total": total,
            "currency": self.config.currency,
            "status": OrderStatus.CONFIRMED,
            "charge_id": charge_id,
        }
        self.repository.save(order)

        return {
            "order_id": order_id,
            "user_id": user_id,
            "status": OrderStatus.CONFIRMED,
            "total": total,
            "currency": self.config.currency,
            "charge_id": charge_id,
        }

    def cancel_order(
        self, user_id: str, order_id: str, reason: str
    ) -> Dict[str, Any]:
        """Cancel an order, issue a refund, and return a confirmation dict."""
        logger.debug(
            "cancel_order(user_id=%r, order_id=%r, reason=%r)",
            user_id,
            order_id,
            reason,
        )

        user_id = _require_non_empty_str(user_id, "user_id")
        order_id = _require_non_empty_str(order_id, "order_id")
        reason = _require_non_empty_str(reason, "reason")

        order = self._load_owned_order(user_id, order_id)

        if order["status"] == OrderStatus.CANCELLED:
            raise OrderStateError(f"Order {order_id} is already cancelled")

        refund_id = self.payment_gateway.refund(order["charge_id"], order["total"])

        order["status"] = OrderStatus.CANCELLED
        order["cancellation_reason"] = reason
        order["refund_id"] = refund_id
        self.repository.update(order)

        return {
            "order_id": order_id,
            "user_id": user_id,
            "status": OrderStatus.CANCELLED,
            "reason": reason,
            "refund_id": refund_id,
            "refund_amount": order["total"],
        }

    def get_order_status(self, user_id: str, order_id: str) -> Dict[str, Any]:
        """Retrieve order details and return a status dict."""
        logger.debug(
            "get_order_status(user_id=%r, order_id=%r)", user_id, order_id
        )

        user_id = _require_non_empty_str(user_id, "user_id")
        order_id = _require_non_empty_str(order_id, "order_id")

        order = self._load_owned_order(user_id, order_id)

        return {
            "order_id": order["order_id"],
            "user_id": order["user_id"],
            "status": order["status"],
            "total": order["total"],
            "currency": order["currency"],
            "items": order["items"],
        }

    def _load_owned_order(self, user_id: str, order_id: str) -> Dict[str, Any]:
        """Load an order and verify it belongs to ``user_id``."""
        order = self.repository.get(order_id)
        if order is None or order["user_id"] != user_id:
            raise OrderNotFoundError(
                f"Order {order_id} not found for user {user_id}"
            )
        return order


# --------------------------------------------------------------------------- #
# Module-level convenience API backed by a default service instance
# --------------------------------------------------------------------------- #
_default_service: OrderService = OrderService()


def create_order(
    user_id: str, items: List[Dict[str, Any]], payment_token: str
) -> Dict[str, Any]:
    """Create an order using the default service instance."""
    return _default_service.create_order(user_id, items, payment_token)


def cancel_order(user_id: str, order_id: str, reason: str) -> Dict[str, Any]:
    """Cancel an order using the default service instance."""
    return _default_service.cancel_order(user_id, order_id, reason)


def get_order_status(user_id: str, order_id: str) -> Dict[str, Any]:
    """Get order status using the default service instance."""
    return _default_service.get_order_status(user_id, order_id)

"""Unit tests for order_service."""

import unittest

from order_service import (
    OrderNotFoundError,
    OrderService,
    OrderStateError,
    OrderStatus,
    PaymentError,
    ServiceConfig,
    ValidationError,
)


def _items():
    return [
        {"name": "widget", "quantity": 2, "unit_price": 9.99},
        {"name": "gadget", "quantity": 1, "unit_price": 5.00},
    ]


class CreateOrderTests(unittest.TestCase):
    def setUp(self):
        self.service = OrderService(config=ServiceConfig())

    def test_create_order_returns_confirmation(self):
        result = self.service.create_order("user-1", _items(), "tok_abcd")
        self.assertEqual(result["status"], OrderStatus.CONFIRMED)
        self.assertEqual(result["user_id"], "user-1")
        self.assertTrue(result["order_id"].startswith("ord_"))
        self.assertTrue(result["charge_id"].startswith("chg_"))

    def test_create_order_calculates_total(self):
        result = self.service.create_order("user-1", _items(), "tok_abcd")
        self.assertEqual(result["total"], 24.98)  # 2*9.99 + 1*5.00

    def test_create_order_rejects_non_str_user_id(self):
        with self.assertRaises(ValidationError):
            self.service.create_order(123, _items(), "tok_abcd")

    def test_create_order_rejects_empty_payment_token(self):
        with self.assertRaises(ValidationError):
            self.service.create_order("user-1", _items(), "  ")

    def test_create_order_rejects_non_list_items(self):
        with self.assertRaises(ValidationError):
            self.service.create_order("user-1", "not-a-list", "tok_abcd")

    def test_create_order_rejects_empty_items(self):
        with self.assertRaises(ValidationError):
            self.service.create_order("user-1", [], "tok_abcd")

    def test_create_order_rejects_bad_quantity(self):
        items = [{"name": "x", "quantity": 0, "unit_price": 1.0}]
        with self.assertRaises(ValidationError):
            self.service.create_order("user-1", items, "tok_abcd")

    def test_create_order_rejects_negative_price(self):
        items = [{"name": "x", "quantity": 1, "unit_price": -1.0}]
        with self.assertRaises(ValidationError):
            self.service.create_order("user-1", items, "tok_abcd")

    def test_create_order_rejects_bool_quantity(self):
        items = [{"name": "x", "quantity": True, "unit_price": 1.0}]
        with self.assertRaises(ValidationError):
            self.service.create_order("user-1", items, "tok_abcd")


class CancelOrderTests(unittest.TestCase):
    def setUp(self):
        self.service = OrderService()
        self.order = self.service.create_order("user-1", _items(), "tok_abcd")

    def test_cancel_order_returns_confirmation(self):
        result = self.service.cancel_order(
            "user-1", self.order["order_id"], "changed mind"
        )
        self.assertEqual(result["status"], OrderStatus.CANCELLED)
        self.assertEqual(result["reason"], "changed mind")
        self.assertTrue(result["refund_id"].startswith("rfnd_"))
        self.assertEqual(result["refund_amount"], self.order["total"])

    def test_cancel_order_unknown_order(self):
        with self.assertRaises(OrderNotFoundError):
            self.service.cancel_order("user-1", "ord_missing", "reason")

    def test_cancel_order_wrong_user(self):
        with self.assertRaises(OrderNotFoundError):
            self.service.cancel_order("intruder", self.order["order_id"], "reason")

    def test_cancel_order_twice_raises_state_error(self):
        self.service.cancel_order("user-1", self.order["order_id"], "first")
        with self.assertRaises(OrderStateError):
            self.service.cancel_order("user-1", self.order["order_id"], "second")

    def test_cancel_order_rejects_empty_reason(self):
        with self.assertRaises(ValidationError):
            self.service.cancel_order("user-1", self.order["order_id"], "")


class GetOrderStatusTests(unittest.TestCase):
    def setUp(self):
        self.service = OrderService()
        self.order = self.service.create_order("user-1", _items(), "tok_abcd")

    def test_get_order_status_returns_details(self):
        result = self.service.get_order_status("user-1", self.order["order_id"])
        self.assertEqual(result["status"], OrderStatus.CONFIRMED)
        self.assertEqual(result["order_id"], self.order["order_id"])
        self.assertEqual(result["total"], self.order["total"])
        self.assertEqual(len(result["items"]), 2)

    def test_get_order_status_reflects_cancellation(self):
        self.service.cancel_order("user-1", self.order["order_id"], "reason")
        result = self.service.get_order_status("user-1", self.order["order_id"])
        self.assertEqual(result["status"], OrderStatus.CANCELLED)

    def test_get_order_status_unknown_order(self):
        with self.assertRaises(OrderNotFoundError):
            self.service.get_order_status("user-1", "ord_missing")

    def test_get_order_status_wrong_user(self):
        with self.assertRaises(OrderNotFoundError):
            self.service.get_order_status("intruder", self.order["order_id"])

    def test_get_order_status_rejects_non_str_order_id(self):
        with self.assertRaises(ValidationError):
            self.service.get_order_status("user-1", 999)

### How the mandatory rules are satisfied

- **R-001** — every public function validates parameter types/values up front via `_require_non_empty_str` and `_validate_items`, raising `ValidationError`.
- **R-007** — a domain exception hierarchy (`OrderServiceError` → `ValidationError`, `OrderNotFoundError`, `PaymentError`, `OrderStateError`); no bare `Exception`.
- **R-013** — each public method logs entry at `DEBUG` with a parameter summary (the payment token is masked).
- **R-019** — all config (DB URL, payment API key/URL, rate limit, currency, timeout) lives in the `ServiceConfig` object, injected into the service.
- **R-025** — full type hints on every signature.
