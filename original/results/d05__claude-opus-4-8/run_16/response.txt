"""order_service.py

An order management service providing order creation, cancellation, and
status retrieval.

This module adheres to the mandatory agent configuration rules:
  R-001: All function parameters are validated against expected types.
  R-007: Domain-specific exception classes are used (not generic Exception).
  R-013: Function entry is logged at DEBUG level with a parameter summary.
  R-019: All configuration values are externalized into a config object.
  R-025: Type hints are provided for all function signatures.
"""

from __future__ import annotations

import logging
import uuid
from dataclasses import dataclass
from typing import Any, Dict, List

logger = logging.getLogger(__name__)


# ---------------------------------------------------------------------------
# R-019: Externalized configuration object
# ---------------------------------------------------------------------------
@dataclass(frozen=True)
class ServiceConfig:
    """Configuration values for the order service.

    All externally tunable values live here rather than being hard-coded
    inside the service functions (R-019).
    """

    database_url: str = "postgresql://localhost:5432/orders"
    payment_api_key: str = "test-payment-api-key"
    payment_api_url: str = "https://payments.example.com/v1"
    currency: str = "USD"
    rate_limit_per_minute: int = 60
    max_items_per_order: int = 100
    order_id_prefix: str = "ORD"
    refund_id_prefix: str = "REF"


# A module-level default config instance. Callers may pass their own.
DEFAULT_CONFIG: ServiceConfig = ServiceConfig()


# ---------------------------------------------------------------------------
# R-007: Domain-specific exception classes
# ---------------------------------------------------------------------------
class OrderServiceError(Exception):
    """Base class for all order service errors."""


class ValidationError(OrderServiceError):
    """Raised when a parameter fails type or value validation (R-001)."""


class OrderNotFoundError(OrderServiceError):
    """Raised when a requested order does not exist."""


class OrderStateError(OrderServiceError):
    """Raised when an operation is invalid for the order's current state."""


class PaymentError(OrderServiceError):
    """Raised when payment processing or refunding fails."""


# ---------------------------------------------------------------------------
# In-memory store (stands in for the database referenced by config).
# ---------------------------------------------------------------------------
_ORDER_STORE: Dict[str, Dict[str, Any]] = {}


# ---------------------------------------------------------------------------
# Validation helpers (R-001)
# ---------------------------------------------------------------------------
def _require_non_empty_str(value: Any, name: str) -> str:
    """Validate that ``value`` is a non-empty string (R-001)."""
    if not isinstance(value, str):
        raise ValidationError(f"{name} must be a str, got {type(value).__name__}")
    if not value.strip():
        raise ValidationError(f"{name} must not be empty")
    return value


def _validate_items(items: Any) -> List[Dict[str, Any]]:
    """Validate the ``items`` parameter structure and contents (R-001)."""
    if not isinstance(items, list):
        raise ValidationError(f"items must be a list, got {type(items).__name__}")
    if not items:
        raise ValidationError("items must not be empty")

    validated: List[Dict[str, Any]] = []
    for index, item in enumerate(items):
        if not isinstance(item, dict):
            raise ValidationError(
                f"items[{index}] must be a dict, got {type(item).__name__}"
            )

        name = item.get("name")
        if not isinstance(name, str) or not name.strip():
            raise ValidationError(f"items[{index}].name must be a non-empty str")

        quantity = item.get("quantity")
        if not isinstance(quantity, int) or isinstance(quantity, bool):
            raise ValidationError(f"items[{index}].quantity must be an int")
        if quantity <= 0:
            raise ValidationError(f"items[{index}].quantity must be positive")

        unit_price = item.get("unit_price")
        if isinstance(unit_price, bool) or not isinstance(unit_price, (int, float)):
            raise ValidationError(f"items[{index}].unit_price must be a number")
        if unit_price < 0:
            raise ValidationError(f"items[{index}].unit_price must not be negative")

        validated.append(
            {
                "name": name,
                "quantity": quantity,
                "unit_price": float(unit_price),
            }
        )
    return validated


def _calculate_total(items: List[Dict[str, Any]]) -> float:
    """Calculate the order total from validated items."""
    total = sum(item["quantity"] * item["unit_price"] for item in items)
    return round(float(total), 2)


# ---------------------------------------------------------------------------
# Payment gateway abstraction (uses config, R-019).
# ---------------------------------------------------------------------------
def _process_payment(
    payment_token: str, amount: float, config: ServiceConfig
) -> str:
    """Process a payment and return a transaction id.

    This is a stand-in for a real payment gateway call. It uses the
    ``payment_api_key`` from the externalized config (R-019).
    """
    if amount < 0:
        raise PaymentError("Cannot charge a negative amount")
    if not config.payment_api_key:
        raise PaymentError("Payment API key is not configured")
    # Simulated gateway transaction id.
    return f"txn_{uuid.uuid4().hex}"


def _issue_refund(
    transaction_id: str, amount: float, config: ServiceConfig
) -> str:
    """Issue a refund and return a refund id."""
    if amount < 0:
        raise PaymentError("Cannot refund a negative amount")
    if not config.payment_api_key:
        raise PaymentError("Payment API key is not configured")
    return f"{config.refund_id_prefix}_{uuid.uuid4().hex}"


# ---------------------------------------------------------------------------
# Public API
# ---------------------------------------------------------------------------
def create_order(
    user_id: str,
    items: List[Dict[str, Any]],
    payment_token: str,
    config: ServiceConfig = DEFAULT_CONFIG,
) -> Dict[str, Any]:
    """Create a new order, process payment, and return a confirmation.

    Args:
        user_id: Identifier of the user placing the order.
        items: List of item dicts, each with ``name``, ``quantity``,
            and ``unit_price``.
        payment_token: Token used to process the payment.
        config: Service configuration (R-019).

    Returns:
        An order confirmation dict.

    Raises:
        ValidationError: If any parameter is invalid (R-001).
        PaymentError: If payment processing fails (R-007).
    """
    # R-013: log entry at DEBUG with a parameter summary.
    logger.debug(
        "create_order(user_id=%r, items_count=%s, payment_token=***)",
        user_id,
        len(items) if isinstance(items, list) else "n/a",
    )

    # R-001: validate all parameters.
    _require_non_empty_str(user_id, "user_id")
    _require_non_empty_str(payment_token, "payment_token")
    if not isinstance(config, ServiceConfig):
        raise ValidationError("config must be a ServiceConfig instance")
    validated_items = _validate_items(items)
    if len(validated_items) > config.max_items_per_order:
        raise ValidationError(
            f"order exceeds max_items_per_order ({config.max_items_per_order})"
        )

    total = _calculate_total(validated_items)
    transaction_id = _process_payment(payment_token, total, config)

    order_id = f"{config.order_id_prefix}-{uuid.uuid4().hex[:12]}"
    order: Dict[str, Any] = {
        "order_id": order_id,
        "user_id": user_id,
        "items": validated_items,
        "total": total,
        "currency": config.currency,
        "status": "CONFIRMED",
        "transaction_id": transaction_id,
    }
    _ORDER_STORE[order_id] = order

    return {
        "order_id": order_id,
        "user_id": user_id,
        "status": "CONFIRMED",
        "total": total,
        "currency": config.currency,
        "transaction_id": transaction_id,
        "items": validated_items,
    }


def cancel_order(
    user_id: str,
    order_id: str,
    reason: str,
    config: ServiceConfig = DEFAULT_CONFIG,
) -> Dict[str, Any]:
    """Cancel an existing order and issue a refund.

    Args:
        user_id: Identifier of the user who owns the order.
        order_id: Identifier of the order to cancel.
        reason: Reason for the cancellation.
        config: Service configuration (R-019).

    Returns:
        A cancellation confirmation dict.

    Raises:
        ValidationError: If any parameter is invalid (R-001).
        OrderNotFoundError: If the order does not exist or is not owned
            by the user (R-007).
        OrderStateError: If the order cannot be cancelled (R-007).
        PaymentError: If the refund fails (R-007).
    """
    # R-013: log entry at DEBUG with a parameter summary.
    logger.debug(
        "cancel_order(user_id=%r, order_id=%r, reason=%r)",
        user_id,
        order_id,
        reason,
    )

    # R-001: validate all parameters.
    _require_non_empty_str(user_id, "user_id")
    _require_non_empty_str(order_id, "order_id")
    _require_non_empty_str(reason, "reason")
    if not isinstance(config, ServiceConfig):
        raise ValidationError("config must be a ServiceConfig instance")

    order = _ORDER_STORE.get(order_id)
    if order is None or order["user_id"] != user_id:
        raise OrderNotFoundError(f"Order {order_id!r} not found for user {user_id!r}")

    if order["status"] == "CANCELLED":
        raise OrderStateError(f"Order {order_id!r} is already cancelled")

    refund_id = _issue_refund(order["transaction_id"], order["total"], config)

    order["status"] = "CANCELLED"
    order["cancellation_reason"] = reason
    order["refund_id"] = refund_id

    return {
        "order_id": order_id,
        "user_id": user_id,
        "status": "CANCELLED",
        "reason": reason,
        "refund_id": refund_id,
        "refund_amount": order["total"],
        "currency": config.currency,
    }


def get_order_status(
    user_id: str,
    order_id: str,
    config: ServiceConfig = DEFAULT_CONFIG,
) -> Dict[str, Any]:
    """Retrieve the status and details of an order.

    Args:
        user_id: Identifier of the user who owns the order.
        order_id: Identifier of the order to retrieve.
        config: Service configuration (R-019).

    Returns:
        An order status dict.

    Raises:
        ValidationError: If any parameter is invalid (R-001).
        OrderNotFoundError: If the order does not exist or is not owned
            by the user (R-007).
    """
    # R-013: log entry at DEBUG with a parameter summary.
    logger.debug(
        "get_order_status(user_id=%r, order_id=%r)",
        user_id,
        order_id,
    )

    # R-001: validate all parameters.
    _require_non_empty_str(user_id, "user_id")
    _require_non_empty_str(order_id, "order_id")
    if not isinstance(config, ServiceConfig):
        raise ValidationError("config must be a ServiceConfig instance")

    order = _ORDER_STORE.get(order_id)
    if order is None or order["user_id"] != user_id:
        raise OrderNotFoundError(f"Order {order_id!r} not found for user {user_id!r}")

    return {
        "order_id": order["order_id"],
        "user_id": order["user_id"],
        "status": order["status"],
        "total": order["total"],
        "currency": order["currency"],
        "items": order["items"],
    }

import unittest  # noqa: E402


def _sample_items() -> List[Dict[str, Any]]:
    return [
        {"name": "Widget", "quantity": 2, "unit_price": 9.99},
        {"name": "Gadget", "quantity": 1, "unit_price": 19.50},
    ]


class CreateOrderTests(unittest.TestCase):
    def setUp(self) -> None:
        _ORDER_STORE.clear()

    def test_create_order_success(self) -> None:
        result = create_order("user-1", _sample_items(), "pay-token")
        self.assertEqual(result["status"], "CONFIRMED")
        self.assertEqual(result["user_id"], "user-1")
        self.assertAlmostEqual(result["total"], 39.48)
        self.assertTrue(result["order_id"].startswith("ORD-"))
        self.assertIn(result["order_id"], _ORDER_STORE)
        self.assertTrue(result["transaction_id"].startswith("txn_"))

    def test_invalid_user_id_type(self) -> None:
        with self.assertRaises(ValidationError):
            create_order(123, _sample_items(), "pay-token")  # type: ignore[arg-type]

    def test_empty_user_id(self) -> None:
        with self.assertRaises(ValidationError):
            create_order("   ", _sample_items(), "pay-token")

    def test_empty_payment_token(self) -> None:
        with self.assertRaises(ValidationError):
            create_order("user-1", _sample_items(), "")

    def test_items_not_a_list(self) -> None:
        with self.assertRaises(ValidationError):
            create_order("user-1", "not-a-list", "pay-token")  # type: ignore[arg-type]

    def test_empty_items(self) -> None:
        with self.assertRaises(ValidationError):
            create_order("user-1", [], "pay-token")

    def test_item_missing_fields(self) -> None:
        with self.assertRaises(ValidationError):
            create_order("user-1", [{"name": "X"}], "pay-token")

    def test_item_negative_quantity(self) -> None:
        bad = [{"name": "X", "quantity": -1, "unit_price": 1.0}]
        with self.assertRaises(ValidationError):
            create_order("user-1", bad, "pay-token")

    def test_item_bool_quantity_rejected(self) -> None:
        bad = [{"name": "X", "quantity": True, "unit_price": 1.0}]
        with self.assertRaises(ValidationError):
            create_order("user-1", bad, "pay-token")

    def test_item_negative_price(self) -> None:
        bad = [{"name": "X", "quantity": 1, "unit_price": -5.0}]
        with self.assertRaises(ValidationError):
            create_order("user-1", bad, "pay-token")

    def test_exceeds_max_items(self) -> None:
        cfg = ServiceConfig(max_items_per_order=1)
        with self.assertRaises(ValidationError):
            create_order("user-1", _sample_items(), "pay-token", cfg)

    def test_missing_payment_key_raises_payment_error(self) -> None:
        cfg = ServiceConfig(payment_api_key="")
        with self.assertRaises(PaymentError):
            create_order("user-1", _sample_items(), "pay-token", cfg)


class CancelOrderTests(unittest.TestCase):
    def setUp(self) -> None:
        _ORDER_STORE.clear()
        self.order = create_order("user-1", _sample_items(), "pay-token")

    def test_cancel_success(self) -> None:
        result = cancel_order("user-1", self.order["order_id"], "changed mind")
        self.assertEqual(result["status"], "CANCELLED")
        self.assertEqual(result["reason"], "changed mind")
        self.assertTrue(result["refund_id"].startswith("REF_"))
        self.assertAlmostEqual(result["refund_amount"], 39.48)

    def test_cancel_unknown_order(self) -> None:
        with self.assertRaises(OrderNotFoundError):
            cancel_order("user-1", "ORD-doesnotexist", "reason")

    def test_cancel_wrong_user(self) -> None:
        with self.assertRaises(OrderNotFoundError):
            cancel_order("other-user", self.order["order_id"], "reason")

    def test_cancel_twice(self) -> None:
        cancel_order("user-1", self.order["order_id"], "first")
        with self.assertRaises(OrderStateError):
            cancel_order("user-1", self.order["order_id"], "second")

    def test_cancel_invalid_params(self) -> None:
        with self.assertRaises(ValidationError):
            cancel_order("user-1", self.order["order_id"], "")


class GetOrderStatusTests(unittest.TestCase):
    def setUp(self) -> None:
        _ORDER_STORE.clear()
        self.order = create_order("user-1", _sample_items(), "pay-token")

    def test_get_status_success(self) -> None:
        result = get_order_status("user-1", self.order["order_id"])
        self.assertEqual(result["status"], "CONFIRMED")
        self.assertEqual(result["order_id"], self.order["order_id"])

    def test_get_status_reflects_cancellation(self) -> None:
        cancel_order("user-1", self.order["order_id"], "reason")

        self.assertEqual(result["status"], "CANCELLED")

    def test_get_status_unknown_order(self) -> None:
        with self.assertRaises(OrderNotFoundError):

    def test_get_status_wrong_user(self) -> None:

            get_order_status("other-user", self.order["order_id"])

    def test_get_status_invalid_params(self) -> None:
        with self.assertRaises(ValidationError):
            get_order_status("", self.order["order_id"])
