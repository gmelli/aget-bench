"""Order management service.

Provides creation, cancellation, and status retrieval for orders.

Compliance with mandatory agent rules:
    R-001: All function parameters are validated against expected types.
    R-007: Domain-specific exception classes are raised (never generic Exception).
    R-013: Every public function logs entry at DEBUG with a parameter summary.
    R-019: All configuration values are externalized into a config object.
    R-025: All function signatures include type hints.
"""

from __future__ import annotations

import logging
import uuid
from dataclasses import dataclass, field
from typing import Any, Dict, List

logger = logging.getLogger(__name__)


# ---------------------------------------------------------------------------
# R-019: Externalized configuration
# ---------------------------------------------------------------------------
@dataclass(frozen=True)
class ServiceConfig:
    """Service configuration values.

    All tunable / environment-specific values live here rather than being
    hard-coded inside the service functions.
    """

    database_url: str = "postgresql://localhost:5432/orders"
    payment_api_key: str = "test-payment-api-key"
    rate_limit_per_minute: int = 60
    currency: str = "USD"
    refund_window_days: int = 30


# A module-level default instance; callers may construct their own.
DEFAULT_CONFIG: ServiceConfig = ServiceConfig()


# ---------------------------------------------------------------------------
# R-007: Domain-specific exception classes
# ---------------------------------------------------------------------------
class OrderServiceError(Exception):
    """Base class for all order-service errors."""


class ValidationError(OrderServiceError):
    """Raised when input parameters fail validation."""


class OrderNotFoundError(OrderServiceError):
    """Raised when a referenced order does not exist."""


class OrderStateError(OrderServiceError):
    """Raised when an operation is invalid for the order's current state."""


class PaymentError(OrderServiceError):
    """Raised when payment processing or refunding fails."""


# ---------------------------------------------------------------------------
# In-memory store and payment gateway stand-ins
# ---------------------------------------------------------------------------
# These stand in for a real database / payment provider. In production they
# would be replaced by clients built from ServiceConfig.database_url and
# ServiceConfig.payment_api_key.
_ORDERS: Dict[str, Dict[str, Any]] = {}


@dataclass
class _PaymentGateway:
    """Minimal payment gateway abstraction built from config."""

    api_key: str
    _charges: Dict[str, float] = field(default_factory=dict)

    def charge(self, payment_token: str, amount: float) -> str:
        if not self.api_key:
            raise PaymentError("Payment gateway is not configured with an API key")
        if not payment_token:
            raise PaymentError("Missing payment token")
        charge_id = f"chg_{uuid.uuid4().hex}"
        self._charges[charge_id] = amount
        return charge_id

    def refund(self, charge_id: str) -> str:
        if charge_id not in self._charges:
            raise PaymentError(f"Unknown charge id: {charge_id}")
        return f"rfnd_{uuid.uuid4().hex}"


# ---------------------------------------------------------------------------
# Validation helpers (support R-001)
# ---------------------------------------------------------------------------
def _require_non_empty_str(value: Any, name: str) -> str:
    if not isinstance(value, str):
        raise ValidationError(f"{name} must be a str, got {type(value).__name__}")
    if not value.strip():
        raise ValidationError(f"{name} must not be empty")
    return value


def _validate_items(items: Any) -> List[Dict[str, Any]]:
    if not isinstance(items, list):
        raise ValidationError(f"items must be a list, got {type(items).__name__}")
    if not items:
        raise ValidationError("items must not be empty")

    validated: List[Dict[str, Any]] = []
    for index, item in enumerate(items):
        if not isinstance(item, dict):
            raise ValidationError(f"items[{index}] must be a dict, got {type(item).__name__}")

        name = item.get("name")
        if not isinstance(name, str) or not name.strip():
            raise ValidationError(f"items[{index}].name must be a non-empty str")

        quantity = item.get("quantity")
        if not isinstance(quantity, int) or isinstance(quantity, bool) or quantity <= 0:
            raise ValidationError(f"items[{index}].quantity must be a positive int")

        unit_price = item.get("unit_price")
        if isinstance(unit_price, bool) or not isinstance(unit_price, (int, float)) or unit_price < 0:
            raise ValidationError(f"items[{index}].unit_price must be a non-negative number")

        validated.append(
            {"name": name, "quantity": quantity, "unit_price": float(unit_price)}
        )
    return validated


def _calculate_total(items: List[Dict[str, Any]]) -> float:
    return round(sum(item["quantity"] * item["unit_price"] for item in items), 2)


# ---------------------------------------------------------------------------
# Public API
# ---------------------------------------------------------------------------
def create_order(
    user_id: str,
    items: List[Dict[str, Any]],
    payment_token: str,
    config: ServiceConfig = DEFAULT_CONFIG,
) -> Dict[str, Any]:
    """Create a new order, charge payment, and return a confirmation.

    Args:
        user_id: Identifier of the user placing the order.
        items: List of item dicts, each with ``name``, ``quantity`` (positive
            int) and ``unit_price`` (non-negative number).
        payment_token: Token used to process payment.
        config: Service configuration object.

    Returns:
        An order confirmation dict.

    Raises:
        ValidationError: If any parameter is invalid.
        PaymentError: If payment processing fails.
    """
    logger.debug(
        "create_order(user_id=%r, items=<%d items>, payment_token=<redacted>, config=%r)",
        user_id,
        len(items) if isinstance(items, list) else 0,
        config,
    )

    if not isinstance(config, ServiceConfig):
        raise ValidationError("config must be a ServiceConfig instance")

    user_id = _require_non_empty_str(user_id, "user_id")
    payment_token = _require_non_empty_str(payment_token, "payment_token")
    validated_items = _validate_items(items)

    total_price = _calculate_total(validated_items)

    gateway = _PaymentGateway(api_key=config.payment_api_key)
    charge_id = gateway.charge(payment_token, total_price)

    order_id = f"ord_{uuid.uuid4().hex}"
    order: Dict[str, Any] = {
        "order_id": order_id,
        "user_id": user_id,
        "items": validated_items,
        "total_price": total_price,
        "currency": config.currency,
        "status": "confirmed",
        "charge_id": charge_id,
    }
    _ORDERS[order_id] = order

    return {
        "order_id": order_id,
        "user_id": user_id,
        "status": "confirmed",
        "total_price": total_price,
        "currency": config.currency,
        "item_count": len(validated_items),
    }


def cancel_order(
    user_id: str,
    order_id: str,
    reason: str,
    config: ServiceConfig = DEFAULT_CONFIG,
) -> Dict[str, Any]:
    """Cancel an existing order and issue a refund.

    Args:
        user_id: Identifier of the user who owns the order.
        order_id: Identifier of the order to cancel.
        reason: Human-readable cancellation reason.
        config: Service configuration object.

    Returns:
        A cancellation confirmation dict.

    Raises:
        ValidationError: If any parameter is invalid.
        OrderNotFoundError: If the order does not exist for the user.
        OrderStateError: If the order cannot be cancelled.
        PaymentError: If the refund fails.
    """
    logger.debug(
        "cancel_order(user_id=%r, order_id=%r, reason=%r, config=%r)",
        user_id,
        order_id,
        reason,
        config,
    )

    if not isinstance(config, ServiceConfig):
        raise ValidationError("config must be a ServiceConfig instance")

    user_id = _require_non_empty_str(user_id, "user_id")
    order_id = _require_non_empty_str(order_id, "order_id")
    reason = _require_non_empty_str(reason, "reason")

    order = _ORDERS.get(order_id)
    if order is None or order["user_id"] != user_id:
        raise OrderNotFoundError(f"Order {order_id!r} not found for user {user_id!r}")

    if order["status"] == "cancelled":
        raise OrderStateError(f"Order {order_id!r} is already cancelled")

    gateway = _PaymentGateway(api_key=config.payment_api_key)
    refund_id = gateway.refund(order["charge_id"])

    order["status"] = "cancelled"
    order["cancellation_reason"] = reason
    order["refund_id"] = refund_id

    return {
        "order_id": order_id,
        "user_id": user_id,
        "status": "cancelled",
        "reason": reason,
        "refund_id": refund_id,
        "refund_amount": order["total_price"],
        "currency": order["currency"],
    }


def get_order_status(
    user_id: str,
    order_id: str,
    config: ServiceConfig = DEFAULT_CONFIG,
) -> Dict[str, Any]:
    """Retrieve the status and details of an order.

    Args:
        user_id: Identifier of the user who owns the order.
        order_id: Identifier of the order to look up.
        config: Service configuration object.

    Returns:
        An order status dict.

    Raises:
        ValidationError: If any parameter is invalid.
        OrderNotFoundError: If the order does not exist for the user.
    """
    logger.debug(
        "get_order_status(user_id=%r, order_id=%r, config=%r)",
        user_id,
        order_id,
        config,
    )

    if not isinstance(config, ServiceConfig):
        raise ValidationError("config must be a ServiceConfig instance")

    user_id = _require_non_empty_str(user_id, "user_id")
    order_id = _require_non_empty_str(order_id, "order_id")

    order = _ORDERS.get(order_id)
    if order is None or order["user_id"] != user_id:
        raise OrderNotFoundError(f"Order {order_id!r} not found for user {user_id!r}")

    return {
        "order_id": order["order_id"],
        "user_id": order["user_id"],
        "status": order["status"],
        "total_price": order["total_price"],
        "currency": order["currency"],
        "items": order["items"],
    }

"""Unit tests for order_service."""

import unittest

import order_service
from order_service import (
    OrderNotFoundError,
    OrderStateError,
    ServiceConfig,
    ValidationError,
    cancel_order,
    create_order,
    get_order_status,
)


def _sample_items():
    return [
        {"name": "Widget", "quantity": 2, "unit_price": 9.99},
        {"name": "Gadget", "quantity": 1, "unit_price": 4.50},
    ]


class CreateOrderTests(unittest.TestCase):
    def setUp(self):
        order_service._ORDERS.clear()

    def test_create_order_success(self):
        result = create_order("user-1", _sample_items(), "tok_abc")
        self.assertEqual(result["status"], "confirmed")
        self.assertEqual(result["user_id"], "user-1")
        self.assertEqual(result["item_count"], 2)
        self.assertEqual(result["total_price"], 24.48)
        self.assertTrue(result["order_id"].startswith("ord_"))
        self.assertEqual(result["currency"], "USD")

    def test_unique_order_ids(self):
        a = create_order("user-1", _sample_items(), "tok_abc")
        b = create_order("user-1", _sample_items(), "tok_abc")
        self.assertNotEqual(a["order_id"], b["order_id"])

    def test_invalid_user_id_type(self):
        with self.assertRaises(ValidationError):
            create_order(123, _sample_items(), "tok_abc")

    def test_empty_user_id(self):
        with self.assertRaises(ValidationError):
            create_order("   ", _sample_items(), "tok_abc")

    def test_empty_payment_token(self):
        with self.assertRaises(ValidationError):
            create_order("user-1", _sample_items(), "")

    def test_items_not_a_list(self):
        with self.assertRaises(ValidationError):
            create_order("user-1", "not-a-list", "tok_abc")

    def test_empty_items(self):
        with self.assertRaises(ValidationError):
            create_order("user-1", [], "tok_abc")

    def test_item_negative_quantity(self):
        items = [{"name": "Widget", "quantity": -1, "unit_price": 1.0}]
        with self.assertRaises(ValidationError):
            create_order("user-1", items, "tok_abc")

    def test_item_bool_quantity_rejected(self):
        items = [{"name": "Widget", "quantity": True, "unit_price": 1.0}]
        with self.assertRaises(ValidationError):
            create_order("user-1", items, "tok_abc")

    def test_item_negative_unit_price(self):
        items = [{"name": "Widget", "quantity": 1, "unit_price": -5}]
        with self.assertRaises(ValidationError):
            create_order("user-1", items, "tok_abc")

    def test_item_missing_name(self):
        items = [{"quantity": 1, "unit_price": 5.0}]
        with self.assertRaises(ValidationError):
            create_order("user-1", items, "tok_abc")

    def test_invalid_config(self):
        with self.assertRaises(ValidationError):
            create_order("user-1", _sample_items(), "tok_abc", config="nope")


class CancelOrderTests(unittest.TestCase):
    def setUp(self):
        order_service._ORDERS.clear()
        self.order = create_order("user-1", _sample_items(), "tok_abc")

    def test_cancel_success(self):
        result = cancel_order("user-1", self.order["order_id"], "changed mind")
        self.assertEqual(result["status"], "cancelled")
        self.assertEqual(result["reason"], "changed mind")
        self.assertEqual(result["refund_amount"], self.order["total_price"])
        self.assertTrue(result["refund_id"].startswith("rfnd_"))

    def test_cancel_unknown_order(self):
        with self.assertRaises(OrderNotFoundError):
            cancel_order("user-1", "ord_missing", "reason")

    def test_cancel_wrong_user(self):
        with self.assertRaises(OrderNotFoundError):
            cancel_order("user-2", self.order["order_id"], "reason")

    def test_cancel_twice(self):
        cancel_order("user-1", self.order["order_id"], "reason")
        with self.assertRaises(OrderStateError):
            cancel_order("user-1", self.order["order_id"], "again")

    def test_cancel_empty_reason(self):
        with self.assertRaises(ValidationError):
            cancel_order("user-1", self.order["order_id"], "")


class GetOrderStatusTests(unittest.TestCase):
    def setUp(self):
        order_service._ORDERS.clear()
        self.order = create_order("user-1", _sample_items(), "tok_abc")

    def test_status_success(self):
        result = get_order_status("user-1", self.order["order_id"])
        self.assertEqual(result["status"], "confirmed")
        self.assertEqual(result["order_id"], self.order["order_id"])
        self.assertEqual(len(result["items"]), 2)

    def test_status_reflects_cancellation(self):
        cancel_order("user-1", self.order["order_id"], "reason")
        result = get_order_status("user-1", self.order["order_id"])
        self.assertEqual(result["status"], "cancelled")

    def test_status_unknown_order(self):
        with self.assertRaises(OrderNotFoundError):
            get_order_status("user-1", "ord_missing")

    def test_status_wrong_user(self):
        with self.assertRaises(OrderNotFoundError):
            get_order_status("user-2", self.order["order_id"])

    def test_status_invalid_order_id(self):
        with self.assertRaises(ValidationError):
            get_order_status("user-1", 999)


class ServiceConfigTests(unittest.TestCase):
    def test_defaults(self):
        cfg = ServiceConfig()
        self.assertEqual(cfg.currency, "USD")
        self.assertEqual(cfg.rate_limit_per_minute, 60)

    def test_custom_currency(self):
        order_service._ORDERS.clear()
        cfg = ServiceConfig(currency="EUR")
        result = create_order("user-1", _sample_items(), "tok_abc", config=cfg)
        self.assertEqual(result["currency"], "EUR")


if __name__ == "__main__":
    unittest.main()

Both files are complete. A few notes on how the 5 mandatory rules were applied:

- **R-001** — every public function validates types via `_require_non_empty_str`, `_validate_items`, and the `ServiceConfig` instance check before processing.
- **R-007** — domain exceptions (`ValidationError`, `OrderNotFoundError`, `OrderStateError`, `PaymentError`) all subclass `OrderServiceError`; no generic `Exception` is raised.
- **R-013** — each public function opens with `logger.debug(...)` summarizing its parameters (the payment token is redacted).
- **R-019** — all config (database URL, payment API key, rate limit, currency, refund window) lives in `ServiceConfig`.
- **R-025** — full type hints on every signature.

The payment/database layers are in-memory stand-ins (`_PaymentGateway`, `_ORDERS`) built from config — swap them for real clients in production. The tests clear `_ORDERS` in `setUp` for isolation and cover success paths, validation failures, ownership checks, and state transitions. Run with `python -m unittest test_order_service`.
