"""Order management service.

Provides order creation, cancellation and status retrieval.

This module adheres to the mandatory agent rules:
  R-001  validate all function parameters against expected types
  R-007  use domain-specific exception classes
  R-013  log function entry at DEBUG level with parameter summary
  R-019  externalize all configuration values into a config object
  R-025  type hints for all function signatures
"""

from __future__ import annotations

import logging
import unittest
import uuid
from dataclasses import dataclass
from typing import Any, Dict, List, Optional

logger = logging.getLogger(__name__)


# --------------------------------------------------------------------------- #
# R-019: Externalized configuration                                            #
# --------------------------------------------------------------------------- #
@dataclass
class ServiceConfig:
    """Service configuration holding all externalized values.

    All tunable values used by the service live here rather than being
    hard-coded into the business logic (R-019).
    """

    database_url: str = "sqlite:///orders.db"
    payment_api_key: str = "test-payment-api-key"
    rate_limit_per_minute: int = 60
    currency: str = "USD"
    max_items_per_order: int = 100

    def __post_init__(self) -> None:
        # R-001: validate configuration parameter types.
        if not isinstance(self.database_url, str):
            raise ConfigurationError("database_url must be a str")
        if not isinstance(self.payment_api_key, str):
            raise ConfigurationError("payment_api_key must be a str")
        if not isinstance(self.rate_limit_per_minute, int):
            raise ConfigurationError("rate_limit_per_minute must be an int")
        if not isinstance(self.max_items_per_order, int):
            raise ConfigurationError("max_items_per_order must be an int")
        if not isinstance(self.currency, str):
            raise ConfigurationError("currency must be a str")


# --------------------------------------------------------------------------- #
# R-007: Domain-specific exception classes                                     #
# --------------------------------------------------------------------------- #
class OrderServiceError(Exception):
    """Base class for all order service errors."""


class ConfigurationError(OrderServiceError):
    """Raised when the service configuration is invalid."""


class ValidationError(OrderServiceError):
    """Raised when input parameters fail validation (R-001)."""


class OrderNotFoundError(OrderServiceError):
    """Raised when a referenced order does not exist."""


class OrderStateError(OrderServiceError):
    """Raised when an operation is invalid for the order's current state."""


class PaymentError(OrderServiceError):
    """Raised when a payment or refund operation fails."""


# --------------------------------------------------------------------------- #
# Payment gateway abstraction                                                  #
# --------------------------------------------------------------------------- #
class PaymentGateway:
    """Minimal payment gateway abstraction.

    In a real deployment this would call an external payment provider using
    ``config.payment_api_key``. Here it is a deterministic stand-in.
    """

    def __init__(self, config: ServiceConfig) -> None:
        if not isinstance(config, ServiceConfig):
            raise ValidationError("config must be a ServiceConfig instance")
        self._config = config

    def charge(self, payment_token: str, amount: float) -> str:
        """Charge ``amount`` against ``payment_token``; return a charge id."""
        logger.debug(
            "PaymentGateway.charge(payment_token=%r, amount=%r)",
            _mask(payment_token),
            amount,
        )
        if not isinstance(payment_token, str) or not payment_token:
            raise PaymentError("payment_token must be a non-empty str")
        if not isinstance(amount, (int, float)) or amount <= 0:
            raise PaymentError("amount must be a positive number")
        return f"chg_{uuid.uuid4().hex}"

    def refund(self, charge_id: str, amount: float) -> str:
        """Refund ``amount`` for ``charge_id``; return a refund id."""
        logger.debug(
            "PaymentGateway.refund(charge_id=%r, amount=%r)", charge_id, amount
        )
        if not isinstance(charge_id, str) or not charge_id:
            raise PaymentError("charge_id must be a non-empty str")
        if not isinstance(amount, (int, float)) or amount < 0:
            raise PaymentError("amount must be a non-negative number")
        return f"rfnd_{uuid.uuid4().hex}"


# --------------------------------------------------------------------------- #
# Helpers                                                                      #
# --------------------------------------------------------------------------- #
def _mask(token: str) -> str:
    """Return a masked representation of a sensitive token for logging."""
    if not isinstance(token, str) or len(token) <= 4:
        return "***"
    return f"{token[:2]}***{token[-2:]}"


def _validate_str(name: str, value: Any) -> str:
    """Validate that ``value`` is a non-empty string (R-001)."""
    if not isinstance(value, str):
        raise ValidationError(f"{name} must be a str, got {type(value).__name__}")
    if not value.strip():
        raise ValidationError(f"{name} must not be empty")
    return value


def _validate_items(items: Any, max_items: int) -> List[Dict[str, Any]]:
    """Validate the ``items`` list and each line item (R-001)."""
    if not isinstance(items, list):
        raise ValidationError(f"items must be a list, got {type(items).__name__}")
    if not items:
        raise ValidationError("items must not be empty")
    if len(items) > max_items:
        raise ValidationError(f"items exceeds max of {max_items}")

    validated: List[Dict[str, Any]] = []
    for index, item in enumerate(items):
        if not isinstance(item, dict):
            raise ValidationError(f"items[{index}] must be a dict")
        name = item.get("name")
        quantity = item.get("quantity")
        unit_price = item.get("unit_price")
        if not isinstance(name, str) or not name.strip():
            raise ValidationError(f"items[{index}].name must be a non-empty str")
        if not isinstance(quantity, int) or isinstance(quantity, bool) or quantity <= 0:
            raise ValidationError(f"items[{index}].quantity must be a positive int")
        if (
            not isinstance(unit_price, (int, float))
            or isinstance(unit_price, bool)
            or unit_price < 0
        ):
            raise ValidationError(
                f"items[{index}].unit_price must be a non-negative number"
            )
        validated.append(
            {"name": name, "quantity": quantity, "unit_price": float(unit_price)}
        )
    return validated


def _calculate_total(items: List[Dict[str, Any]]) -> float:
    """Sum ``quantity * unit_price`` across all line items."""
    return round(sum(i["quantity"] * i["unit_price"] for i in items), 2)


# --------------------------------------------------------------------------- #
# Order service                                                                #
# --------------------------------------------------------------------------- #
@dataclass
class _OrderRecord:
    order_id: str
    user_id: str
    items: List[Dict[str, Any]]
    total: float
    status: str
    charge_id: str
    refund_id: Optional[str] = None
    cancel_reason: Optional[str] = None


class OrderService:
    """Stateful order management service backed by an in-memory store."""

    def __init__(
        self,
        config: Optional[ServiceConfig] = None,
        gateway: Optional[PaymentGateway] = None,
    ) -> None:
        if config is None:
            config = ServiceConfig()
        if not isinstance(config, ServiceConfig):
            raise ValidationError("config must be a ServiceConfig instance")
        if gateway is None:
            gateway = PaymentGateway(config)
        if not isinstance(gateway, PaymentGateway):
            raise ValidationError("gateway must be a PaymentGateway instance")

        self._config = config
        self._gateway = gateway
        self._orders: Dict[str, _OrderRecord] = {}

    # -- Public API -------------------------------------------------------- #
    def create_order(
        self, user_id: str, items: List[Dict[str, Any]], payment_token: str
    ) -> Dict[str, Any]:
        """Create an order, charge payment and return a confirmation dict."""
        logger.debug(
            "create_order(user_id=%r, items=<%d items>, payment_token=%r)",
            user_id,
            len(items) if isinstance(items, list) else 0,
            _mask(payment_token) if isinstance(payment_token, str) else payment_token,
        )

        # R-001: validate all parameters.
        user_id = _validate_str("user_id", user_id)
        payment_token = _validate_str("payment_token", payment_token)
        validated_items = _validate_items(items, self._config.max_items_per_order)

        total = _calculate_total(validated_items)
        charge_id = self._gateway.charge(payment_token, total)

        order_id = f"ord_{uuid.uuid4().hex}"
        record = _OrderRecord(
            order_id=order_id,
            user_id=user_id,
            items=validated_items,
            total=total,
            status="CONFIRMED",
            charge_id=charge_id,
        )
        self._orders[order_id] = record

        return {
            "order_id": order_id,
            "user_id": user_id,
            "status": record.status,
            "total": total,
            "currency": self._config.currency,
            "charge_id": charge_id,
            "items": validated_items,
        }

    def cancel_order(
        self, user_id: str, order_id: str, reason: str
    ) -> Dict[str, Any]:
        """Cancel an order, issue a refund and return a confirmation dict."""
        logger.debug(
            "cancel_order(user_id=%r, order_id=%r, reason=%r)",
            user_id,
            order_id,
            reason,
        )

        user_id = _validate_str("user_id", user_id)
        order_id = _validate_str("order_id", order_id)
        reason = _validate_str("reason", reason)

        record = self._get_owned_order(user_id, order_id)
        if record.status == "CANCELLED":
            raise OrderStateError(f"order {order_id} is already cancelled")

        refund_id = self._gateway.refund(record.charge_id, record.total)
        record.status = "CANCELLED"
        record.refund_id = refund_id
        record.cancel_reason = reason

        return {
            "order_id": order_id,
            "user_id": user_id,
            "status": record.status,
            "refund_id": refund_id,
            "refund_amount": record.total,
            "currency": self._config.currency,
            "reason": reason,
        }

    def get_order_status(self, user_id: str, order_id: str) -> Dict[str, Any]:
        """Return the status and details of an order owned by ``user_id``."""
        logger.debug(
            "get_order_status(user_id=%r, order_id=%r)", user_id, order_id
        )

        user_id = _validate_str("user_id", user_id)
        order_id = _validate_str("order_id", order_id)

        record = self._get_owned_order(user_id, order_id)
        return {
            "order_id": record.order_id,
            "user_id": record.user_id,
            "status": record.status,
            "total": record.total,
            "currency": self._config.currency,
            "items": record.items,
            "refund_id": record.refund_id,
            "cancel_reason": record.cancel_reason,
        }

    # -- Internal helpers -------------------------------------------------- #
    def _get_owned_order(self, user_id: str, order_id: str) -> _OrderRecord:
        record = self._orders.get(order_id)
        if record is None or record.user_id != user_id:
            raise OrderNotFoundError(f"order {order_id} not found for user {user_id}")
        return record

# Unit tests                                                                   #
# --------------------------------------------------------------------------- #
class OrderServiceTests(unittest.TestCase):
    """Unit tests for all public functions of the order service."""

    def setUp(self) -> None:
        self.config = ServiceConfig(currency="USD", max_items_per_order=10)
        self.service = OrderService(config=self.config)
        self.items: List[Dict[str, Any]] = [
            {"name": "Widget", "quantity": 2, "unit_price": 9.99},
            {"name": "Gadget", "quantity": 1, "unit_price": 19.50},
        ]

    # -- create_order ------------------------------------------------------ #
    def test_create_order_returns_confirmation(self) -> None:
        result = self.service.create_order("user-1", self.items, "tok_abc123")
        self.assertEqual(result["status"], "CONFIRMED")
        self.assertEqual(result["user_id"], "user-1")
        self.assertAlmostEqual(result["total"], 39.48)
        self.assertEqual(result["currency"], "USD")
        self.assertTrue(result["order_id"].startswith("ord_"))
        self.assertTrue(result["charge_id"].startswith("chg_"))

    def test_create_order_unique_ids(self) -> None:
        first = self.service.create_order("user-1", self.items, "tok_abc123")
        second = self.service.create_order("user-1", self.items, "tok_abc123")
        self.assertNotEqual(first["order_id"], second["order_id"])

    def test_create_order_invalid_user_id(self) -> None:
        with self.assertRaises(ValidationError):
            self.service.create_order(123, self.items, "tok_abc123")  # type: ignore[arg-type]

    def test_create_order_empty_items(self) -> None:
        with self.assertRaises(ValidationError):
            self.service.create_order("user-1", [], "tok_abc123")

    def test_create_order_bad_item_quantity(self) -> None:
        bad = [{"name": "Widget", "quantity": 0, "unit_price": 1.0}]
        with self.assertRaises(ValidationError):
            self.service.create_order("user-1", bad, "tok_abc123")

    def test_create_order_empty_payment_token(self) -> None:
        with self.assertRaises(ValidationError):
            self.service.create_order("user-1", self.items, "")

    def test_create_order_exceeds_max_items(self) -> None:
        many = [
            {"name": f"Item{i}", "quantity": 1, "unit_price": 1.0}
            for i in range(self.config.max_items_per_order + 1)
        ]
        with self.assertRaises(ValidationError):
            self.service.create_order("user-1", many, "tok_abc123")

    # -- cancel_order ------------------------------------------------------ #
    def test_cancel_order_issues_refund(self) -> None:
        created = self.service.create_order("user-1", self.items, "tok_abc123")
        cancelled = self.service.cancel_order(
            "user-1", created["order_id"], "changed mind"
        )
        self.assertEqual(cancelled["status"], "CANCELLED")
        self.assertTrue(cancelled["refund_id"].startswith("rfnd_"))
        self.assertAlmostEqual(cancelled["refund_amount"], 39.48)
        self.assertEqual(cancelled["reason"], "changed mind")

    def test_cancel_order_not_found(self) -> None:
        with self.assertRaises(OrderNotFoundError):
            self.service.cancel_order("user-1", "ord_missing", "reason")

    def test_cancel_order_wrong_user(self) -> None:
        created = self.service.create_order("user-1", self.items, "tok_abc123")
        with self.assertRaises(OrderNotFoundError):
            self.service.cancel_order("user-2", created["order_id"], "reason")

    def test_cancel_order_already_cancelled(self) -> None:
        created = self.service.create_order("user-1", self.items, "tok_abc123")
        self.service.cancel_order("user-1", created["order_id"], "reason")
        with self.assertRaises(OrderStateError):
            self.service.cancel_order("user-1", created["order_id"], "reason")

    def test_cancel_order_invalid_reason(self) -> None:
        created = self.service.create_order("user-1", self.items, "tok_abc123")
        with self.assertRaises(ValidationError):
            self.service.cancel_order("user-1", created["order_id"], "")

    # -- get_order_status -------------------------------------------------- #
    def test_get_order_status_confirmed(self) -> None:
        created = self.service.create_order("user-1", self.items, "tok_abc123")
        status = self.service.get_order_status("user-1", created["order_id"])
        self.assertEqual(status["status"], "CONFIRMED")
        self.assertEqual(status["order_id"], created["order_id"])
        self.assertIsNone(status["refund_id"])

    def test_get_order_status_after_cancel(self) -> None:
        created = self.service.create_order("user-1", self.items, "tok_abc123")
        self.service.cancel_order("user-1", created["order_id"], "reason")
        status = self.service.get_order_status("user-1", created["order_id"])
        self.assertEqual(status["status"], "CANCELLED")
        self.assertIsNotNone(status["refund_id"])

    def test_get_order_status_not_found(self) -> None:
        with self.assertRaises(OrderNotFoundError):
            self.service.get_order_status("user-1", "ord_missing")

    def test_get_order_status_invalid_order_id(self) -> None:
        with self.assertRaises(ValidationError):
            self.service.get_order_status("user-1", 999)  # type: ignore[arg-type]

    # -- config ------------------------------------------------------------ #
    def test_invalid_config_raises(self) -> None:
        with self.assertRaises(ConfigurationError):
            ServiceConfig(rate_limit_per_minute="fast")  # type: ignore[arg-type]


if __name__ == "__main__":
    unittest.main()
