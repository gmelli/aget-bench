"""order_service.py — Order management service.

Implements creation, cancellation, and status retrieval for orders.

Compliance with mandatory agent rules:
    R-001: All public function parameters are validated against expected
           types before processing (see ``_validate_*`` helpers).
    R-007: Domain-specific exceptions (``OrderError`` hierarchy) are raised
           instead of generic ``Exception``.
    R-013: Every public function logs entry at DEBUG level with a summary
           of its parameters.
    R-019: All configuration values are externalized into ``ServiceConfig``.
    R-025: Type hints are provided for every function signature.
"""

from __future__ import annotations

import logging
import uuid
from dataclasses import dataclass, field
from typing import Any, Dict, List, Optional

logger = logging.getLogger(__name__)


# ---------------------------------------------------------------------------
# R-019: Externalized configuration
# ---------------------------------------------------------------------------
@dataclass(frozen=True)
class ServiceConfig:
    """Service configuration holding all externalized values."""

    database_url: str = "postgresql://localhost:5432/orders"
    payment_api_key: str = "test-payment-api-key"
    payment_api_url: str = "https://payments.example.com/v1"
    rate_limit_per_minute: int = 60
    currency: str = "USD"
    max_items_per_order: int = 100

    @classmethod
    def from_dict(cls, data: Dict[str, Any]) -> "ServiceConfig":
        """Build a config from a plain dict, ignoring unknown keys."""
        if not isinstance(data, dict):
            raise ConfigError(f"config data must be a dict, got {type(data).__name__}")
        known = set(cls.__dataclass_fields__)  # type: ignore[attr-defined]
        filtered = {k: v for k, v in data.items() if k in known}
        return cls(**filtered)


# Default, module-level configuration instance.
DEFAULT_CONFIG: ServiceConfig = ServiceConfig()


# ---------------------------------------------------------------------------
# R-007: Domain-specific exception classes
# ---------------------------------------------------------------------------
class OrderError(Exception):
    """Base class for all order service errors."""


class ConfigError(OrderError):
    """Raised when service configuration is invalid."""


class ValidationError(OrderError):
    """Raised when an argument fails type or value validation."""


class OrderNotFoundError(OrderError):
    """Raised when an order cannot be located for a user."""


class PaymentError(OrderError):
    """Raised when payment processing or refunding fails."""


# ---------------------------------------------------------------------------
# Order status constants
# ---------------------------------------------------------------------------
STATUS_CONFIRMED = "CONFIRMED"
STATUS_CANCELLED = "CANCELLED"


# ---------------------------------------------------------------------------
# In-memory order store (stands in for a database addressed by config URL).
# ---------------------------------------------------------------------------
@dataclass
class _OrderStore:
    """Simple in-memory order store keyed by order id."""

    orders: Dict[str, Dict[str, Any]] = field(default_factory=dict)

    def save(self, order: Dict[str, Any]) -> None:
        self.orders[order["order_id"]] = order

    def get(self, order_id: str) -> Optional[Dict[str, Any]]:
        return self.orders.get(order_id)


_store = _OrderStore()


# ---------------------------------------------------------------------------
# R-001: Validation helpers
# ---------------------------------------------------------------------------
def _validate_str(name: str, value: Any) -> str:
    """Validate that ``value`` is a non-empty string."""
    if not isinstance(value, str):
        raise ValidationError(f"{name} must be a str, got {type(value).__name__}")
    if not value.strip():
        raise ValidationError(f"{name} must not be empty")
    return value


def _validate_items(items: Any, max_items: int) -> List[Dict[str, Any]]:
    """Validate the ``items`` list and each item's structure/types."""
    if not isinstance(items, list):
        raise ValidationError(f"items must be a list, got {type(items).__name__}")
    if not items:
        raise ValidationError("items must not be empty")
    if len(items) > max_items:
        raise ValidationError(f"items exceeds max of {max_items}")

    for index, item in enumerate(items):
        if not isinstance(item, dict):
            raise ValidationError(f"item[{index}] must be a dict, got {type(item).__name__}")
        if not isinstance(item.get("name"), str) or not item["name"].strip():
            raise ValidationError(f"item[{index}].name must be a non-empty str")
        if not isinstance(item.get("quantity"), int) or isinstance(item.get("quantity"), bool):
            raise ValidationError(f"item[{index}].quantity must be an int")
        if item["quantity"] <= 0:
            raise ValidationError(f"item[{index}].quantity must be positive")
        unit_price = item.get("unit_price")
        if isinstance(unit_price, bool) or not isinstance(unit_price, (int, float)):
            raise ValidationError(f"item[{index}].unit_price must be a number")
        if unit_price < 0:
            raise ValidationError(f"item[{index}].unit_price must not be negative")
    return items


def _calculate_total(items: List[Dict[str, Any]]) -> float:
    """Calculate the order total from validated items."""
    return round(sum(item["quantity"] * item["unit_price"] for item in items), 2)


# ---------------------------------------------------------------------------
# Payment gateway (stub addressed via config).
# ---------------------------------------------------------------------------
def _process_payment(token: str, amount: float, config: ServiceConfig) -> str:
    """Charge ``amount`` using ``token``; returns a payment reference."""
    if amount <= 0:
        raise PaymentError("payment amount must be positive")
    # A real implementation would call config.payment_api_url with the key.
    return f"pay_{uuid.uuid4().hex[:16]}"


def _issue_refund(payment_ref: str, amount: float, config: ServiceConfig) -> str:
    """Refund a prior payment; returns a refund reference."""
    if not payment_ref:
        raise PaymentError("missing payment reference for refund")
    return f"ref_{uuid.uuid4().hex[:16]}"


# ---------------------------------------------------------------------------
# Public API
# ---------------------------------------------------------------------------
def create_order(
    user_id: str,
    items: List[Dict[str, Any]],
    payment_token: str,
    config: ServiceConfig = DEFAULT_CONFIG,
) -> Dict[str, Any]:
    """Create an order, charge payment, and return a confirmation dict.

    Args:
        user_id: Identifier of the ordering user.
        items: List of items, each a dict with ``name``, ``quantity``,
            and ``unit_price``.
        payment_token: Token used to process payment.
        config: Service configuration.

    Returns:
        An order confirmation dict.

    Raises:
        ValidationError: If any argument fails validation.
        PaymentError: If payment processing fails.
    """
    logger.debug(
        "create_order(user_id=%r, items=<%d items>, payment_token=%r)",
        user_id,
        len(items) if isinstance(items, list) else 0,
        "***" if isinstance(payment_token, str) else payment_token,
    )

    _validate_str("user_id", user_id)
    _validate_str("payment_token", payment_token)
    validated_items = _validate_items(items, config.max_items_per_order)

    total = _calculate_total(validated_items)
    payment_ref = _process_payment(payment_token, total, config)

    order: Dict[str, Any] = {
        "order_id": f"ord_{uuid.uuid4().hex}",
        "user_id": user_id,
        "items": validated_items,
        "total": total,
        "currency": config.currency,
        "status": STATUS_CONFIRMED,
        "payment_ref": payment_ref,
    }
    _store.save(order)

    return {
        "order_id": order["order_id"],
        "user_id": user_id,
        "status": STATUS_CONFIRMED,
        "total": total,
        "currency": config.currency,
        "payment_ref": payment_ref,
    }


def cancel_order(
    user_id: str,
    order_id: str,
    reason: str,
    config: ServiceConfig = DEFAULT_CONFIG,
) -> Dict[str, Any]:
    """Cancel an existing order and issue a refund.

    Args:
        user_id: Identifier of the user who owns the order.
        order_id: Identifier of the order to cancel.
        reason: Free-text cancellation reason.
        config: Service configuration.

    Returns:
        A cancellation confirmation dict.

    Raises:
        ValidationError: If any argument fails validation.
        OrderNotFoundError: If the order does not exist for the user.
        PaymentError: If the refund fails.
    """
    logger.debug(
        "cancel_order(user_id=%r, order_id=%r, reason=%r)",
        user_id,
        order_id,
        reason,
    )

    _validate_str("user_id", user_id)
    _validate_str("order_id", order_id)
    _validate_str("reason", reason)

    order = _store.get(order_id)
    if order is None or order["user_id"] != user_id:
        raise OrderNotFoundError(f"order {order_id!r} not found for user {user_id!r}")

    if order["status"] == STATUS_CANCELLED:
        raise OrderError(f"order {order_id!r} is already cancelled")

    refund_ref = _issue_refund(order["payment_ref"], order["total"], config)
    order["status"] = STATUS_CANCELLED
    order["cancel_reason"] = reason
    order["refund_ref"] = refund_ref
    _store.save(order)

    return {
        "order_id": order_id,
        "user_id": user_id,
        "status": STATUS_CANCELLED,
        "reason": reason,
        "refund_ref": refund_ref,
        "refund_amount": order["total"],
        "currency": config.currency,
    }


def get_order_status(
    user_id: str,
    order_id: str,
    config: ServiceConfig = DEFAULT_CONFIG,
) -> Dict[str, Any]:
    """Retrieve the status and details of an order.

    Args:
        user_id: Identifier of the user who owns the order.
        order_id: Identifier of the order to look up.
        config: Service configuration.

    Returns:
        An order status dict.

    Raises:
        ValidationError: If any argument fails validation.
        OrderNotFoundError: If the order does not exist for the user.
    """
    logger.debug("get_order_status(user_id=%r, order_id=%r)", user_id, order_id)

    _validate_str("user_id", user_id)
    _validate_str("order_id", order_id)

    order = _store.get(order_id)
    if order is None or order["user_id"] != user_id:
        raise OrderNotFoundError(f"order {order_id!r} not found for user {user_id!r}")

    return {
        "order_id": order_id,
        "user_id": user_id,
        "status": order["status"],
        "total": order["total"],
        "currency": order["currency"],
        "items": order["items"],
    }

"""Unit tests for order_service."""

import unittest

import order_service as svc


def _sample_items():
    return [
        {"name": "widget", "quantity": 2, "unit_price": 9.99},
        {"name": "gadget", "quantity": 1, "unit_price": 19.50},
    ]


class CreateOrderTests(unittest.TestCase):
    def setUp(self):
        svc._store.orders.clear()

    def test_create_order_returns_confirmation(self):
        result = svc.create_order("user1", _sample_items(), "tok_abc")
        self.assertEqual(result["user_id"], "user1")
        self.assertEqual(result["status"], svc.STATUS_CONFIRMED)
        self.assertEqual(result["total"], 39.48)
        self.assertTrue(result["order_id"].startswith("ord_"))
        self.assertTrue(result["payment_ref"].startswith("pay_"))

    def test_create_order_persists_order(self):
        result = svc.create_order("user1", _sample_items(), "tok_abc")
        self.assertIn(result["order_id"], svc._store.orders)

    def test_invalid_user_id_type(self):
        with self.assertRaises(svc.ValidationError):
            svc.create_order(123, _sample_items(), "tok")

    def test_empty_payment_token(self):
        with self.assertRaises(svc.ValidationError):
            svc.create_order("user1", _sample_items(), "   ")

    def test_items_not_a_list(self):
        with self.assertRaises(svc.ValidationError):
            svc.create_order("user1", "not-a-list", "tok")

    def test_empty_items(self):
        with self.assertRaises(svc.ValidationError):
            svc.create_order("user1", [], "tok")

    def test_item_negative_quantity(self):
        items = [{"name": "x", "quantity": -1, "unit_price": 1.0}]
        with self.assertRaises(svc.ValidationError):
            svc.create_order("user1", items, "tok")

    def test_item_bool_quantity_rejected(self):
        items = [{"name": "x", "quantity": True, "unit_price": 1.0}]
        with self.assertRaises(svc.ValidationError):
            svc.create_order("user1", items, "tok")

    def test_item_negative_unit_price(self):
        items = [{"name": "x", "quantity": 1, "unit_price": -5.0}]
        with self.assertRaises(svc.ValidationError):
            svc.create_order("user1", items, "tok")


class CancelOrderTests(unittest.TestCase):
    def setUp(self):
        svc._store.orders.clear()
        self.order = svc.create_order("user1", _sample_items(), "tok_abc")

    def test_cancel_order_success(self):
        result = svc.cancel_order("user1", self.order["order_id"], "changed mind")
        self.assertEqual(result["status"], svc.STATUS_CANCELLED)
        self.assertEqual(result["reason"], "changed mind")
        self.assertEqual(result["refund_amount"], 39.48)
        self.assertTrue(result["refund_ref"].startswith("ref_"))

    def test_cancel_unknown_order(self):
        with self.assertRaises(svc.OrderNotFoundError):
            svc.cancel_order("user1", "ord_missing", "reason")

    def test_cancel_wrong_user(self):
        with self.assertRaises(svc.OrderNotFoundError):
            svc.cancel_order("intruder", self.order["order_id"], "reason")

    def test_double_cancel_rejected(self):
        svc.cancel_order("user1", self.order["order_id"], "first")
        with self.assertRaises(svc.OrderError):
            svc.cancel_order("user1", self.order["order_id"], "second")

    def test_invalid_reason(self):
        with self.assertRaises(svc.ValidationError):
            svc.cancel_order("user1", self.order["order_id"], "")


class GetOrderStatusTests(unittest.TestCase):
    def setUp(self):
        svc._store.orders.clear()
        self.order = svc.create_order("user1", _sample_items(), "tok_abc")

    def test_get_status_success(self):
        result = svc.get_order_status("user1", self.order["order_id"])
        self.assertEqual(result["status"], svc.STATUS_CONFIRMED)
        self.assertEqual(result["total"], 39.48)
        self.assertEqual(len(result["items"]), 2)

    def test_get_status_unknown_order(self):
        with self.assertRaises(svc.OrderNotFoundError):
            svc.get_order_status("user1", "ord_missing")

    def test_get_status_wrong_user(self):
        with self.assertRaises(svc.OrderNotFoundError):
            svc.get_order_status("intruder", self.order["order_id"])

    def test_get_status_reflects_cancellation(self):
        svc.cancel_order("user1", self.order["order_id"], "reason")
        result = svc.get_order_status("user1", self.order["order_id"])
        self.assertEqual(result["status"], svc.STATUS_CANCELLED)

    def test_invalid_order_id_type(self):
        with self.assertRaises(svc.ValidationError):
            svc.get_order_status("user1", 999)


class ConfigTests(unittest.TestCase):
    def test_from_dict_filters_unknown_keys(self):
        cfg = svc.ServiceConfig.from_dict({"currency": "EUR", "bogus": 1})
        self.assertEqual(cfg.currency, "EUR")

    def test_from_dict_rejects_non_dict(self):
        with self.assertRaises(svc.ConfigError):
            svc.ServiceConfig.from_dict(["not", "a", "dict"])


if __name__ == "__main__":
    unittest.main()
