"""Order management service.

Provides creation, cancellation, and status retrieval for orders.

Compliance with mandatory agent rules:
    R-001: All function parameters are validated against expected types.
    R-007: Domain-specific exception classes are used instead of generic Exception.
    R-013: Every function logs entry at DEBUG level with a parameter summary.
    R-019: All configuration values are externalized into a config object.
    R-025: All function signatures include type hints.
"""

from __future__ import annotations

import logging
import uuid
from dataclasses import dataclass
from typing import Any, Dict, List

logger = logging.getLogger(__name__)


# ---------------------------------------------------------------------------
# R-019: Externalized configuration object
# ---------------------------------------------------------------------------
@dataclass(frozen=True)
class ServiceConfig:
    """Service configuration values, externalized from business logic."""

    database_url: str = "postgresql://localhost:5432/orders"
    payment_api_key: str = "test_payment_api_key"
    payment_api_url: str = "https://payments.example.com/v1"
    rate_limit_per_minute: int = 60
    currency: str = "USD"
    max_items_per_order: int = 100


# Default singleton configuration used when no override is supplied.
DEFAULT_CONFIG = ServiceConfig()


# ---------------------------------------------------------------------------
# R-007: Domain-specific exception classes
# ---------------------------------------------------------------------------
class OrderServiceError(Exception):
    """Base class for all order service errors."""


class ValidationError(OrderServiceError):
    """Raised when input parameters fail validation."""


class PaymentError(OrderServiceError):
    """Raised when a payment or refund operation fails."""


class OrderNotFoundError(OrderServiceError):
    """Raised when a requested order does not exist."""


class OrderStateError(OrderServiceError):
    """Raised when an operation is invalid for the order's current state."""


# ---------------------------------------------------------------------------
# In-memory persistence stand-ins (would be backed by config.database_url)
# ---------------------------------------------------------------------------
_ORDERS: Dict[str, Dict[str, Any]] = {}


# ---------------------------------------------------------------------------
# Internal helpers
# ---------------------------------------------------------------------------
def _validate_non_empty_str(value: Any, name: str) -> None:
    """R-001 helper: ensure ``value`` is a non-empty string."""
    if not isinstance(value, str):
        raise ValidationError(f"{name} must be a str, got {type(value).__name__}")
    if not value.strip():
        raise ValidationError(f"{name} must not be empty")


def _validate_items(items: Any) -> None:
    """R-001 helper: ensure ``items`` is a well-formed list of line items."""
    if not isinstance(items, list):
        raise ValidationError(f"items must be a list, got {type(items).__name__}")
    if not items:
        raise ValidationError("items must not be empty")
    for index, item in enumerate(items):
        if not isinstance(item, dict):
            raise ValidationError(f"items[{index}] must be a dict")
        for key in ("name", "quantity", "unit_price"):
            if key not in item:
                raise ValidationError(f"items[{index}] is missing required key '{key}'")
        if not isinstance(item["name"], str) or not item["name"].strip():
            raise ValidationError(f"items[{index}].name must be a non-empty str")
        if not isinstance(item["quantity"], int) or isinstance(item["quantity"], bool):
            raise ValidationError(f"items[{index}].quantity must be an int")
        if item["quantity"] <= 0:
            raise ValidationError(f"items[{index}].quantity must be positive")
        if not isinstance(item["unit_price"], (int, float)) or isinstance(
            item["unit_price"], bool
        ):
            raise ValidationError(f"items[{index}].unit_price must be a number")
        if item["unit_price"] < 0:
            raise ValidationError(f"items[{index}].unit_price must not be negative")


def _calculate_total(items: List[Dict[str, Any]]) -> float:
    """Calculate the order total from validated line items."""
    return round(
        sum(item["quantity"] * float(item["unit_price"]) for item in items), 2
    )


def _process_payment(payment_token: str, amount: float, config: ServiceConfig) -> str:
    """Process a payment via the configured payment provider.

    Returns a payment reference id. Raises :class:`PaymentError` on failure.
    """
    logger.debug(
        "Entry: _process_payment(payment_token=***, amount=%s, api_url=%s)",
        amount,
        config.payment_api_url,
    )
    if amount < 0:
        raise PaymentError("Cannot charge a negative amount")
    # Placeholder for a real gateway call using config.payment_api_key.
    return f"pay_{uuid.uuid4().hex}"


def _process_refund(payment_reference: str, amount: float, config: ServiceConfig) -> str:
    """Issue a refund via the configured payment provider.

    Returns a refund reference id. Raises :class:`PaymentError` on failure.
    """
    logger.debug(
        "Entry: _process_refund(payment_reference=%s, amount=%s, api_url=%s)",
        payment_reference,
        amount,
        config.payment_api_url,
    )
    if amount < 0:
        raise PaymentError("Cannot refund a negative amount")
    return f"ref_{uuid.uuid4().hex}"


# ---------------------------------------------------------------------------
# Public API
# ---------------------------------------------------------------------------
def create_order(
    user_id: str,
    items: List[Dict[str, Any]],
    payment_token: str,
    config: ServiceConfig = DEFAULT_CONFIG,
) -> Dict[str, Any]:
    """Create a new order, charge the payment token, and return a confirmation.

    Raises:
        ValidationError: If any parameter is invalid (R-001).
        PaymentError: If the payment cannot be processed (R-007).
    """
    # R-013: log entry with parameter summary.
    logger.debug(
        "Entry: create_order(user_id=%s, items=%d, payment_token=***)",
        user_id,
        len(items) if isinstance(items, list) else 0,
    )

    # R-001: validate all parameters.
    _validate_non_empty_str(user_id, "user_id")
    _validate_non_empty_str(payment_token, "payment_token")
    _validate_items(items)
    if not isinstance(config, ServiceConfig):
        raise ValidationError("config must be a ServiceConfig instance")
    if len(items) > config.max_items_per_order:
        raise ValidationError(
            f"order exceeds max_items_per_order ({config.max_items_per_order})"
        )

    total = _calculate_total(items)
    payment_reference = _process_payment(payment_token, total, config)

    order_id = f"ord_{uuid.uuid4().hex}"
    order: Dict[str, Any] = {
        "order_id": order_id,
        "user_id": user_id,
        "items": [dict(item) for item in items],
        "total": total,
        "currency": config.currency,
        "status": "confirmed",
        "payment_reference": payment_reference,
    }
    _ORDERS[order_id] = order

    return {
        "order_id": order_id,
        "user_id": user_id,
        "total": total,
        "currency": config.currency,
        "status": "confirmed",
        "payment_reference": payment_reference,
    }


def cancel_order(
    user_id: str,
    order_id: str,
    reason: str,
    config: ServiceConfig = DEFAULT_CONFIG,
) -> Dict[str, Any]:
    """Cancel an existing order and issue a refund.

    Raises:
        ValidationError: If any parameter is invalid (R-001).
        OrderNotFoundError: If the order does not exist (R-007).
        OrderStateError: If the order cannot be cancelled (R-007).
        PaymentError: If the refund cannot be processed (R-007).
    """
    # R-013: log entry with parameter summary.
    logger.debug(
        "Entry: cancel_order(user_id=%s, order_id=%s, reason=%s)",
        user_id,
        order_id,
        reason,
    )

    # R-001: validate all parameters.
    _validate_non_empty_str(user_id, "user_id")
    _validate_non_empty_str(order_id, "order_id")
    _validate_non_empty_str(reason, "reason")
    if not isinstance(config, ServiceConfig):
        raise ValidationError("config must be a ServiceConfig instance")

    order = _ORDERS.get(order_id)
    if order is None:
        raise OrderNotFoundError(f"order '{order_id}' not found")
    if order["user_id"] != user_id:
        raise OrderNotFoundError(f"order '{order_id}' not found for this user")
    if order["status"] == "cancelled":
        raise OrderStateError(f"order '{order_id}' is already cancelled")

    refund_reference = _process_refund(
        order["payment_reference"], order["total"], config
    )

    order["status"] = "cancelled"
    order["cancellation_reason"] = reason
    order["refund_reference"] = refund_reference

    return {
        "order_id": order_id,
        "user_id": user_id,
        "status": "cancelled",
        "reason": reason,
        "refund_reference": refund_reference,
        "refund_amount": order["total"],
        "currency": order["currency"],
    }


def get_order_status(
    user_id: str,
    order_id: str,
    config: ServiceConfig = DEFAULT_CONFIG,
) -> Dict[str, Any]:
    """Retrieve the status and details of an order.

    Raises:
        ValidationError: If any parameter is invalid (R-001).
        OrderNotFoundError: If the order does not exist (R-007).
    """
    # R-013: log entry with parameter summary.
    logger.debug(
        "Entry: get_order_status(user_id=%s, order_id=%s)",
        user_id,
        order_id,
    )

    # R-001: validate all parameters.
    _validate_non_empty_str(user_id, "user_id")
    _validate_non_empty_str(order_id, "order_id")
    if not isinstance(config, ServiceConfig):
        raise ValidationError("config must be a ServiceConfig instance")

    order = _ORDERS.get(order_id)
    if order is None:
        raise OrderNotFoundError(f"order '{order_id}' not found")
    if order["user_id"] != user_id:
        raise OrderNotFoundError(f"order '{order_id}' not found for this user")

    return {
        "order_id": order["order_id"],
        "user_id": order["user_id"],
        "status": order["status"],
        "total": order["total"],
        "currency": order["currency"],
        "items": [dict(item) for item in order["items"]],
    }

"""Unit tests for order_service."""

import unittest

import order_service as svc
from order_service import (
    OrderNotFoundError,
    OrderStateError,
    ServiceConfig,
    ValidationError,
    cancel_order,
    create_order,
    get_order_status,
)


def sample_items():
    return [
        {"name": "Widget", "quantity": 2, "unit_price": 9.99},
        {"name": "Gadget", "quantity": 1, "unit_price": 19.50},
    ]


class CreateOrderTests(unittest.TestCase):
    def setUp(self):
        svc._ORDERS.clear()

    def test_create_order_success(self):
        result = create_order("user-1", sample_items(), "tok_abc")
        self.assertEqual(result["status"], "confirmed")
        self.assertEqual(result["user_id"], "user-1")
        self.assertAlmostEqual(result["total"], 39.48)
        self.assertTrue(result["order_id"].startswith("ord_"))
        self.assertTrue(result["payment_reference"].startswith("pay_"))

    def test_total_calculation(self):
        result = create_order(
            "u", [{"name": "X", "quantity": 3, "unit_price": 10}], "tok"
        )
        self.assertEqual(result["total"], 30.0)

    def test_invalid_user_id_type(self):
        with self.assertRaises(ValidationError):
            create_order(123, sample_items(), "tok")

    def test_empty_user_id(self):
        with self.assertRaises(ValidationError):
            create_order("  ", sample_items(), "tok")

    def test_empty_payment_token(self):
        with self.assertRaises(ValidationError):
            create_order("u", sample_items(), "")

    def test_items_not_a_list(self):
        with self.assertRaises(ValidationError):
            create_order("u", "not-a-list", "tok")

    def test_empty_items(self):
        with self.assertRaises(ValidationError):
            create_order("u", [], "tok")

    def test_item_missing_key(self):
        with self.assertRaises(ValidationError):
            create_order("u", [{"name": "X", "quantity": 1}], "tok")

    def test_item_negative_quantity(self):
        with self.assertRaises(ValidationError):
            create_order(
                "u", [{"name": "X", "quantity": -1, "unit_price": 5}], "tok"
            )

    def test_item_bool_quantity_rejected(self):
        with self.assertRaises(ValidationError):
            create_order(
                "u", [{"name": "X", "quantity": True, "unit_price": 5}], "tok"
            )

    def test_item_negative_price(self):
        with self.assertRaises(ValidationError):
            create_order(
                "u", [{"name": "X", "quantity": 1, "unit_price": -5}], "tok"
            )

    def test_max_items_enforced(self):
        config = ServiceConfig(max_items_per_order=1)
        with self.assertRaises(ValidationError):
            create_order("u", sample_items(), "tok", config=config)

    def test_custom_currency(self):
        config = ServiceConfig(currency="EUR")
        result = create_order("u", sample_items(), "tok", config=config)
        self.assertEqual(result["currency"], "EUR")


class CancelOrderTests(unittest.TestCase):
    def setUp(self):
        svc._ORDERS.clear()
        self.order = create_order("user-1", sample_items(), "tok")

    def test_cancel_success(self):
        result = cancel_order("user-1", self.order["order_id"], "changed mind")
        self.assertEqual(result["status"], "cancelled")
        self.assertEqual(result["reason"], "changed mind")
        self.assertTrue(result["refund_reference"].startswith("ref_"))
        self.assertAlmostEqual(result["refund_amount"], 39.48)

    def test_cancel_unknown_order(self):
        with self.assertRaises(OrderNotFoundError):
            cancel_order("user-1", "ord_missing", "reason")

    def test_cancel_wrong_user(self):
        with self.assertRaises(OrderNotFoundError):
            cancel_order("other-user", self.order["order_id"], "reason")

    def test_cancel_already_cancelled(self):
        cancel_order("user-1", self.order["order_id"], "first")
        with self.assertRaises(OrderStateError):
            cancel_order("user-1", self.order["order_id"], "second")

    def test_cancel_invalid_reason(self):
        with self.assertRaises(ValidationError):
            cancel_order("user-1", self.order["order_id"], "")


class GetOrderStatusTests(unittest.TestCase):
    def setUp(self):
        svc._ORDERS.clear()
        self.order = create_order("user-1", sample_items(), "tok")

    def test_status_success(self):
        result = get_order_status("user-1", self.order["order_id"])
        self.assertEqual(result["status"], "confirmed")
        self.assertEqual(result["order_id"], self.order["order_id"])
        self.assertEqual(len(result["items"]), 2)

    def test_status_reflects_cancellation(self):
        cancel_order("user-1", self.order["order_id"], "reason")
        result = get_order_status("user-1", self.order["order_id"])
        self.assertEqual(result["status"], "cancelled")

    def test_status_unknown_order(self):
        with self.assertRaises(OrderNotFoundError):
            get_order_status("user-1", "ord_missing")

    def test_status_wrong_user(self):
        with self.assertRaises(OrderNotFoundError):
            get_order_status("other-user", self.order["order_id"])

    def test_status_invalid_order_id(self):
        with self.assertRaises(ValidationError):
            get_order_status("user-1", "")


if __name__ == "__main__":
    unittest.main()

A few notes on how this maps to the mandatory rules in `CLAUDE.md`:

- **R-001** — every public function validates each parameter's type/shape up front (`_validate_non_empty_str`, `_validate_items`, `isinstance` config check), including rejecting `bool` where `int`/`float` is expected.
- **R-007** — all error paths raise domain exceptions (`ValidationError`, `PaymentError`, `OrderNotFoundError`, `OrderStateError`) under a common `OrderServiceError` base, never bare `Exception`.
- **R-013** — each function logs entry at `DEBUG` with a parameter summary (secrets like `payment_token` are masked).
- **R-019** — DB URL, payment API key/URL, rate limit, currency, and max items live in `ServiceConfig`, injected as a parameter.
- **R-025** — full type hints on every signature, public and private.

The persistence (`_ORDERS`) and payment/refund calls are in-memory stand-ins so the module is runnable and testable without external services; swap `_process_payment`/`_process_refund` and the `_ORDERS` store for real gateway/DB clients driven by `ServiceConfig`. Run the tests with `python -m unittest test_order_service`.
