"""Order management service.

Provides creation, cancellation, and status retrieval for orders.

This module adheres to the mandatory agent rules:
    R-001: validate all function parameters against expected types.
    R-007: use domain-specific exception classes.
    R-013: log function entry at DEBUG level with a parameter summary.
    R-019: externalize all configuration values into a config object.
    R-025: include type hints for all function signatures.
"""

from __future__ import annotations

import logging
import uuid
from dataclasses import dataclass, field
from typing import Any, Dict, List, Optional

logger = logging.getLogger(__name__)


# ---------------------------------------------------------------------------
# R-019: Configuration externalized into a config object.
# ---------------------------------------------------------------------------
@dataclass
class ServiceConfig:
    """Holds all externalized configuration values for the order service."""

    database_url: str = "postgresql://localhost:5432/orders"
    payment_api_key: str = "test-payment-api-key"
    payment_api_url: str = "https://payments.example.com/v1"
    rate_limit_per_minute: int = 60
    refund_window_days: int = 30
    currency: str = "USD"


# Default config instance used when callers do not supply their own.
DEFAULT_CONFIG: ServiceConfig = ServiceConfig()


# ---------------------------------------------------------------------------
# R-007: Domain-specific exception classes (not generic Exception).
# ---------------------------------------------------------------------------
class OrderServiceError(Exception):
    """Base class for all order-service errors."""


class ValidationError(OrderServiceError):
    """Raised when a function parameter fails validation."""


class OrderNotFoundError(OrderServiceError):
    """Raised when a referenced order does not exist."""


class PaymentError(OrderServiceError):
    """Raised when payment processing fails."""


class RefundError(OrderServiceError):
    """Raised when a refund cannot be issued."""


# ---------------------------------------------------------------------------
# In-memory data store (stand-in for a real database at config.database_url).
# ---------------------------------------------------------------------------
@dataclass
class _OrderRecord:
    """Internal representation of a stored order."""

    order_id: str
    user_id: str
    items: List[Dict[str, Any]]
    total: float
    currency: str
    status: str
    payment_token: str
    cancel_reason: Optional[str] = field(default=None)


_ORDERS: Dict[str, _OrderRecord] = {}


# ---------------------------------------------------------------------------
# Validation helpers (R-001).
# ---------------------------------------------------------------------------
def _require_non_empty_str(value: Any, name: str) -> str:
    """Validate that ``value`` is a non-empty string."""
    if not isinstance(value, str):
        raise ValidationError(f"{name} must be a str, got {type(value).__name__}")
    if not value.strip():
        raise ValidationError(f"{name} must not be empty")
    return value


def _validate_items(items: Any) -> List[Dict[str, Any]]:
    """Validate that ``items`` is a list of well-formed item dicts."""
    if not isinstance(items, list):
        raise ValidationError(f"items must be a list, got {type(items).__name__}")
    if not items:
        raise ValidationError("items must not be empty")

    for index, item in enumerate(items):
        if not isinstance(item, dict):
            raise ValidationError(f"items[{index}] must be a dict")
        if not isinstance(item.get("name"), str) or not item["name"].strip():
            raise ValidationError(f"items[{index}].name must be a non-empty str")
        quantity = item.get("quantity")
        if not isinstance(quantity, int) or isinstance(quantity, bool) or quantity <= 0:
            raise ValidationError(f"items[{index}].quantity must be a positive int")
        unit_price = item.get("unit_price")
        if isinstance(unit_price, bool) or not isinstance(unit_price, (int, float)):
            raise ValidationError(f"items[{index}].unit_price must be a number")
        if unit_price < 0:
            raise ValidationError(f"items[{index}].unit_price must not be negative")
    return items


# ---------------------------------------------------------------------------
# Payment helpers (stand-ins for a real payment gateway).
# ---------------------------------------------------------------------------
def _process_payment(
    payment_token: str, amount: float, config: ServiceConfig
) -> str:
    """Process a payment and return a transaction id.

    Raises:
        PaymentError: if the payment cannot be processed.
    """
    if amount <= 0:
        raise PaymentError("payment amount must be positive")
    # A real implementation would call config.payment_api_url with the api key.
    return f"txn_{uuid.uuid4().hex}"


def _issue_refund(
    payment_token: str, amount: float, config: ServiceConfig
) -> str:
    """Issue a refund and return a refund id.

    Raises:
        RefundError: if the refund cannot be issued.
    """
    if amount <= 0:
        raise RefundError("refund amount must be positive")
    return f"rfnd_{uuid.uuid4().hex}"


# ---------------------------------------------------------------------------
# Public API.
# ---------------------------------------------------------------------------
def create_order(
    user_id: str,
    items: List[Dict[str, Any]],
    payment_token: str,
    config: ServiceConfig = DEFAULT_CONFIG,
) -> Dict[str, Any]:
    """Create a new order, charge the payment token, and confirm.

    Args:
        user_id: Identifier of the user placing the order.
        items: List of dicts, each with ``name``, ``quantity``, ``unit_price``.
        payment_token: Token used to charge the customer.
        config: Service configuration object.

    Returns:
        An order confirmation dict.

    Raises:
        ValidationError: if any parameter is invalid.
        PaymentError: if payment processing fails.
    """
    logger.debug(
        "Entering create_order(user_id=%s, items_count=%s, payment_token=***)",
        user_id,
        len(items) if isinstance(items, list) else "n/a",
    )

    _require_non_empty_str(user_id, "user_id")
    _validate_items(items)
    _require_non_empty_str(payment_token, "payment_token")
    if not isinstance(config, ServiceConfig):
        raise ValidationError("config must be a ServiceConfig instance")

    total: float = round(
        sum(item["quantity"] * item["unit_price"] for item in items), 2
    )

    transaction_id: str = _process_payment(payment_token, total, config)
    order_id: str = f"ord_{uuid.uuid4().hex}"

    _ORDERS[order_id] = _OrderRecord(
        order_id=order_id,
        user_id=user_id,
        items=list(items),
        total=total,
        currency=config.currency,
        status="confirmed",
        payment_token=payment_token,
    )

    return {
        "order_id": order_id,
        "user_id": user_id,
        "status": "confirmed",
        "total": total,
        "currency": config.currency,
        "transaction_id": transaction_id,
        "items": list(items),
    }


def cancel_order(
    user_id: str,
    order_id: str,
    reason: str,
    config: ServiceConfig = DEFAULT_CONFIG,
) -> Dict[str, Any]:
    """Cancel an existing order and issue a refund.

    Args:
        user_id: Identifier of the user who owns the order.
        order_id: Identifier of the order to cancel.
        reason: Human-readable cancellation reason.
        config: Service configuration object.

    Returns:
        A cancellation confirmation dict.

    Raises:
        ValidationError: if any parameter is invalid.
        OrderNotFoundError: if the order does not exist for the user.
        RefundError: if the refund cannot be issued.
    """
    logger.debug(
        "Entering cancel_order(user_id=%s, order_id=%s, reason=%s)",
        user_id,
        order_id,
        reason,
    )

    _require_non_empty_str(user_id, "user_id")
    _require_non_empty_str(order_id, "order_id")
    _require_non_empty_str(reason, "reason")
    if not isinstance(config, ServiceConfig):
        raise ValidationError("config must be a ServiceConfig instance")

    record = _ORDERS.get(order_id)
    if record is None or record.user_id != user_id:
        raise OrderNotFoundError(f"order {order_id} not found for user {user_id}")

    if record.status == "cancelled":
        raise ValidationError(f"order {order_id} is already cancelled")

    refund_id: str = _issue_refund(record.payment_token, record.total, config)

    record.status = "cancelled"
    record.cancel_reason = reason

    return {
        "order_id": order_id,
        "user_id": user_id,
        "status": "cancelled",
        "reason": reason,
        "refund_id": refund_id,
        "refund_amount": record.total,
        "currency": record.currency,
    }


def get_order_status(
    user_id: str,
    order_id: str,
    config: ServiceConfig = DEFAULT_CONFIG,
) -> Dict[str, Any]:
    """Retrieve the status and details of an order.

    Args:
        user_id: Identifier of the user who owns the order.
        order_id: Identifier of the order to look up.
        config: Service configuration object.

    Returns:
        An order status dict.

    Raises:
        ValidationError: if any parameter is invalid.
        OrderNotFoundError: if the order does not exist for the user.
    """
    logger.debug(
        "Entering get_order_status(user_id=%s, order_id=%s)",
        user_id,
        order_id,
    )

    _require_non_empty_str(user_id, "user_id")
    _require_non_empty_str(order_id, "order_id")
    if not isinstance(config, ServiceConfig):
        raise ValidationError("config must be a ServiceConfig instance")

    record = _ORDERS.get(order_id)
    if record is None or record.user_id != user_id:
        raise OrderNotFoundError(f"order {order_id} not found for user {user_id}")

    return {
        "order_id": record.order_id,
        "user_id": record.user_id,
        "status": record.status,
        "total": record.total,
        "currency": record.currency,
        "items": list(record.items),
        "cancel_reason": record.cancel_reason,
    }


# ===========================================================================
# Unit tests
# ===========================================================================
import unittest


def _sample_items() -> List[Dict[str, Any]]:
    return [
        {"name": "Widget", "quantity": 2, "unit_price": 9.99},
        {"name": "Gadget", "quantity": 1, "unit_price": 19.50},
    ]


class OrderServiceTestCase(unittest.TestCase):
    """Tests for the public order-service API."""

    def setUp(self) -> None:
        _ORDERS.clear()

    # --- create_order ----------------------------------------------------
    def test_create_order_success(self) -> None:
        result = create_order("user-1", _sample_items(), "tok_abc")
        self.assertEqual(result["status"], "confirmed")
        self.assertEqual(result["user_id"], "user-1")
        self.assertEqual(result["total"], round(2 * 9.99 + 19.50, 2))
        self.assertTrue(result["order_id"].startswith("ord_"))
        self.assertTrue(result["transaction_id"].startswith("txn_"))

    def test_create_order_invalid_user_id(self) -> None:
        with self.assertRaises(ValidationError):
            create_order("", _sample_items(), "tok_abc")
        with self.assertRaises(ValidationError):
            create_order(123, _sample_items(), "tok_abc")  # type: ignore[arg-type]

    def test_create_order_empty_items(self) -> None:
        with self.assertRaises(ValidationError):
            create_order("user-1", [], "tok_abc")

    def test_create_order_bad_item_quantity(self) -> None:
        bad = [{"name": "X", "quantity": 0, "unit_price": 1.0}]
        with self.assertRaises(ValidationError):
            create_order("user-1", bad, "tok_abc")

    def test_create_order_bad_item_price(self) -> None:
        bad = [{"name": "X", "quantity": 1, "unit_price": -5.0}]
        with self.assertRaises(ValidationError):
            create_order("user-1", bad, "tok_abc")

    def test_create_order_missing_payment_token(self) -> None:
        with self.assertRaises(ValidationError):
            create_order("user-1", _sample_items(), "")

    # --- cancel_order ----------------------------------------------------
    def test_cancel_order_success(self) -> None:
        created = create_order("user-1", _sample_items(), "tok_abc")
        result = cancel_order("user-1", created["order_id"], "changed mind")
        self.assertEqual(result["status"], "cancelled")
        self.assertEqual(result["reason"], "changed mind")
        self.assertTrue(result["refund_id"].startswith("rfnd_"))
        self.assertEqual(result["refund_amount"], created["total"])

    def test_cancel_order_not_found(self) -> None:
        with self.assertRaises(OrderNotFoundError):
            cancel_order("user-1", "ord_missing", "reason")

    def test_cancel_order_wrong_user(self) -> None:
        created = create_order("user-1", _sample_items(), "tok_abc")
        with self.assertRaises(OrderNotFoundError):
            cancel_order("user-2", created["order_id"], "reason")

    def test_cancel_order_already_cancelled(self) -> None:
        created = create_order("user-1", _sample_items(), "tok_abc")
        cancel_order("user-1", created["order_id"], "first")
        with self.assertRaises(ValidationError):
            cancel_order("user-1", created["order_id"], "second")

    def test_cancel_order_invalid_reason(self) -> None:
        created = create_order("user-1", _sample_items(), "tok_abc")
        with self.assertRaises(ValidationError):
            cancel_order("user-1", created["order_id"], "")

    # --- get_order_status ------------------------------------------------
    def test_get_order_status_success(self) -> None:
        created = create_order("user-1", _sample_items(), "tok_abc")
        status = get_order_status("user-1", created["order_id"])
        self.assertEqual(status["status"], "confirmed")
        self.assertEqual(status["order_id"], created["order_id"])
        self.assertIsNone(status["cancel_reason"])

    def test_get_order_status_reflects_cancellation(self) -> None:
        created = create_order("user-1", _sample_items(), "tok_abc")
        cancel_order("user-1", created["order_id"], "changed mind")
        status = get_order_status("user-1", created["order_id"])
        self.assertEqual(status["status"], "cancelled")
        self.assertEqual(status["cancel_reason"], "changed mind")

    def test_get_order_status_not_found(self) -> None:
        with self.assertRaises(OrderNotFoundError):
            get_order_status("user-1", "ord_missing")

    def test_get_order_status_invalid_params(self) -> None:
        with self.assertRaises(ValidationError):
            get_order_status("", "ord_x")

    # --- config ----------------------------------------------------------
    def test_custom_config_currency(self) -> None:
        cfg = ServiceConfig(currency="EUR")
        result = create_order("user-1", _sample_items(), "tok_abc", config=cfg)
        self.assertEqual(result["currency"], "EUR")


if __name__ == "__main__":
    unittest.main()
