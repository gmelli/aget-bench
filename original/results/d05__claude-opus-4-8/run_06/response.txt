"""Order management service.

Provides creation, cancellation, and status retrieval for orders.

Mandatory rule compliance:
    R-001  All public function parameters are validated against their expected
           types before processing.
    R-007  Domain-specific exception classes are used instead of generic
           Exception.
    R-013  Every function logs entry at DEBUG level with a parameter summary.
    R-019  All configuration values are externalized into a config object.
    R-025  Type hints are present on all function signatures.
"""

from __future__ import annotations

import logging
import uuid
from dataclasses import dataclass
from typing import Any, Dict, List, Optional

logger = logging.getLogger(__name__)


# ---------------------------------------------------------------------------
# R-019: Externalized configuration
# ---------------------------------------------------------------------------
@dataclass(frozen=True)
class ServiceConfig:
    """Externalized configuration for the order service (R-019)."""

    database_url: str = "postgresql://localhost:5432/orders"
    payment_api_key: str = "sk_test_change_me"
    payment_api_url: str = "https://payments.example.com/v1"
    rate_limit_per_minute: int = 60
    currency: str = "USD"
    refund_window_days: int = 30

    def __post_init__(self) -> None:
        # R-001: validate configuration values on construction.
        if not isinstance(self.database_url, str) or not self.database_url:
            raise ConfigurationError("database_url must be a non-empty string")
        if not isinstance(self.payment_api_key, str) or not self.payment_api_key:
            raise ConfigurationError("payment_api_key must be a non-empty string")
        if not isinstance(self.rate_limit_per_minute, int) or self.rate_limit_per_minute <= 0:
            raise ConfigurationError("rate_limit_per_minute must be a positive int")


# ---------------------------------------------------------------------------
# R-007: Domain-specific exception classes
# ---------------------------------------------------------------------------
class OrderServiceError(Exception):
    """Base class for all order service errors (R-007)."""


class ConfigurationError(OrderServiceError):
    """Raised when the service configuration is invalid."""


class ValidationError(OrderServiceError):
    """Raised when a function parameter fails validation (R-001)."""


class OrderNotFoundError(OrderServiceError):
    """Raised when an order cannot be located for the given user."""


class OrderStateError(OrderServiceError):
    """Raised when an operation is invalid for the order's current state."""


class PaymentError(OrderServiceError):
    """Raised when payment processing or refunding fails."""


# ---------------------------------------------------------------------------
# Internal validation helpers (R-001)
# ---------------------------------------------------------------------------
def _require_non_empty_str(value: Any, name: str) -> str:
    """Validate that ``value`` is a non-empty string (R-001)."""
    if not isinstance(value, str):
        raise ValidationError(f"{name} must be a str, got {type(value).__name__}")
    if not value.strip():
        raise ValidationError(f"{name} must not be empty")
    return value


def _validate_items(items: Any) -> List[Dict[str, Any]]:
    """Validate the ``items`` list and each item's shape (R-001)."""
    if not isinstance(items, list):
        raise ValidationError(f"items must be a list, got {type(items).__name__}")
    if not items:
        raise ValidationError("items must contain at least one item")

    validated: List[Dict[str, Any]] = []
    for index, item in enumerate(items):
        if not isinstance(item, dict):
            raise ValidationError(f"items[{index}] must be a dict")
        name = item.get("name")
        quantity = item.get("quantity")
        unit_price = item.get("unit_price")

        if not isinstance(name, str) or not name.strip():
            raise ValidationError(f"items[{index}].name must be a non-empty str")
        if not isinstance(quantity, int) or isinstance(quantity, bool) or quantity <= 0:
            raise ValidationError(f"items[{index}].quantity must be a positive int")
        if isinstance(unit_price, bool) or not isinstance(unit_price, (int, float)) or unit_price < 0:
            raise ValidationError(f"items[{index}].unit_price must be a non-negative number")

        validated.append({"name": name, "quantity": quantity, "unit_price": float(unit_price)})
    return validated


# ---------------------------------------------------------------------------
# Payment gateway abstraction (kept injectable for testing)
# ---------------------------------------------------------------------------
class PaymentGateway:
    """Thin payment gateway wrapper.

    The default implementation is a stub; in production it would call
    ``config.payment_api_url`` using ``config.payment_api_key``.
    """

    def __init__(self, config: ServiceConfig) -> None:
        self._config = config

    def charge(self, payment_token: str, amount: float) -> str:
        """Charge ``amount`` and return a transaction id."""
        _require_non_empty_str(payment_token, "payment_token")
        if amount <= 0:
            raise PaymentError("charge amount must be positive")
        # Stubbed gateway interaction.
        return f"txn_{uuid.uuid4().hex}"

    def refund(self, transaction_id: str, amount: float) -> str:
        """Refund ``amount`` for ``transaction_id`` and return a refund id."""
        _require_non_empty_str(transaction_id, "transaction_id")
        if amount <= 0:
            raise PaymentError("refund amount must be positive")
        return f"rfnd_{uuid.uuid4().hex}"


# ---------------------------------------------------------------------------
# Order service
# ---------------------------------------------------------------------------
@dataclass
class _OrderRecord:
    """Internal in-memory representation of an order."""

    order_id: str
    user_id: str
    items: List[Dict[str, Any]]
    total: float
    status: str
    transaction_id: str
    currency: str
    refund_id: Optional[str] = None
    cancel_reason: Optional[str] = None


class OrderService:
    """Stateful order management service.

    Storage is in-memory for simplicity; ``config.database_url`` documents where
    a real implementation would persist records.
    """

    def __init__(
        self,
        config: Optional[ServiceConfig] = None,
        payment_gateway: Optional[PaymentGateway] = None,
    ) -> None:
        self._config: ServiceConfig = config or ServiceConfig()
        self._payments: PaymentGateway = payment_gateway or PaymentGateway(self._config)
        self._orders: Dict[str, _OrderRecord] = {}

    # -- public API ---------------------------------------------------------
    def create_order(
        self, user_id: str, items: List[Dict[str, Any]], payment_token: str
    ) -> Dict[str, Any]:
        """Create a new order, charge payment, and return a confirmation dict."""
        logger.debug(
            "Entering create_order(user_id=%r, items=<%d items>, payment_token=%s)",
            user_id,
            len(items) if isinstance(items, list) else 0,
            "***" if isinstance(payment_token, str) and payment_token else payment_token,
        )

        # R-001: validate parameters.
        user_id = _require_non_empty_str(user_id, "user_id")
        payment_token = _require_non_empty_str(payment_token, "payment_token")
        validated_items = _validate_items(items)

        total = round(sum(i["quantity"] * i["unit_price"] for i in validated_items), 2)

        try:
            transaction_id = self._payments.charge(payment_token, total)
        except OrderServiceError:
            raise
        except Exception as exc:  # noqa: BLE001 - wrap unexpected gateway failures
            raise PaymentError(f"payment processing failed: {exc}") from exc

        order_id = f"ord_{uuid.uuid4().hex}"
        record = _OrderRecord(
            order_id=order_id,
            user_id=user_id,
            items=validated_items,
            total=total,
            status="CONFIRMED",
            transaction_id=transaction_id,
            currency=self._config.currency,
        )
        self._orders[order_id] = record

        return {
            "order_id": order_id,
            "user_id": user_id,
            "status": record.status,
            "total": total,
            "currency": record.currency,
            "transaction_id": transaction_id,
            "items": validated_items,
        }

    def cancel_order(self, user_id: str, order_id: str, reason: str) -> Dict[str, Any]:
        """Cancel an existing order, issue a refund, and confirm cancellation."""
        logger.debug(
            "Entering cancel_order(user_id=%r, order_id=%r, reason=%r)",
            user_id,
            order_id,
            reason,
        )

        # R-001: validate parameters.
        user_id = _require_non_empty_str(user_id, "user_id")
        order_id = _require_non_empty_str(order_id, "order_id")
        reason = _require_non_empty_str(reason, "reason")

        record = self._get_owned_order(user_id, order_id)

        if record.status == "CANCELLED":
            raise OrderStateError(f"order {order_id} is already cancelled")
        if record.status != "CONFIRMED":
            raise OrderStateError(
                f"order {order_id} cannot be cancelled from status {record.status}"
            )

        try:
            refund_id = self._payments.refund(record.transaction_id, record.total)
        except OrderServiceError:
            raise
        except Exception as exc:  # noqa: BLE001 - wrap unexpected gateway failures
            raise PaymentError(f"refund failed: {exc}") from exc

        record.status = "CANCELLED"
        record.refund_id = refund_id
        record.cancel_reason = reason

        return {
            "order_id": order_id,
            "user_id": user_id,
            "status": record.status,
            "refund_id": refund_id,
            "refund_amount": record.total,
            "reason": reason,
        }

    def get_order_status(self, user_id: str, order_id: str) -> Dict[str, Any]:
        """Retrieve order details and return its current status dict."""
        logger.debug(
            "Entering get_order_status(user_id=%r, order_id=%r)", user_id, order_id
        )

        # R-001: validate parameters.
        user_id = _require_non_empty_str(user_id, "user_id")
        order_id = _require_non_empty_str(order_id, "order_id")

        record = self._get_owned_order(user_id, order_id)

        return {
            "order_id": record.order_id,
            "user_id": record.user_id,
            "status": record.status,
            "total": record.total,
            "currency": record.currency,
            "items": record.items,
            "refund_id": record.refund_id,
            "cancel_reason": record.cancel_reason,
        }

    # -- internals ----------------------------------------------------------
    def _get_owned_order(self, user_id: str, order_id: str) -> _OrderRecord:
        """Fetch an order, enforcing that it belongs to ``user_id``."""
        record = self._orders.get(order_id)
        if record is None or record.user_id != user_id:
            raise OrderNotFoundError(f"order {order_id} not found for user {user_id}")
        return record


# ---------------------------------------------------------------------------
# Module-level convenience API backed by a shared default service instance
# ---------------------------------------------------------------------------
_default_service: OrderService = OrderService()


def create_order(
    user_id: str, items: List[Dict[str, Any]], payment_token: str
) -> Dict[str, Any]:
    """Create an order using the shared default service."""
    logger.debug(
        "Entering create_order(user_id=%r, items=<%s items>, payment_token=***)",
        user_id,
        len(items) if isinstance(items, list) else 0,
    )
    return _default_service.create_order(user_id, items, payment_token)


def cancel_order(user_id: str, order_id: str, reason: str) -> Dict[str, Any]:
    """Cancel an order using the shared default service."""
    logger.debug(
        "Entering cancel_order(user_id=%r, order_id=%r, reason=%r)",
        user_id,
        order_id,
        reason,
    )
    return _default_service.cancel_order(user_id, order_id, reason)


def get_order_status(user_id: str, order_id: str) -> Dict[str, Any]:
    """Get an order's status using the shared default service."""
    logger.debug(
        "Entering get_order_status(user_id=%r, order_id=%r)", user_id, order_id
    )
    return _default_service.get_order_status(user_id, order_id)


# ===========================================================================
# Unit tests (R-025: type hints on all test signatures)
# ===========================================================================
import unittest  # noqa: E402


class OrderServiceTests(unittest.TestCase):
    """Unit tests covering all public service functions."""

    def setUp(self) -> None:
        self.config: ServiceConfig = ServiceConfig(currency="USD")
        self.service: OrderService = OrderService(config=self.config)
        self.items: List[Dict[str, Any]] = [
            {"name": "widget", "quantity": 2, "unit_price": 9.99},
            {"name": "gadget", "quantity": 1, "unit_price": 19.50},
        ]

    # -- create_order -------------------------------------------------------
    def test_create_order_success(self) -> None:
        result = self.service.create_order("user-1", self.items, "tok_abc")
        self.assertEqual(result["status"], "CONFIRMED")
        self.assertEqual(result["total"], round(2 * 9.99 + 19.50, 2))
        self.assertTrue(result["order_id"].startswith("ord_"))
        self.assertTrue(result["transaction_id"].startswith("txn_"))

    def test_create_order_invalid_user_id(self) -> None:
        with self.assertRaises(ValidationError):
            self.service.create_order("", self.items, "tok_abc")
        with self.assertRaises(ValidationError):
            self.service.create_order(123, self.items, "tok_abc")  # type: ignore[arg-type]

    def test_create_order_invalid_items(self) -> None:
        with self.assertRaises(ValidationError):
            self.service.create_order("user-1", [], "tok_abc")
        with self.assertRaises(ValidationError):
            self.service.create_order("user-1", "not-a-list", "tok_abc")  # type: ignore[arg-type]
        with self.assertRaises(ValidationError):
            self.service.create_order(
                "user-1", [{"name": "x", "quantity": 0, "unit_price": 1.0}], "tok_abc"
            )

    def test_create_order_invalid_payment_token(self) -> None:
        with self.assertRaises(ValidationError):
            self.service.create_order("user-1", self.items, "")

    # -- cancel_order -------------------------------------------------------
    def test_cancel_order_success(self) -> None:
        created = self.service.create_order("user-1", self.items, "tok_abc")
        result = self.service.cancel_order("user-1", created["order_id"], "changed mind")
        self.assertEqual(result["status"], "CANCELLED")
        self.assertEqual(result["refund_amount"], created["total"])
        self.assertTrue(result["refund_id"].startswith("rfnd_"))

    def test_cancel_order_not_found(self) -> None:
        with self.assertRaises(OrderNotFoundError):
            self.service.cancel_order("user-1", "ord_missing", "reason")

    def test_cancel_order_wrong_user(self) -> None:
        created = self.service.create_order("user-1", self.items, "tok_abc")
        with self.assertRaises(OrderNotFoundError):
            self.service.cancel_order("user-2", created["order_id"], "reason")

    def test_cancel_order_already_cancelled(self) -> None:
        created = self.service.create_order("user-1", self.items, "tok_abc")
        self.service.cancel_order("user-1", created["order_id"], "reason")
        with self.assertRaises(OrderStateError):
            self.service.cancel_order("user-1", created["order_id"], "reason again")

    def test_cancel_order_invalid_reason(self) -> None:
        created = self.service.create_order("user-1", self.items, "tok_abc")
        with self.assertRaises(ValidationError):
            self.service.cancel_order("user-1", created["order_id"], "")

    # -- get_order_status ---------------------------------------------------
    def test_get_order_status_success(self) -> None:
        created = self.service.create_order("user-1", self.items, "tok_abc")
        status = self.service.get_order_status("user-1", created["order_id"])
        self.assertEqual(status["status"], "CONFIRMED")
        self.assertEqual(status["order_id"], created["order_id"])
        self.assertIsNone(status["refund_id"])

    def test_get_order_status_reflects_cancellation(self) -> None:
        created = self.service.create_order("user-1", self.items, "tok_abc")
        self.service.cancel_order("user-1", created["order_id"], "reason")
        status = self.service.get_order_status("user-1", created["order_id"])
        self.assertEqual(status["status"], "CANCELLED")
        self.assertEqual(status["cancel_reason"], "reason")

    def test_get_order_status_not_found(self) -> None:
        with self.assertRaises(OrderNotFoundError):
            self.service.get_order_status("user-1", "ord_missing")

    # -- config -------------------------------------------------------------
    def test_invalid_config_rejected(self) -> None:
        with self.assertRaises(ConfigurationError):
            ServiceConfig(database_url="")
        with self.assertRaises(ConfigurationError):
            ServiceConfig(rate_limit_per_minute=0)

    # -- module-level API ---------------------------------------------------
    def test_module_level_functions(self) -> None:
        created = create_order("user-9", self.items, "tok_xyz")
        self.assertEqual(get_order_status("user-9", created["order_id"])["status"], "CONFIRMED")
        cancelled = cancel_order("user-9", created["order_id"], "test")
        self.assertEqual(cancelled["status"], "CANCELLED")


if __name__ == "__main__":
    unittest.main()

A few design notes:

- **Rule compliance** — Every public function validates its parameters (R-001) raising domain-specific `ValidationError`/`OrderNotFoundError`/`OrderStateError`/`PaymentError` (R-007), logs entry at DEBUG with a parameter summary (R-013, with the payment token masked), reads all settings from `ServiceConfig` (R-019), and carries full type hints (R-025).
- **Both APIs** — There's a stateful `OrderService` class plus module-level `create_order`/`cancel_order`/`get_order_status` wrappers backed by a shared default instance, matching the exact signatures you specified.
- **Testability** — The `PaymentGateway` is injectable, so payment/refund behavior can be stubbed in tests without network calls. Storage is in-memory (the `database_url` documents where a real backend would go).
