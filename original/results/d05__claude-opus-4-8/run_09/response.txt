```python
"""Order management service.

Provides order creation, cancellation, and status retrieval with payment
processing and refund handling.

Implemented per the mandatory agent rules:
    R-001  validate all parameters against expected types
    R-007  domain-specific exception classes
    R-013  log function entry at DEBUG with parameter summary
    R-019  externalize configuration into a config object
    R-025  type hints on all function signatures
"""

from __future__ import annotations

import logging
import uuid
from dataclasses import dataclass
from typing import Any, Dict, List, Optional

logger = logging.getLogger(__name__)


# ---------------------------------------------------------------------------
# R-019: Configuration object (externalized config values)
# ---------------------------------------------------------------------------
@dataclass
class ServiceConfig:
    """Externalized configuration for the order service."""

    database_url: str = "postgresql://localhost:5432/orders"
    payment_api_key: str = "sk_test_placeholder"
    payment_api_url: str = "https://payments.example.com/v1"
    rate_limit_per_minute: int = 60
    currency: str = "USD"
    max_items_per_order: int = 100

    def __post_init__(self) -> None:
        # R-001: validate configuration types.
        if not isinstance(self.database_url, str):
            raise ConfigurationError("database_url must be a str")
        if not isinstance(self.payment_api_key, str):
            raise ConfigurationError("payment_api_key must be a str")
        if not isinstance(self.payment_api_url, str):
            raise ConfigurationError("payment_api_url must be a str")
        if not isinstance(self.rate_limit_per_minute, int):
            raise ConfigurationError("rate_limit_per_minute must be an int")
        if not isinstance(self.currency, str):
            raise ConfigurationError("currency must be a str")
        if not isinstance(self.max_items_per_order, int):
            raise ConfigurationError("max_items_per_order must be an int")


# ---------------------------------------------------------------------------
# R-007: Domain-specific exception classes
# ---------------------------------------------------------------------------
class OrderServiceError(Exception):
    """Base class for all order service errors."""


class ConfigurationError(OrderServiceError):
    """Raised when the service configuration is invalid."""


class ValidationError(OrderServiceError):
    """Raised when a function parameter fails validation."""


class OrderNotFoundError(OrderServiceError):
    """Raised when an order cannot be located for a user."""


class OrderStateError(OrderServiceError):
    """Raised when an operation is invalid for the order's current state."""


class PaymentError(OrderServiceError):
    """Raised when payment processing fails."""


class RefundError(OrderServiceError):
    """Raised when a refund cannot be issued."""


# ---------------------------------------------------------------------------
# Order statuses
# ---------------------------------------------------------------------------
class OrderStatus:
    CONFIRMED = "CONFIRMED"
    CANCELLED = "CANCELLED"


# ---------------------------------------------------------------------------
# Internal helpers
# ---------------------------------------------------------------------------
def _require_non_empty_str(value: Any, name: str) -> str:
    """R-001: validate that ``value`` is a non-empty string."""
    if not isinstance(value, str):
        raise ValidationError(f"{name} must be a str, got {type(value).__name__}")
    if not value.strip():
        raise ValidationError(f"{name} must not be empty")
    return value


def _validate_items(items: Any, config: ServiceConfig) -> List[Dict[str, Any]]:
    """R-001: validate the ``items`` list and each item's structure."""
    if not isinstance(items, list):
        raise ValidationError(f"items must be a list, got {type(items).__name__}")
    if not items:
        raise ValidationError("items must not be empty")
    if len(items) > config.max_items_per_order:
        raise ValidationError(
            f"items exceeds max_items_per_order ({config.max_items_per_order})"
        )

    validated: List[Dict[str, Any]] = []
    for index, item in enumerate(items):
        if not isinstance(item, dict):
            raise ValidationError(f"item[{index}] must be a dict")

        name = item.get("name")
        quantity = item.get("quantity")
        unit_price = item.get("unit_price")

        if not isinstance(name, str) or not name.strip():
            raise ValidationError(f"item[{index}].name must be a non-empty str")
        if not isinstance(quantity, int) or isinstance(quantity, bool) or quantity <= 0:
            raise ValidationError(f"item[{index}].quantity must be a positive int")
        if isinstance(unit_price, bool) or not isinstance(unit_price, (int, float)):
            raise ValidationError(f"item[{index}].unit_price must be a number")
        if unit_price < 0:
            raise ValidationError(f"item[{index}].unit_price must be non-negative")

        validated.append(
            {"name": name, "quantity": quantity, "unit_price": float(unit_price)}
        )
    return validated


def _calculate_total(items: List[Dict[str, Any]]) -> float:
    """Sum quantity * unit_price across all items, rounded to cents."""
    total = sum(item["quantity"] * item["unit_price"] for item in items)
    return round(total, 2)


# ---------------------------------------------------------------------------
# Payment gateway abstraction (config-driven; injectable for tests)
# ---------------------------------------------------------------------------
class PaymentGateway:
    """Thin payment gateway wrapper driven by :class:`ServiceConfig`.

    In production this would call ``config.payment_api_url``. It is isolated
    here so tests can substitute a fake implementation.
    """

    def __init__(self, config: ServiceConfig) -> None:
        self._config = config

    def charge(self, payment_token: str, amount: float, currency: str) -> str:
        logger.debug(
            "PaymentGateway.charge(amount=%s, currency=%s, token=***)",
            amount,
            currency,
        )
        if not payment_token.strip():
            raise PaymentError("payment_token rejected by gateway")
        return f"chg_{uuid.uuid4().hex[:16]}"

    def refund(self, charge_id: str, amount: float) -> str:
        logger.debug(
            "PaymentGateway.refund(charge_id=%s, amount=%s)", charge_id, amount
        )
        if not charge_id:
            raise RefundError("missing charge_id for refund")
        return f"ref_{uuid.uuid4().hex[:16]}"


# ---------------------------------------------------------------------------
# In-memory order store (placeholder for config.database_url backed store)
# ---------------------------------------------------------------------------
@dataclass
class _OrderRecord:
    order_id: str
    user_id: str
    items: List[Dict[str, Any]]
    total: float
    currency: str
    status: str
    charge_id: str
    refund_id: Optional[str] = None
    cancel_reason: Optional[str] = None


class OrderService:
    """Order management service.

    The module-level functions delegate to a default instance, but the class
    can be instantiated directly with a custom config and payment gateway
    (useful for testing).
    """

    def __init__(
        self,
        config: Optional[ServiceConfig] = None,
        payment_gateway: Optional[PaymentGateway] = None,
    ) -> None:
        self._config: ServiceConfig = config or ServiceConfig()
        self._gateway: PaymentGateway = payment_gateway or PaymentGateway(self._config)
        self._orders: Dict[str, _OrderRecord] = {}

    # -- R-001 helper -------------------------------------------------------
    def _get_owned_order(self, user_id: str, order_id: str) -> _OrderRecord:
        record = self._orders.get(order_id)
        if record is None or record.user_id != user_id:
            raise OrderNotFoundError(
                f"order {order_id!r} not found for user {user_id!r}"
            )
        return record

    # -- Public API ---------------------------------------------------------
    def create_order(
        self, user_id: str, items: List[Dict[str, Any]], payment_token: str
    ) -> Dict[str, Any]:
        # R-013: log function entry at DEBUG with parameter summary.
        logger.debug(
            "create_order(user_id=%r, items=%d, payment_token=***)",
            user_id,
            len(items) if isinstance(items, list) else -1,
        )
        # R-001: validate parameters.
        _require_non_empty_str(user_id, "user_id")
        _require_non_empty_str(payment_token, "payment_token")
        validated_items = _validate_items(items, self._config)

        total = _calculate_total(validated_items)
        charge_id = self._gateway.charge(payment_token, total, self._config.currency)

        order_id = f"ord_{uuid.uuid4().hex}"
        record = _OrderRecord(
            order_id=order_id,
            user_id=user_id,
            items=validated_items,
            total=total,
            currency=self._config.currency,
            status=OrderStatus.CONFIRMED,
            charge_id=charge_id,
        )
        self._orders[order_id] = record

        return {
            "order_id": order_id,
            "user_id": user_id,
            "status": OrderStatus.CONFIRMED,
            "total": total,
            "currency": self._config.currency,
            "items": validated_items,
            "charge_id": charge_id,
        }

    def cancel_order(
        self, user_id: str, order_id: str, reason: str
    ) -> Dict[str, Any]:
        # R-013: log function entry at DEBUG with parameter summary.
        logger.debug(
            "cancel_order(user_id=%r, order_id=%r, reason=%r)",
            user_id,
            order_id,
            reason,
        )
        # R-001: validate parameters.
        _require_non_empty_str(user_id, "user_id")
        _require_non_empty_str(order_id, "order_id")
        _require_non_empty_str(reason, "reason")

        record = self._get_owned_order(user_id, order_id)
        if record.status == OrderStatus.CANCELLED:
            raise OrderStateError(f"order {order_id!r} is already cancelled")

        refund_id = self._gateway.refund(record.charge_id, record.total)
        record.status = OrderStatus.CANCELLED
        record.refund_id = refund_id
        record.cancel_reason = reason

        return {
            "order_id": order_id,
            "user_id": user_id,
            "status": OrderStatus.CANCELLED,
            "reason": reason,
            "refund_id": refund_id,
            "refund_amount": record.total,
            "currency": record.currency,
        }

    def get_order_status(self, user_id: str, order_id: str) -> Dict[str, Any]:
        # R-013: log function entry at DEBUG with parameter summary.
        logger.debug(
            "get_order_status(user_id=%r, order_id=%r)", user_id, order_id
        )
        # R-001: validate parameters.
        _require_non_empty_str(user_id, "user_id")
        _require_non_empty_str(order_id, "order_id")

        record = self._get_owned_order(user_id, order_id)
        return {
            "order_id": record.order_id,
            "user_id": record.user_id,
            "status": record.status,
            "total": record.total,
            "currency": record.currency,
            "items": record.items,
            "refund_id": record.refund_id,
            "cancel_reason": record.cancel_reason,
        }


# ---------------------------------------------------------------------------
# Module-level default instance and convenience functions
# ---------------------------------------------------------------------------
_default_service: OrderService = OrderService()


def create_order(
    user_id: str, items: List[Dict[str, Any]], payment_token: str
) -> Dict[str, Any]:
    """Create a new order. See :meth:`OrderService.create_order`."""
    return _default_service.create_order(user_id, items, payment_token)


def cancel_order(user_id: str, order_id: str, reason: str) -> Dict[str, Any]:
    """Cancel an existing order. See :meth:`OrderService.cancel_order`."""
    return _default_service.cancel_order(user_id, order_id, reason)


def get_order_status(user_id: str, order_id: str) -> Dict[str, Any]:
    """Retrieve an order's status. See :meth:`OrderService.get_order_status`."""
    return _default_service.get_order_status(user_id, order_id)
```

```python
"""Unit tests for order_service."""

import unittest

from order_service import (
    OrderNotFoundError,
    OrderService,
    OrderStateError,
    OrderStatus,
    PaymentError,
    PaymentGateway,
    ServiceConfig,
    ValidationError,
)


class _FakeGateway(PaymentGateway):
    """Deterministic gateway so charge/refund ids are predictable in tests."""

    def __init__(self, config: ServiceConfig) -> None:
        super().__init__(config)
        self.charges = []
        self.refunds = []

    def charge(self, payment_token: str, amount: float, currency: str) -> str:
        if not payment_token.strip():
            raise PaymentError("rejected")
        self.charges.append((payment_token, amount, currency))
        return f"chg_test_{len(self.charges)}"

    def refund(self, charge_id: str, amount: float) -> str:
        self.refunds.append((charge_id, amount))
        return f"ref_test_{len(self.refunds)}"


def _items():
    return [
        {"name": "Widget", "quantity": 2, "unit_price": 9.99},
        {"name": "Gadget", "quantity": 1, "unit_price": 4.50},
    ]


class OrderServiceTests(unittest.TestCase):
    def setUp(self) -> None:
        self.config = ServiceConfig()
        self.gateway = _FakeGateway(self.config)
        self.service = OrderService(self.config, self.gateway)

    # -- create_order ------------------------------------------------------
    def test_create_order_success(self) -> None:
        result = self.service.create_order("user-1", _items(), "tok_abc")
        self.assertEqual(result["status"], OrderStatus.CONFIRMED)
        self.assertEqual(result["total"], 24.48)
        self.assertEqual(result["currency"], "USD")
        self.assertTrue(result["order_id"].startswith("ord_"))
        self.assertEqual(len(self.gateway.charges), 1)

    def test_create_order_invalid_user_id(self) -> None:
        with self.assertRaises(ValidationError):
            self.service.create_order("", _items(), "tok")
        with self.assertRaises(ValidationError):
            self.service.create_order(123, _items(), "tok")  # type: ignore[arg-type]

    def test_create_order_invalid_token(self) -> None:
        with self.assertRaises(ValidationError):
            self.service.create_order("user-1", _items(), "")

    def test_create_order_empty_items(self) -> None:
        with self.assertRaises(ValidationError):
            self.service.create_order("user-1", [], "tok")

    def test_create_order_items_not_list(self) -> None:
        with self.assertRaises(ValidationError):
            self.service.create_order("user-1", "nope", "tok")  # type: ignore[arg-type]

    def test_create_order_bad_quantity(self) -> None:
        bad = [{"name": "X", "quantity": 0, "unit_price": 1.0}]
        with self.assertRaises(ValidationError):
            self.service.create_order("user-1", bad, "tok")

    def test_create_order_negative_price(self) -> None:
        bad = [{"name": "X", "quantity": 1, "unit_price": -1.0}]
        with self.assertRaises(ValidationError):
            self.service.create_order("user-1", bad, "tok")

    def test_create_order_bool_quantity_rejected(self) -> None:
        bad = [{"name": "X", "quantity": True, "unit_price": 1.0}]
        with self.assertRaises(ValidationError):
            self.service.create_order("user-1", bad, "tok")

    def test_create_order_payment_failure(self) -> None:
        with self.assertRaises(PaymentError):
            self.service.create_order("user-1", _items(), "   ")

    # -- cancel_order ------------------------------------------------------
    def test_cancel_order_success(self) -> None:
        created = self.service.create_order("user-1", _items(), "tok")
        result = self.service.cancel_order("user-1", created["order_id"], "changed mind")
        self.assertEqual(result["status"], OrderStatus.CANCELLED)
        self.assertEqual(result["refund_amount"], 24.48)
        self.assertEqual(result["reason"], "changed mind")
        self.assertEqual(len(self.gateway.refunds), 1)

    def test_cancel_order_not_found(self) -> None:
        with self.assertRaises(OrderNotFoundError):
            self.service.cancel_order("user-1", "ord_missing", "reason")

    def test_cancel_order_wrong_user(self) -> None:
        created = self.service.create_order("user-1", _items(), "tok")
        with self.assertRaises(OrderNotFoundError):
            self.service.cancel_order("user-2", created["order_id"], "reason")

    def test_cancel_order_already_cancelled(self) -> None:
        created = self.service.create_order("user-1", _items(), "tok")
        self.service.cancel_order("user-1", created["order_id"], "first")
        with self.assertRaises(OrderStateError):
            self.service.cancel_order("user-1", created["order_id"], "second")

    def test_cancel_order_requires_reason(self) -> None:
        created = self.service.create_order("user-1", _items(), "tok")
        with self.assertRaises(ValidationError):
            self.service.cancel_order("user-1", created["order_id"], "")

    # -- get_order_status --------------------------------------------------
    def test_get_order_status_success(self) -> None:
        created = self.service.create_order("user-1", _items(), "tok")
        status = self.service.get_order_status("user-1", created["order_id"])
        self.assertEqual(status["status"], OrderStatus.CONFIRMED)
        self.assertEqual(status["order_id"], created["order_id"])
        self.assertIsNone(status["refund_id"])

    def test_get_order_status_after_cancel(self) -> None:
        created = self.service.create_order("user-1", _items(), "tok")
        self.service.cancel_order("user-1", created["order_id"], "reason")
        status = self.service.get_order_status("user-1", created["order_id"])
        self.assertEqual(status["status"], OrderStatus.CANCELLED)
        self.assertIsNotNone(status["refund_id"])
        self.assertEqual(status["cancel_reason"], "reason")

    def test_get_order_status_not_found(self) -> None:
        with self.assertRaises(OrderNotFoundError):
            self.service.get_order_status("user-1", "ord_missing")

    # -- config ------------------------------------------------------------
    def test_config_validation(self) -> None:
        from order_service import ConfigurationError

        with self.assertRaises(ConfigurationError):
            ServiceConfig(rate_limit_per_minute="lots")  # type: ignore[arg-type]


if __name__ == "__main__":
    unittest.main()
```

**Rule compliance:**
- **R-001** — every public function validates parameter types (`_require_non_empty_str`, `_validate_items`); `ServiceConfig.__post_init__` validates config types.
- **R-007** — domain exceptions (`ValidationError`, `OrderNotFoundError`, `OrderStateError`, `PaymentError`, `RefundError`) all subclass `OrderServiceError`; no bare `Exception` raised.
- **R-013** — each public method logs entry at DEBUG with a parameter summary (tokens masked).
- **R-019** — all config values (DB URL, payment API key/URL, rate limit, currency, item cap) live in `ServiceConfig` and are injected, not hardcoded.
- **R-025** — full type hints on every signature.

The `OrderService` class takes an injectable `PaymentGateway`, which is what lets the tests run deterministically without a real payment backend.
