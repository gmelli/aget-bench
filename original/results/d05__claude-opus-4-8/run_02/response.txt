```python
"""Order management service.

Provides creation, cancellation and status retrieval for orders, with payment
and refund processing. Behaviour is governed by an externalized configuration
object (see ``ServiceConfig``).
"""

from __future__ import annotations

import logging
import uuid
from dataclasses import dataclass, field
from datetime import datetime, timezone
from typing import Any, Dict, List, Optional

logger = logging.getLogger(__name__)


# ---------------------------------------------------------------------------
# Configuration (R-019: all configuration values externalized into a config
# object rather than hard-coded at point of use).
# ---------------------------------------------------------------------------
@dataclass(frozen=True)
class ServiceConfig:
    """Service-wide configuration.

    Attributes:
        database_url: Connection string for the order datastore.
        payment_api_key: API key used to authenticate against the payment gateway.
        rate_limit_per_minute: Maximum number of order operations per user per minute.
        currency: ISO currency code used for monetary amounts.
        max_items_per_order: Upper bound on distinct line items in a single order.
    """

    database_url: str = "postgresql://localhost:5432/orders"
    payment_api_key: str = "test-payment-api-key"
    rate_limit_per_minute: int = 60
    currency: str = "USD"
    max_items_per_order: int = 100


# Default config instance; callers may inject their own ServiceConfig.
DEFAULT_CONFIG: ServiceConfig = ServiceConfig()


# ---------------------------------------------------------------------------
# Domain-specific exceptions (R-007: never raise generic Exception).
# ---------------------------------------------------------------------------
class OrderServiceError(Exception):
    """Base class for all order service errors."""


class ValidationError(OrderServiceError):
    """Raised when input parameters fail validation."""


class OrderNotFoundError(OrderServiceError):
    """Raised when a requested order does not exist for the given user."""


class OrderStateError(OrderServiceError):
    """Raised when an operation is invalid for the order's current state."""


class PaymentError(OrderServiceError):
    """Raised when payment or refund processing fails."""


# ---------------------------------------------------------------------------
# Lightweight in-memory store and external gateway stand-ins.
# In a real system these would talk to the database / payment provider
# identified by ServiceConfig.
# ---------------------------------------------------------------------------
@dataclass
class _OrderRecord:
    order_id: str
    user_id: str
    items: List[Dict[str, Any]]
    total: float
    currency: str
    status: str
    payment_reference: str
    created_at: str
    cancelled_at: Optional[str] = None
    refund_reference: Optional[str] = None
    cancel_reason: Optional[str] = None


@dataclass
class _OrderStore:
    """Trivial in-memory persistence keyed by order_id."""

    _orders: Dict[str, _OrderRecord] = field(default_factory=dict)

    def save(self, record: _OrderRecord) -> None:
        self._orders[record.order_id] = record

    def get(self, order_id: str) -> Optional[_OrderRecord]:
        return self._orders.get(order_id)

    def clear(self) -> None:
        self._orders.clear()


# Module-level store instance.
_STORE: _OrderStore = _OrderStore()


def _mask(token: str) -> str:
    """Return a redacted form of a sensitive token for logging."""
    if not isinstance(token, str) or len(token) <= 4:
        return "****"
    return f"****{token[-4:]}"


def _process_payment(payment_token: str, amount: float, config: ServiceConfig) -> str:
    """Charge the payment token. Returns a payment reference.

    Raises:
        PaymentError: If the amount is non-positive or the token is rejected.
    """
    logger.debug(
        "Entering _process_payment(payment_token=%r, amount=%.2f)",
        _mask(payment_token),
        amount,
    )
    if amount <= 0:
        raise PaymentError("Cannot charge a non-positive amount")
    if not payment_token or payment_token.lower().startswith("invalid"):
        raise PaymentError("Payment token was rejected by the gateway")
    # A real implementation would call the gateway using config.payment_api_key.
    return f"pay_{uuid.uuid4().hex[:16]}"


def _process_refund(payment_reference: str, amount: float, config: ServiceConfig) -> str:
    """Refund a previous charge. Returns a refund reference.

    Raises:
        PaymentError: If the refund cannot be issued.
    """
    logger.debug(
        "Entering _process_refund(payment_reference=%r, amount=%.2f)",
        payment_reference,
        amount,
    )
    if not payment_reference:
        raise PaymentError("Missing payment reference for refund")
    if amount <= 0:
        raise PaymentError("Cannot refund a non-positive amount")
    return f"ref_{uuid.uuid4().hex[:16]}"


# ---------------------------------------------------------------------------
# Validation helpers (R-001: validate all parameters against expected types).
# ---------------------------------------------------------------------------
def _require_non_empty_str(value: Any, name: str) -> str:
    if not isinstance(value, str):
        raise ValidationError(f"{name} must be a str, got {type(value).__name__}")
    if not value.strip():
        raise ValidationError(f"{name} must not be empty")
    return value


def _validate_items(items: Any, config: ServiceConfig) -> List[Dict[str, Any]]:
    """Validate the items list and each line item's shape and types."""
    if not isinstance(items, list):
        raise ValidationError(f"items must be a list, got {type(items).__name__}")
    if not items:
        raise ValidationError("items must contain at least one entry")
    if len(items) > config.max_items_per_order:
        raise ValidationError(
            f"items exceeds max_items_per_order ({config.max_items_per_order})"
        )

    validated: List[Dict[str, Any]] = []
    for index, item in enumerate(items):
        if not isinstance(item, dict):
            raise ValidationError(f"items[{index}] must be a dict")

        name = item.get("name")
        quantity = item.get("quantity")
        unit_price = item.get("unit_price")

        if not isinstance(name, str) or not name.strip():
            raise ValidationError(f"items[{index}].name must be a non-empty str")
        # bool is a subclass of int; reject it explicitly.
        if not isinstance(quantity, int) or isinstance(quantity, bool) or quantity <= 0:
            raise ValidationError(f"items[{index}].quantity must be a positive int")
        if isinstance(unit_price, bool) or not isinstance(unit_price, (int, float)):
            raise ValidationError(f"items[{index}].unit_price must be a number")
        if unit_price < 0:
            raise ValidationError(f"items[{index}].unit_price must not be negative")

        validated.append(
            {"name": name, "quantity": quantity, "unit_price": float(unit_price)}
        )
    return validated


def _calculate_total(items: List[Dict[str, Any]]) -> float:
    """Sum quantity * unit_price across all line items, rounded to cents."""
    total = sum(item["quantity"] * item["unit_price"] for item in items)
    return round(float(total), 2)


def _now_iso() -> str:
    return datetime.now(timezone.utc).isoformat()


# ---------------------------------------------------------------------------
# Public API
# ---------------------------------------------------------------------------
def create_order(
    user_id: str,
    items: List[Dict[str, Any]],
    payment_token: str,
    config: ServiceConfig = DEFAULT_CONFIG,
) -> Dict[str, Any]:
    """Create an order, charge payment, and return a confirmation.

    Args:
        user_id: Identifier of the user placing the order.
        items: Line items; each a dict with ``name`` (str), ``quantity`` (int),
            and ``unit_price`` (number).
        payment_token: Token authorizing the charge.
        config: Service configuration.

    Returns:
        Order confirmation dict with order id, status, total and payment reference.

    Raises:
        ValidationError: If any parameter is malformed.
        PaymentError: If the charge cannot be completed.
    """
    logger.debug(
        "Entering create_order(user_id=%r, items=<%d items>, payment_token=%r)",
        user_id,
        len(items) if isinstance(items, list) else 0,
        _mask(payment_token) if isinstance(payment_token, str) else payment_token,
    )

    _require_non_empty_str(user_id, "user_id")
    _require_non_empty_str(payment_token, "payment_token")
    validated_items = _validate_items(items, config)

    total = _calculate_total(validated_items)
    payment_reference = _process_payment(payment_token, total, config)

    order_id = f"ord_{uuid.uuid4().hex}"
    record = _OrderRecord(
        order_id=order_id,
        user_id=user_id,
        items=validated_items,
        total=total,
        currency=config.currency,
        status="confirmed",
        payment_reference=payment_reference,
        created_at=_now_iso(),
    )
    _STORE.save(record)

    return {
        "order_id": order_id,
        "user_id": user_id,
        "status": record.status,
        "total": total,
        "currency": record.currency,
        "items": validated_items,
        "payment_reference": payment_reference,
        "created_at": record.created_at,
    }


def cancel_order(
    user_id: str,
    order_id: str,
    reason: str,
    config: ServiceConfig = DEFAULT_CONFIG,
) -> Dict[str, Any]:
    """Cancel an existing order and issue a refund.

    Args:
        user_id: Identifier of the user who owns the order.
        order_id: Identifier of the order to cancel.
        reason: Human-readable cancellation reason.
        config: Service configuration.

    Returns:
        Cancellation confirmation dict.

    Raises:
        ValidationError: If any parameter is malformed.
        OrderNotFoundError: If the order does not exist for this user.
        OrderStateError: If the order is already cancelled.
        PaymentError: If the refund cannot be issued.
    """
    logger.debug(
        "Entering cancel_order(user_id=%r, order_id=%r, reason=%r)",
        user_id,
        order_id,
        reason,
    )

    _require_non_empty_str(user_id, "user_id")
    _require_non_empty_str(order_id, "order_id")
    _require_non_empty_str(reason, "reason")

    record = _STORE.get(order_id)
    if record is None or record.user_id != user_id:
        raise OrderNotFoundError(f"Order {order_id!r} not found for user {user_id!r}")
    if record.status == "cancelled":
        raise OrderStateError(f"Order {order_id!r} is already cancelled")

    refund_reference = _process_refund(record.payment_reference, record.total, config)

    record.status = "cancelled"
    record.cancelled_at = _now_iso()
    record.refund_reference = refund_reference
    record.cancel_reason = reason
    _STORE.save(record)

    return {
        "order_id": order_id,
        "user_id": user_id,
        "status": record.status,
        "reason": reason,
        "refund_reference": refund_reference,
        "refunded_amount": record.total,
        "currency": record.currency,
        "cancelled_at": record.cancelled_at,
    }


def get_order_status(
    user_id: str,
    order_id: str,
    config: ServiceConfig = DEFAULT_CONFIG,
) -> Dict[str, Any]:
    """Retrieve the status and details of an order.

    Args:
        user_id: Identifier of the user who owns the order.
        order_id: Identifier of the order to look up.
        config: Service configuration.

    Returns:
        Order status dict.

    Raises:
        ValidationError: If any parameter is malformed.
        OrderNotFoundError: If the order does not exist for this user.
    """
    logger.debug(
        "Entering get_order_status(user_id=%r, order_id=%r)", user_id, order_id
    )

    _require_non_empty_str(user_id, "user_id")
    _require_non_empty_str(order_id, "order_id")

    record = _STORE.get(order_id)
    if record is None or record.user_id != user_id:
        raise OrderNotFoundError(f"Order {order_id!r} not found for user {user_id!r}")

    return {
        "order_id": record.order_id,
        "user_id": record.user_id,
        "status": record.status,
        "total": record.total,
        "currency": record.currency,
        "items": record.items,
        "created_at": record.created_at,
        "cancelled_at": record.cancelled_at,
        "cancel_reason": record.cancel_reason,
    }
```

```python
"""Unit tests for order_service."""

import unittest

import order_service
from order_service import (
    OrderNotFoundError,
    OrderStateError,
    PaymentError,
    ValidationError,
    cancel_order,
    create_order,
    get_order_status,
)


def _items():
    return [
        {"name": "Widget", "quantity": 2, "unit_price": 9.99},
        {"name": "Gadget", "quantity": 1, "unit_price": 5.00},
    ]


class OrderServiceTestCase(unittest.TestCase):
    def setUp(self):
        # Isolate each test from shared module state.
        order_service._STORE.clear()

    # --- create_order -----------------------------------------------------
    def test_create_order_success(self):
        result = create_order("user-1", _items(), "tok_valid")
        self.assertEqual(result["status"], "confirmed")
        self.assertEqual(result["total"], 24.98)  # 2*9.99 + 1*5.00
        self.assertTrue(result["order_id"].startswith("ord_"))
        self.assertTrue(result["payment_reference"].startswith("pay_"))
        self.assertEqual(result["currency"], "USD")

    def test_create_order_rejects_bad_user_id(self):
        with self.assertRaises(ValidationError):
            create_order("", _items(), "tok_valid")
        with self.assertRaises(ValidationError):
            create_order(None, _items(), "tok_valid")  # type: ignore[arg-type]

    def test_create_order_rejects_empty_items(self):
        with self.assertRaises(ValidationError):
            create_order("user-1", [], "tok_valid")

    def test_create_order_rejects_malformed_item(self):
        with self.assertRaises(ValidationError):
            create_order("user-1", [{"name": "X", "quantity": 0, "unit_price": 1}], "tok_valid")
        with self.assertRaises(ValidationError):
            create_order("user-1", [{"name": "X", "quantity": True, "unit_price": 1}], "tok_valid")
        with self.assertRaises(ValidationError):
            create_order("user-1", [{"name": "X", "quantity": 1, "unit_price": -1}], "tok_valid")

    def test_create_order_payment_rejected(self):
        with self.assertRaises(PaymentError):
            create_order("user-1", _items(), "invalid-token")

    # --- get_order_status -------------------------------------------------
    def test_get_order_status_success(self):
        created = create_order("user-1", _items(), "tok_valid")
        status = get_order_status("user-1", created["order_id"])
        self.assertEqual(status["status"], "confirmed")
        self.assertEqual(status["order_id"], created["order_id"])

    def test_get_order_status_not_found(self):
        with self.assertRaises(OrderNotFoundError):
            get_order_status("user-1", "ord_missing")

    def test_get_order_status_wrong_user(self):
        created = create_order("user-1", _items(), "tok_valid")
        with self.assertRaises(OrderNotFoundError):
            get_order_status("user-2", created["order_id"])

    def test_get_order_status_validation(self):
        with self.assertRaises(ValidationError):
            get_order_status("", "ord_x")

    # --- cancel_order -----------------------------------------------------
    def test_cancel_order_success(self):
        created = create_order("user-1", _items(), "tok_valid")
        result = cancel_order("user-1", created["order_id"], "changed mind")
        self.assertEqual(result["status"], "cancelled")
        self.assertEqual(result["refunded_amount"], 24.98)
        self.assertTrue(result["refund_reference"].startswith("ref_"))

        status = get_order_status("user-1", created["order_id"])
        self.assertEqual(status["status"], "cancelled")
        self.assertEqual(status["cancel_reason"], "changed mind")

    def test_cancel_order_not_found(self):
        with self.assertRaises(OrderNotFoundError):
            cancel_order("user-1", "ord_missing", "reason")

    def test_cancel_order_already_cancelled(self):
        created = create_order("user-1", _items(), "tok_valid")
        cancel_order("user-1", created["order_id"], "first")
        with self.assertRaises(OrderStateError):
            cancel_order("user-1", created["order_id"], "again")

    def test_cancel_order_validation(self):
        created = create_order("user-1", _items(), "tok_valid")
        with self.assertRaises(ValidationError):
            cancel_order("user-1", created["order_id"], "")


if __name__ == "__main__":
    unittest.main()
```
