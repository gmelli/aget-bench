"""Order management service.

Provides creation, cancellation, and status retrieval for customer orders.

Implementation adheres to the mandatory agent rules:
    R-001  Validate all function parameters against expected types.
    R-007  Use domain-specific exception classes rather than generic Exception.
    R-013  Log function entry at DEBUG level with a parameter summary.
    R-019  Externalize all configuration values into a config object.
    R-025  Include type hints for all function signatures.
"""

from __future__ import annotations

import logging
import uuid
from dataclasses import dataclass
from typing import Any, Dict, List, Optional

logger = logging.getLogger(__name__)


# --------------------------------------------------------------------------- #
# R-019: Externalized configuration
# --------------------------------------------------------------------------- #
@dataclass(frozen=True)
class ServiceConfig:
    """Configuration values for the order service.

    All tunable / environment-specific values live here rather than being
    hard-coded throughout the module (R-019).
    """

    database_url: str = "postgresql://localhost:5432/orders"
    payment_api_key: str = "test-payment-api-key"
    rate_limit_per_minute: int = 60
    max_items_per_order: int = 100
    currency: str = "USD"


# Default config instance used when callers do not supply their own.
DEFAULT_CONFIG: ServiceConfig = ServiceConfig()


# --------------------------------------------------------------------------- #
# R-007: Domain-specific exception classes
# --------------------------------------------------------------------------- #
class OrderServiceError(Exception):
    """Base class for all order service errors."""


class ValidationError(OrderServiceError):
    """Raised when an input parameter fails validation (R-001)."""


class OrderNotFoundError(OrderServiceError):
    """Raised when an order cannot be located for the given user."""


class OrderStateError(OrderServiceError):
    """Raised when an operation is invalid for the order's current state."""


class PaymentError(OrderServiceError):
    """Raised when payment processing or refunding fails."""


# --------------------------------------------------------------------------- #
# Internal helpers
# --------------------------------------------------------------------------- #
def _validate_non_empty_str(value: Any, name: str) -> str:
    """Validate that ``value`` is a non-empty string (R-001)."""
    if not isinstance(value, str):
        raise ValidationError(f"{name} must be a str, got {type(value).__name__}")
    if not value.strip():
        raise ValidationError(f"{name} must not be empty")
    return value


def _validate_items(items: Any, max_items: int) -> List[Dict[str, Any]]:
    """Validate the ``items`` parameter against the expected structure (R-001)."""
    if not isinstance(items, list):
        raise ValidationError(f"items must be a list, got {type(items).__name__}")
    if not items:
        raise ValidationError("items must contain at least one entry")
    if len(items) > max_items:
        raise ValidationError(f"items may not exceed {max_items} entries")

    for index, item in enumerate(items):
        if not isinstance(item, dict):
            raise ValidationError(f"items[{index}] must be a dict")
        _validate_non_empty_str(item.get("name"), f"items[{index}].name")

        quantity = item.get("quantity")
        if not isinstance(quantity, int) or isinstance(quantity, bool):
            raise ValidationError(f"items[{index}].quantity must be an int")
        if quantity <= 0:
            raise ValidationError(f"items[{index}].quantity must be positive")

        unit_price = item.get("unit_price")
        if isinstance(unit_price, bool) or not isinstance(unit_price, (int, float)):
            raise ValidationError(f"items[{index}].unit_price must be a number")
        if unit_price < 0:
            raise ValidationError(f"items[{index}].unit_price must not be negative")

    return items


def _calculate_total(items: List[Dict[str, Any]]) -> float:
    """Sum ``quantity * unit_price`` across all items."""
    total = sum(item["quantity"] * item["unit_price"] for item in items)
    return round(float(total), 2)


# --------------------------------------------------------------------------- #
# Storage abstraction
# --------------------------------------------------------------------------- #
class OrderRepository:
    """In-memory order store standing in for a real database.

    The connection target is taken from :class:`ServiceConfig.database_url`
    (R-019). In production this would be backed by a real database client.
    """

    def __init__(self, config: ServiceConfig) -> None:
        self._config: ServiceConfig = config
        self._orders: Dict[str, Dict[str, Any]] = {}

    def save(self, order: Dict[str, Any]) -> None:
        self._orders[order["order_id"]] = order

    def get(self, order_id: str) -> Optional[Dict[str, Any]]:
        return self._orders.get(order_id)


class PaymentGateway:
    """Minimal payment gateway abstraction.

    Uses the API key from :class:`ServiceConfig` (R-019). The methods here are
    deliberately simple so they can be mocked in tests.
    """

    def __init__(self, config: ServiceConfig) -> None:
        self._config: ServiceConfig = config

    def charge(self, payment_token: str, amount: float) -> str:
        """Charge ``amount`` against ``payment_token``; return a charge id."""
        if amount <= 0:
            raise PaymentError("charge amount must be positive")
        return f"chg_{uuid.uuid4().hex}"

    def refund(self, charge_id: str, amount: float) -> str:
        """Refund a prior charge; return a refund id."""
        if not charge_id:
            raise PaymentError("charge_id is required for a refund")
        return f"rfnd_{uuid.uuid4().hex}"


# --------------------------------------------------------------------------- #
# Service
# --------------------------------------------------------------------------- #
class OrderService:
    """Order management service holding its collaborators and config."""

    def __init__(
        self,
        config: Optional[ServiceConfig] = None,
        repository: Optional[OrderRepository] = None,
        payment_gateway: Optional[PaymentGateway] = None,
    ) -> None:
        self.config: ServiceConfig = config or DEFAULT_CONFIG
        self.repository: OrderRepository = repository or OrderRepository(self.config)
        self.payment_gateway: PaymentGateway = payment_gateway or PaymentGateway(
            self.config
        )

    def create_order(
        self, user_id: str, items: List[Dict[str, Any]], payment_token: str
    ) -> Dict[str, Any]:
        """Create a new order, charge payment, and return a confirmation."""
        logger.debug(
            "Entering create_order(user_id=%r, items=<%d items>, payment_token=***)",
            user_id,
            len(items) if isinstance(items, list) else 0,
        )

        # R-001: validate parameters.
        _validate_non_empty_str(user_id, "user_id")
        _validate_non_empty_str(payment_token, "payment_token")
        validated_items = _validate_items(items, self.config.max_items_per_order)

        total = _calculate_total(validated_items)
        order_id = f"ord_{uuid.uuid4().hex}"

        charge_id = self.payment_gateway.charge(payment_token, total)

        order: Dict[str, Any] = {
            "order_id": order_id,
            "user_id": user_id,
            "items": validated_items,
            "total": total,
            "currency": self.config.currency,
            "status": "confirmed",
            "charge_id": charge_id,
        }
        self.repository.save(order)

        return {
            "order_id": order_id,
            "user_id": user_id,
            "total": total,
            "currency": self.config.currency,
            "status": "confirmed",
        }

    def cancel_order(
        self, user_id: str, order_id: str, reason: str
    ) -> Dict[str, Any]:
        """Cancel an existing order and issue a refund."""
        logger.debug(
            "Entering cancel_order(user_id=%r, order_id=%r, reason=%r)",
            user_id,
            order_id,
            reason,
        )

        # R-001: validate parameters.
        _validate_non_empty_str(user_id, "user_id")
        _validate_non_empty_str(order_id, "order_id")
        _validate_non_empty_str(reason, "reason")

        order = self._load_owned_order(user_id, order_id)

        if order["status"] == "cancelled":
            raise OrderStateError(f"order {order_id} is already cancelled")

        refund_id = self.payment_gateway.refund(order["charge_id"], order["total"])

        order["status"] = "cancelled"
        order["cancellation_reason"] = reason
        order["refund_id"] = refund_id
        self.repository.save(order)

        return {
            "order_id": order_id,
            "user_id": user_id,
            "status": "cancelled",
            "reason": reason,
            "refund_id": refund_id,
            "refund_amount": order["total"],
        }

    def get_order_status(self, user_id: str, order_id: str) -> Dict[str, Any]:
        """Retrieve order details and return a status dict."""
        logger.debug(
            "Entering get_order_status(user_id=%r, order_id=%r)",
            user_id,
            order_id,
        )

        # R-001: validate parameters.
        _validate_non_empty_str(user_id, "user_id")
        _validate_non_empty_str(order_id, "order_id")

        order = self._load_owned_order(user_id, order_id)

        return {
            "order_id": order["order_id"],
            "user_id": order["user_id"],
            "status": order["status"],
            "total": order["total"],
            "currency": order["currency"],
        }

    def _load_owned_order(self, user_id: str, order_id: str) -> Dict[str, Any]:
        """Fetch an order and confirm it belongs to ``user_id``."""
        order = self.repository.get(order_id)
        if order is None or order["user_id"] != user_id:
            raise OrderNotFoundError(
                f"order {order_id} not found for user {user_id}"
            )
        return order


# --------------------------------------------------------------------------- #
# Module-level convenience API
# --------------------------------------------------------------------------- #
_default_service: OrderService = OrderService()


def create_order(

"""Unit tests for order_service."""

import unittest

from order_service import (
    OrderNotFoundError,
    OrderService,
    OrderStateError,
    PaymentError,
    ServiceConfig,
    ValidationError,
)


def _items():
    return [
        {"name": "Widget", "quantity": 2, "unit_price": 9.99},
        {"name": "Gadget", "quantity": 1, "unit_price": 4.50},
    ]


class CreateOrderTests(unittest.TestCase):
    def setUp(self):
        self.service = OrderService(config=ServiceConfig())

    def test_create_order_success(self):
        result = self.service.create_order("user-1", _items(), "tok_abc")
        self.assertEqual(result["status"], "confirmed")
        self.assertEqual(result["user_id"], "user-1")
        self.assertEqual(result["total"], 24.48)
        self.assertEqual(result["currency"], "USD")
        self.assertTrue(result["order_id"].startswith("ord_"))

    def test_create_order_unique_ids(self):
        a = self.service.create_order("user-1", _items(), "tok")
        b = self.service.create_order("user-1", _items(), "tok")
        self.assertNotEqual(a["order_id"], b["order_id"])

    def test_create_order_rejects_non_string_user_id(self):
        with self.assertRaises(ValidationError):
            self.service.create_order(123, _items(), "tok")

    def test_create_order_rejects_empty_payment_token(self):
        with self.assertRaises(ValidationError):
            self.service.create_order("user-1", _items(), "   ")

    def test_create_order_rejects_empty_items(self):
        with self.assertRaises(ValidationError):
            self.service.create_order("user-1", [], "tok")

    def test_create_order_rejects_non_list_items(self):
        with self.assertRaises(ValidationError):
            self.service.create_order("user-1", "not-a-list", "tok")

    def test_create_order_rejects_bad_quantity(self):
        items = [{"name": "X", "quantity": 0, "unit_price": 1.0}]
        with self.assertRaises(ValidationError):
            self.service.create_order("user-1", items, "tok")

    def test_create_order_rejects_negative_price(self):
        items = [{"name": "X", "quantity": 1, "unit_price": -1.0}]
        with self.assertRaises(ValidationError):
            self.service.create_order("user-1", items, "tok")

    def test_create_order_respects_max_items(self):
        service = OrderService(config=ServiceConfig(max_items_per_order=1))
        with self.assertRaises(ValidationError):
            service.create_order("user-1", _items(), "tok")


class CancelOrderTests(unittest.TestCase):
    def setUp(self):
        self.service = OrderService(config=ServiceConfig())
        self.order = self.service.create_order("user-1", _items(), "tok")

    def test_cancel_order_success(self):
        result = self.service.cancel_order(
            "user-1", self.order["order_id"], "changed mind"
        )
        self.assertEqual(result["status"], "cancelled")
        self.assertEqual(result["reason"], "changed mind")
        self.assertEqual(result["refund_amount"], self.order["total"])
        self.assertTrue(result["refund_id"].startswith("rfnd_"))

    def test_cancel_order_unknown_id(self):
        with self.assertRaises(OrderNotFoundError):
            self.service.cancel_order("user-1", "ord_missing", "reason")

    def test_cancel_order_wrong_user(self):
        with self.assertRaises(OrderNotFoundError):
            self.service.cancel_order("user-2", self.order["order_id"], "reason")

    def test_cancel_order_already_cancelled(self):
        self.service.cancel_order("user-1", self.order["order_id"], "reason")
        with self.assertRaises(OrderStateError):
            self.service.cancel_order("user-1", self.order["order_id"], "again")

    def test_cancel_order_requires_reason(self):
        with self.assertRaises(ValidationError):
            self.service.cancel_order("user-1", self.order["order_id"], "")


class GetOrderStatusTests(unittest.TestCase):
    def setUp(self):
        self.service = OrderService(config=ServiceConfig())
        self.order = self.service.create_order("user-1", _items(), "tok")

    def test_get_order_status_success(self):
        result = self.service.get_order_status("user-1", self.order["order_id"])
        self.assertEqual(result["status"], "confirmed")
        self.assertEqual(result["order_id"], self.order["order_id"])
        self.assertEqual(result["total"], self.order["total"])

    def test_get_order_status_reflects_cancellation(self):
        self.service.cancel_order("user-1", self.order["order_id"], "reason")
        result = self.service.get_order_status("user-1", self.order["order_id"])
        self.assertEqual(result["status"], "cancelled")

    def test_get_order_status_unknown_id(self):
        with self.assertRaises(OrderNotFoundError):
            self.service.get_order_status("user-1", "ord_missing")

    def test_get_order_status_validates_input(self):
        with self.assertRaises(ValidationError):
            self.service.get_order_status("", self.order["order_id"])


class PaymentGatewayTests(unittest.TestCase):
    def test_refund_requires_charge_id(self):
        from order_service import PaymentGateway

        gateway = PaymentGateway(ServiceConfig())
        with self.assertRaises(PaymentError):
            gateway.refund("", 10.0)
