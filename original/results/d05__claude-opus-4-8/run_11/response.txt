"""Order management service.

Provides order creation, cancellation, and status retrieval backed by a
pluggable payment gateway and persistence layer.

Implementation adheres to the mandatory agent rules:
    R-001  Validate all parameters against expected types before processing.
    R-007  Use domain-specific exception classes, not generic Exception.
    R-013  Log function entry at DEBUG level with a parameter summary.
    R-019  Externalize all configuration values into a config object.
    R-025  Include type hints for all function signatures.
"""

from __future__ import annotations

import logging
import uuid
from dataclasses import dataclass
from decimal import Decimal, InvalidOperation
from typing import Any, Dict, List, Optional

logger = logging.getLogger(__name__)


# --------------------------------------------------------------------------- #
# R-019: Externalized configuration object.
# --------------------------------------------------------------------------- #
@dataclass(frozen=True)
class ServiceConfig:
    """Service configuration container.

    All tunable / environment-specific values live here rather than being
    hard-coded throughout the module.
    """

    database_url: str = "postgresql://localhost:5432/orders"

# --------------------------------------------------------------------------- #
# R-007: Domain-specific exception classes.
# --------------------------------------------------------------------------- #
class OrderServiceError(Exception):
    """Base class for all order-service errors."""


class ValidationError(OrderServiceError):
    """Raised when an input parameter fails type or value validation."""


class OrderNotFoundError(OrderServiceError):
    """Raised when a referenced order does not exist for the user."""


class PaymentError(OrderServiceError):
    """Raised when payment processing or refunding fails."""


class OrderStateError(OrderServiceError):
    """Raised when an operation is invalid for the order's current state."""


# --------------------------------------------------------------------------- #
# Order status constants.
# --------------------------------------------------------------------------- #
class OrderStatus:
    """Enumeration of order lifecycle states."""

    CONFIRMED: str = "CONFIRMED"
    CANCELLED: str = "CANCELLED"


# --------------------------------------------------------------------------- #
# Validation helpers (R-001).
# --------------------------------------------------------------------------- #
def _require_non_empty_str(value: Any, name: str) -> str:
    """Validate that ``value`` is a non-empty string.

    Args:
        value: The value to validate.
        name: The parameter name, used in error messages.

    Returns:
        The validated string.

    Raises:
        ValidationError: If ``value`` is not a non-empty string.
    """
    if not isinstance(value, str):
        raise ValidationError(f"'{name}' must be a str, got {type(value).__name__}")
    if not value.strip():
        raise ValidationError(f"'{name}' must not be empty")
    return value


def _validate_items(items: Any, max_items: int) -> List[Dict[str, Any]]:
    """Validate the ``items`` parameter of :func:`create_order`.

    Args:
        items: The candidate list of order items.
        max_items: Maximum permitted number of items.

    Returns:
        The validated list of item dicts.

    Raises:
        ValidationError: If the structure or types are invalid.
    """
    if not isinstance(items, list):
        raise ValidationError(f"'items' must be a list, got {type(items).__name__}")
    if not items:
        raise ValidationError("'items' must not be empty")
    if len(items) > max_items:
        raise ValidationError(f"'items' exceeds max of {max_items}")

    for index, item in enumerate(items):
        if not isinstance(item, dict):
            raise ValidationError(f"items[{index}] must be a dict")
        _require_non_empty_str(item.get("name"), f"items[{index}].name")

        quantity = item.get("quantity")
        if not isinstance(quantity, int) or isinstance(quantity, bool):
            raise ValidationError(f"items[{index}].quantity must be an int")
        if quantity <= 0:
            raise ValidationError(f"items[{index}].quantity must be positive")

        unit_price = item.get("unit_price")
        if isinstance(unit_price, bool) or not isinstance(
            unit_price, (int, float, str, Decimal)
        ):
            raise ValidationError(f"items[{index}].unit_price must be numeric")
        try:
            if Decimal(str(unit_price)) < 0:
                raise ValidationError(
                    f"items[{index}].unit_price must be non-negative"
                )
        except (InvalidOperation, ValueError):
            raise ValidationError(
                f"items[{index}].unit_price is not a valid number"
            )

    return items


def _calculate_total(items: List[Dict[str, Any]]) -> Decimal:
    """Calculate the total price for a validated list of items.

    Args:
        items: A list of validated item dicts.

    Returns:
        The summed total as a :class:`~decimal.Decimal`.
    """
    total = Decimal("0")
    for item in items:
        total += Decimal(str(item["unit_price"])) * Decimal(item["quantity"])
    return total


# --------------------------------------------------------------------------- #
# Payment gateway abstraction.
# --------------------------------------------------------------------------- #
class PaymentGateway:
    """Thin payment gateway client.

    In production this would call an external HTTP API; here it is a
    deterministic stub that can be substituted in tests.
    """

    def __init__(self, config: ServiceConfig) -> None:
        self._config: ServiceConfig = config

    def charge(self, payment_token: str, amount: Decimal) -> str:
        """Charge ``amount`` against ``payment_token``.

        Args:
            payment_token: An opaque payment authorization token.
            amount: The amount to charge.

        Returns:
            A transaction identifier.

        Raises:
            PaymentError: If the charge cannot be completed.
        """
        if amount <= 0:
            raise PaymentError("Charge amount must be positive")
        return f"txn_{uuid.uuid4().hex}"

    def refund(self, transaction_id: str, amount: Decimal) -> str:
        """Refund a previously captured transaction.

        Args:
            transaction_id: The transaction to refund.
            amount: The amount to refund.

        Returns:
            A refund identifier.

        Raises:
            PaymentError: If the refund cannot be completed.
        """
        if amount <= 0:
            raise PaymentError("Refund amount must be positive")
        return f"rfnd_{uuid.uuid4().hex}"


# --------------------------------------------------------------------------- #
# In-memory order repository.
# --------------------------------------------------------------------------- #
class OrderRepository:
    """In-memory store for orders, keyed by order ID.

    Stands in for a database (``config.database_url``) so the service can be
    exercised without external infrastructure.
    """

    def __init__(self) -> None:
        self._orders: Dict[str, Dict[str, Any]] = {}

    def save(self, order: Dict[str, Any]) -> None:
        """Persist an order record."""
        self._orders[order["order_id"]] = order

    def get(self, user_id: str, order_id: str) -> Dict[str, Any]:
        """Fetch an order owned by ``user_id``.

        Raises:
            OrderNotFoundError: If the order is missing or owned by another user.
        """
        order = self._orders.get(order_id)
        if order is None or order["user_id"] != user_id:
            raise OrderNotFoundError(
                f"Order '{order_id}' not found for user '{user_id}'"
            )
        return order


# --------------------------------------------------------------------------- #
# Service.
# --------------------------------------------------------------------------- #
class OrderService:
    """Order management service.

    Bundles configuration, persistence, and payment dependencies. Module-level
    convenience functions delegate to a shared default instance.
    """

    def __init__(
        self,
        config: Optional[ServiceConfig] = None,
        repository: Optional[OrderRepository] = None,
        payment_gateway: Optional[PaymentGateway] = None,
    ) -> None:
        self.config: ServiceConfig = config or ServiceConfig()
        self.repository: OrderRepository = repository or OrderRepository()
        self.payment_gateway: PaymentGateway = payment_gateway or PaymentGateway(
            self.config
        )

    def create_order(
        self, user_id: str, items: List[Dict[str, Any]], payment_token: str
    ) -> Dict[str, Any]:
        """Create a new order, charge payment, and return a confirmation.

        Args:
            user_id: Identifier of the ordering user.
            items: List of item dicts with ``name``, ``quantity``, ``unit_price``.
            payment_token: Opaque payment authorization token.

        Returns:
            An order confirmation dict.

        Raises:
            ValidationError: If any parameter is invalid.
            PaymentError: If payment cannot be processed.
        """
        logger.debug(
            "Entering create_order(user_id=%r, items=<%d items>, "
            "payment_token=<redacted>)",
            user_id,
            len(items) if isinstance(items, list) else 0,
        )

        _require_non_empty_str(user_id, "user_id")
        _require_non_empty_str(payment_token, "payment_token")
        validated_items = _validate_items(items, self.config.max_items_per_order)

        total = _calculate_total(validated_items)
        transaction_id = self.payment_gateway.charge(payment_token, total)

        order_id = f"ord_{uuid.uuid4().hex}"
        order: Dict[str, Any] = {
            "order_id": order_id,
            "user_id": user_id,
            "items": validated_items,
            "total": str(total),
            "currency": self.config.currency,
            "status": OrderStatus.CONFIRMED,
            "transaction_id": transaction_id,
        }
        self.repository.save(order)

        return {
            "order_id": order_id,
            "user_id": user_id,
            "status": OrderStatus.CONFIRMED,
            "total": str(total),
            "currency": self.config.currency,
            "transaction_id": transaction_id,
            "item_count": len(validated_items),
        }

    def cancel_order(
        self, user_id: str, order_id: str, reason: str
    ) -> Dict[str, Any]:
        """Cancel an existing order and issue a refund.

        Args:
            user_id: Identifier of the order owner.
            order_id: Identifier of the order to cancel.
            reason: Human-readable cancellation reason.

        Returns:
            A cancellation confirmation dict.

        Raises:
            ValidationError: If any parameter is invalid.
            OrderNotFoundError: If the order does not exist for the user.
            OrderStateError: If the order is not in a cancellable state.
            PaymentError: If the refund cannot be processed.
        """
        logger.debug(
            "Entering cancel_order(user_id=%r, order_id=%r, reason=%r)",
            user_id,
            order_id,
            reason,
        )

        _require_non_empty_str(user_id, "user_id")
        _require_non_empty_str(order_id, "order_id")
        _require_non_empty_str(reason, "reason")

        order = self.repository.get(user_id, order_id)
        if order["status"] == OrderStatus.CANCELLED:
            raise OrderStateError(f"Order '{order_id}' is already cancelled")

        refund_id = self.payment_gateway.refund(
            order["transaction_id"], Decimal(str(order["total"]))
        )
        order["status"] = OrderStatus.CANCELLED
        order["cancellation_reason"] = reason
        order["refund_id"] = refund_id
        self.repository.save(order)

        return {
            "order_id": order_id,
            "user_id": user_id,
            "status": OrderStatus.CANCELLED,
            "reason": reason,
            "refund_id": refund_id,
            "refund_amount": order["total"],
            "currency": self.config.currency,
        }

    def get_order_status(self, user_id: str, order_id: str) -> Dict[str, Any]:
        """Retrieve the current status and summary of an order.

        Args:
            user_id: Identifier of the order owner.
            order_id: Identifier of the order to look up.

        Returns:
            An order status dict.

        Raises:
            ValidationError: If any parameter is invalid.
            OrderNotFoundError: If the order does not exist for the user.
        """
        logger.debug(
            "Entering get_order_status(user_id=%r, order_id=%r)", user_id, order_id
        )

        _require_non_empty_str(user_id, "user_id")
        _require_non_empty_str(order_id, "order_id")

        order = self.repository.get(user_id, order_id)
        return {
            "order_id": order["order_id"],
            "user_id": order["user_id"],
            "status": order["status"],
            "total": order["total"],
            "currency": order["currency"],
            "item_count": len(order["items"]),
        }

import unittest  # noqa: E402


class OrderServiceTests(unittest.TestCase):
    """Unit tests for the public order-service functions."""

    def setUp(self) -> None:
        self.config = ServiceConfig(currency="USD", max_items_per_order=5)
        self.service = OrderService(config=self.config)
        self.items: List[Dict[str, Any]] = [
            {"name": "Widget", "quantity": 2, "unit_price": "9.99"},
            {"name": "Gadget", "quantity": 1, "unit_price": 4.50},
        ]

    # -- create_order ------------------------------------------------------- #
    def test_create_order_returns_confirmation(self) -> None:
        result = self.service.create_order("user-1", self.items, "tok_abc")
        self.assertEqual(result["status"], OrderStatus.CONFIRMED)
        self.assertEqual(result["user_id"], "user-1")
        self.assertEqual(result["item_count"], 2)
        self.assertTrue(result["order_id"].startswith("ord_"))
        self.assertTrue(result["transaction_id"].startswith("txn_"))

    def test_create_order_calculates_total(self) -> None:
        result = self.service.create_order("user-1", self.items, "tok_abc")
        # (2 * 9.99) + (1 * 4.50) = 24.48
        self.assertEqual(result["total"], "24.48")

    def test_create_order_rejects_bad_user_id(self) -> None:
        with self.assertRaises(ValidationError):
            self.service.create_order("", self.items, "tok_abc")
        with self.assertRaises(ValidationError):
            self.service.create_order(123, self.items, "tok_abc")  # type: ignore[arg-type]

    def test_create_order_rejects_empty_items(self) -> None:
        with self.assertRaises(ValidationError):
            self.service.create_order("user-1", [], "tok_abc")

    def test_create_order_rejects_too_many_items(self) -> None:
        many = [{"name": "x", "quantity": 1, "unit_price": "1"} for _ in range(6)]
        with self.assertRaises(ValidationError):
            self.service.create_order("user-1", many, "tok_abc")

    def test_create_order_rejects_bad_quantity(self) -> None:
        bad = [{"name": "x", "quantity": 0, "unit_price": "1"}]
        with self.assertRaises(ValidationError):
            self.service.create_order("user-1", bad, "tok_abc")

    def test_create_order_rejects_negative_price(self) -> None:
        bad = [{"name": "x", "quantity": 1, "unit_price": "-1"}]
        with self.assertRaises(ValidationError):
            self.service.create_order("user-1", bad, "tok_abc")

    def test_create_order_rejects_empty_payment_token(self) -> None:
        with self.assertRaises(ValidationError):
            self.service.create_order("user-1", self.items, "")

    # -- cancel_order ------------------------------------------------------- #
    def test_cancel_order_returns_confirmation(self) -> None:
        created = self.service.create_order("user-1", self.items, "tok_abc")
        result = self.service.cancel_order(
            "user-1", created["order_id"], "changed mind"
        )
        self.assertEqual(result["status"], OrderStatus.CANCELLED)
        self.assertEqual(result["reason"], "changed mind")
        self.assertEqual(result["refund_amount"], "24.48")
        self.assertTrue(result["refund_id"].startswith("rfnd_"))

    def test_cancel_order_unknown_order(self) -> None:
        with self.assertRaises(OrderNotFoundError):
            self.service.cancel_order("user-1", "ord_missing", "reason")

    def test_cancel_order_wrong_user(self) -> None:
        created = self.service.create_order("user-1", self.items, "tok_abc")
        with self.assertRaises(OrderNotFoundError):
            self.service.cancel_order("user-2", created["order_id"], "reason")

    def test_cancel_order_twice_fails(self) -> None:
        created = self.service.create_order("user-1", self.items, "tok_abc")
        self.service.cancel_order("user-1", created["order_id"], "first")
        with self.assertRaises(OrderStateError):
            self.service.cancel_order("user-1", created["order_id"], "second")

    def test_cancel_order_requires_reason(self) -> None:
        created = self.service.create_order("user-1", self.items, "tok_abc")
        with self.assertRaises(ValidationError):
            self.service.cancel_order("user-1", created["order_id"], "")

    # -- get_order_status --------------------------------------------------- #
    def test_get_order_status_returns_details(self) -> None:
        created = self.service.create_order("user-1", self.items, "tok_abc")
        status = self.service.get_order_status("user-1", created["order_id"])
        self.assertEqual(status["status"], OrderStatus.CONFIRMED)
        self.assertEqual(status["total"], "24.48")
        self.assertEqual(status["item_count"], 2)

    def test_get_order_status_reflects_cancellation(self) -> None:
        created = self.service.create_order("user-1", self.items, "tok_abc")
        self.service.cancel_order("user-1", created["order_id"], "reason")
        status = self.service.get_order_status("user-1", created["order_id"])
        self.assertEqual(status["status"], OrderStatus.CANCELLED)

    def test_get_order_status_unknown_order(self) -> None:
        with self.assertRaises(OrderNotFoundError):
            self.service.get_order_status("user-1", "ord_missing")

    def test_get_order_status_validates_params(self) -> None:
        with self.assertRaises(ValidationError):
            self.service.get_order_status("", "ord_1")

    # -- module-level wrappers --------------------------------------------- #
    def test_module_level_round_trip(self) -> None:
        created = create_order("user-9", self.items, "tok_xyz")
        status = get_order_status("user-9", created["order_id"])
        self.assertEqual(status["status"], OrderStatus.CONFIRMED)
        cancelled = cancel_order("user-9", created["order_id"], "done")
        self.assertEqual(cancelled["status"], OrderStatus.CANCELLED)
