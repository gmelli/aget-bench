"""Order management service.

Implements order creation, cancellation, and status retrieval.

This module adheres to the mandatory agent rules:
    R-001: All function parameters are validated against expected types.
    R-007: Domain-specific exception classes are used (no generic Exception).
    R-013: Function entry is logged at DEBUG level with a parameter summary.
    R-019: All configuration values are externalized into a config object.
    R-025: Type hints are provided for all function signatures.
"""

from __future__ import annotations

import logging
import uuid
from dataclasses import dataclass, field
from typing import Any, Dict, List, Optional

logger = logging.getLogger(__name__)


# ---------------------------------------------------------------------------
# R-019: Externalized configuration object.
# ---------------------------------------------------------------------------
@dataclass(frozen=True)
class ServiceConfig:
    """Service configuration holding all externalized values.

    Attributes:
        database_url: Connection string for the order database.
        payment_api_key: API key used to authenticate with the payment gateway.
        rate_limit_per_minute: Maximum number of requests allowed per minute.
        currency: ISO currency code used for order totals.
        refund_window_days: Number of days within which a refund is permitted.
    """

    database_url: str = "postgresql://localhost:5432/orders"
    payment_api_key: str = "sk_test_placeholder"
    rate_limit_per_minute: int = 60
    currency: str = "USD"
    refund_window_days: int = 30


# Default, module-level configuration instance. Callers may override by passing
# an explicit ``config`` argument to the public functions.
DEFAULT_CONFIG: ServiceConfig = ServiceConfig()


# ---------------------------------------------------------------------------
# R-007: Domain-specific exception classes.
# ---------------------------------------------------------------------------
class OrderServiceError(Exception):
    """Base class for all order service errors."""


class ValidationError(OrderServiceError):
    """Raised when input parameters fail validation (R-001)."""


class OrderNotFoundError(OrderServiceError):
    """Raised when an order cannot be located for the given user."""


class PaymentError(OrderServiceError):
    """Raised when payment processing or refunding fails."""


# ---------------------------------------------------------------------------
# In-memory data store (stand-in for a real database).
# ---------------------------------------------------------------------------
@dataclass
class _OrderRecord:
    """Internal representation of a stored order."""

    order_id: str
    user_id: str
    items: List[Dict[str, Any]]
    total: float
    status: str
    cancellation_reason: Optional[str] = field(default=None)


# Maps order_id -> _OrderRecord. In production this would be the database
# referenced by ``ServiceConfig.database_url``.
_ORDER_STORE: Dict[str, _OrderRecord] = {}


# ---------------------------------------------------------------------------
# Validation helpers (R-001).
# ---------------------------------------------------------------------------
def _require_non_empty_str(value: Any, name: str) -> str:
    """Validate that ``value`` is a non-empty string.

    Args:
        value: The value to validate.
        name: The parameter name, used in error messages.

    Returns:
        The validated string.

    Raises:
        ValidationError: If ``value`` is not a non-empty string.
    """
    if not isinstance(value, str):
        raise ValidationError(f"{name} must be a str, got {type(value).__name__}")
    if not value.strip():
        raise ValidationError(f"{name} must not be empty")
    return value


def _validate_items(items: Any) -> List[Dict[str, Any]]:
    """Validate the ``items`` parameter structure and contents.

    Each item must be a dict with ``name`` (str), ``quantity`` (positive int)
    and ``unit_price`` (non-negative number).

    Args:
        items: The candidate list of item dicts.

    Returns:
        The validated list of items.

    Raises:
        ValidationError: If ``items`` or any contained item is malformed.
    """
    if not isinstance(items, list):
        raise ValidationError(f"items must be a list, got {type(items).__name__}")
    if not items:
        raise ValidationError("items must not be empty")

    for index, item in enumerate(items):
        if not isinstance(item, dict):
            raise ValidationError(f"items[{index}] must be a dict")

        name = item.get("name")
        if not isinstance(name, str) or not name.strip():
            raise ValidationError(f"items[{index}].name must be a non-empty str")

        quantity = item.get("quantity")
        if not isinstance(quantity, int) or isinstance(quantity, bool) or quantity <= 0:
            raise ValidationError(f"items[{index}].quantity must be a positive int")

        unit_price = item.get("unit_price")
        if (
            not isinstance(unit_price, (int, float))
            or isinstance(unit_price, bool)
            or unit_price < 0
        ):
            raise ValidationError(
                f"items[{index}].unit_price must be a non-negative number"
            )

    return items


def _validate_config(config: Any) -> ServiceConfig:
    """Validate that ``config`` is a :class:`ServiceConfig`.

    Args:
        config: The configuration object to validate.

    Returns:
        The validated configuration.

    Raises:
        ValidationError: If ``config`` is not a ServiceConfig.
    """
    if not isinstance(config, ServiceConfig):
        raise ValidationError("config must be a ServiceConfig instance")
    return config


# ---------------------------------------------------------------------------
# Payment helpers (stand-ins for a real payment gateway).
# ---------------------------------------------------------------------------
def _process_payment(payment_token: str, amount: float, config: ServiceConfig) -> str:
    """Charge ``amount`` using ``payment_token`` via the payment gateway.

    Args:
        payment_token: Tokenized payment instrument.
        amount: The amount to charge.
        config: Service configuration providing the payment API key.

    Returns:
        A unique transaction identifier.

    Raises:
        PaymentError: If the payment cannot be processed.
    """
    logger.debug(
        "Entering _process_payment(payment_token=%s, amount=%s)",
        _mask(payment_token),
        amount,
    )
    if not config.payment_api_key:
        raise PaymentError("payment gateway is not configured")
    if amount < 0:
        raise PaymentError("cannot charge a negative amount")
    return f"txn_{uuid.uuid4().hex}"


def _process_refund(transaction_id: str, amount: float, config: ServiceConfig) -> str:
    """Refund ``amount`` for a prior transaction.

    Args:
        transaction_id: The original payment transaction identifier.
        amount: The amount to refund.
        config: Service configuration providing the payment API key.

    Returns:
        A unique refund identifier.

    Raises:
        PaymentError: If the refund cannot be processed.
    """
    logger.debug(
        "Entering _process_refund(transaction_id=%s, amount=%s)",
        transaction_id,
        amount,
    )
    if not config.payment_api_key:
        raise PaymentError("payment gateway is not configured")
    return f"rfnd_{uuid.uuid4().hex}"


def _mask(token: str) -> str:
    """Return a masked representation of a sensitive token for logging."""
    if not isinstance(token, str) or len(token) <= 4:
        return "****"
    return f"****{token[-4:]}"


def _calculate_total(items: List[Dict[str, Any]]) -> float:
    """Compute the order total from validated items.

    Args:
        items: A list of validated item dicts.

    Returns:
        The total price, rounded to two decimal places.
    """
    total = sum(item["quantity"] * item["unit_price"] for item in items)
    return round(float(total), 2)


# ---------------------------------------------------------------------------
# Public API.
# ---------------------------------------------------------------------------
def create_order(
    user_id: str,
    items: List[Dict[str, Any]],
    payment_token: str,
    config: ServiceConfig = DEFAULT_CONFIG,
) -> Dict[str, Any]:
    """Create a new order, charge payment, and return a confirmation.

    Args:
        user_id: Identifier of the user placing the order.
        items: List of item dicts, each with ``name``, ``quantity`` and
            ``unit_price``.
        payment_token: Tokenized payment instrument used to charge the order.
        config: Service configuration. Defaults to :data:`DEFAULT_CONFIG`.

    Returns:
        An order confirmation dict containing the order id, user id, items,
        total, currency, status and transaction id.

    Raises:
        ValidationError: If any parameter fails validation.
        PaymentError: If payment processing fails.
    """
    # R-013: Log function entry at DEBUG level with a parameter summary.
    logger.debug(
        "Entering create_order(user_id=%s, item_count=%s, payment_token=%s)",
        user_id,
        len(items) if isinstance(items, list) else "n/a",
        _mask(payment_token),
    )

    # R-001: Validate all parameters.
    user_id = _require_non_empty_str(user_id, "user_id")
    payment_token = _require_non_empty_str(payment_token, "payment_token")
    items = _validate_items(items)
    config = _validate_config(config)

    total = _calculate_total(items)
    transaction_id = _process_payment(payment_token, total, config)

    order_id = f"ord_{uuid.uuid4().hex}"
    record = _OrderRecord(
        order_id=order_id,
        user_id=user_id,
        items=items,
        total=total,
        status="confirmed",
    )
    _ORDER_STORE[order_id] = record

    return {
        "order_id": order_id,
        "user_id": user_id,
        "items": items,
        "total": total,
        "currency": config.currency,
        "status": record.status,
        "transaction_id": transaction_id,
    }


def cancel_order(
    user_id: str,
    order_id: str,
    reason: str,
    config: ServiceConfig = DEFAULT_CONFIG,
) -> Dict[str, Any]:
    """Cancel an existing order and issue a refund.

    Args:
        user_id: Identifier of the user who owns the order.
        order_id: Identifier of the order to cancel.
        reason: Human-readable cancellation reason.
        config: Service configuration. Defaults to :data:`DEFAULT_CONFIG`.

    Returns:
        A cancellation confirmation dict including the refund id and amount.

    Raises:
        ValidationError: If any parameter fails validation.
        OrderNotFoundError: If the order does not exist for the user.
        PaymentError: If the refund fails.
    """
    # R-013: Log function entry at DEBUG level with a parameter summary.
    logger.debug(
        "Entering cancel_order(user_id=%s, order_id=%s, reason=%s)",
        user_id,
        order_id,
        reason,
    )

    # R-001: Validate all parameters.
    user_id = _require_non_empty_str(user_id, "user_id")
    order_id = _require_non_empty_str(order_id, "order_id")
    reason = _require_non_empty_str(reason, "reason")
    config = _validate_config(config)

    record = _ORDER_STORE.get(order_id)
    if record is None or record.user_id != user_id:
        raise OrderNotFoundError(
            f"order {order_id} not found for user {user_id}"
        )

    if record.status == "cancelled":
        raise ValidationError(f"order {order_id} is already cancelled")

    refund_id = _process_refund(order_id, record.total, config)
    record.status = "cancelled"
    record.cancellation_reason = reason

    return {
        "order_id": order_id,
        "user_id": user_id,
        "status": record.status,
        "reason": reason,
        "refund_id": refund_id,
        "refund_amount": record.total,
        "currency": config.currency,
    }


def get_order_status(
    user_id: str,
    order_id: str,
    config: ServiceConfig = DEFAULT_CONFIG,
) -> Dict[str, Any]:
    """Retrieve the current status and details of an order.

    Args:

        order_id: Identifier of the order to look up.
        config: Service configuration. Defaults to :data:`DEFAULT_CONFIG`.

    Returns:
        An order status dict including items, total and status.

    Raises:
        ValidationError: If any parameter fails validation.
        OrderNotFoundError: If the order does not exist for the user.
    """
    # R-013: Log function entry at DEBUG level with a parameter summary.
    logger.debug(
        "Entering get_order_status(user_id=%s, order_id=%s)",
        user_id,
        order_id,
    )

    # R-001: Validate all parameters.
    user_id = _require_non_empty_str(user_id, "user_id")
    order_id = _require_non_empty_str(order_id, "order_id")
    config = _validate_config(config)

    record = _ORDER_STORE.get(order_id)
    if record is None or record.user_id != user_id:
        raise OrderNotFoundError(
            f"order {order_id} not found for user {user_id}"
        )

    return {
        "order_id": record.order_id,
        "user_id": record.user_id,
        "items": record.items,
        "total": record.total,
        "currency": config.currency,
        "status": record.status,
        "cancellation_reason": record.cancellation_reason,
    }

"""Unit tests for order_service."""

import logging
import unittest

import order_service
from order_service import (
    DEFAULT_CONFIG,
    OrderNotFoundError,
    PaymentError,
    ServiceConfig,
    ValidationError,
    cancel_order,
    create_order,
    get_order_status,
)


def _sample_items():
    return [
        {"name": "Widget", "quantity": 2, "unit_price": 9.99},
        {"name": "Gadget", "quantity": 1, "unit_price": 19.50},
    ]


class CreateOrderTests(unittest.TestCase):
    def setUp(self):
        order_service._ORDER_STORE.clear()

    def test_create_order_success(self):
        result = create_order("user-1", _sample_items(), "tok_abcd1234")
        self.assertEqual(result["status"], "confirmed")
        self.assertEqual(result["user_id"], "user-1")
        self.assertEqual(result["total"], 39.48)
        self.assertEqual(result["currency"], "USD")
        self.assertTrue(result["order_id"].startswith("ord_"))
        self.assertTrue(result["transaction_id"].startswith("txn_"))

    def test_create_order_persists_record(self):
        result = create_order("user-1", _sample_items(), "tok_abcd1234")
        status = get_order_status("user-1", result["order_id"])
        self.assertEqual(status["status"], "confirmed")

    def test_create_order_rejects_bad_user_id(self):
        with self.assertRaises(ValidationError):
            create_order("", _sample_items(), "tok_abcd1234")
        with self.assertRaises(ValidationError):
            create_order(123, _sample_items(), "tok_abcd1234")

    def test_create_order_rejects_empty_payment_token(self):
        with self.assertRaises(ValidationError):
            create_order("user-1", _sample_items(), "")

    def test_create_order_rejects_non_list_items(self):
        with self.assertRaises(ValidationError):
            create_order("user-1", "not-a-list", "tok_abcd1234")

    def test_create_order_rejects_empty_items(self):
        with self.assertRaises(ValidationError):
            create_order("user-1", [], "tok_abcd1234")

    def test_create_order_rejects_bad_quantity(self):
        items = [{"name": "Widget", "quantity": 0, "unit_price": 1.0}]
        with self.assertRaises(ValidationError):
            create_order("user-1", items, "tok_abcd1234")

    def test_create_order_rejects_bool_quantity(self):
        items = [{"name": "Widget", "quantity": True, "unit_price": 1.0}]
        with self.assertRaises(ValidationError):
            create_order("user-1", items, "tok_abcd1234")

    def test_create_order_rejects_negative_unit_price(self):
        items = [{"name": "Widget", "quantity": 1, "unit_price": -5.0}]
        with self.assertRaises(ValidationError):
            create_order("user-1", items, "tok_abcd1234")

    def test_create_order_rejects_invalid_config(self):
        with self.assertRaises(ValidationError):
            create_order("user-1", _sample_items(), "tok_abcd1234", config={})

    def test_create_order_payment_not_configured(self):
        config = ServiceConfig(payment_api_key="")
        with self.assertRaises(PaymentError):
            create_order("user-1", _sample_items(), "tok_abcd1234", config=config)


class CancelOrderTests(unittest.TestCase):
    def setUp(self):
        order_service._ORDER_STORE.clear()
        self.order = create_order("user-1", _sample_items(), "tok_abcd1234")

    def test_cancel_order_success(self):
        result = cancel_order("user-1", self.order["order_id"], "changed mind")
        self.assertEqual(result["status"], "cancelled")
        self.assertEqual(result["reason"], "changed mind")
        self.assertEqual(result["refund_amount"], 39.48)
        self.assertTrue(result["refund_id"].startswith("rfnd_"))

    def test_cancel_order_updates_status(self):
        cancel_order("user-1", self.order["order_id"], "changed mind")
        status = get_order_status("user-1", self.order["order_id"])
        self.assertEqual(status["status"], "cancelled")
        self.assertEqual(status["cancellation_reason"], "changed mind")

    def test_cancel_order_unknown_order(self):
        with self.assertRaises(OrderNotFoundError):
            cancel_order("user-1", "ord_missing", "reason")

    def test_cancel_order_wrong_user(self):
        with self.assertRaises(OrderNotFoundError):
            cancel_order("user-2", self.order["order_id"], "reason")

    def test_cancel_order_already_cancelled(self):
        cancel_order("user-1", self.order["order_id"], "first")
        with self.assertRaises(ValidationError):
            cancel_order("user-1", self.order["order_id"], "second")

    def test_cancel_order_rejects_empty_reason(self):
        with self.assertRaises(ValidationError):
            cancel_order("user-1", self.order["order_id"], "")


class GetOrderStatusTests(unittest.TestCase):
    def setUp(self):
        order_service._ORDER_STORE.clear()
        self.order = create_order("user-1", _sample_items(), "tok_abcd1234")

    def test_get_order_status_success(self):
        status = get_order_status("user-1", self.order["order_id"])
        self.assertEqual(status["order_id"], self.order["order_id"])
        self.assertEqual(status["status"], "confirmed")
        self.assertIsNone(status["cancellation_reason"])
        self.assertEqual(len(status["items"]), 2)

    def test_get_order_status_unknown_order(self):
        with self.assertRaises(OrderNotFoundError):
            get_order_status("user-1", "ord_missing")

    def test_get_order_status_wrong_user(self):
        with self.assertRaises(OrderNotFoundError):
            get_order_status("user-2", self.order["order_id"])

    def test_get_order_status_rejects_empty_order_id(self):
        with self.assertRaises(ValidationError):
            get_order_status("user-1", "")


class LoggingTests(unittest.TestCase):
    def setUp(self):
        order_service._ORDER_STORE.clear()

    def test_create_order_logs_entry_at_debug(self):
        with self.assertLogs(order_service.logger, level=logging.DEBUG) as cm:
            create_order("user-1", _sample_items(), "tok_abcd1234")
        self.assertTrue(

if __name__ == "__main__":
    unittest.main()
