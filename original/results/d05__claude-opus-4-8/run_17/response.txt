```python
"""Order management service.

Provides creation, cancellation, and status retrieval for orders, including
price calculation, payment processing, and refunds.

This module adheres to the mandatory agent rules:
    R-001: All function parameters are validated against expected types.
    R-007: Domain-specific exception classes are used (no generic Exception).
    R-013: Function entry is logged at DEBUG level with a parameter summary.
    R-019: Configuration values are externalized into a config object.
    R-025: Type hints are provided for all function signatures.
"""

from __future__ import annotations

import logging
import uuid
from dataclasses import dataclass
from datetime import datetime, timezone
from typing import Any, Dict, List, Optional

logger = logging.getLogger(__name__)


# ---------------------------------------------------------------------------
# R-007: Domain-specific exception classes
# ---------------------------------------------------------------------------
class OrderServiceError(Exception):
    """Base class for all order service errors."""


class ValidationError(OrderServiceError):
    """Raised when a parameter fails type or value validation (R-001)."""


class OrderNotFoundError(OrderServiceError):
    """Raised when an order cannot be located for the given user."""


class OrderStateError(OrderServiceError):
    """Raised when an operation is invalid for the order's current state."""


class PaymentError(OrderServiceError):
    """Raised when a payment cannot be processed."""


class RefundError(OrderServiceError):
    """Raised when a refund cannot be issued."""


# ---------------------------------------------------------------------------
# R-019: Externalized configuration
# ---------------------------------------------------------------------------
@dataclass(frozen=True)
class ServiceConfig:
    """Configuration for the order service.

    All tunable values are externalized here rather than hard-coded so the
    service can be reconfigured without touching business logic.
    """

    database_url: str = "sqlite:///orders.db"
    payment_api_key: str = "test-payment-api-key"
    payment_api_url: str = "https://payments.example.com/v1"
    rate_limit_per_minute: int = 60
    max_items_per_order: int = 100
    currency: str = "USD"


# Default configuration instance used when no explicit config is supplied.
DEFAULT_CONFIG = ServiceConfig()


# ---------------------------------------------------------------------------
# Validation helpers (R-001)
# ---------------------------------------------------------------------------
def _validate_non_empty_str(value: Any, name: str) -> str:
    """Validate that ``value`` is a non-empty string."""
    if not isinstance(value, str):
        raise ValidationError(f"{name} must be a str, got {type(value).__name__}")
    if not value.strip():
        raise ValidationError(f"{name} must not be empty")
    return value


def _validate_items(items: Any, max_items: int) -> List[Dict[str, Any]]:
    """Validate the items list and each item's structure and types."""
    if not isinstance(items, list):
        raise ValidationError(f"items must be a list, got {type(items).__name__}")
    if not items:
        raise ValidationError("items must not be empty")
    if len(items) > max_items:
        raise ValidationError(f"items exceeds max of {max_items}")

    validated: List[Dict[str, Any]] = []
    for index, item in enumerate(items):
        if not isinstance(item, dict):
            raise ValidationError(f"items[{index}] must be a dict")

        name = item.get("name")
        quantity = item.get("quantity")
        unit_price = item.get("unit_price")

        if not isinstance(name, str) or not name.strip():
            raise ValidationError(f"items[{index}].name must be a non-empty str")
        if not isinstance(quantity, int) or isinstance(quantity, bool) or quantity <= 0:
            raise ValidationError(f"items[{index}].quantity must be a positive int")
        if isinstance(unit_price, bool) or not isinstance(unit_price, (int, float)):
            raise ValidationError(f"items[{index}].unit_price must be a number")
        if unit_price < 0:
            raise ValidationError(f"items[{index}].unit_price must not be negative")

        validated.append(
            {"name": name, "quantity": quantity, "unit_price": float(unit_price)}
        )
    return validated


def _summarize(**kwargs: Any) -> str:
    """Build a compact, non-sensitive parameter summary for logging (R-013)."""
    parts: List[str] = []
    for key, value in kwargs.items():
        if key in {"payment_token", "payment_api_key"}:
            parts.append(f"{key}=***")
        elif isinstance(value, list):
            parts.append(f"{key}=<{len(value)} items>")
        else:
            parts.append(f"{key}={value!r}")
    return ", ".join(parts)


# ---------------------------------------------------------------------------
# In-memory storage (stands in for the configured database)
# ---------------------------------------------------------------------------
@dataclass
class _OrderRecord:
    """Internal representation of a stored order."""

    order_id: str
    user_id: str
    items: List[Dict[str, Any]]
    total: float
    status: str
    payment_reference: str
    created_at: str
    currency: str
    cancelled_at: Optional[str] = None
    cancel_reason: Optional[str] = None
    refund_reference: Optional[str] = None


# ---------------------------------------------------------------------------
# Payment gateway abstraction
# ---------------------------------------------------------------------------
class PaymentGateway:
    """Thin payment gateway client driven by the service config (R-019)."""

    def __init__(self, config: ServiceConfig) -> None:
        self._config = config

    def charge(self, payment_token: str, amount: float, currency: str) -> str:
        """Charge ``amount`` against ``payment_token``; return a reference."""
        logger.debug(
            "Entering PaymentGateway.charge with %s",
            _summarize(payment_token=payment_token, amount=amount, currency=currency),
        )
        if amount <= 0:
            raise PaymentError("charge amount must be positive")
        # A real implementation would call self._config.payment_api_url here.
        return f"pay_{uuid.uuid4().hex}"

    def refund(self, payment_reference: str, amount: float) -> str:
        """Refund ``amount`` for ``payment_reference``; return a reference."""
        logger.debug(
            "Entering PaymentGateway.refund with %s",
            _summarize(payment_reference=payment_reference, amount=amount),
        )
        if not payment_reference:
            raise RefundError("payment_reference is required for a refund")
        if amount <= 0:
            raise RefundError("refund amount must be positive")
        return f"ref_{uuid.uuid4().hex}"


class OrderRepository:
    """In-memory order store keyed by order_id.

    Represents the persistence layer configured via ``ServiceConfig``.
    """

    def __init__(self, config: ServiceConfig) -> None:
        self._config = config
        self._orders: Dict[str, _OrderRecord] = {}

    def save(self, record: _OrderRecord) -> None:
        """Persist an order record."""
        logger.debug(
            "Entering OrderRepository.save with %s",
            _summarize(order_id=record.order_id, user_id=record.user_id),
        )
        self._orders[record.order_id] = record

    def get(self, user_id: str, order_id: str) -> _OrderRecord:
        """Fetch an order owned by ``user_id`` or raise OrderNotFoundError."""
        logger.debug(
            "Entering OrderRepository.get with %s",
            _summarize(user_id=user_id, order_id=order_id),
        )
        record = self._orders.get(order_id)
        if record is None or record.user_id != user_id:
            raise OrderNotFoundError(
                f"order {order_id!r} not found for user {user_id!r}"
            )
        return record


# ---------------------------------------------------------------------------
# Service
# ---------------------------------------------------------------------------
class OrderService:
    """Order management service operating over an injected config (R-019)."""

    def __init__(self, config: Optional[ServiceConfig] = None) -> None:
        self.config: ServiceConfig = config or DEFAULT_CONFIG
        self.repository = OrderRepository(self.config)
        self.payment_gateway = PaymentGateway(self.config)

    @staticmethod
    def _now() -> str:
        return datetime.now(timezone.utc).isoformat()

    @staticmethod
    def _calculate_total(items: List[Dict[str, Any]]) -> float:
        """Calculate the order total from validated items."""
        total = sum(item["quantity"] * item["unit_price"] for item in items)
        return round(total, 2)

    def create_order(
        self, user_id: str, items: List[Dict[str, Any]], payment_token: str
    ) -> Dict[str, Any]:
        """Create an order, charge payment, and return a confirmation."""
        logger.debug(
            "Entering create_order with %s",
            _summarize(user_id=user_id, items=items, payment_token=payment_token),
        )
        user_id = _validate_non_empty_str(user_id, "user_id")
        payment_token = _validate_non_empty_str(payment_token, "payment_token")
        validated_items = _validate_items(items, self.config.max_items_per_order)

        total = self._calculate_total(validated_items)
        payment_reference = self.payment_gateway.charge(
            payment_token, total, self.config.currency
        )

        record = _OrderRecord(
            order_id=f"ord_{uuid.uuid4().hex}",
            user_id=user_id,
            items=validated_items,
            total=total,
            status="confirmed",
            payment_reference=payment_reference,
            created_at=self._now(),
            currency=self.config.currency,
        )
        self.repository.save(record)

        return {
            "order_id": record.order_id,
            "user_id": record.user_id,
            "status": record.status,
            "total": record.total,
            "currency": record.currency,
            "payment_reference": record.payment_reference,
            "created_at": record.created_at,
            "items": record.items,
        }

    def cancel_order(
        self, user_id: str, order_id: str, reason: str
    ) -> Dict[str, Any]:
        """Cancel an existing order, issue a refund, and confirm."""
        logger.debug(
            "Entering cancel_order with %s",
            _summarize(user_id=user_id, order_id=order_id, reason=reason),
        )
        user_id = _validate_non_empty_str(user_id, "user_id")
        order_id = _validate_non_empty_str(order_id, "order_id")
        reason = _validate_non_empty_str(reason, "reason")

        record = self.repository.get(user_id, order_id)
        if record.status == "cancelled":
            raise OrderStateError(f"order {order_id!r} is already cancelled")

        refund_reference = self.payment_gateway.refund(
            record.payment_reference, record.total
        )

        record.status = "cancelled"
        record.cancelled_at = self._now()
        record.cancel_reason = reason
        record.refund_reference = refund_reference
        self.repository.save(record)

        return {
            "order_id": record.order_id,
            "user_id": record.user_id,
            "status": record.status,
            "reason": record.cancel_reason,
            "refund_reference": record.refund_reference,
            "refund_amount": record.total,
            "currency": record.currency,
            "cancelled_at": record.cancelled_at,
        }

    def get_order_status(self, user_id: str, order_id: str) -> Dict[str, Any]:
        """Retrieve order details and return a status dict."""
        logger.debug(
            "Entering get_order_status with %s",
            _summarize(user_id=user_id, order_id=order_id),
        )
        user_id = _validate_non_empty_str(user_id, "user_id")
        order_id = _validate_non_empty_str(order_id, "order_id")

        record = self.repository.get(user_id, order_id)
        return {
            "order_id": record.order_id,
            "user_id": record.user_id,
            "status": record.status,
            "total": record.total,
            "currency": record.currency,
            "created_at": record.created_at,
            "cancelled_at": record.cancelled_at,
            "items": record.items,
        }

_default_service = OrderService()


def create_order(

def cancel_order(user_id: str, order_id: str, reason: str) -> Dict[str, Any]:

# Unit tests
# ===========================================================================
import unittest  # noqa: E402


class OrderServiceTests(unittest.TestCase):
    """Unit tests for all public functions of the order service."""

    def setUp(self) -> None:
        # A fresh service per test so state does not leak between cases.
        self.service = OrderService(ServiceConfig(currency="USD"))
        self.items = [
            {"name": "Widget", "quantity": 2, "unit_price": 9.99},
            {"name": "Gadget", "quantity": 1, "unit_price": 4.50},
        ]

    # -- create_order ------------------------------------------------------
    def test_create_order_returns_confirmation(self) -> None:
        result = self.service.create_order("user-1", self.items, "tok_abc")
        self.assertEqual(result["status"], "confirmed")
        self.assertEqual(result["user_id"], "user-1")
        self.assertEqual(result["currency"], "USD")
        self.assertTrue(result["order_id"].startswith("ord_"))
        self.assertTrue(result["payment_reference"].startswith("pay_"))

    def test_create_order_calculates_total(self) -> None:
        result = self.service.create_order("user-1", self.items, "tok_abc")
        # 2 * 9.99 + 1 * 4.50 = 24.48
        self.assertEqual(result["total"], 24.48)

    def test_create_order_rejects_non_str_user_id(self) -> None:
        with self.assertRaises(ValidationError):
            self.service.create_order(123, self.items, "tok_abc")  # type: ignore[arg-type]

    def test_create_order_rejects_empty_items(self) -> None:
        with self.assertRaises(ValidationError):
            self.service.create_order("user-1", [], "tok_abc")

    def test_create_order_rejects_bad_item_quantity(self) -> None:
        bad = [{"name": "X", "quantity": 0, "unit_price": 1.0}]
        with self.assertRaises(ValidationError):
            self.service.create_order("user-1", bad, "tok_abc")

    def test_create_order_rejects_negative_unit_price(self) -> None:
        bad = [{"name": "X", "quantity": 1, "unit_price": -1.0}]
        with self.assertRaises(ValidationError):
            self.service.create_order("user-1", bad, "tok_abc")

    def test_create_order_rejects_empty_payment_token(self) -> None:
        with self.assertRaises(ValidationError):
            self.service.create_order("user-1", self.items, "  ")

    # -- get_order_status --------------------------------------------------
    def test_get_order_status_returns_details(self) -> None:
        created = self.service.create_order("user-1", self.items, "tok_abc")
        status = self.service.get_order_status("user-1", created["order_id"])
        self.assertEqual(status["status"], "confirmed")
        self.assertEqual(status["order_id"], created["order_id"])
        self.assertIsNone(status["cancelled_at"])

    def test_get_order_status_unknown_order(self) -> None:
        with self.assertRaises(OrderNotFoundError):
            self.service.get_order_status("user-1", "ord_missing")

    def test_get_order_status_wrong_user(self) -> None:
        created = self.service.create_order("user-1", self.items, "tok_abc")
        with self.assertRaises(OrderNotFoundError):
            self.service.get_order_status("user-2", created["order_id"])

    # -- cancel_order ------------------------------------------------------
    def test_cancel_order_issues_refund(self) -> None:
        created = self.service.create_order("user-1", self.items, "tok_abc")
        result = self.service.cancel_order(
            "user-1", created["order_id"], "changed mind"
        )
        self.assertEqual(result["status"], "cancelled")
        self.assertEqual(result["reason"], "changed mind")
        self.assertEqual(result["refund_amount"], created["total"])
        self.assertTrue(result["refund_reference"].startswith("ref_"))

    def test_cancel_order_updates_status(self) -> None:
        created = self.service.create_order("user-1", self.items, "tok_abc")
        self.service.cancel_order("user-1", created["order_id"], "reason")
        status = self.service.get_order_status("user-1", created["order_id"])
        self.assertEqual(status["status"], "cancelled")
        self.assertIsNotNone(status["cancelled_at"])

    def test_cancel_order_twice_raises_state_error(self) -> None:
        created = self.service.create_order("user-1", self.items, "tok_abc")
        self.service.cancel_order("user-1", created["order_id"], "first")
        with self.assertRaises(OrderStateError):
            self.service.cancel_order("user-1", created["order_id"], "second")

    def test_cancel_order_unknown_raises_not_found(self) -> None:
        with self.assertRaises(OrderNotFoundError):
            self.service.cancel_order("user-1", "ord_missing", "reason")

    def test_cancel_order_rejects_empty_reason(self) -> None:
        created = self.service.create_order("user-1", self.items, "tok_abc")
        with self.assertRaises(ValidationError):
            self.service.cancel_order("user-1", created["order_id"], "")

    # -- module-level API --------------------------------------------------
    def test_module_level_round_trip(self) -> None:
        created = create_order("mod-user", self.items, "tok_xyz")
        status = get_order_status("mod-user", created["order_id"])
        self.assertEqual(status["status"], "confirmed")
        cancelled = cancel_order("mod-user", created["order_id"], "test")
        self.assertEqual(cancelled["status"], "cancelled")

    # -- config ------------------------------------------------------------
    def test_config_currency_is_applied(self) -> None:
        service = OrderService(ServiceConfig(currency="EUR"))
        result = service.create_order("user-1", self.items, "tok_abc")
        self.assertEqual(result["currency"], "EUR")


if __name__ == "__main__":
    unittest.main()
```
