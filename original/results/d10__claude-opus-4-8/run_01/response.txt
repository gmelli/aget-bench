"""Order management service.

Provides functions to create, cancel, and inspect orders. Inputs are validated
against expected types, payments/refunds run through pluggable gateways, and all
configuration is externalized into a dedicated config object.

Mandatory-rule compliance:
  R-001 type validation, R-002 reject negative quantities, R-007 domain
  exceptions, R-008 preserved error context, R-013/R-014 DEBUG entry/exit
  logging, R-019 externalized config, R-020 permission checks, R-025 type
  hints, R-026 docstrings with Args/Returns/Raises.
"""

from __future__ import annotations

import logging
import uuid
from dataclasses import dataclass, field
from typing import Any, Callable, Dict, List, Optional

logger = logging.getLogger(__name__)


# ---------------------------------------------------------------------------
# Configuration (R-019)
# ---------------------------------------------------------------------------
@dataclass
class ServiceConfig:
    """Service configuration container.

    Externalizes all configuration values so no operational constant is
    hard-coded in the business logic.

    Attributes:
        database_url: Connection string for the order datastore.
        payment_api_key: API key used to authenticate with the payment gateway.
        rate_limit_per_minute: Max operations permitted per minute per user.
        currency: ISO 4217 currency code used for monetary amounts.
        max_items_per_order: Max number of line items allowed in one order.
    """

    database_url: str = "sqlite:///orders.db"
    payment_api_key: str = "test-payment-api-key"
    rate_limit_per_minute: int = 60
    currency: str = "USD"
    max_items_per_order: int = 100


# ---------------------------------------------------------------------------
# Domain-specific exceptions (R-007)
# ---------------------------------------------------------------------------
class OrderServiceError(Exception):
    """Base class for all order service errors."""


class ValidationError(OrderServiceError):
    """Raised when a parameter fails type or value validation."""


class PermissionDeniedError(OrderServiceError):
    """Raised when a caller lacks permission for a privileged operation."""


class OrderNotFoundError(OrderServiceError):
    """Raised when a referenced order does not exist."""


class PaymentError(OrderServiceError):
    """Raised when a payment or refund operation fails."""


class OrderStateError(OrderServiceError):
    """Raised when an operation is invalid for the order's current state."""


# ---------------------------------------------------------------------------
# Validation helpers (R-001, R-002)
# ---------------------------------------------------------------------------
def _validate_str(value: Any, name: str) -> str:
    """Validate that a value is a non-empty string.

    Args:
        value: The value to validate.
        name: The parameter name, used in error messages.

    Returns:
        The validated string.

    Raises:
        ValidationError: If the value is not a non-empty string.
    """
    if not isinstance(value, str):
        raise ValidationError(
            f"{name} must be of type str, got {type(value).__name__}"
        )
    if not value.strip():
        raise ValidationError(f"{name} must not be empty")
    return value


def _validate_items(items: Any) -> List[Dict[str, Any]]:
    """Validate a list of order line items.

    Each item must be a mapping with a string ``name``, an integer
    ``quantity`` that is not negative, and a numeric, non-negative
    ``unit_price``.

    Args:
        items: The candidate list of item mappings.

    Returns:
        The validated list of items.

    Raises:
        ValidationError: If items is not a list, is empty, or any item fails
            type or value validation.
    """
    if not isinstance(items, list):
        raise ValidationError(
            f"items must be of type list, got {type(items).__name__}"
        )
    if not items:
        raise ValidationError("items must contain at least one item")

    for index, item in enumerate(items):
        if not isinstance(item, dict):
            raise ValidationError(
                f"items[{index}] must be of type dict, "
                f"got {type(item).__name__}"
            )

        name = item.get("name")
        if not isinstance(name, str) or not name.strip():
            raise ValidationError(
                f"items[{index}].name must be a non-empty str"
            )

        quantity = item.get("quantity")
        if isinstance(quantity, bool) or not isinstance(quantity, int):
            raise ValidationError(
                f"items[{index}].quantity must be of type int, "
                f"got {type(quantity).__name__}"
            )
        if quantity < 0:  # R-002
            raise ValidationError(
                f"items[{index}].quantity must not be negative"
            )

        unit_price = item.get("unit_price")
        if isinstance(unit_price, bool) or not isinstance(
            unit_price, (int, float)
        ):
            raise ValidationError(
                f"items[{index}].unit_price must be a number, "
                f"got {type(unit_price).__name__}"
            )
        if unit_price < 0:
            raise ValidationError(
                f"items[{index}].unit_price must not be negative"
            )

    return items


def _calculate_total(items: List[Dict[str, Any]]) -> float:
    """Calculate the total price of a list of validated items.

    Args:
        items: A list of validated item mappings.

    Returns:
        The total price as a float, rounded to two decimal places.
    """
    total = sum(item["quantity"] * item["unit_price"] for item in items)
    return round(float(total), 2)


# ---------------------------------------------------------------------------
# Permission check (R-020)
# ---------------------------------------------------------------------------
def _verify_permission(
    user_id: str,
    order: Optional[Dict[str, Any]],
    operation: str,
) -> None:
    """Verify that a caller may perform a privileged operation.

    Args:
        user_id: The identifier of the calling user.
        order: The order the operation targets, or None for creation.
        operation: A short label describing the operation, used in errors.

    Raises:
        PermissionDeniedError: If the user does not own the target order.
    """
    if order is not None and order.get("user_id") != user_id:
        raise PermissionDeniedError(
            f"user {user_id!r} is not permitted to {operation} "
            f"order {order.get('order_id')!r}"
        )


# ---------------------------------------------------------------------------
# Default payment / refund adapters
# ---------------------------------------------------------------------------
def _default_payment_processor(
    payment_token: str, amount: float, config: ServiceConfig
) -> str:
    """Process a payment using the configured gateway.

    Default stub that simulates a successful charge and returns a transaction
    id. Replace with a real gateway integration in production.

    Args:
        payment_token: The tokenized payment instrument.
        amount: The amount to charge.
        config: The active service configuration.

    Returns:
        A transaction identifier string.

    Raises:
        PaymentError: If the payment cannot be processed.
    """
    if amount < 0:
        raise PaymentError("cannot charge a negative amount")
    return f"txn_{uuid.uuid4().hex}"


def _default_refund_processor(
    transaction_id: str, amount: float, config: ServiceConfig
) -> str:
    """Issue a refund using the configured gateway.

    Default stub that simulates a successful refund and returns a refund id.

    Args:
        transaction_id: The original transaction identifier to refund.
        amount: The amount to refund.
        config: The active service configuration.

    Returns:
        A refund identifier string.

    Raises:
        PaymentError: If the refund cannot be processed.
    """
    if amount < 0:
        raise PaymentError("cannot refund a negative amount")
    return f"rfnd_{uuid.uuid4().hex}"


# ---------------------------------------------------------------------------
# Service
# ---------------------------------------------------------------------------
@dataclass
class OrderService:
    """In-memory order management service.

    Stores orders in memory and delegates payment/refund handling to injectable
    callables, keeping the business logic testable and side-effect free by
    default.

    Attributes:
        config: The active service configuration.
        payment_processor: Callable charging a token for an amount.
        refund_processor: Callable refunding a transaction for an amount.
    """

    config: ServiceConfig = field(default_factory=ServiceConfig)
    payment_processor: Callable[[str, float, ServiceConfig], str] = (
        _default_payment_processor
    )
    refund_processor: Callable[[str, float, ServiceConfig], str] = (
        _default_refund_processor
    )
    _orders: Dict[str, Dict[str, Any]] = field(default_factory=dict)

    def create_order(
        self, user_id: str, items: List[Dict[str, Any]], payment_token: str
    ) -> Dict[str, Any]:
        """Create a new order, charge payment, and return a confirmation.

        Args:
            user_id: The identifier of the user placing the order.
            items: A list of line items; each is a mapping with a string
                ``name``, an integer ``quantity`` (non-negative), and a numeric
                ``unit_price`` (non-negative).
            payment_token: A tokenized payment instrument used for the charge.

        Returns:
            An order confirmation mapping with order id, user id, items, total,
            currency, status, and transaction id.

        Raises:
            ValidationError: If any parameter fails type or value validation.
            PaymentError: If the payment cannot be processed.
        """
        logger.debug(
            "Entering create_order(user_id=%s, items_count=%s, "
            "payment_token=***)",
            user_id,
            len(items) if isinstance(items, list) else "n/a",
        )

        user_id = _validate_str(user_id, "user_id")
        payment_token = _validate_str(payment_token, "payment_token")
        items = _validate_items(items)
        if len(items) > self.config.max_items_per_order:
            raise ValidationError(
                f"items must not exceed {self.config.max_items_per_order} "
                "entries"
            )

        _verify_permission(user_id, None, "create")

        total = _calculate_total(items)

        try:
            transaction_id = self.payment_processor(
                payment_token, total, self.config
            )
        except PaymentError:
            raise
        except Exception as exc:  # R-008: preserve original context
            raise PaymentError(
                f"payment processing failed for user {user_id!r}"
            ) from exc

        order_id = f"ord_{uuid.uuid4().hex}"
        order: Dict[str, Any] = {
            "order_id": order_id,
            "user_id": user_id,
            "items": list(items),
            "total": total,
            "currency": self.config.currency,
            "status": "confirmed",
            "transaction_id": transaction_id,
        }
        self._orders[order_id] = order

        confirmation = dict(order)
        logger.debug(
            "Exiting create_order with return type %s",
            type(confirmation).__name__,
        )
        return confirmation

    def cancel_order(
        self, user_id: str, order_id: str, reason: str
    ) -> Dict[str, Any]:
        """Cancel an existing order and issue a refund.

        Args:
            user_id: The identifier of the user requesting cancellation.
            order_id: The identifier of the order to cancel.
            reason: A human-readable cancellation reason.

        Returns:
            A cancellation confirmation mapping with order id, status, refund
            id, refunded amount, currency, and reason.

        Raises:
            ValidationError: If any parameter fails type or value validation.
            OrderNotFoundError: If no order matches the given id.
            PermissionDeniedError: If the user does not own the order.
            OrderStateError: If the order is already cancelled.
            PaymentError: If the refund cannot be processed.
        """
        logger.debug(
            "Entering cancel_order(user_id=%s, order_id=%s, reason=%s)",
            user_id,
            order_id,
            reason,
        )

        user_id = _validate_str(user_id, "user_id")
        order_id = _validate_str(order_id, "order_id")
        reason = _validate_str(reason, "reason")

        order = self._orders.get(order_id)
        if order is None:
            raise OrderNotFoundError(f"order {order_id!r} was not found")

        _verify_permission(user_id, order, "cancel")

        if order["status"] == "cancelled":
            raise OrderStateError(f"order {order_id!r} is already cancelled")

        try:
            refund_id = self.refund_processor(
                order["transaction_id"], order["total"], self.config
            )
        except PaymentError:
            raise
        except Exception as exc:  # R-008: preserve original context
            raise PaymentError(
                f"refund failed for order {order_id!r}"
            ) from exc

        order["status"] = "cancelled"
        order["cancellation_reason"] = reason
        order["refund_id"] = refund_id

        confirmation: Dict[str, Any] = {
            "order_id": order_id,
            "status": order["status"],
            "refund_id": refund_id,
            "refunded_amount": order["total"],
            "currency": order["currency"],
            "reason": reason,
        }
        logger.debug(
            "Exiting cancel_order with return type %s",
            type(confirmation).__name__,
        )
        return confirmation

    def get_order_status(
        self, user_id: str, order_id: str
    ) -> Dict[str, Any]:
        """Retrieve the current status and details of an order.

        Args:
            user_id: The identifier of the user requesting the status.
            order_id: The identifier of the order to inspect.

        Returns:
            An order status mapping with order id, status, total, currency, and
            item count.

        Raises:
            ValidationError: If any parameter fails type or value validation.
            OrderNotFoundError: If no order matches the given id.
            PermissionDeniedError: If the user does not own the order.
        """
        logger.debug(
            "Entering get_order_status(user_id=%s, order_id=%s)",
            user_id,
            order_id,
        )

        user_id = _validate_str(user_id, "user_id")
        order_id = _validate_str(order_id, "order_id")

        order = self._orders.get(order_id)
        if order is None:
            raise OrderNotFoundError(f"order {order_id!r} was not found")

        _verify_permission(user_id, order, "view")

        status: Dict[str, Any] = {
            "order_id": order["order_id"],
            "status": order["status"],
            "total": order["total"],
            "currency": order["currency"],
            "item_count": len(order["items"]),
        }
        logger.debug(
            "Exiting get_order_status with return type %s",
            type(status).__name__,
        )
        return status


# ---------------------------------------------------------------------------

_default_service = OrderService()

# Unit tests (R: tests for all public functions)
# Run with: python -m unittest order_service
# ===========================================================================
import unittest  # noqa: E402


class OrderServiceTests(unittest.TestCase):
    """Unit tests covering create, cancel, and status retrieval flows."""

    def setUp(self) -> None:
        """Create a fresh, isolated service instance per test."""
        self.service = OrderService(config=ServiceConfig())
        self.user = "user-1"
        self.items = [
            {"name": "Widget", "quantity": 2, "unit_price": 10.0},
            {"name": "Gadget", "quantity": 1, "unit_price": 5.5},
        ]

    # --- create_order ----------------------------------------------------
    def test_create_order_success(self) -> None:
        order = self.service.create_order(self.user, self.items, "tok_abc")
        self.assertEqual(order["status"], "confirmed")
        self.assertEqual(order["total"], 25.5)
        self.assertTrue(order["order_id"].startswith("ord_"))
        self.assertTrue(order["transaction_id"].startswith("txn_"))

    def test_create_order_rejects_non_str_user(self) -> None:
        with self.assertRaises(ValidationError):
            self.service.create_order(123, self.items, "tok_abc")  # type: ignore[arg-type]

    def test_create_order_rejects_negative_quantity(self) -> None:
        bad = [{"name": "X", "quantity": -1, "unit_price": 1.0}]
        with self.assertRaises(ValidationError):
            self.service.create_order(self.user, bad, "tok_abc")

    def test_create_order_rejects_empty_items(self) -> None:
        with self.assertRaises(ValidationError):
            self.service.create_order(self.user, [], "tok_abc")

    def test_create_order_rejects_bad_item_type(self) -> None:
        with self.assertRaises(ValidationError):
            self.service.create_order(self.user, ["nope"], "tok_abc")  # type: ignore[list-item]

    def test_create_order_wraps_payment_failure(self) -> None:
        def boom(token: str, amount: float, config: ServiceConfig) -> str:
            raise RuntimeError("gateway down")

        svc = OrderService(payment_processor=boom)
        with self.assertRaises(PaymentError) as ctx:
            svc.create_order(self.user, self.items, "tok_abc")
        self.assertIsInstance(ctx.exception.__cause__, RuntimeError)  # R-008

    # --- cancel_order ----------------------------------------------------
    def test_cancel_order_success(self) -> None:
        order = self.service.create_order(self.user, self.items, "tok_abc")
        result = self.service.cancel_order(
            self.user, order["order_id"], "changed mind"
        )
        self.assertEqual(result["status"], "cancelled")
        self.assertEqual(result["refunded_amount"], 25.5)
        self.assertTrue(result["refund_id"].startswith("rfnd_"))

    def test_cancel_order_not_found(self) -> None:
        with self.assertRaises(OrderNotFoundError):
            self.service.cancel_order(self.user, "ord_missing", "reason")

    def test_cancel_order_permission_denied(self) -> None:
        order = self.service.create_order(self.user, self.items, "tok_abc")
        with self.assertRaises(PermissionDeniedError):
            self.service.cancel_order("intruder", order["order_id"], "x")

    def test_cancel_order_already_cancelled(self) -> None:
        order = self.service.create_order(self.user, self.items, "tok_abc")
        self.service.cancel_order(self.user, order["order_id"], "first")
        with self.assertRaises(OrderStateError):
            self.service.cancel_order(self.user, order["order_id"], "again")

    # --- get_order_status ------------------------------------------------
    def test_get_order_status_success(self) -> None:
        order = self.service.create_order(self.user, self.items, "tok_abc")
        status = self.service.get_order_status(self.user, order["order_id"])
        self.assertEqual(status["status"], "confirmed")
        self.assertEqual(status["item_count"], 2)
        self.assertEqual(status["total"], 25.5)

    def test_get_order_status_not_found(self) -> None:
        with self.assertRaises(OrderNotFoundError):
            self.service.get_order_status(self.user, "ord_missing")

    def test_get_order_status_permission_denied(self) -> None:
        order = self.service.create_order(self.user, self.items, "tok_abc")
        with self.assertRaises(PermissionDeniedError):
            self.service.get_order_status("intruder", order["order_id"])

    # --- module-level convenience API ------------------------------------
    def test_module_level_round_trip(self) -> None:
        order = create_order("u9", self.items, "tok_z")
        status = get_order_status("u9", order["order_id"])
        self.assertEqual(status["status"], "confirmed")
        cancelled = cancel_order("u9", order["order_id"], "done")
        self.assertEqual(cancelled["status"], "cancelled")


if __name__ == "__main__":
    unittest.main()
