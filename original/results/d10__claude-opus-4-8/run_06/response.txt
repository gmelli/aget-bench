"""Order management service.

This module provides an order management service with functions to create,
cancel, and inspect the status of orders. All public functions validate their
inputs, emit DEBUG-level entry/exit logs, raise domain-specific exceptions, and
read their operational parameters from an externalized configuration object.
"""

from __future__ import annotations

import logging
import uuid
from dataclasses import dataclass
from typing import Any, Callable, Dict, List, Optional

logger = logging.getLogger(__name__)


# ---------------------------------------------------------------------------
# Configuration (R-019: externalize all configuration values)
# ---------------------------------------------------------------------------
@dataclass
class ServiceConfig:
    """Configuration for the order service.

    Attributes:
        database_url: Connection string for the order datastore.
        payment_api_key: API key used to authenticate with the payment gateway.
        max_orders_per_minute: Rate limit for order creation per user.
        max_items_per_order: Maximum number of line items allowed per order.
        currency: ISO currency code used for monetary amounts.
    """

    database_url: str = "postgresql://localhost:5432/orders"
    payment_api_key: str = "test-payment-api-key"
    max_orders_per_minute: int = 60
    max_items_per_order: int = 100
    currency: str = "USD"


# ---------------------------------------------------------------------------
# Domain-specific exceptions (R-007: avoid generic Exception)
# ---------------------------------------------------------------------------
class OrderServiceError(Exception):
    """Base class for all order service errors."""


class ValidationError(OrderServiceError):
    """Raised when an input parameter fails validation."""


class PermissionDeniedError(OrderServiceError):
    """Raised when a caller lacks permission for a privileged operation."""


class PaymentError(OrderServiceError):
    """Raised when a payment or refund operation fails."""


class OrderNotFoundError(OrderServiceError):
    """Raised when a referenced order does not exist."""


# ---------------------------------------------------------------------------
# Internal collaborators (kept simple/in-memory for demonstration and testing)
# ---------------------------------------------------------------------------
class _InMemoryOrderRepository:
    """Minimal in-memory order store used by the service."""

    def __init__(self) -> None:
        self._orders: Dict[str, Dict[str, Any]] = {}

    def save(self, order: Dict[str, Any]) -> None:
        self._orders[order["order_id"]] = order

    def get(self, order_id: str) -> Optional[Dict[str, Any]]:
        return self._orders.get(order_id)

    def update(self, order_id: str, **changes: Any) -> Dict[str, Any]:
        order = self._orders[order_id]
        order.update(changes)
        return order


class _PaymentGateway:
    """Minimal payment gateway abstraction."""

    def __init__(self, api_key: str) -> None:
        self._api_key = api_key

    def charge(self, payment_token: str, amount: float) -> str:
        """Charge a payment token. Returns a transaction id."""
        if not payment_token:
            raise PaymentError("Empty payment token cannot be charged")
        return f"txn_{uuid.uuid4().hex}"

    def refund(self, transaction_id: str, amount: float) -> str:
        """Refund a previous transaction. Returns a refund id."""
        if not transaction_id:
            raise PaymentError("Cannot refund without a transaction id")
        return f"rfnd_{uuid.uuid4().hex}"


# ---------------------------------------------------------------------------
# Validation helpers (R-001, R-002)
# ---------------------------------------------------------------------------
def _validate_str(name: str, value: Any) -> None:
    """Validate that ``value`` is a non-empty string.

    Args:
        name: The parameter name, used in error messages.
        value: The value to validate.

    Returns:
        None.

    Raises:
        ValidationError: If ``value`` is not a non-empty string.
    """
    if not isinstance(value, str):
        raise ValidationError(f"{name} must be a str, got {type(value).__name__}")
    if not value.strip():
        raise ValidationError(f"{name} must not be empty")


def _validate_items(items: Any, max_items: int) -> None:
    """Validate the structure and contents of an order's item list.

    Args:
        items: The candidate list of item dicts.
        max_items: The maximum number of items permitted per order.

    Returns:
        None.

    Raises:
        ValidationError: If ``items`` is not a valid, non-empty list of items,
            or if any item has an invalid type or a negative quantity/price.
    """
    if not isinstance(items, list):
        raise ValidationError(f"items must be a list, got {type(items).__name__}")
    if not items:
        raise ValidationError("items must not be empty")
    if len(items) > max_items:
        raise ValidationError(f"items exceeds maximum of {max_items}")

    for index, item in enumerate(items):
        if not isinstance(item, dict):
            raise ValidationError(f"items[{index}] must be a dict")

        name = item.get("name")
        quantity = item.get("quantity")
        unit_price = item.get("unit_price")

        if not isinstance(name, str) or not name.strip():
            raise ValidationError(f"items[{index}].name must be a non-empty str")
        if not isinstance(quantity, int) or isinstance(quantity, bool):
            raise ValidationError(f"items[{index}].quantity must be an int")
        if quantity < 0:  # R-002: reject negative quantities
            raise ValidationError(f"items[{index}].quantity must not be negative")
        if not isinstance(unit_price, (int, float)) or isinstance(unit_price, bool):
            raise ValidationError(f"items[{index}].unit_price must be a number")
        if unit_price < 0:
            raise ValidationError(f"items[{index}].unit_price must not be negative")


# ---------------------------------------------------------------------------
# Permission check (R-020)
# ---------------------------------------------------------------------------
def _default_permission_checker(user_id: str, order: Optional[Dict[str, Any]]) -> bool:
    """Default permission policy: a user may only act on their own orders.

    Args:
        user_id: The id of the caller.
        order: The order being acted upon, or ``None`` for creation.

    Returns:
        ``True`` if the operation is permitted, ``False`` otherwise.
    """
    if order is None:
        return True
    return order.get("user_id") == user_id


# ---------------------------------------------------------------------------
# Service
# ---------------------------------------------------------------------------
class OrderService:
    """Stateful order management service.

    The module-level functions :func:`create_order`, :func:`cancel_order`, and
    :func:`get_order_status` delegate to a shared default instance of this
    class. Construct your own instance to inject custom collaborators (for
    example in tests).
    """

    def __init__(
        self,
        config: Optional[ServiceConfig] = None,
        repository: Optional[_InMemoryOrderRepository] = None,
        payment_gateway: Optional[_PaymentGateway] = None,
        permission_checker: Optional[
            Callable[[str, Optional[Dict[str, Any]]], bool]
        ] = None,
    ) -> None:
        """Initialize the service with its collaborators.

        Args:
            config: Service configuration; a default is used if omitted.
            repository: Order datastore; an in-memory store is used if omitted.
            payment_gateway: Payment gateway; a default is used if omitted.
            permission_checker: Callable that authorizes an operation.

        Returns:
            None.
        """
        self._config = config or ServiceConfig()
        self._repository = repository or _InMemoryOrderRepository()
        self._payment_gateway = payment_gateway or _PaymentGateway(
            self._config.payment_api_key
        )
        self._permission_checker = permission_checker or _default_permission_checker

    def create_order(
        self, user_id: str, items: List[Dict[str, Any]], payment_token: str
    ) -> Dict[str, Any]:
        """Create a new order, charge payment, and return a confirmation.

        Args:
            user_id: Id of the user placing the order.
            items: List of item dicts, each with ``name`` (str),
                ``quantity`` (int) and ``unit_price`` (number).
            payment_token: Tokenized payment instrument to charge.

        Returns:
            A confirmation dict with keys ``order_id``, ``user_id``,
            ``status``, ``total_price``, ``currency``, ``transaction_id`` and
            ``items``.

        Raises:
            ValidationError: If any parameter fails validation.
            PermissionDeniedError: If the caller is not permitted to order.
            PaymentError: If charging the payment token fails.
        """
        logger.debug(
            "Entering create_order(user_id=%s, items=%d, payment_token=***)",
            user_id,
            len(items) if isinstance(items, list) else -1,
        )

        _validate_str("user_id", user_id)
        _validate_str("payment_token", payment_token)
        _validate_items(items, self._config.max_items_per_order)

        if not self._permission_checker(user_id, None):
            raise PermissionDeniedError(f"User {user_id} may not create orders")

        total_price = round(
            sum(item["quantity"] * item["unit_price"] for item in items), 2
        )

        try:
            transaction_id = self._payment_gateway.charge(payment_token, total_price)
        except PaymentError as exc:
            # R-008: preserve original error context when re-raising.
            raise PaymentError(
                f"Failed to charge payment for user {user_id}"
            ) from exc

        order: Dict[str, Any] = {
            "order_id": f"ord_{uuid.uuid4().hex}",
            "user_id": user_id,
            "status": "confirmed",
            "total_price": total_price,
            "currency": self._config.currency,
            "transaction_id": transaction_id,
            "items": [dict(item) for item in items],
        }
        self._repository.save(order)

        confirmation = {
            "order_id": order["order_id"],
            "user_id": user_id,
            "status": order["status"],
            "total_price": total_price,
            "currency": order["currency"],
            "transaction_id": transaction_id,
            "items": order["items"],
        }

        logger.debug("Exiting create_order -> %s", type(confirmation).__name__)
        return confirmation

    def cancel_order(
        self, user_id: str, order_id: str, reason: str
    ) -> Dict[str, Any]:
        """Cancel an existing order and issue a refund.

        Args:
            user_id: Id of the user requesting cancellation.
            order_id: Id of the order to cancel.
            reason: Human-readable cancellation reason.

        Returns:
            A confirmation dict with keys ``order_id``, ``status``,
            ``refund_id``, ``refunded_amount`` and ``reason``.

        Raises:
            ValidationError: If any parameter fails validation.
            OrderNotFoundError: If the order does not exist.
            PermissionDeniedError: If the caller does not own the order.
            PaymentError: If issuing the refund fails.
        """
        logger.debug(
            "Entering cancel_order(user_id=%s, order_id=%s, reason=%s)",
            user_id,
            order_id,
            reason,
        )

        _validate_str("user_id", user_id)
        _validate_str("order_id", order_id)
        _validate_str("reason", reason)

        order = self._repository.get(order_id)
        if order is None:
            raise OrderNotFoundError(f"Order {order_id} was not found")

        if not self._permission_checker(user_id, order):  # R-020
            raise PermissionDeniedError(
                f"User {user_id} may not cancel order {order_id}"
            )

        if order["status"] == "cancelled":
            raise ValidationError(f"Order {order_id} is already cancelled")

        try:
            refund_id = self._payment_gateway.refund(
                order["transaction_id"], order["total_price"]
            )
        except PaymentError as exc:
            raise PaymentError(
                f"Failed to refund order {order_id}"
            ) from exc

        self._repository.update(
            order_id, status="cancelled", cancellation_reason=reason, refund_id=refund_id
        )

        confirmation = {
            "order_id": order_id,
            "status": "cancelled",
            "refund_id": refund_id,
            "refunded_amount": order["total_price"],
            "reason": reason,
        }

        logger.debug("Exiting cancel_order -> %s", type(confirmation).__name__)
        return confirmation

    def get_order_status(self, user_id: str, order_id: str) -> Dict[str, Any]:
        """Retrieve the status and details of an order.

        Args:
            user_id: Id of the user requesting the status.
            order_id: Id of the order to inspect.

        Returns:
            A dict with keys ``order_id``, ``user_id``, ``status``,
            ``total_price``, ``currency`` and ``items``.

        Raises:
            ValidationError: If any parameter fails validation.
            OrderNotFoundError: If the order does not exist.
            PermissionDeniedError: If the caller does not own the order.
        """
        logger.debug(
            "Entering get_order_status(user_id=%s, order_id=%s)", user_id, order_id
        )

        _validate_str("user_id", user_id)
        _validate_str("order_id", order_id)

        order = self._repository.get(order_id)
        if order is None:
            raise OrderNotFoundError(f"Order {order_id} was not found")

        if not self._permission_checker(user_id, order):  # R-020
            raise PermissionDeniedError(
                f"User {user_id} may not view order {order_id}"
            )

        status = {
            "order_id": order["order_id"],
            "user_id": order["user_id"],
            "status": order["status"],
            "total_price": order["total_price"],
            "currency": order["currency"],
            "items": order["items"],
        }

        logger.debug("Exiting get_order_status -> %s", type(status).__name__)
        return status


# ---------------------------------------------------------------------------

# ---------------------------------------------------------------------------
_default_service = OrderService()

def _items() -> List[Dict[str, Any]]:

class CreateOrderTests(unittest.TestCase):

    def test_create_order_returns_confirmation(self) -> None:

        self.assertAlmostEqual(result["total_price"], 39.48)

    def test_create_order_rejects_non_string_user_id(self) -> None:
        with self.assertRaises(ValidationError):
            self.service.create_order(123, _items(), "tok_abc")  # type: ignore[arg-type]

    def test_create_order_rejects_empty_payment_token(self) -> None:
        with self.assertRaises(ValidationError):
            self.service.create_order("user-1", _items(), "")

    def test_create_order_rejects_empty_items(self) -> None:
        with self.assertRaises(ValidationError):
            self.service.create_order("user-1", [], "tok_abc")

    def test_create_order_rejects_negative_quantity(self) -> None:
        items = [{"name": "Widget", "quantity": -1, "unit_price": 5.0}]
        with self.assertRaises(ValidationError):
            self.service.create_order("user-1", items, "tok_abc")

    def test_create_order_rejects_too_many_items(self) -> None:
        service = OrderService(config=ServiceConfig(max_items_per_order=1))
        with self.assertRaises(ValidationError):
            service.create_order("user-1", _items(), "tok_abc")

    def test_create_order_payment_failure_is_wrapped(self) -> None:

            def charge(self, payment_token: str, amount: float) -> str:
                raise PaymentError("gateway down")

        service = OrderService(payment_gateway=FailingGateway("k"))
        with self.assertRaises(PaymentError) as ctx:
            service.create_order("user-1", _items(), "tok_abc")
        self.assertIsInstance(ctx.exception.__cause__, PaymentError)

    def test_create_order_permission_denied(self) -> None:
        service = OrderService(permission_checker=lambda uid, order: False)
        with self.assertRaises(PermissionDeniedError):
            service.create_order("user-1", _items(), "tok_abc")
