```python
"""Order management service.

This module provides a small order-management service with functions to
create, cancel, and inspect orders. All configuration is externalized into
:class:`ServiceConfig` (R-019), every public function validates its inputs
(R-001, R-002), raises domain-specific exceptions (R-007, R-008), logs entry
and exit at DEBUG level (R-013, R-014), verifies caller permissions for
privileged operations (R-020), and carries full type hints and docstrings
(R-025, R-026).
"""

from __future__ import annotations

import logging
import uuid
from dataclasses import dataclass, field
from typing import Any, Callable, Dict, List, Optional

logger = logging.getLogger(__name__)


# ---------------------------------------------------------------------------
# Configuration (R-019: externalize all configuration values)
# ---------------------------------------------------------------------------
@dataclass(frozen=True)
class ServiceConfig:
    """Configuration for the order service.

    Attributes:
        database_url: Connection string for the order datastore.
        payment_api_key: API key used to authenticate with the payment gateway.
        rate_limit_per_minute: Maximum number of operations permitted per minute.
        currency: ISO currency code used for order totals.
    """

    database_url: str = "postgresql://localhost:5432/orders"
    payment_api_key: str = "test-payment-api-key"
    rate_limit_per_minute: int = 60
    currency: str = "USD"


# ---------------------------------------------------------------------------
# Domain-specific exceptions (R-007: no generic Exception)
# ---------------------------------------------------------------------------
class OrderServiceError(Exception):
    """Base class for all order-service errors."""


class ValidationError(OrderServiceError):
    """Raised when input parameters fail validation."""


class PermissionDeniedError(OrderServiceError):
    """Raised when the caller lacks permission for a privileged operation."""


class PaymentError(OrderServiceError):
    """Raised when a payment or refund cannot be processed."""


class OrderNotFoundError(OrderServiceError):
    """Raised when the requested order does not exist."""


# ---------------------------------------------------------------------------
# Collaborators (small, in-memory implementations for demonstration/testing)
# ---------------------------------------------------------------------------
class PaymentGateway:
    """Minimal payment gateway abstraction.

    A real implementation would call an external payment provider; this
    in-memory version is sufficient for unit testing.
    """

    def __init__(self, api_key: str) -> None:
        """Initialize the gateway.

        Args:
            api_key: API key used to authenticate with the provider.
        """
        self._api_key = api_key

    def charge(self, payment_token: str, amount: float, currency: str) -> str:
        """Charge a payment token.

        Args:
            payment_token: Opaque token identifying the payment method.
            amount: Amount to charge; must be non-negative.
            currency: ISO currency code.

        Returns:
            A unique transaction identifier.

        Raises:
            PaymentError: If the charge cannot be completed.
        """
        if not payment_token:
            raise PaymentError("Missing payment token")
        if amount < 0:
            raise PaymentError("Charge amount cannot be negative")
        return f"txn_{uuid.uuid4().hex}"

    def refund(self, transaction_id: str, amount: float, currency: str) -> str:
        """Refund a previous charge.

        Args:
            transaction_id: Identifier of the original charge.
            amount: Amount to refund; must be non-negative.
            currency: ISO currency code.

        Returns:
            A unique refund identifier.

        Raises:
            PaymentError: If the refund cannot be completed.
        """
        if not transaction_id:
            raise PaymentError("Missing transaction id for refund")
        if amount < 0:
            raise PaymentError("Refund amount cannot be negative")
        return f"rfnd_{uuid.uuid4().hex}"


class OrderRepository:
    """In-memory order repository keyed by order id."""

    def __init__(self) -> None:
        """Initialize an empty repository."""
        self._orders: Dict[str, Dict[str, Any]] = {}

    def save(self, order: Dict[str, Any]) -> None:
        """Persist an order.

        Args:
            order: Order record containing at least an ``order_id`` key.
        """
        self._orders[order["order_id"]] = order

    def get(self, order_id: str) -> Optional[Dict[str, Any]]:
        """Fetch an order by id.

        Args:
            order_id: Identifier of the order to retrieve.

        Returns:
            The order record, or ``None`` if it does not exist.
        """
        return self._orders.get(order_id)


# ---------------------------------------------------------------------------
# Service
# ---------------------------------------------------------------------------
@dataclass
class OrderService:
    """Order management service.

    The service owns its configuration and collaborators so that callers can
    inject test doubles. A ``permission_checker`` callable verifies caller
    permissions before privileged operations are performed (R-020).
    """

    config: ServiceConfig = field(default_factory=ServiceConfig)
    repository: OrderRepository = field(default_factory=OrderRepository)
    gateway: Optional[PaymentGateway] = None
    permission_checker: Callable[[str, str], bool] = field(
        default=lambda user_id, action: True
    )

    def __post_init__(self) -> None:
        """Wire up default collaborators that depend on configuration."""
        if self.gateway is None:
            self.gateway = PaymentGateway(self.config.payment_api_key)

    # -- internal helpers ---------------------------------------------------
    def _require_permission(self, user_id: str, action: str) -> None:
        """Verify the caller may perform a privileged action (R-020).

        Args:
            user_id: Identifier of the calling user.
            action: Name of the privileged action being attempted.

        Raises:
            PermissionDeniedError: If the caller lacks permission.
        """
        if not self.permission_checker(user_id, action):
            raise PermissionDeniedError(
                f"User {user_id!r} is not permitted to perform {action!r}"
            )

    @staticmethod
    def _validate_non_empty_str(value: Any, name: str) -> str:
        """Validate that a value is a non-empty string (R-001).

        Args:
            value: Value to validate.
            name: Parameter name, used in error messages.

        Returns:
            The validated string.

        Raises:
            ValidationError: If the value is not a non-empty string.
        """
        if not isinstance(value, str):
            raise ValidationError(f"{name} must be a str, got {type(value).__name__}")
        if not value:
            raise ValidationError(f"{name} must not be empty")
        return value

    def _calculate_total(self, items: List[Dict[str, Any]]) -> float:
        """Validate items and calculate the order total.

        Args:
            items: List of item dicts with ``name``, ``quantity`` and
                ``unit_price`` keys.

        Returns:
            The total price across all items.

        Raises:
            ValidationError: If items are malformed or contain negative
                quantities (R-001, R-002).
        """
        if not isinstance(items, list) or not items:
            raise ValidationError("items must be a non-empty list")

        total = 0.0
        for index, item in enumerate(items):
            if not isinstance(item, dict):
                raise ValidationError(f"items[{index}] must be a dict")

            name = item.get("name")
            quantity = item.get("quantity")
            unit_price = item.get("unit_price")

            self._validate_non_empty_str(name, f"items[{index}].name")

            if not isinstance(quantity, int) or isinstance(quantity, bool):
                raise ValidationError(f"items[{index}].quantity must be an int")
            if quantity < 0:  # R-002: reject negative quantities
                raise ValidationError(
                    f"items[{index}].quantity must not be negative"
                )

            if isinstance(unit_price, bool) or not isinstance(
                unit_price, (int, float)
            ):
                raise ValidationError(f"items[{index}].unit_price must be a number")
            if unit_price < 0:
                raise ValidationError(
                    f"items[{index}].unit_price must not be negative"
                )

            total += quantity * float(unit_price)

        return total

    # -- public API ---------------------------------------------------------
    def create_order(
        self, user_id: str, items: List[Dict[str, Any]], payment_token: str
    ) -> Dict[str, Any]:
        """Create a new order and process its payment.

        Args:
            user_id: Identifier of the user placing the order.
            items: List of item dicts, each with ``name`` (str),
                ``quantity`` (int) and ``unit_price`` (number).
            payment_token: Opaque token identifying the payment method.

        Returns:
            An order confirmation dict containing the order id, status, total,
            currency and the payment transaction id.

        Raises:
            ValidationError: If any parameter is invalid (R-001, R-002).
            PermissionDeniedError: If the caller may not create orders (R-020).
            PaymentError: If the payment cannot be processed.
        """
        logger.debug(
            "Entering create_order(user_id=%r, items=<%d items>, payment_token=<redacted>)",
            user_id,
            len(items) if isinstance(items, list) else 0,
        )  # R-013

        self._validate_non_empty_str(user_id, "user_id")
        self._validate_non_empty_str(payment_token, "payment_token")
        self._require_permission(user_id, "create_order")

        total = self._calculate_total(items)
        order_id = f"ord_{uuid.uuid4().hex}"

        try:
            assert self.gateway is not None  # set in __post_init__
            transaction_id = self.gateway.charge(
                payment_token, total, self.config.currency
            )
        except PaymentError as exc:
            # R-008: preserve original error context when re-raising.
            raise PaymentError(
                f"Payment failed for order {order_id}: {exc}"
            ) from exc

        order: Dict[str, Any] = {
            "order_id": order_id,
            "user_id": user_id,
            "items": items,
            "total": total,
            "currency": self.config.currency,
            "status": "confirmed",
            "transaction_id": transaction_id,
        }
        self.repository.save(order)

        confirmation: Dict[str, Any] = {
            "order_id": order_id,
            "status": "confirmed",
            "total": total,
            "currency": self.config.currency,
            "transaction_id": transaction_id,
        }

        logger.debug(
            "Exiting create_order -> %s", type(confirmation).__name__
        )  # R-014
        return confirmation

    def cancel_order(
        self, user_id: str, order_id: str, reason: str
    ) -> Dict[str, Any]:
        """Cancel an existing order and issue a refund.

        Args:
            user_id: Identifier of the user cancelling the order.
            order_id: Identifier of the order to cancel.
            reason: Human-readable cancellation reason.

        Returns:
            A cancellation confirmation dict containing the order id, status,
            refund id and the cancellation reason.

        Raises:
            ValidationError: If any parameter is invalid (R-001).
            PermissionDeniedError: If the caller may not cancel the order (R-020).
            OrderNotFoundError: If the order does not exist.
            PaymentError: If the refund cannot be processed.
        """
        logger.debug(
            "Entering cancel_order(user_id=%r, order_id=%r, reason=%r)",
            user_id,
            order_id,
            reason,
        )  # R-013

        self._validate_non_empty_str(user_id, "user_id")
        self._validate_non_empty_str(order_id, "order_id")
        self._validate_non_empty_str(reason, "reason")
        self._require_permission(user_id, "cancel_order")

        order = self.repository.get(order_id)
        if order is None:
            raise OrderNotFoundError(f"Order {order_id!r} not found")
        if order["user_id"] != user_id:
            raise PermissionDeniedError(
                f"User {user_id!r} does not own order {order_id!r}"
            )
        if order["status"] == "cancelled":
            raise ValidationError(f"Order {order_id!r} is already cancelled")

        try:
            assert self.gateway is not None
            refund_id = self.gateway.refund(
                order["transaction_id"], order["total"], self.config.currency
            )
        except PaymentError as exc:
            raise PaymentError(
                f"Refund failed for order {order_id}: {exc}"
            ) from exc  # R-008

        order["status"] = "cancelled"
        order["cancellation_reason"] = reason
        order["refund_id"] = refund_id
        self.repository.save(order)

        confirmation: Dict[str, Any] = {
            "order_id": order_id,
            "status": "cancelled",
            "refund_id": refund_id,
            "reason": reason,
        }

        logger.debug(
            "Exiting cancel_order -> %s", type(confirmation).__name__
        )  # R-014
        return confirmation

    def get_order_status(self, user_id: str, order_id: str) -> Dict[str, Any]:
        """Retrieve the status of an existing order.

        Args:
            user_id: Identifier of the user requesting the status.
            order_id: Identifier of the order to inspect.

        Returns:
            An order status dict containing the order id, status, total and
            currency.

        Raises:
            ValidationError: If any parameter is invalid (R-001).
            PermissionDeniedError: If the caller does not own the order (R-020).
            OrderNotFoundError: If the order does not exist.
        """
        logger.debug(
            "Entering get_order_status(user_id=%r, order_id=%r)",
            user_id,
            order_id,
        )  # R-013

        self._validate_non_empty_str(user_id, "user_id")
        self._validate_non_empty_str(order_id, "order_id")
        self._require_permission(user_id, "get_order_status")

        order = self.repository.get(order_id)
        if order is None:
            raise OrderNotFoundError(f"Order {order_id!r} not found")
        if order["user_id"] != user_id:
            raise PermissionDeniedError(
                f"User {user_id!r} does not own order {order_id!r}"
            )

        status: Dict[str, Any] = {
            "order_id": order_id,
            "status": order["status"],
            "total": order["total"],
            "currency": order["currency"],
        }

        logger.debug(
            "Exiting get_order_status -> %s", type(status).__name__
        )  # R-014
        return status


# ---------------------------------------------------------------------------

# ---------------------------------------------------------------------------
_default_service = OrderService()
```

```python
"""Unit tests for order_service."""

import unittest

from order_service import (
    OrderNotFoundError,
    OrderService,
    PaymentError,
    PaymentGateway,
    PermissionDeniedError,
    ServiceConfig,
    ValidationError,
)


def make_items():
    """Return a valid sample item list."""
    return [
        {"name": "widget", "quantity": 2, "unit_price": 5.0},
        {"name": "gadget", "quantity": 1, "unit_price": 3.5},
    ]


class CreateOrderTests(unittest.TestCase):
    """Tests for OrderService.create_order."""

    def setUp(self):
        self.service = OrderService()

    def test_create_order_returns_confirmation(self):
        conf = self.service.create_order("user-1", make_items(), "tok_abc")
        self.assertEqual(conf["status"], "confirmed")
        self.assertEqual(conf["total"], 13.5)
        self.assertEqual(conf["currency"], "USD")
        self.assertTrue(conf["order_id"].startswith("ord_"))
        self.assertTrue(conf["transaction_id"].startswith("txn_"))

    def test_order_ids_are_unique(self):
        a = self.service.create_order("user-1", make_items(), "tok")
        b = self.service.create_order("user-1", make_items(), "tok")
        self.assertNotEqual(a["order_id"], b["order_id"])

    def test_invalid_user_id_type_raises(self):
        with self.assertRaises(ValidationError):
            self.service.create_order(123, make_items(), "tok")

    def test_empty_payment_token_raises(self):
        with self.assertRaises(ValidationError):
            self.service.create_order("user-1", make_items(), "")

    def test_empty_items_raises(self):
        with self.assertRaises(ValidationError):
            self.service.create_order("user-1", [], "tok")

    def test_negative_quantity_rejected(self):  # R-002
        items = [{"name": "widget", "quantity": -1, "unit_price": 5.0}]
        with self.assertRaises(ValidationError):
            self.service.create_order("user-1", items, "tok")

    def test_negative_unit_price_rejected(self):
        items = [{"name": "widget", "quantity": 1, "unit_price": -5.0}]
        with self.assertRaises(ValidationError):
            self.service.create_order("user-1", items, "tok")

    def test_bool_quantity_rejected(self):
        items = [{"name": "widget", "quantity": True, "unit_price": 5.0}]
        with self.assertRaises(ValidationError):
            self.service.create_order("user-1", items, "tok")

    def test_permission_denied(self):  # R-020
        service = OrderService(permission_checker=lambda u, a: False)
        with self.assertRaises(PermissionDeniedError):
            service.create_order("user-1", make_items(), "tok")

    def test_payment_failure_wraps_context(self):  # R-008
        class FailingGateway(PaymentGateway):
            def charge(self, payment_token, amount, currency):
                raise PaymentError("declined")

        service = OrderService(gateway=FailingGateway("k"))
        with self.assertRaises(PaymentError) as ctx:
            service.create_order("user-1", make_items(), "tok")
        self.assertIsInstance(ctx.exception.__cause__, PaymentError)


class CancelOrderTests(unittest.TestCase):
    """Tests for OrderService.cancel_order."""

    def setUp(self):
        self.service = OrderService()
        self.order = self.service.create_order("user-1", make_items(), "tok")

    def test_cancel_returns_confirmation(self):
        conf = self.service.cancel_order(
            "user-1", self.order["order_id"], "changed mind"
        )
        self.assertEqual(conf["status"], "cancelled")
        self.assertEqual(conf["reason"], "changed mind")
        self.assertTrue(conf["refund_id"].startswith("rfnd_"))

    def test_cancel_unknown_order_raises(self):
        with self.assertRaises(OrderNotFoundError):
            self.service.cancel_order("user-1", "ord_missing", "reason")

    def test_cancel_other_users_order_denied(self):  # R-020
        with self.assertRaises(PermissionDeniedError):
            self.service.cancel_order("user-2", self.order["order_id"], "reason")

    def test_double_cancel_raises(self):
        self.service.cancel_order("user-1", self.order["order_id"], "reason")
        with self.assertRaises(ValidationError):
            self.service.cancel_order("user-1", self.order["order_id"], "again")

    def test_empty_reason_raises(self):
        with self.assertRaises(ValidationError):
            self.service.cancel_order("user-1", self.order["order_id"], "")


class GetOrderStatusTests(unittest.TestCase):
    """Tests for OrderService.get_order_status."""

    def setUp(self):
        self.service = OrderService()
        self.order = self.service.create_order("user-1", make_items(), "tok")

    def test_get_status_returns_dict(self):
        status = self.service.get_order_status("user-1", self.order["order_id"])
        self.assertEqual(status["status"], "confirmed")
        self.assertEqual(status["total"], 13.5)
        self.assertEqual(status["order_id"], self.order["order_id"])

    def test_status_reflects_cancellation(self):
        self.service.cancel_order("user-1", self.order["order_id"], "reason")
        status = self.service.get_order_status("user-1", self.order["order_id"])
        self.assertEqual(status["status"], "cancelled")

    def test_get_unknown_order_raises(self):
        with self.assertRaises(OrderNotFoundError):
            self.service.get_order_status("user-1", "ord_missing")

    def test_get_other_users_order_denied(self):  # R-020
        with self.assertRaises(PermissionDeniedError):
            self.service.get_order_status("user-2", self.order["order_id"])

    def test_empty_order_id_raises(self):
        with self.assertRaises(ValidationError):
            self.service.get_order_status("user-1", "")


class ConfigTests(unittest.TestCase):
    """Tests for ServiceConfig wiring."""

    def test_custom_currency_used(self):
        service = OrderService(config=ServiceConfig(currency="EUR"))
        conf = service.create_order("user-1", make_items(), "tok")
        self.assertEqual(conf["currency"], "EUR")

    unittest.main()
```
