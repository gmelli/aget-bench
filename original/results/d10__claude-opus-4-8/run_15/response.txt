```python
"""Order management service.

This module provides a small order management service with functions to
create, cancel and query orders. It is intentionally self-contained: the
payment gateway, datastore and permission checks are represented by simple
in-memory / stub implementations so the public API can be exercised in tests.

The implementation adheres to the mandatory agent rules:

* All public functions validate their parameters (R-001) and reject negative
  quantities (R-002).
* Domain-specific exceptions are used and original context is preserved
  when re-raising (R-007, R-008).
* Function entry/exit is logged at DEBUG level (R-013, R-014).
* All configuration lives in :class:`ServiceConfig` (R-019).
* Privileged operations verify caller permissions (R-020).
* Every function carries type hints and a structured docstring (R-025, R-026).
"""

from __future__ import annotations

import logging
import uuid
from dataclasses import dataclass
from typing import Any, Dict, List, Optional

logger = logging.getLogger(__name__)


# ---------------------------------------------------------------------------
# Configuration (R-019)
# ---------------------------------------------------------------------------
@dataclass(frozen=True)
class ServiceConfig:
    """Externalized configuration for the order service.

    Attributes:
        database_url: Connection string for the order datastore.
        payment_api_key: API key used to authenticate against the payment
            gateway.
        rate_limit_per_minute: Maximum number of privileged operations a
            single caller may perform per minute.
        currency: ISO currency code used for monetary amounts.
    """

    database_url: str = "sqlite:///orders.db"
    payment_api_key: str = "test-payment-api-key"
    rate_limit_per_minute: int = 60
    currency: str = "USD"


# Default configuration instance used when a caller does not supply one.
DEFAULT_CONFIG = ServiceConfig()


# ---------------------------------------------------------------------------
# Domain-specific exceptions (R-007)
# ---------------------------------------------------------------------------
class OrderServiceError(Exception):
    """Base class for all order service errors."""


class ValidationError(OrderServiceError):
    """Raised when a parameter fails type or value validation."""


class PermissionDeniedError(OrderServiceError):
    """Raised when a caller lacks permission for a privileged operation."""


class OrderNotFoundError(OrderServiceError):
    """Raised when an order cannot be located for the given identifiers."""


class PaymentError(OrderServiceError):
    """Raised when payment processing or refunding fails."""


# ---------------------------------------------------------------------------
# Internal collaborators (datastore / payment gateway / permissions)
# ---------------------------------------------------------------------------
class _InMemoryOrderStore:
    """A minimal in-memory order store standing in for a real database."""

    def __init__(self) -> None:
        self._orders: Dict[str, Dict[str, Any]] = {}

    def save(self, order: Dict[str, Any]) -> None:
        """Persist an order keyed by its order ID."""
        self._orders[order["order_id"]] = order

    def get(self, order_id: str) -> Optional[Dict[str, Any]]:
        """Return the stored order or ``None`` if it does not exist."""
        return self._orders.get(order_id)


class _PaymentGateway:
    """A stub payment gateway.

    A real implementation would call out to an external provider using the
    configured API key. Here we simulate success while rejecting an obviously
    invalid token so behaviour can be tested.
    """

    def __init__(self, config: ServiceConfig) -> None:
        self._config = config

    def charge(self, payment_token: str, amount: float) -> str:
        """Charge ``amount`` against ``payment_token``.

        Args:
            payment_token: Tokenized payment instrument.
            amount: Positive amount to charge.

        Returns:
            A transaction identifier string.

        Raises:
            PaymentError: If the gateway rejects the charge.
        """
        if not payment_token or payment_token == "invalid":
            raise PaymentError("Payment gateway rejected the payment token")
        return f"txn_{uuid.uuid4().hex}"

    def refund(self, transaction_id: str, amount: float) -> str:
        """Refund a previously captured transaction.

        Args:
            transaction_id: Identifier of the original charge.
            amount: Positive amount to refund.

        Returns:
            A refund identifier string.

        Raises:
            PaymentError: If the gateway rejects the refund.
        """
        if not transaction_id:
            raise PaymentError("Cannot refund without a transaction id")
        return f"rfnd_{uuid.uuid4().hex}"


# A module-level store so the public functions share state within a process.
_STORE = _InMemoryOrderStore()


def _check_permission(user_id: str, operation: str) -> None:
    """Verify that ``user_id`` may perform ``operation`` (R-020).

    Args:
        user_id: Identifier of the calling user.
        operation: Name of the privileged operation being attempted.

    Raises:
        PermissionDeniedError: If the caller is not permitted.
    """
    # A real system would consult an authorization service or ACL. Here we
    # reject a sentinel "anonymous" user to demonstrate the enforced check.
    if user_id == "anonymous":
        raise PermissionDeniedError(
            f"User '{user_id}' is not permitted to perform '{operation}'"
        )


# ---------------------------------------------------------------------------
# Validation helpers (R-001, R-002)
# ---------------------------------------------------------------------------
def _require_non_empty_str(value: Any, name: str) -> str:
    """Validate that ``value`` is a non-empty string.

    Args:
        value: The value to validate.
        name: Parameter name used in error messages.

    Returns:
        The validated string.

    Raises:
        ValidationError: If ``value`` is not a non-empty string.
    """
    if not isinstance(value, str):
        raise ValidationError(f"'{name}' must be a str, got {type(value).__name__}")
    if not value.strip():
        raise ValidationError(f"'{name}' must not be empty")
    return value


def _validate_items(items: List[Dict[str, Any]]) -> List[Dict[str, Any]]:
    """Validate a list of order items.

    Each item must be a mapping containing a ``name`` (non-empty str), a
    ``quantity`` (non-negative int) and a ``unit_price`` (non-negative number).

    Args:
        items: The list of item mappings to validate.

    Returns:
        The validated list of items.

    Raises:
        ValidationError: If the list or any contained item is invalid
            (R-001), including rejection of negative quantities (R-002).
    """
    if not isinstance(items, list):
        raise ValidationError(f"'items' must be a list, got {type(items).__name__}")
    if not items:
        raise ValidationError("'items' must contain at least one item")

    for index, item in enumerate(items):
        if not isinstance(item, dict):
            raise ValidationError(f"items[{index}] must be a dict")

        name = item.get("name")
        if not isinstance(name, str) or not name.strip():
            raise ValidationError(f"items[{index}].name must be a non-empty str")

        quantity = item.get("quantity")
        if not isinstance(quantity, int) or isinstance(quantity, bool):
            raise ValidationError(f"items[{index}].quantity must be an int")
        if quantity < 0:
            # R-002: negative quantities are rejected.
            raise ValidationError(f"items[{index}].quantity must not be negative")

        unit_price = item.get("unit_price")
        if isinstance(unit_price, bool) or not isinstance(unit_price, (int, float)):
            raise ValidationError(f"items[{index}].unit_price must be a number")
        if unit_price < 0:
            raise ValidationError(f"items[{index}].unit_price must not be negative")

    return items


def _calculate_total(items: List[Dict[str, Any]]) -> float:
    """Sum ``quantity * unit_price`` across validated items.

    Args:
        items: A list of already-validated item mappings.

    Returns:
        The total price as a float, rounded to two decimal places.
    """
    total = sum(item["quantity"] * item["unit_price"] for item in items)
    return round(float(total), 2)


# ---------------------------------------------------------------------------
# Public API
# ---------------------------------------------------------------------------
def create_order(
    user_id: str,
    items: List[Dict[str, Any]],
    payment_token: str,
    config: ServiceConfig = DEFAULT_CONFIG,
) -> Dict[str, Any]:
    """Create a new order, charge payment and return a confirmation.

    Args:
        user_id: Identifier of the user placing the order.
        items: List of item mappings, each with ``name``, ``quantity`` and
            ``unit_price``.
        payment_token: Tokenized payment instrument to charge.
        config: Service configuration (defaults to :data:`DEFAULT_CONFIG`).

    Returns:
        A confirmation dict with keys ``order_id``, ``user_id``, ``status``,
        ``items``, ``total``, ``currency`` and ``transaction_id``.

    Raises:
        ValidationError: If any parameter fails validation.
        PermissionDeniedError: If the caller is not permitted to create orders.
        PaymentError: If payment processing fails.
    """
    logger.debug(
        "Entering create_order(user_id=%s, items=%d, payment_token=***)",
        user_id,
        len(items) if isinstance(items, list) else 0,
    )  # R-013

    user_id = _require_non_empty_str(user_id, "user_id")
    payment_token = _require_non_empty_str(payment_token, "payment_token")
    validated_items = _validate_items(items)

    _check_permission(user_id, "create_order")  # R-020

    total = _calculate_total(validated_items)
    gateway = _PaymentGateway(config)

    try:
        transaction_id = gateway.charge(payment_token, total)
    except PaymentError as exc:
        # R-008: preserve the original error context when re-raising.
        raise PaymentError(f"Failed to process payment for user '{user_id}'") from exc

    order = {
        "order_id": f"ord_{uuid.uuid4().hex}",
        "user_id": user_id,
        "status": "CONFIRMED",
        "items": validated_items,
        "total": total,
        "currency": config.currency,
        "transaction_id": transaction_id,
    }
    _STORE.save(order)

    logger.debug("Exiting create_order -> %s", type(order).__name__)  # R-014
    return order


def cancel_order(
    user_id: str,
    order_id: str,
    reason: str,
    config: ServiceConfig = DEFAULT_CONFIG,
) -> Dict[str, Any]:
    """Cancel an existing order and issue a refund.

    Args:
        user_id: Identifier of the user requesting cancellation.
        order_id: Identifier of the order to cancel.
        reason: Free-text reason for the cancellation.
        config: Service configuration (defaults to :data:`DEFAULT_CONFIG`).

    Returns:
        A cancellation confirmation dict with keys ``order_id``, ``user_id``,
        ``status``, ``reason`` and ``refund_id``.

    Raises:
        ValidationError: If any parameter fails validation.
        PermissionDeniedError: If the caller is not permitted to cancel.
        OrderNotFoundError: If the order does not exist or belongs to another
            user.
        PaymentError: If the refund fails.
    """
    logger.debug(
        "Entering cancel_order(user_id=%s, order_id=%s, reason=%r)",
        user_id,
        order_id,
        reason,
    )  # R-013

    user_id = _require_non_empty_str(user_id, "user_id")
    order_id = _require_non_empty_str(order_id, "order_id")
    reason = _require_non_empty_str(reason, "reason")

    _check_permission(user_id, "cancel_order")  # R-020

    order = _STORE.get(order_id)
    if order is None or order["user_id"] != user_id:
        raise OrderNotFoundError(
            f"Order '{order_id}' not found for user '{user_id}'"
        )

    gateway = _PaymentGateway(config)
    try:
        refund_id = gateway.refund(order["transaction_id"], order["total"])
    except PaymentError as exc:
        # R-008: preserve original context.
        raise PaymentError(f"Failed to refund order '{order_id}'") from exc

    order["status"] = "CANCELLED"
    order["cancellation_reason"] = reason
    order["refund_id"] = refund_id
    _STORE.save(order)

    confirmation = {
        "order_id": order_id,
        "user_id": user_id,
        "status": "CANCELLED",
        "reason": reason,
        "refund_id": refund_id,
    }

    logger.debug("Exiting cancel_order -> %s", type(confirmation).__name__)  # R-014
    return confirmation


def get_order_status(
    user_id: str,
    order_id: str,
    config: ServiceConfig = DEFAULT_CONFIG,
) -> Dict[str, Any]:
    """Retrieve the status of an existing order.

    Args:
        user_id: Identifier of the user requesting the status.
        order_id: Identifier of the order to query.
        config: Service configuration (defaults to :data:`DEFAULT_CONFIG`).

    Returns:
        A status dict with keys ``order_id``, ``user_id``, ``status``,
        ``total`` and ``currency``.

    Raises:
        ValidationError: If any parameter fails validation.
        PermissionDeniedError: If the caller is not permitted to read the order.
        OrderNotFoundError: If the order does not exist or belongs to another
            user.
    """
    logger.debug(
        "Entering get_order_status(user_id=%s, order_id=%s)", user_id, order_id
    )  # R-013

    user_id = _require_non_empty_str(user_id, "user_id")
    order_id = _require_non_empty_str(order_id, "order_id")

    _check_permission(user_id, "get_order_status")  # R-020

    order = _STORE.get(order_id)
    if order is None or order["user_id"] != user_id:
        raise OrderNotFoundError(
            f"Order '{order_id}' not found for user '{user_id}'"
        )

    status = {
        "order_id": order["order_id"],
        "user_id": order["user_id"],
        "status": order["status"],
        "total": order["total"],
        "currency": order["currency"],
    }

    logger.debug("Exiting get_order_status -> %s", type(status).__name__)  # R-014
    return status
```

```python
"""Unit tests for :mod:`order_service`."""

import unittest

import order_service as svc


def _sample_items():
    """Return a fresh list of valid sample items."""
    return [
        {"name": "Widget", "quantity": 2, "unit_price": 9.99},
        {"name": "Gadget", "quantity": 1, "unit_price": 19.00},
    ]


class CreateOrderTests(unittest.TestCase):
    def test_create_order_success(self):
        order = svc.create_order("user-1", _sample_items(), "tok_valid")
        self.assertEqual(order["status"], "CONFIRMED")
        self.assertEqual(order["user_id"], "user-1")
        self.assertEqual(order["total"], round(2 * 9.99 + 19.00, 2))
        self.assertTrue(order["order_id"].startswith("ord_"))
        self.assertTrue(order["transaction_id"].startswith("txn_"))

    def test_unique_order_ids(self):
        a = svc.create_order("user-1", _sample_items(), "tok_valid")
        b = svc.create_order("user-1", _sample_items(), "tok_valid")
        self.assertNotEqual(a["order_id"], b["order_id"])

    def test_rejects_non_string_user_id(self):
        with self.assertRaises(svc.ValidationError):
            svc.create_order(123, _sample_items(), "tok_valid")

    def test_rejects_empty_payment_token(self):
        with self.assertRaises(svc.ValidationError):
            svc.create_order("user-1", _sample_items(), "")

    def test_rejects_empty_items(self):
        with self.assertRaises(svc.ValidationError):
            svc.create_order("user-1", [], "tok_valid")

    def test_rejects_negative_quantity(self):
        items = [{"name": "Widget", "quantity": -1, "unit_price": 9.99}]
        with self.assertRaises(svc.ValidationError):
            svc.create_order("user-1", items, "tok_valid")

    def test_rejects_negative_unit_price(self):
        items = [{"name": "Widget", "quantity": 1, "unit_price": -5}]
        with self.assertRaises(svc.ValidationError):
            svc.create_order("user-1", items, "tok_valid")

    def test_permission_denied(self):
        with self.assertRaises(svc.PermissionDeniedError):
            svc.create_order("anonymous", _sample_items(), "tok_valid")

    def test_payment_failure_wraps_context(self):
        with self.assertRaises(svc.PaymentError) as ctx:
            svc.create_order("user-1", _sample_items(), "invalid")
        self.assertIsNotNone(ctx.exception.__cause__)


class CancelOrderTests(unittest.TestCase):
    def setUp(self):
        self.order = svc.create_order("user-1", _sample_items(), "tok_valid")

    def test_cancel_success(self):
        result = svc.cancel_order("user-1", self.order["order_id"], "changed mind")
        self.assertEqual(result["status"], "CANCELLED")
        self.assertEqual(result["reason"], "changed mind")
        self.assertTrue(result["refund_id"].startswith("rfnd_"))

    def test_cancel_unknown_order(self):
        with self.assertRaises(svc.OrderNotFoundError):
            svc.cancel_order("user-1", "ord_missing", "reason")

    def test_cancel_other_users_order(self):
        with self.assertRaises(svc.OrderNotFoundError):
            svc.cancel_order("user-2", self.order["order_id"], "reason")

    def test_cancel_requires_reason(self):
        with self.assertRaises(svc.ValidationError):
            svc.cancel_order("user-1", self.order["order_id"], "")

    def test_cancel_permission_denied(self):
        with self.assertRaises(svc.PermissionDeniedError):
            svc.cancel_order("anonymous", self.order["order_id"], "reason")


class GetOrderStatusTests(unittest.TestCase):
    def setUp(self):
        self.order = svc.create_order("user-1", _sample_items(), "tok_valid")

    def test_status_success(self):
        status = svc.get_order_status("user-1", self.order["order_id"])
        self.assertEqual(status["order_id"], self.order["order_id"])
        self.assertEqual(status["status"], "CONFIRMED")
        self.assertEqual(status["currency"], "USD")

    def test_status_reflects_cancellation(self):
        svc.cancel_order("user-1", self.order["order_id"], "reason")
        status = svc.get_order_status("user-1", self.order["order_id"])
        self.assertEqual(status["status"], "CANCELLED")

    def test_status_unknown_order(self):
        with self.assertRaises(svc.OrderNotFoundError):
            svc.get_order_status("user-1", "ord_missing")

    def test_status_invalid_order_id(self):
        with self.assertRaises(svc.ValidationError):
            svc.get_order_status("user-1", "")

    def test_status_permission_denied(self):
        with self.assertRaises(svc.PermissionDeniedError):
            svc.get_order_status("anonymous", self.order["order_id"])


if __name__ == "__main__":
    unittest.main()
```
