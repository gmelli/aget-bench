"""Order management service.

Provides order creation, cancellation, and status retrieval backed by a
payment gateway. All configuration is externalized into :class:`ServiceConfig`
(R-019) and all privileged operations verify caller permissions (R-020).
"""

from __future__ import annotations

import logging
import uuid
from dataclasses import dataclass
from typing import Any, Dict, List, Optional

logger = logging.getLogger(__name__)


# ---------------------------------------------------------------------------
# Configuration (R-019)
# ---------------------------------------------------------------------------
@dataclass
class ServiceConfig:
    """Externalized configuration for the order service.

    Attributes:
        database_url: Connection string for the order store.
        payment_api_key: API key used to authenticate with the payment gateway.
        rate_limit_per_minute: Maximum number of requests permitted per minute.
        max_items_per_order: Upper bound on the number of line items per order.
        currency: ISO currency code used for all monetary amounts.
    """

    database_url: str = "postgresql://localhost:5432/orders"
    payment_api_key: str = "REPLACE_ME"
    rate_limit_per_minute: int = 60
    max_items_per_order: int = 100
    currency: str = "USD"


# Default configuration instance; callers may override by passing their own.
DEFAULT_CONFIG = ServiceConfig()


# ---------------------------------------------------------------------------
# Domain-specific exceptions (R-007)
# ---------------------------------------------------------------------------
class OrderServiceError(Exception):
    """Base class for all order service errors."""


class ValidationError(OrderServiceError):
    """Raised when input parameters fail validation."""


class PermissionDeniedError(OrderServiceError):
    """Raised when a caller lacks permission for a privileged operation."""


class PaymentError(OrderServiceError):
    """Raised when a payment or refund operation fails."""


class OrderNotFoundError(OrderServiceError):
    """Raised when a requested order does not exist."""


# ---------------------------------------------------------------------------
# In-memory order store (stands in for the database referenced by config)
# ---------------------------------------------------------------------------
_ORDERS: Dict[str, Dict[str, Any]] = {}


# ---------------------------------------------------------------------------
# Internal helpers
# ---------------------------------------------------------------------------
def _require_non_empty_str(value: Any, name: str) -> str:
    """Validate that ``value`` is a non-empty string (R-001).

    Args:
        value: The value to validate.
        name: The parameter name, used in error messages.

    Returns:
        The validated string.

    Raises:
        ValidationError: If ``value`` is not a non-empty string.
    """
    if not isinstance(value, str):
        raise ValidationError(f"{name} must be a str, got {type(value).__name__}")
    if not value.strip():
        raise ValidationError(f"{name} must not be empty")
    return value


def _check_permission(user_id: str, order: Optional[Dict[str, Any]] = None) -> None:
    """Verify the caller may operate on the order (R-020).

    Args:
        user_id: The identifier of the calling user.
        order: The order being accessed, if one exists. When provided, the
            order's owner must match ``user_id``.

    Raises:
        PermissionDeniedError: If the caller is not permitted.
    """
    if order is not None and order.get("user_id") != user_id:
        raise PermissionDeniedError(
            f"User {user_id!r} is not permitted to access order {order.get('order_id')!r}"
        )


def _calculate_total(items: List[Dict[str, Any]], config: ServiceConfig) -> float:
    """Validate line items and compute the order total.

    Args:
        items: List of item dicts, each with ``name``, ``quantity`` and
            ``unit_price`` keys.
        config: Service configuration providing item limits.

    Returns:
        The total price across all items.

    Raises:
        ValidationError: If items are malformed, exceed limits, or contain
            negative quantities (R-001, R-002).
    """
    if not isinstance(items, list):
        raise ValidationError(f"items must be a list, got {type(items).__name__}")
    if not items:
        raise ValidationError("items must not be empty")
    if len(items) > config.max_items_per_order:
        raise ValidationError(
            f"items exceeds max_items_per_order ({config.max_items_per_order})"
        )

    total = 0.0
    for index, item in enumerate(items):
        if not isinstance(item, dict):
            raise ValidationError(f"items[{index}] must be a dict")

        quantity = item.get("quantity")
        unit_price = item.get("unit_price")

        if not isinstance(quantity, int):
            raise ValidationError(
                f"items[{index}].quantity must be an int, "
                f"got {type(quantity).__name__}"
            )
        if not isinstance(unit_price, (int, float)):
            raise ValidationError(
                f"items[{index}].unit_price must be a number, "
                f"got {type(unit_price).__name__}"
            )
        # R-002: reject negative quantities.
        if quantity < 0:
            raise ValidationError(f"items[{index}].quantity must not be negative")
        if unit_price < 0:
            raise ValidationError(f"items[{index}].unit_price must not be negative")

        total += quantity * float(unit_price)

    return total


def _process_payment(payment_token: str, amount: float, config: ServiceConfig) -> str:
    """Charge the payment token via the payment gateway.

    Args:
        payment_token: Tokenized payment instrument.
        amount: Amount to charge.
        config: Service configuration providing the payment API key.

    Returns:
        A transaction identifier.

    Raises:
        PaymentError: If the gateway rejects the charge.
    """
    try:
        # Placeholder for a real payment gateway call using config.payment_api_key.
        if amount < 0:
            raise ValueError("amount must not be negative")
        return f"txn_{uuid.uuid4().hex}"
    except Exception as exc:  # noqa: BLE001 - re-raised with context below.
        # R-008: preserve the original error context via ``from exc``.
        raise PaymentError(f"Payment failed for amount {amount}: {exc}") from exc


def _issue_refund(transaction_id: str, amount: float, config: ServiceConfig) -> str:
    """Issue a refund for a prior transaction.

    Args:
        transaction_id: The original payment transaction identifier.
        amount: Amount to refund.
        config: Service configuration providing the payment API key.

    Returns:
        A refund identifier.

    Raises:
        PaymentError: If the gateway rejects the refund.
    """
    try:
        if amount < 0:
            raise ValueError("amount must not be negative")
        return f"rfnd_{uuid.uuid4().hex}"
    except Exception as exc:  # noqa: BLE001 - re-raised with context below.
        # R-008: preserve the original error context.
        raise PaymentError(
            f"Refund failed for transaction {transaction_id!r}: {exc}"
        ) from exc


# ---------------------------------------------------------------------------
# Public API
# ---------------------------------------------------------------------------
def create_order(
    user_id: str,
    items: List[Dict[str, Any]],
    payment_token: str,
    config: ServiceConfig = DEFAULT_CONFIG,
) -> Dict[str, Any]:
    """Create a new order, charge payment, and return a confirmation.

    Args:
        user_id: Identifier of the user placing the order.
        items: Line items, each a dict with ``name``, ``quantity`` and
            ``unit_price`` keys.
        payment_token: Tokenized payment instrument used to charge the order.
        config: Service configuration. Defaults to :data:`DEFAULT_CONFIG`.

    Returns:
        An order confirmation dict containing ``order_id``, ``user_id``,
        ``status``, ``total``, ``currency``, ``transaction_id`` and ``items``.

    Raises:
        ValidationError: If any parameter fails validation (R-001, R-002).
        PaymentError: If the payment cannot be processed.
    """
    # R-013: log entry with parameter summary.
    logger.debug(
        "create_order(user_id=%s, item_count=%s, payment_token=***)",
        user_id,
        len(items) if isinstance(items, list) else "invalid",
    )

    _require_non_empty_str(user_id, "user_id")
    _require_non_empty_str(payment_token, "payment_token")
    _check_permission(user_id)

    total = _calculate_total(items, config)
    transaction_id = _process_payment(payment_token, total, config)

    order_id = f"ord_{uuid.uuid4().hex}"
    order: Dict[str, Any] = {
        "order_id": order_id,
        "user_id": user_id,
        "status": "confirmed",
        "total": total,
        "currency": config.currency,
        "transaction_id": transaction_id,
        "items": items,
    }
    _ORDERS[order_id] = order

    confirmation = dict(order)
    # R-014: log exit with return value type.
    logger.debug("create_order -> %s", type(confirmation).__name__)
    return confirmation


def cancel_order(
    user_id: str,
    order_id: str,
    reason: str,
    config: ServiceConfig = DEFAULT_CONFIG,
) -> Dict[str, Any]:
    """Cancel an existing order and issue a refund.

    Args:
        user_id: Identifier of the user requesting cancellation.
        order_id: Identifier of the order to cancel.
        reason: Human-readable cancellation reason.
        config: Service configuration. Defaults to :data:`DEFAULT_CONFIG`.

    Returns:
        A cancellation confirmation dict containing ``order_id``, ``status``,
        ``reason``, ``refund_id`` and ``refunded_amount``.

    Raises:
        ValidationError: If any parameter fails validation (R-001).
        PermissionDeniedError: If the caller does not own the order (R-020).
        OrderNotFoundError: If the order does not exist.
        PaymentError: If the refund cannot be processed.
    """
    # R-013: log entry with parameter summary.
    logger.debug(
        "cancel_order(user_id=%s, order_id=%s, reason=%s)",
        user_id,
        order_id,
        reason,
    )

    _require_non_empty_str(user_id, "user_id")
    _require_non_empty_str(order_id, "order_id")
    _require_non_empty_str(reason, "reason")

    order = _ORDERS.get(order_id)
    if order is None:
        raise OrderNotFoundError(f"Order {order_id!r} not found")

    # R-020: verify the caller owns the order before refunding.
    _check_permission(user_id, order)

    refund_id = _issue_refund(order["transaction_id"], order["total"], config)
    order["status"] = "cancelled"
    order["cancellation_reason"] = reason

    confirmation: Dict[str, Any] = {
        "order_id": order_id,
        "status": "cancelled",
        "reason": reason,
        "refund_id": refund_id,
        "refunded_amount": order["total"],
        "currency": config.currency,
    }
    # R-014: log exit with return value type.
    logger.debug("cancel_order -> %s", type(confirmation).__name__)
    return confirmation


def get_order_status(
    user_id: str,
    order_id: str,
    config: ServiceConfig = DEFAULT_CONFIG,
) -> Dict[str, Any]:
    """Retrieve the status and details of an order.

    Args:
        user_id: Identifier of the user requesting the order.
        order_id: Identifier of the order to retrieve.
        config: Service configuration. Defaults to :data:`DEFAULT_CONFIG`.

    Returns:
        An order status dict containing ``order_id``, ``user_id``, ``status``,
        ``total``, ``currency`` and ``items``.

    Raises:
        ValidationError: If any parameter fails validation (R-001).
        PermissionDeniedError: If the caller does not own the order (R-020).
        OrderNotFoundError: If the order does not exist.
    """
    # R-013: log entry with parameter summary.
    logger.debug("get_order_status(user_id=%s, order_id=%s)", user_id, order_id)

    _require_non_empty_str(user_id, "user_id")
    _require_non_empty_str(order_id, "order_id")

    order = _ORDERS.get(order_id)
    if order is None:
        raise OrderNotFoundError(f"Order {order_id!r} not found")

    # R-020: verify the caller owns the order before disclosing details.
    _check_permission(user_id, order)

    status: Dict[str, Any] = {
        "order_id": order["order_id"],
        "user_id": order["user_id"],
        "status": order["status"],
        "total": order["total"],
        "currency": config.currency,
        "items": order["items"],
    }
    # R-014: log exit with return value type.
    logger.debug("get_order_status -> %s", type(status).__name__)
    return status

"""Unit tests for order_service."""

import unittest

import order_service
from order_service import (
    OrderNotFoundError,
    PermissionDeniedError,
    ServiceConfig,
    ValidationError,
    cancel_order,
    create_order,
    get_order_status,
)


def _items():
    return [
        {"name": "widget", "quantity": 2, "unit_price": 10.0},
        {"name": "gadget", "quantity": 1, "unit_price": 5.5},
    ]


class CreateOrderTests(unittest.TestCase):
    def setUp(self):
        order_service._ORDERS.clear()

    def test_create_order_returns_confirmation(self):
        result = create_order("user-1", _items(), "tok_abc")
        self.assertEqual(result["status"], "confirmed")
        self.assertEqual(result["user_id"], "user-1")
        self.assertEqual(result["total"], 25.5)
        self.assertTrue(result["order_id"].startswith("ord_"))
        self.assertTrue(result["transaction_id"].startswith("txn_"))

    def test_create_order_rejects_negative_quantity(self):
        items = [{"name": "widget", "quantity": -1, "unit_price": 10.0}]
        with self.assertRaises(ValidationError):
            create_order("user-1", items, "tok_abc")

    def test_create_order_rejects_bad_user_id_type(self):
        with self.assertRaises(ValidationError):
            create_order(123, _items(), "tok_abc")

    def test_create_order_rejects_empty_payment_token(self):
        with self.assertRaises(ValidationError):
            create_order("user-1", _items(), "  ")

    def test_create_order_rejects_empty_items(self):
        with self.assertRaises(ValidationError):
            create_order("user-1", [], "tok_abc")

    def test_create_order_rejects_non_int_quantity(self):
        items = [{"name": "widget", "quantity": 2.0, "unit_price": 10.0}]
        with self.assertRaises(ValidationError):
            create_order("user-1", items, "tok_abc")

    def test_create_order_respects_max_items_config(self):
        config = ServiceConfig(max_items_per_order=1)
        with self.assertRaises(ValidationError):
            create_order("user-1", _items(), "tok_abc", config=config)


class CancelOrderTests(unittest.TestCase):
    def setUp(self):
        order_service._ORDERS.clear()
        self.order = create_order("user-1", _items(), "tok_abc")

    def test_cancel_order_returns_confirmation(self):
        result = cancel_order("user-1", self.order["order_id"], "changed mind")
        self.assertEqual(result["status"], "cancelled")
        self.assertEqual(result["reason"], "changed mind")
        self.assertEqual(result["refunded_amount"], 25.5)
        self.assertTrue(result["refund_id"].startswith("rfnd_"))

    def test_cancel_order_unknown_order_raises(self):
        with self.assertRaises(OrderNotFoundError):
            cancel_order("user-1", "ord_missing", "reason")

    def test_cancel_order_wrong_user_denied(self):
        with self.assertRaises(PermissionDeniedError):
            cancel_order("user-2", self.order["order_id"], "reason")

    def test_cancel_order_requires_reason(self):
        with self.assertRaises(ValidationError):
            cancel_order("user-1", self.order["order_id"], "")


class GetOrderStatusTests(unittest.TestCase):
    def setUp(self):
        order_service._ORDERS.clear()
        self.order = create_order("user-1", _items(), "tok_abc")

    def test_get_order_status_returns_details(self):
        result = get_order_status("user-1", self.order["order_id"])
        self.assertEqual(result["order_id"], self.order["order_id"])
        self.assertEqual(result["status"], "confirmed")
        self.assertEqual(result["total"], 25.5)

    def test_get_order_status_reflects_cancellation(self):
        cancel_order("user-1", self.order["order_id"], "reason")
        result = get_order_status("user-1", self.order["order_id"])
        self.assertEqual(result["status"], "cancelled")

    def test_get_order_status_unknown_raises(self):
        with self.assertRaises(OrderNotFoundError):
            get_order_status("user-1", "ord_missing")

    def test_get_order_status_wrong_user_denied(self):
        with self.assertRaises(PermissionDeniedError):
            get_order_status("user-2", self.order["order_id"])


if __name__ == "__main__":
    unittest.main()
