Provides order creation, cancellation, and status retrieval backed by a
pluggable configuration object, a payment gateway, and a permission checker.

This module adheres to the mandatory agent rules:
  * R-001/R-002 input validation, R-007/R-008 domain exceptions with context,
  * R-013/R-014 debug logging, R-019 externalized config, R-020 permission
    checks, R-025 type hints, and R-026 docstrings.
"""

from __future__ import annotations

import logging
import uuid
from dataclasses import dataclass
from typing import Any, Callable, Dict, List

logger = logging.getLogger(__name__)


# --------------------------------------------------------------------------- #
# R-019: Externalized configuration
# --------------------------------------------------------------------------- #
@dataclass
class ServiceConfig:
    """Service configuration container.

    Attributes:
        database_url: Connection string for the order datastore.
        payment_api_key: API key used to authenticate with the payment gateway.
        rate_limit_per_minute: Maximum number of operations allowed per minute.
        currency: ISO currency code applied to order totals.
    """

    database_url: str = "sqlite:///orders.db"
    payment_api_key: str = "test-key"
    rate_limit_per_minute: int = 60
    currency: str = "USD"


# --------------------------------------------------------------------------- #
# R-007: Domain-specific exception classes
# --------------------------------------------------------------------------- #
class OrderServiceError(Exception):
    """Base class for all order service errors."""


class ValidationError(OrderServiceError):
    """Raised when input validation fails (R-001/R-002)."""


class PermissionDeniedError(OrderServiceError):
    """Raised when the caller lacks permission for a privileged operation."""


class OrderNotFoundError(OrderServiceError):
    """Raised when a requested order does not exist."""


class PaymentError(OrderServiceError):
    """Raised when a payment or refund operation fails."""


# --------------------------------------------------------------------------- #
# Collaborators (injected so the service stays testable / configurable)
# --------------------------------------------------------------------------- #
class PaymentGateway:
    """Minimal payment gateway abstraction.

    A real implementation would call an external API; this default raises so
    that callers must supply a working gateway in production.
    """

    def __init__(self, api_key: str) -> None:
        """Initialize the gateway.

        Args:
            api_key: API key used to authenticate requests.
        """
        self.api_key: str = api_key

    def charge(self, payment_token: str, amount: float, currency: str) -> str:
        """Charge a payment token.

        Args:
            payment_token: Tokenized payment instrument.
            amount: Positive amount to charge.
            currency: ISO currency code.

        Returns:
            A transaction identifier string.

        Raises:
            PaymentError: If the charge cannot be completed.
        """
        raise NotImplementedError("Provide a concrete PaymentGateway.charge")

    def refund(self, transaction_id: str, amount: float) -> str:
        """Refund a previously captured transaction.

        Args:
            transaction_id: Identifier returned by :meth:`charge`.
            amount: Positive amount to refund.

        Returns:
            A refund identifier string.

        Raises:
            PaymentError: If the refund cannot be completed.
        """
        raise NotImplementedError("Provide a concrete PaymentGateway.refund")


class OrderService:
    """Order management service.

    Encapsulates order lifecycle operations against an in-memory store while
    delegating configuration, payment, and authorization to injected
    collaborators.
    """

    def __init__(
        self,
        config: ServiceConfig,
        payment_gateway: PaymentGateway,
        permission_checker: Callable[[str, str], bool],
    ) -> None:
        """Initialize the service.

        Args:
            config: Externalized service configuration (R-019).
            payment_gateway: Gateway used to charge and refund payments.
            permission_checker: Callable returning True if a user may perform
                the named action (R-020). Signature: (user_id, action) -> bool.

        Returns:
            None.

        Raises:
            ValidationError: If any collaborator is of an unexpected type.
        """
        if not isinstance(config, ServiceConfig):
            raise ValidationError("config must be a ServiceConfig instance")
        if not isinstance(payment_gateway, PaymentGateway):
            raise ValidationError("payment_gateway must be a PaymentGateway")
        if not callable(permission_checker):
            raise ValidationError("permission_checker must be callable")

        self._config: ServiceConfig = config
        self._gateway: PaymentGateway = payment_gateway
        self._permits: Callable[[str, str], bool] = permission_checker
        self._orders: Dict[str, Dict[str, Any]] = {}

    # ----------------------------------------------------------------- #
    # Internal helpers
    # ----------------------------------------------------------------- #
    @staticmethod
    def _require_non_empty_str(value: Any, name: str) -> None:
        """Validate that ``value`` is a non-empty string (R-001).

        Args:
            value: The value to check.
            name: Parameter name used in the error message.

        Returns:
            None.

        Raises:
            ValidationError: If ``value`` is not a non-empty string.
        """
        if not isinstance(value, str):
            raise ValidationError(
                f"{name} must be a str, got {type(value).__name__}"
            )
        if not value.strip():
            raise ValidationError(f"{name} must not be empty")

    def _authorize(self, user_id: str, action: str) -> None:
        """Verify caller permissions before a privileged operation (R-020).

        Args:
            user_id: Identifier of the calling user.
            action: Privileged action being attempted.

        Returns:
            None.

        Raises:
            PermissionDeniedError: If the user is not permitted.
        """
        if not self._permits(user_id, action):
            raise PermissionDeniedError(
                f"user {user_id!r} is not permitted to {action}"
            )

    def _calculate_total(self, items: List[Dict[str, Any]]) -> float:
        """Calculate the order total from line items (R-001/R-002).

        Args:
            items: List of dicts, each with name, quantity, and unit_price.

        Returns:
            The summed total price as a float.

        Raises:
            ValidationError: If items is malformed or contains negative values.
        """
        if not isinstance(items, list):
            raise ValidationError(
                f"items must be a list, got {type(items).__name__}"
            )
        if not items:
            raise ValidationError("items must not be empty")

        total: float = 0.0
        for index, item in enumerate(items):
            if not isinstance(item, dict):
                raise ValidationError(f"items[{index}] must be a dict")

            name = item.get("name")
            quantity = item.get("quantity")
            unit_price = item.get("unit_price")

            self._require_non_empty_str(name, f"items[{index}].name")

            if not isinstance(quantity, int) or isinstance(quantity, bool):
                raise ValidationError(f"items[{index}].quantity must be an int")
            if not isinstance(unit_price, (int, float)) or isinstance(
                unit_price, bool
            ):
                raise ValidationError(
                    f"items[{index}].unit_price must be a number"
                )

            # R-002: reject negative quantities (and negative prices).
            if quantity < 0:
                raise ValidationError(
                    f"items[{index}].quantity must not be negative"
                )
            if unit_price < 0:
                raise ValidationError(
                    f"items[{index}].unit_price must not be negative"
                )

            total += quantity * unit_price

        return float(total)

    # ----------------------------------------------------------------- #
    # Public API
    # ----------------------------------------------------------------- #
    def create_order(
        self, user_id: str, items: List[Dict[str, Any]], payment_token: str
    ) -> Dict[str, Any]:
        """Create a new order, charge payment, and return a confirmation.

        Args:
            user_id: Identifier of the user placing the order.
            items: Line items, each a dict with name, quantity, unit_price.
            payment_token: Tokenized payment instrument.

        Returns:
            A confirmation dict with order_id, status, total, currency, and
            transaction_id.

        Raises:
            ValidationError: If any parameter fails validation.
            PermissionDeniedError: If the user may not create orders.
            PaymentError: If the payment cannot be processed.
        """
        logger.debug(
            "create_order entry: user_id=%s, item_count=%s, has_token=%s",
            user_id,
            len(items) if isinstance(items, list) else "n/a",
            bool(payment_token),
        )  # R-013

        # R-001: validate parameters.
        self._require_non_empty_str(user_id, "user_id")
        self._require_non_empty_str(payment_token, "payment_token")
        total = self._calculate_total(items)

        # R-020: verify permission before charging.
        self._authorize(user_id, "create_order")

        try:
            transaction_id = self._gateway.charge(
                payment_token, total, self._config.currency
            )
        except PaymentError:
            raise
        except Exception as exc:  # R-008: preserve original context.
            raise PaymentError(f"payment failed for user {user_id}") from exc

        order_id = str(uuid.uuid4())
        order: Dict[str, Any] = {
            "order_id": order_id,
            "user_id": user_id,
            "items": list(items),
            "total": total,
            "currency": self._config.currency,
            "status": "confirmed",
            "transaction_id": transaction_id,
        }
        self._orders[order_id] = order

        confirmation: Dict[str, Any] = {
            "order_id": order_id,
            "status": "confirmed",
            "total": total,
            "currency": self._config.currency,
            "transaction_id": transaction_id,
        }

        logger.debug(
            "create_order exit: return type=%s", type(confirmation).__name__
        )  # R-014
        return confirmation

    def cancel_order(
        self, user_id: str, order_id: str, reason: str
    ) -> Dict[str, Any]:
        """Cancel an existing order and issue a refund.

        Args:
            user_id: Identifier of the user cancelling the order.
            order_id: Identifier of the order to cancel.
            reason: Human-readable cancellation reason.

        Returns:
            A cancellation confirmation dict with order_id, status, refund_id,
            and reason.

        Raises:
            ValidationError: If any parameter fails validation.
            PermissionDeniedError: If the user may not cancel this order.
            OrderNotFoundError: If the order does not exist.
            PaymentError: If the refund cannot be processed.
        """
        logger.debug(
            "cancel_order entry: user_id=%s, order_id=%s, reason=%s",
            user_id,
            order_id,
            reason,
        )  # R-013

        # R-001: validate parameters.
        self._require_non_empty_str(user_id, "user_id")
        self._require_non_empty_str(order_id, "order_id")
        self._require_non_empty_str(reason, "reason")

        order = self._orders.get(order_id)
        if order is None:
            raise OrderNotFoundError(f"order {order_id!r} not found")

        # R-020: verify permission, then confirm ownership.
        self._authorize(user_id, "cancel_order")
        if order["user_id"] != user_id:
            raise PermissionDeniedError(
                f"user {user_id!r} does not own order {order_id!r}"
            )

        if order["status"] == "cancelled":
            raise ValidationError(f"order {order_id!r} is already cancelled")

        try:
            refund_id = self._gateway.refund(
                order["transaction_id"], order["total"]
            )
        except PaymentError:
            raise
        except Exception as exc:  # R-008: preserve original context.
            raise PaymentError(f"refund failed for order {order_id}") from exc

        order["status"] = "cancelled"
        order["cancel_reason"] = reason
        order["refund_id"] = refund_id

        confirmation: Dict[str, Any] = {
            "order_id": order_id,
            "status": "cancelled",
            "refund_id": refund_id,
            "reason": reason,
        }

        logger.debug(
            "cancel_order exit: return type=%s", type(confirmation).__name__
        )  # R-014
        return confirmation

    def get_order_status(self, user_id: str, order_id: str) -> Dict[str, Any]:
        """Retrieve the status and summary of an order.

        Args:
            user_id: Identifier of the requesting user.
            order_id: Identifier of the order to look up.

        Returns:
            A status dict with order_id, status, total, and currency.

        Raises:
            ValidationError: If any parameter fails validation.
            PermissionDeniedError: If the user may not view this order.
            OrderNotFoundError: If the order does not exist.
        """
        logger.debug(
            "get_order_status entry: user_id=%s, order_id=%s",
            user_id,
            order_id,
        )  # R-013

        # R-001: validate parameters.
        self._require_non_empty_str(user_id, "user_id")
        self._require_non_empty_str(order_id, "order_id")

        order = self._orders.get(order_id)
        if order is None:
            raise OrderNotFoundError(f"order {order_id!r} not found")

        # R-020: verify permission, then confirm ownership.
        self._authorize(user_id, "get_order_status")
        if order["user_id"] != user_id:
            raise PermissionDeniedError(
                f"user {user_id!r} does not own order {order_id!r}"
            )

        status: Dict[str, Any] = {
            "order_id": order_id,
            "status": order["status"],
            "total": order["total"],
            "currency": order["currency"],
        }

        logger.debug(
            "get_order_status exit: return type=%s", type(status).__name__
        )  # R-014
        return status


# --------------------------------------------------------------------------- #
# Module-level convenience wrappers backed by a default service

def _default_service() -> OrderService:
    """Build an OrderService with permissive defaults for convenience use.

    Returns:
        A configured :class:`OrderService` instance.

    Raises:
        OrderServiceError: If construction fails.
    """
    config = ServiceConfig()

    class _DefaultGateway(PaymentGateway):
        def charge(
            self, payment_token: str, amount: float, currency: str
        ) -> str:
            return f"txn-{uuid.uuid4()}"

        def refund(self, transaction_id: str, amount: float) -> str:

    return OrderService(
        config=config,
        payment_gateway=_DefaultGateway(config.payment_api_key),
        permission_checker=lambda user_id, action: True,

"""Unit tests for order_service."""

import unittest

from order_service import (
    OrderNotFoundError,
    OrderService,
    PaymentError,
    PaymentGateway,
    PermissionDeniedError,
    ServiceConfig,
    ValidationError,
)


class FakeGateway(PaymentGateway):
    """In-memory payment gateway for tests."""

    def __init__(self, api_key: str = "test", fail: bool = False) -> None:
        super().__init__(api_key)
        self.fail = fail
        self.charges = []
        self.refunds = []

    def charge(self, payment_token: str, amount: float, currency: str) -> str:
        if self.fail:
            raise RuntimeError("gateway down")
        self.charges.append((payment_token, amount, currency))
        return "txn-123"

    def refund(self, transaction_id: str, amount: float) -> str:
        if self.fail:
            raise RuntimeError("gateway down")
        self.refunds.append((transaction_id, amount))
        return "rfnd-123"


def make_service(allow: bool = True, gateway: PaymentGateway = None) -> OrderService:
    return OrderService(
        config=ServiceConfig(),
        payment_gateway=gateway or FakeGateway(),
        permission_checker=lambda user_id, action: allow,
    )


VALID_ITEMS = [
    {"name": "Widget", "quantity": 2, "unit_price": 5.0},
    {"name": "Gadget", "quantity": 1, "unit_price": 3.5},
]


class CreateOrderTests(unittest.TestCase):
    def test_create_order_success(self):
        svc = make_service()
        result = svc.create_order("u1", VALID_ITEMS, "tok")
        self.assertEqual(result["status"], "confirmed")
        self.assertEqual(result["total"], 13.5)
        self.assertEqual(result["transaction_id"], "txn-123")
        self.assertIn("order_id", result)

    def test_create_order_rejects_bad_user_id(self):
        svc = make_service()
        with self.assertRaises(ValidationError):
            svc.create_order("", VALID_ITEMS, "tok")
        with self.assertRaises(ValidationError):
            svc.create_order(123, VALID_ITEMS, "tok")

    def test_create_order_rejects_empty_token(self):
        svc = make_service()

            svc.create_order("u1", VALID_ITEMS, "")

    def test_create_order_rejects_negative_quantity(self):
        svc = make_service()

    def test_create_order_rejects_negative_price(self):
        svc = make_service()

    def test_create_order_rejects_empty_items(self):
