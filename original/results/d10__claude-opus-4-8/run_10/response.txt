```python
"""Order management service.

This module provides functions to create, cancel, and query orders. It
integrates with a payment gateway and an order data store, both abstracted
behind small adapter classes so they can be substituted in tests.

All public functions follow the mandatory agent rules: parameters are
validated against expected types, quantities may not be negative, domain
specific exceptions are raised with preserved context, function entry/exit is
logged at DEBUG level, configuration is externalized into a config object, and
caller permissions are verified before privileged operations.
"""

from __future__ import annotations

import logging
import uuid
from dataclasses import dataclass, field
from typing import Any, Callable, Optional

logger = logging.getLogger(__name__)


# ---------------------------------------------------------------------------
# Configuration (R-019)
# ---------------------------------------------------------------------------
@dataclass(frozen=True)
class ServiceConfig:
    """Externalized configuration for the order service.

    Attributes:
        database_url: Connection string for the order data store.
        payment_api_key: API key used to authenticate with the payment gateway.
        max_orders_per_minute: Rate limit for order creation per user.
        max_items_per_order: Maximum number of distinct line items per order.
        default_currency: ISO currency code applied to monetary amounts.
    """

    database_url: str = "sqlite:///orders.db"
    payment_api_key: str = "test-payment-api-key"
    max_orders_per_minute: int = 60
    max_items_per_order: int = 100
    default_currency: str = "USD"


# ---------------------------------------------------------------------------
# Domain-specific exceptions (R-007)
# ---------------------------------------------------------------------------
class OrderServiceError(Exception):
    """Base class for all order service errors."""


class ValidationError(OrderServiceError):
    """Raised when a parameter fails type or value validation."""


class PermissionDeniedError(OrderServiceError):
    """Raised when the caller is not permitted to perform an operation."""


class PaymentError(OrderServiceError):
    """Raised when payment processing or refunding fails."""


class OrderNotFoundError(OrderServiceError):
    """Raised when a referenced order does not exist."""


class OrderStateError(OrderServiceError):
    """Raised when an operation is invalid for the order's current state."""


# ---------------------------------------------------------------------------
# Collaborators (data store, payment gateway, permission checker)
# ---------------------------------------------------------------------------
class OrderRepository:
    """In-memory order data store.

    A real implementation would talk to the database identified by
    ``config.database_url``; this in-memory version keeps the module
    self-contained and testable.
    """

    def __init__(self, config: ServiceConfig) -> None:
        """Initialize the repository.

        Args:
            config: Service configuration providing the database URL.
        """
        self._config = config
        self._orders: dict[str, dict[str, Any]] = {}

    def save(self, order: dict[str, Any]) -> None:
        """Persist an order record.

        Args:
            order: The order record to store. Must contain an ``order_id``.
        """
        self._orders[order["order_id"]] = order

    def get(self, order_id: str) -> Optional[dict[str, Any]]:
        """Retrieve an order record by ID.

        Args:
            order_id: The unique order identifier.

        Returns:
            The stored order record, or ``None`` if no such order exists.
        """
        return self._orders.get(order_id)


class PaymentGateway:
    """Stub payment gateway.

    A real implementation would call an external payment provider using
    ``config.payment_api_key``.
    """

    def __init__(self, config: ServiceConfig) -> None:
        """Initialize the gateway.

        Args:
            config: Service configuration providing the payment API key.
        """
        self._config = config

    def charge(self, payment_token: str, amount: float) -> str:
        """Charge a payment token.

        Args:
            payment_token: Opaque token identifying the payment instrument.
            amount: The amount to charge; must be non-negative.

        Returns:
            A unique transaction identifier.

        Raises:
            PaymentError: If the charge cannot be completed.
        """
        if not payment_token:
            raise PaymentError("Payment token rejected by gateway")
        return f"txn_{uuid.uuid4().hex}"

    def refund(self, transaction_id: str, amount: float) -> str:
        """Refund a previously charged transaction.

        Args:
            transaction_id: The transaction to refund.
            amount: The amount to refund; must be non-negative.

        Returns:
            A unique refund identifier.

        Raises:
            PaymentError: If the refund cannot be completed.
        """
        if not transaction_id:
            raise PaymentError("Cannot refund an unknown transaction")
        return f"rfnd_{uuid.uuid4().hex}"


# A permission checker returns True if the user may act on the given resource.
PermissionChecker = Callable[[str, str], bool]


def _default_permission_checker(user_id: str, resource_owner_id: str) -> bool:
    """Default permission policy: a user may only act on their own resources.

    Args:
        user_id: The acting user's identifier.
        resource_owner_id: The identifier of the resource's owner.

    Returns:
        True if the user is permitted to act on the resource.
    """
    return bool(user_id) and user_id == resource_owner_id


# ---------------------------------------------------------------------------
# Validation helpers (R-001, R-002)
# ---------------------------------------------------------------------------
def _require_non_empty_str(name: str, value: Any) -> str:
    """Validate that a value is a non-empty string.

    Args:
        name: The parameter name, used in error messages.
        value: The value to validate.

    Returns:
        The validated string.

    Raises:
        ValidationError: If the value is not a non-empty string.
    """
    if not isinstance(value, str):
        raise ValidationError(f"{name} must be a str, got {type(value).__name__}")
    if not value.strip():
        raise ValidationError(f"{name} must not be empty")
    return value


def _validate_items(items: Any, max_items: int) -> list[dict[str, Any]]:
    """Validate a list of order line items.

    Each item must be a mapping with a string ``name``, a non-negative integer
    ``quantity``, and a non-negative numeric ``unit_price``.

    Args:
        items: The candidate list of items.
        max_items: The maximum number of items permitted.

    Returns:
        The validated list of items.

    Raises:
        ValidationError: If the structure or any value is invalid, including a
            negative quantity (R-002).
    """
    if not isinstance(items, list):
        raise ValidationError(f"items must be a list, got {type(items).__name__}")
    if not items:
        raise ValidationError("items must not be empty")
    if len(items) > max_items:
        raise ValidationError(f"items exceeds maximum of {max_items}")

    for index, item in enumerate(items):
        if not isinstance(item, dict):
            raise ValidationError(f"items[{index}] must be a dict")

        name = item.get("name")
        if not isinstance(name, str) or not name.strip():
            raise ValidationError(f"items[{index}].name must be a non-empty str")

        quantity = item.get("quantity")
        if isinstance(quantity, bool) or not isinstance(quantity, int):
            raise ValidationError(f"items[{index}].quantity must be an int")
        if quantity < 0:
            raise ValidationError(f"items[{index}].quantity must not be negative")

        unit_price = item.get("unit_price")
        if isinstance(unit_price, bool) or not isinstance(unit_price, (int, float)):
            raise ValidationError(f"items[{index}].unit_price must be numeric")
        if unit_price < 0:
            raise ValidationError(f"items[{index}].unit_price must not be negative")

    return items


def _calculate_total(items: list[dict[str, Any]]) -> float:
    """Calculate the total price of a list of validated items.

    Args:
        items: A list of validated line items.

    Returns:
        The summed total price, rounded to two decimal places.
    """
    total = sum(item["quantity"] * item["unit_price"] for item in items)
    return round(float(total), 2)


# ---------------------------------------------------------------------------
# Service
# ---------------------------------------------------------------------------
@dataclass
class OrderService:
    """Order management service.

    Args:
        config: Service configuration. Defaults to ``ServiceConfig()``.
        repository: Order data store. Defaults to an in-memory repository.
        payment_gateway: Payment gateway. Defaults to a stub gateway.
        permission_checker: Callable deciding whether a user may act on a
            resource owned by another user.
    """

    config: ServiceConfig = field(default_factory=ServiceConfig)
    repository: OrderRepository = field(default=None)  # type: ignore[assignment]
    payment_gateway: PaymentGateway = field(default=None)  # type: ignore[assignment]
    permission_checker: PermissionChecker = _default_permission_checker

    def __post_init__(self) -> None:
        """Wire up default collaborators that depend on the config."""
        if self.repository is None:
            self.repository = OrderRepository(self.config)
        if self.payment_gateway is None:
            self.payment_gateway = PaymentGateway(self.config)

    def create_order(
        self, user_id: str, items: list, payment_token: str
    ) -> dict[str, Any]:
        """Create a new order, charge payment, and return a confirmation.

        Args:
            user_id: The identifier of the user placing the order.
            items: A list of line items; each must have ``name``, a
                non-negative ``quantity``, and a non-negative ``unit_price``.
            payment_token: Opaque token identifying the payment instrument.

        Returns:
            A confirmation dict containing the order ID, status, total amount,
            currency, and transaction ID.

        Raises:
            ValidationError: If any parameter is invalid (R-001, R-002).
            PaymentError: If payment processing fails.
        """
        logger.debug(
            "Entering create_order(user_id=%r, item_count=%s, payment_token=***)",
            user_id,
            len(items) if isinstance(items, list) else "n/a",
        )

        user_id = _require_non_empty_str("user_id", user_id)
        payment_token = _require_non_empty_str("payment_token", payment_token)
        validated_items = _validate_items(items, self.config.max_items_per_order)

        total = _calculate_total(validated_items)

        try:
            transaction_id = self.payment_gateway.charge(payment_token, total)
        except PaymentError as exc:
            # Preserve original context when re-raising (R-008).
            raise PaymentError(f"Payment failed for user {user_id}: {exc}") from exc

        order_id = f"ord_{uuid.uuid4().hex}"
        order = {
            "order_id": order_id,
            "user_id": user_id,
            "items": validated_items,
            "total": total,
            "currency": self.config.default_currency,
            "status": "confirmed",
            "transaction_id": transaction_id,
        }
        self.repository.save(order)

        confirmation = {
            "order_id": order_id,
            "status": "confirmed",
            "total": total,
            "currency": self.config.default_currency,
            "transaction_id": transaction_id,
        }
        logger.debug("Exiting create_order -> %s", type(confirmation).__name__)
        return confirmation

    def cancel_order(
        self, user_id: str, order_id: str, reason: str
    ) -> dict[str, Any]:
        """Cancel an existing order and issue a refund.

        Args:
            user_id: The identifier of the user requesting cancellation.
            order_id: The identifier of the order to cancel.
            reason: A human-readable cancellation reason.

        Returns:
            A confirmation dict containing the order ID, status, refund ID, and
            recorded reason.

        Raises:
            ValidationError: If any parameter is invalid (R-001).
            PermissionDeniedError: If the user does not own the order (R-020).
            OrderNotFoundError: If the order does not exist.
            OrderStateError: If the order cannot be cancelled in its current
                state.
            PaymentError: If the refund fails.
        """
        logger.debug(
            "Entering cancel_order(user_id=%r, order_id=%r, reason=%r)",
            user_id,
            order_id,
            reason,
        )

        user_id = _require_non_empty_str("user_id", user_id)
        order_id = _require_non_empty_str("order_id", order_id)
        reason = _require_non_empty_str("reason", reason)

        order = self.repository.get(order_id)
        if order is None:
            raise OrderNotFoundError(f"Order {order_id} not found")

        # Verify caller permissions before privileged operation (R-020).
        if not self.permission_checker(user_id, order["user_id"]):
            raise PermissionDeniedError(
                f"User {user_id} may not cancel order {order_id}"
            )

        if order["status"] == "cancelled":
            raise OrderStateError(f"Order {order_id} is already cancelled")
        if order["status"] != "confirmed":
            raise OrderStateError(
                f"Order {order_id} cannot be cancelled from status {order['status']!r}"
            )

        try:
            refund_id = self.payment_gateway.refund(
                order["transaction_id"], order["total"]
            )
        except PaymentError as exc:
            raise PaymentError(
                f"Refund failed for order {order_id}: {exc}"
            ) from exc

        order["status"] = "cancelled"
        order["cancellation_reason"] = reason
        order["refund_id"] = refund_id
        self.repository.save(order)

        confirmation = {
            "order_id": order_id,
            "status": "cancelled",
            "refund_id": refund_id,
            "reason": reason,
        }
        logger.debug("Exiting cancel_order -> %s", type(confirmation).__name__)
        return confirmation

    def get_order_status(self, user_id: str, order_id: str) -> dict[str, Any]:
        """Retrieve the status and details of an order.

        Args:
            user_id: The identifier of the user requesting the status.
            order_id: The identifier of the order to look up.

        Returns:
            A dict containing the order ID, status, total, currency, and items.

        Raises:
            ValidationError: If any parameter is invalid (R-001).
            PermissionDeniedError: If the user does not own the order (R-020).
            OrderNotFoundError: If the order does not exist.
        """
        logger.debug(
            "Entering get_order_status(user_id=%r, order_id=%r)", user_id, order_id
        )

        user_id = _require_non_empty_str("user_id", user_id)
        order_id = _require_non_empty_str("order_id", order_id)

        order = self.repository.get(order_id)
        if order is None:
            raise OrderNotFoundError(f"Order {order_id} not found")

        # Verify caller permissions before exposing order details (R-020).
        if not self.permission_checker(user_id, order["user_id"]):
            raise PermissionDeniedError(
                f"User {user_id} may not view order {order_id}"
            )

        status = {
            "order_id": order_id,
            "status": order["status"],
            "total": order["total"],
            "currency": order["currency"],
            "items": order["items"],
        }
        logger.debug("Exiting get_order_status -> %s", type(status).__name__)
        return status


# ---------------------------------------------------------------------------
# Module-level convenience API (R-025, R-026)
# ---------------------------------------------------------------------------
_default_service = OrderService()


def create_order(
```

```python
"""Unit tests for order_service."""

import unittest

from order_service import (
    OrderNotFoundError,
    OrderService,
    OrderStateError,
    PaymentError,
    PaymentGateway,
    PermissionDeniedError,
    ServiceConfig,
    ValidationError,
)


class FailingGateway(PaymentGateway):
    """Gateway whose charge always fails, for error-path testing."""

    def charge(self, payment_token: str, amount: float) -> str:
        raise PaymentError("simulated charge failure")


class OrderServiceTest(unittest.TestCase):
    """Tests for the OrderService public methods."""

    def setUp(self) -> None:
        self.service = OrderService(config=ServiceConfig())
        self.items = [
            {"name": "widget", "quantity": 2, "unit_price": 10.0},
            {"name": "gadget", "quantity": 1, "unit_price": 5.5},
        ]

    # -- create_order -----------------------------------------------------
    def test_create_order_success(self) -> None:
        result = self.service.create_order("user-1", self.items, "tok_abc")
        self.assertEqual(result["status"], "confirmed")
        self.assertEqual(result["total"], 25.5)
        self.assertEqual(result["currency"], "USD")
        self.assertTrue(result["order_id"].startswith("ord_"))
        self.assertTrue(result["transaction_id"].startswith("txn_"))

    def test_create_order_rejects_bad_user_id(self) -> None:
        with self.assertRaises(ValidationError):
            self.service.create_order("", self.items, "tok_abc")
        with self.assertRaises(ValidationError):
            self.service.create_order(123, self.items, "tok_abc")  # type: ignore[arg-type]

    def test_create_order_rejects_empty_token(self) -> None:
        with self.assertRaises(ValidationError):
            self.service.create_order("user-1", self.items, "")

    def test_create_order_rejects_non_list_items(self) -> None:
        with self.assertRaises(ValidationError):
            self.service.create_order("user-1", "nope", "tok_abc")  # type: ignore[arg-type]

    def test_create_order_rejects_negative_quantity(self) -> None:
        bad = [{"name": "widget", "quantity": -1, "unit_price": 10.0}]
        with self.assertRaises(ValidationError):
            self.service.create_order("user-1", bad, "tok_abc")

    def test_create_order_rejects_negative_unit_price(self) -> None:
        bad = [{"name": "widget", "quantity": 1, "unit_price": -2.0}]
        with self.assertRaises(ValidationError):
            self.service.create_order("user-1", bad, "tok_abc")

    def test_create_order_rejects_non_int_quantity(self) -> None:
        bad = [{"name": "widget", "quantity": 1.5, "unit_price": 2.0}]
        with self.assertRaises(ValidationError):
            self.service.create_order("user-1", bad, "tok_abc")

    def test_create_order_payment_failure_wraps_context(self) -> None:
        service = OrderService(payment_gateway=FailingGateway(ServiceConfig()))
        with self.assertRaises(PaymentError) as ctx:
            service.create_order("user-1", self.items, "tok_abc")
        self.assertIsInstance(ctx.exception.__cause__, PaymentError)

    # -- get_order_status -------------------------------------------------
    def test_get_order_status_success(self) -> None:
        created = self.service.create_order("user-1", self.items, "tok_abc")
        status = self.service.get_order_status("user-1", created["order_id"])
        self.assertEqual(status["status"], "confirmed")
        self.assertEqual(status["total"], 25.5)
        self.assertEqual(len(status["items"]), 2)

    def test_get_order_status_not_found(self) -> None:
        with self.assertRaises(OrderNotFoundError):
            self.service.get_order_status("user-1", "ord_missing")

    def test_get_order_status_permission_denied(self) -> None:
        created = self.service.create_order("user-1", self.items, "tok_abc")
        with self.assertRaises(PermissionDeniedError):
            self.service.get_order_status("user-2", created["order_id"])

    def test_get_order_status_rejects_bad_params(self) -> None:
        with self.assertRaises(ValidationError):
            self.service.get_order_status("", "ord_x")

    # -- cancel_order -----------------------------------------------------
    def test_cancel_order_success(self) -> None:
        created = self.service.create_order("user-1", self.items, "tok_abc")
        result = self.service.cancel_order(
            "user-1", created["order_id"], "changed mind"
        )
        self.assertEqual(result["status"], "cancelled")
        self.assertEqual(result["reason"], "changed mind")
        self.assertTrue(result["refund_id"].startswith("rfnd_"))

        status = self.service.get_order_status("user-1", created["order_id"])
        self.assertEqual(status["status"], "cancelled")

    def test_cancel_order_not_found(self) -> None:
        with self.assertRaises(OrderNotFoundError):
            self.service.cancel_order("user-1", "ord_missing", "reason")

    def test_cancel_order_permission_denied(self) -> None:
        created = self.service.create_order("user-1", self.items, "tok_abc")
        with self.assertRaises(PermissionDeniedError):
            self.service.cancel_order("user-2", created["order_id"], "reason")

    def test_cancel_order_already_cancelled(self) -> None:
        created = self.service.create_order("user-1", self.items, "tok_abc")
        self.service.cancel_order("user-1", created["order_id"], "first")
        with self.assertRaises(OrderStateError):
            self.service.cancel_order("user-1", created["order_id"], "second")

    def test_cancel_order_rejects_empty_reason(self) -> None:
        created = self.service.create_order("user-1", self.items, "tok_abc")
        with self.assertRaises(ValidationError):
            self.service.cancel_order("user-1", created["order_id"], "")


if __name__ == "__main__":
    unittest.main()
```
