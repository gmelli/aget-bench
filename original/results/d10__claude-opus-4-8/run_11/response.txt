"""Order management service.

Provides creation, cancellation, and status retrieval for orders, including
price calculation, payment processing, and refunds. All configuration values
are externalized into :class:`ServiceConfig` (R-019), and every privileged
operation verifies caller permissions before executing (R-020).
"""

from __future__ import annotations

import logging
import uuid
from dataclasses import dataclass
from typing import Any, Callable, Dict, List, Optional

logger = logging.getLogger(__name__)


# ---------------------------------------------------------------------------
# Domain-specific exceptions (R-007)
# ---------------------------------------------------------------------------
class OrderServiceError(Exception):
    """Base class for all order service errors."""


class ValidationError(OrderServiceError):
    """Raised when input parameters fail validation."""


class PermissionDeniedError(OrderServiceError):
    """Raised when a caller lacks permission for a privileged operation."""


class PaymentError(OrderServiceError):
    """Raised when payment processing or refunding fails."""


class OrderNotFoundError(OrderServiceError):
    """Raised when a requested order does not exist."""


class OrderStateError(OrderServiceError):
    """Raised when an operation is invalid for the order's current state."""


# ---------------------------------------------------------------------------
# Configuration (R-019)
# ---------------------------------------------------------------------------
@dataclass
class ServiceConfig:
    """Configuration for the order service.

    Attributes:
        database_url: Connection string for the order datastore.
        payment_api_key: API key used to authenticate with the payment gateway.
        max_orders_per_minute: Rate limit for order creation per user.
        max_items_per_order: Maximum number of line items allowed in an order.
        currency: ISO 4217 currency code used for all monetary amounts.
    """

    database_url: str = "sqlite:///orders.db"
    payment_api_key: str = "CHANGE_ME"
    max_orders_per_minute: int = 60
    max_items_per_order: int = 100
    currency: str = "USD"


# ---------------------------------------------------------------------------
# Collaborators (injectable for testing)
# ---------------------------------------------------------------------------
class PaymentGateway:
    """Thin wrapper around an external payment provider."""

    def __init__(self, api_key: str) -> None:
        """Initialize the gateway.

        Args:
            api_key: Credential used to authenticate with the provider.
        """
        self._api_key = api_key

    def charge(self, payment_token: str, amount: float, currency: str) -> str:
        """Charge a payment token.

        Args:
            payment_token: Tokenized payment instrument supplied by the caller.
            amount: Amount to charge; must be non-negative.
            currency: ISO 4217 currency code.

        Returns:
            A transaction identifier string.

        Raises:
            PaymentError: If the charge cannot be completed.
        """
        if not payment_token:
            raise PaymentError("Empty payment token supplied")
        return f"txn_{uuid.uuid4().hex}"

    def refund(self, transaction_id: str, amount: float, currency: str) -> str:
        """Refund a previously charged transaction.

        Args:
            transaction_id: Identifier of the transaction to refund.
            amount: Amount to refund; must be non-negative.
            currency: ISO 4217 currency code.

        Returns:
            A refund identifier string.

        Raises:
            PaymentError: If the refund cannot be completed.
        """
        if not transaction_id:
            raise PaymentError("Empty transaction id supplied")
        return f"rfnd_{uuid.uuid4().hex}"


# ---------------------------------------------------------------------------
# Validation helpers (R-001, R-002)
# ---------------------------------------------------------------------------
def _require_non_empty_str(value: Any, name: str) -> str:
    """Validate that ``value`` is a non-empty string.

    Args:
        value: The value to validate.
        name: Parameter name used in error messages.

    Returns:
        The validated string.

    Raises:
        ValidationError: If ``value`` is not a non-empty string.
    """
    if not isinstance(value, str):
        raise ValidationError(f"{name} must be a str, got {type(value).__name__}")
    if not value.strip():
        raise ValidationError(f"{name} must not be empty")
    return value


def _validate_items(items: Any, max_items: int) -> List[Dict[str, Any]]:
    """Validate a list of order line items.

    Args:
        items: Candidate list of item dicts, each with ``name``, ``quantity``,
            and ``unit_price``.
        max_items: Maximum number of items permitted.

    Returns:
        The validated list of items.

    Raises:
        ValidationError: If the structure or any field is invalid, including
            negative quantities (R-002) or negative unit prices.
    """
    if not isinstance(items, list):
        raise ValidationError(f"items must be a list, got {type(items).__name__}")
    if not items:
        raise ValidationError("items must contain at least one entry")
    if len(items) > max_items:
        raise ValidationError(f"items exceeds maximum of {max_items}")

    for index, item in enumerate(items):
        if not isinstance(item, dict):
            raise ValidationError(f"items[{index}] must be a dict")

        name = item.get("name")
        if not isinstance(name, str) or not name.strip():
            raise ValidationError(f"items[{index}].name must be a non-empty str")

        quantity = item.get("quantity")
        if not isinstance(quantity, int) or isinstance(quantity, bool):
            raise ValidationError(f"items[{index}].quantity must be an int")
        if quantity < 0:
            raise ValidationError(f"items[{index}].quantity must not be negative")

        unit_price = item.get("unit_price")
        if not isinstance(unit_price, (int, float)) or isinstance(unit_price, bool):
            raise ValidationError(f"items[{index}].unit_price must be numeric")
        if unit_price < 0:
            raise ValidationError(f"items[{index}].unit_price must not be negative")

    return items


def _calculate_total(items: List[Dict[str, Any]]) -> float:
    """Compute the total price for a list of validated items.

    Args:
        items: A list of validated item dicts.

    Returns:
        The summed total price rounded to two decimal places.
    """
    total = sum(item["quantity"] * item["unit_price"] for item in items)
    return round(float(total), 2)


# ---------------------------------------------------------------------------
# Permission checks (R-020)
# ---------------------------------------------------------------------------
def _verify_permission(user_id: str, order: Optional[Dict[str, Any]] = None) -> None:
    """Verify that ``user_id`` may operate on ``order``.

    Args:
        user_id: Identifier of the calling user.
        order: The order being operated on, or ``None`` for creation.

    Raises:
        PermissionDeniedError: If the caller is not authorized.
    """
    if order is not None and order.get("user_id") != user_id:
        raise PermissionDeniedError(
            f"User {user_id!r} is not permitted to access order "
            f"{order.get('order_id')!r}"
        )


# ---------------------------------------------------------------------------
# Service
# ---------------------------------------------------------------------------
class OrderService:
    """Stateful order management service.

    The service keeps orders in an in-memory store keyed by order id. In a
    production deployment the store would be backed by ``config.database_url``.
    """

    def __init__(
        self,
        config: Optional[ServiceConfig] = None,
        payment_gateway: Optional[PaymentGateway] = None,
        id_factory: Optional[Callable[[], str]] = None,
    ) -> None:
        """Initialize the service.

        Args:
            config: Service configuration; defaults to :class:`ServiceConfig`.
            payment_gateway: Payment collaborator; defaults to one built from
                the configured API key.
            id_factory: Callable returning unique order ids; injectable for
                deterministic testing.
        """
        self._config: ServiceConfig = config or ServiceConfig()
        self._gateway: PaymentGateway = payment_gateway or PaymentGateway(
            self._config.payment_api_key
        )
        self._id_factory: Callable[[], str] = id_factory or (
            lambda: f"ord_{uuid.uuid4().hex}"
        )
        self._orders: Dict[str, Dict[str, Any]] = {}

    def create_order(
        self, user_id: str, items: List[Dict[str, Any]], payment_token: str
    ) -> Dict[str, Any]:
        """Create a new order, charge payment, and return a confirmation.

        Args:
            user_id: Identifier of the user placing the order.
            items: List of line items, each a dict with ``name`` (str),
                ``quantity`` (non-negative int), and ``unit_price`` (non-negative
                number).
            payment_token: Tokenized payment instrument to charge.

        Returns:
            A confirmation dict with keys ``order_id``, ``user_id``, ``status``,
            ``total``, ``currency``, ``transaction_id``, and ``items``.

        Raises:
            ValidationError: If any parameter fails validation.
            PaymentError: If the payment cannot be processed.
        """
        logger.debug(
            "Entering create_order(user_id=%r, item_count=%s, payment_token=***)",
            user_id,
            len(items) if isinstance(items, list) else "n/a",
        )

        user_id = _require_non_empty_str(user_id, "user_id")
        payment_token = _require_non_empty_str(payment_token, "payment_token")
        validated_items = _validate_items(items, self._config.max_items_per_order)

        _verify_permission(user_id, order=None)

        total = _calculate_total(validated_items)

        try:
            transaction_id = self._gateway.charge(
                payment_token, total, self._config.currency
            )
        except PaymentError as exc:
            raise PaymentError(
                f"Failed to charge payment for user {user_id!r}"
            ) from exc  # R-008

        order_id = self._id_factory()
        order: Dict[str, Any] = {
            "order_id": order_id,
            "user_id": user_id,
            "status": "CONFIRMED",
            "total": total,
            "currency": self._config.currency,
            "transaction_id": transaction_id,
            "items": validated_items,
        }
        self._orders[order_id] = order

        confirmation = dict(order)
        logger.debug(
            "Exiting create_order with return type %s", type(confirmation).__name__
        )
        return confirmation

    def cancel_order(
        self, user_id: str, order_id: str, reason: str
    ) -> Dict[str, Any]:
        """Cancel an existing order and issue a refund.

        Args:
            user_id: Identifier of the user requesting cancellation.
            order_id: Identifier of the order to cancel.
            reason: Free-text cancellation reason.

        Returns:
            A confirmation dict with keys ``order_id``, ``user_id``, ``status``,
            ``reason``, ``refund_id``, and ``refunded_amount``.

        Raises:
            ValidationError: If any parameter fails validation.
            OrderNotFoundError: If the order does not exist.
            PermissionDeniedError: If the caller does not own the order.
            OrderStateError: If the order is already cancelled.
            PaymentError: If the refund cannot be processed.
        """
        logger.debug(
            "Entering cancel_order(user_id=%r, order_id=%r, reason=%r)",
            user_id,
            order_id,
            reason,
        )

        user_id = _require_non_empty_str(user_id, "user_id")
        order_id = _require_non_empty_str(order_id, "order_id")
        reason = _require_non_empty_str(reason, "reason")

        order = self._orders.get(order_id)
        if order is None:
            raise OrderNotFoundError(f"Order {order_id!r} not found")

        _verify_permission(user_id, order)

        if order["status"] == "CANCELLED":
            raise OrderStateError(f"Order {order_id!r} is already cancelled")

        try:
            refund_id = self._gateway.refund(
                order["transaction_id"], order["total"], order["currency"]
            )
        except PaymentError as exc:
            raise PaymentError(
                f"Failed to refund order {order_id!r}"
            ) from exc  # R-008

        order["status"] = "CANCELLED"
        order["cancel_reason"] = reason

        confirmation: Dict[str, Any] = {
            "order_id": order_id,
            "user_id": user_id,
            "status": "CANCELLED",
            "reason": reason,
            "refund_id": refund_id,
            "refunded_amount": order["total"],
        }
        logger.debug(
            "Exiting cancel_order with return type %s", type(confirmation).__name__
        )
        return confirmation

    def get_order_status(self, user_id: str, order_id: str) -> Dict[str, Any]:
        """Retrieve the status of an existing order.

        Args:
            user_id: Identifier of the user requesting status.
            order_id: Identifier of the order to inspect.

        Returns:
            A status dict with keys ``order_id``, ``user_id``, ``status``,
            ``total``, and ``currency``.

        Raises:
            ValidationError: If any parameter fails validation.
            OrderNotFoundError: If the order does not exist.
            PermissionDeniedError: If the caller does not own the order.
        """
        logger.debug(
            "Entering get_order_status(user_id=%r, order_id=%r)", user_id, order_id
        )

        user_id = _require_non_empty_str(user_id, "user_id")
        order_id = _require_non_empty_str(order_id, "order_id")

        order = self._orders.get(order_id)
        if order is None:
            raise OrderNotFoundError(f"Order {order_id!r} not found")

        _verify_permission(user_id, order)

        status: Dict[str, Any] = {
            "order_id": order["order_id"],
            "user_id": order["user_id"],
            "status": order["status"],
            "total": order["total"],
            "currency": order["currency"],
        }
        logger.debug(
            "Exiting get_order_status with return type %s", type(status).__name__
        )
        return status


# ---------------------------------------------------------------------------

# ---------------------------------------------------------------------------
_default_service = OrderService()

"""Unit tests for order_service."""

import unittest

from order_service import (
    OrderNotFoundError,
    OrderService,
    OrderStateError,
    PaymentError,
    PaymentGateway,
    PermissionDeniedError,
    ServiceConfig,
    ValidationError,
)


class _StubGateway(PaymentGateway):
    """Deterministic gateway stub for assertions."""

    def __init__(self, fail_charge: bool = False, fail_refund: bool = False) -> None:
        super().__init__(api_key="test")
        self.fail_charge = fail_charge
        self.fail_refund = fail_refund
        self.charged = []
        self.refunded = []

    def charge(self, payment_token, amount, currency):
        if self.fail_charge:
            raise PaymentError("charge failed")
        self.charged.append((payment_token, amount, currency))
        return "txn_test"

    def refund(self, transaction_id, amount, currency):
        if self.fail_refund:
            raise PaymentError("refund failed")
        self.refunded.append((transaction_id, amount, currency))
        return "rfnd_test"


def _items():
    return [
        {"name": "widget", "quantity": 2, "unit_price": 9.99},
        {"name": "gadget", "quantity": 1, "unit_price": 4.50},
    ]


class CreateOrderTests(unittest.TestCase):
    def setUp(self):
        self.gateway = _StubGateway()
        counter = {"n": 0}

        def ids():
            counter["n"] += 1
            return f"ord_{counter['n']}"

        self.service = OrderService(
            config=ServiceConfig(), payment_gateway=self.gateway, id_factory=ids
        )

    def test_create_order_success(self):
        result = self.service.create_order("u1", _items(), "tok_123")
        self.assertEqual(result["status"], "CONFIRMED")
        self.assertEqual(result["order_id"], "ord_1")
        self.assertEqual(result["total"], 24.48)
        self.assertEqual(result["transaction_id"], "txn_test")
        self.assertEqual(self.gateway.charged[0][1], 24.48)

    def test_create_order_rejects_negative_quantity(self):
        items = [{"name": "x", "quantity": -1, "unit_price": 5.0}]
        with self.assertRaises(ValidationError):
            self.service.create_order("u1", items, "tok")

    def test_create_order_rejects_negative_unit_price(self):
        items = [{"name": "x", "quantity": 1, "unit_price": -5.0}]
        with self.assertRaises(ValidationError):
            self.service.create_order("u1", items, "tok")

    def test_create_order_rejects_bad_types(self):
        with self.assertRaises(ValidationError):
            self.service.create_order(123, _items(), "tok")
        with self.assertRaises(ValidationError):
            self.service.create_order("u1", "not-a-list", "tok")
        with self.assertRaises(ValidationError):
            self.service.create_order("u1", _items(), "")

    def test_create_order_empty_items(self):
        with self.assertRaises(ValidationError):
            self.service.create_order("u1", [], "tok")

    def test_create_order_payment_failure_wraps_context(self):
        service = OrderService(payment_gateway=_StubGateway(fail_charge=True))
        with self.assertRaises(PaymentError) as ctx:
            service.create_order("u1", _items(), "tok")
        self.assertIsInstance(ctx.exception.__cause__, PaymentError)


class CancelOrderTests(unittest.TestCase):
    def setUp(self):
        self.gateway = _StubGateway()
        self.service = OrderService(payment_gateway=self.gateway)
        self.order = self.service.create_order("u1", _items(), "tok")
        self.order_id = self.order["order_id"]

    def test_cancel_order_success(self):
        result = self.service.cancel_order("u1", self.order_id, "changed mind")
        self.assertEqual(result["status"], "CANCELLED")
        self.assertEqual(result["refund_id"], "rfnd_test")
        self.assertEqual(result["refunded_amount"], self.order["total"])
        self.assertEqual(result["reason"], "changed mind")

    def test_cancel_order_not_found(self):
        with self.assertRaises(OrderNotFoundError):
            self.service.cancel_order("u1", "missing", "reason")

    def test_cancel_order_permission_denied(self):
        with self.assertRaises(PermissionDeniedError):
            self.service.cancel_order("other", self.order_id, "reason")

    def test_cancel_order_already_cancelled(self):
        self.service.cancel_order("u1", self.order_id, "first")
        with self.assertRaises(OrderStateError):
            self.service.cancel_order("u1", self.order_id, "again")

    def test_cancel_order_refund_failure_wraps_context(self):
        gateway = _StubGateway(fail_refund=True)
        service = OrderService(payment_gateway=gateway)
        order = service.create_order("u1", _items(), "tok")

            service.cancel_order("u1", order["order_id"], "reason")
        self.assertIsInstance(ctx.exception.__cause__, PaymentError)

    def test_cancel_order_validates_reason(self):
        with self.assertRaises(ValidationError):
            self.service.cancel_order("u1", self.order_id, "")
