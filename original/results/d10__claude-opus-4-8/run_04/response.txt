```python
"""Order management service.

This module provides a small order-management service with functions to
create, cancel, and inspect orders. It is intentionally self-contained:
payment processing, persistence, and permission checks are modeled with
lightweight in-memory stand-ins so the public API and its contracts can be
exercised by the accompanying unit tests.

The implementation adheres to the project's mandatory rules: parameters are
validated, negative quantities are rejected, domain-specific exceptions carry
original error context, configuration is externalized, caller permissions are
verified for privileged operations, and every public function is fully typed,
documented, and logged on entry and exit.
"""

from __future__ import annotations

import logging
import uuid
from dataclasses import dataclass
from typing import Any, Callable, Dict, List, Optional

logger = logging.getLogger(__name__)


# ---------------------------------------------------------------------------
# Configuration (R-019: externalize all configuration values)
# ---------------------------------------------------------------------------
@dataclass
class ServiceConfig:
    """Configuration for the order service.

    Attributes:
        database_url: Connection string for the order datastore.
        payment_api_key: API key used to authenticate with the payment gateway.
        rate_limit_per_minute: Maximum number of operations permitted per
            caller per minute.
        currency: ISO currency code used for monetary amounts.
        refund_window_days: Number of days during which an order may be
            cancelled and refunded.
    """

    database_url: str = "sqlite:///orders.db"
    payment_api_key: str = "test-payment-key"
    rate_limit_per_minute: int = 60
    currency: str = "USD"
    refund_window_days: int = 30


# ---------------------------------------------------------------------------
# Domain-specific exceptions (R-007: no generic Exception)
# ---------------------------------------------------------------------------
class OrderServiceError(Exception):
    """Base class for all order-service errors."""


class ValidationError(OrderServiceError):
    """Raised when input parameters fail validation."""


class PermissionDeniedError(OrderServiceError):
    """Raised when a caller lacks permission for a privileged operation."""


class PaymentError(OrderServiceError):
    """Raised when payment processing or refunding fails."""


class OrderNotFoundError(OrderServiceError):
    """Raised when a referenced order does not exist."""


class OrderStateError(OrderServiceError):
    """Raised when an operation is invalid for an order's current state."""


# ---------------------------------------------------------------------------
# In-memory collaborators (stand-ins for real infrastructure)
# ---------------------------------------------------------------------------
@dataclass
class _PaymentGateway:
    """Minimal payment gateway stand-in.

    A real implementation would call an external API authenticated with
    ``config.payment_api_key``. Here we simulate success/failure based on the
    token so the behavior is deterministic and testable.
    """

    api_key: str

    def charge(self, payment_token: str, amount: float) -> str:
        """Charge the given amount against a payment token.

        Args:
            payment_token: Tokenized payment instrument.
            amount: Positive amount to charge.

        Returns:
            A transaction identifier for the successful charge.

        Raises:
            PaymentError: If the token is invalid or the charge is declined.
        """
        if not payment_token or payment_token.startswith("invalid"):
            raise PaymentError(f"Payment declined for token '{payment_token}'.")
        return f"txn_{uuid.uuid4().hex}"

    def refund(self, transaction_id: str, amount: float) -> str:
        """Refund a previously charged transaction.

        Args:
            transaction_id: Identifier of the original charge.
            amount: Positive amount to refund.

        Returns:
            A refund identifier for the successful refund.

        Raises:
            PaymentError: If the original transaction is missing.
        """
        if not transaction_id:
            raise PaymentError("Cannot refund without a transaction id.")
        return f"rfnd_{uuid.uuid4().hex}"


# Module-level state used by the service functions. In production these would
# be backed by a database (config.database_url) and an external gateway.
_CONFIG = ServiceConfig()
_ORDERS: Dict[str, Dict[str, Any]] = {}
_GATEWAY = _PaymentGateway(api_key=_CONFIG.payment_api_key)


def _default_permission_checker(user_id: str, operation: str) -> bool:
    """Default permission policy: any non-empty, non-blocked user may act.

    Args:
        user_id: Identifier of the calling user.
        operation: Name of the privileged operation being attempted.

    Returns:
        True if the operation is permitted, otherwise False.
    """
    return bool(user_id) and not user_id.startswith("blocked")


_PERMISSION_CHECKER: Callable[[str, str], bool] = _default_permission_checker


def configure(
    config: Optional[ServiceConfig] = None,
    permission_checker: Optional[Callable[[str, str], bool]] = None,
) -> None:
    """Reconfigure the service.

    Args:
        config: Replacement configuration. If ``None``, configuration is left
            unchanged.
        permission_checker: Callable taking ``(user_id, operation)`` and
            returning whether the caller is permitted. If ``None``, the
            existing checker is left unchanged.

    Returns:
        None.

    Raises:
        ValidationError: If ``config`` is provided but is not a ServiceConfig.
    """
    global _CONFIG, _GATEWAY, _PERMISSION_CHECKER
    if config is not None:
        if not isinstance(config, ServiceConfig):
            raise ValidationError("config must be a ServiceConfig instance.")
        _CONFIG = config
        _GATEWAY = _PaymentGateway(api_key=config.payment_api_key)
    if permission_checker is not None:
        _PERMISSION_CHECKER = permission_checker


# ---------------------------------------------------------------------------
# Validation helpers (R-001, R-002)
# ---------------------------------------------------------------------------
def _require_non_empty_str(value: Any, name: str) -> str:
    """Validate that a value is a non-empty string.

    Args:
        value: The value to validate.
        name: Parameter name used in error messages.

    Returns:
        The validated string.

    Raises:
        ValidationError: If the value is not a non-empty string.
    """
    if not isinstance(value, str):
        raise ValidationError(f"{name} must be a str, got {type(value).__name__}.")
    if not value.strip():
        raise ValidationError(f"{name} must not be empty.")
    return value


def _validate_items(items: List[Dict[str, Any]]) -> float:
    """Validate order items and compute their total price.

    Args:
        items: List of item dicts, each with ``name`` (str),
            ``quantity`` (int) and ``unit_price`` (int or float).

    Returns:
        The computed total price as a float.

    Raises:
        ValidationError: If ``items`` is not a non-empty list, or any item is
            malformed, has a negative quantity (R-002), or a negative price.
    """
    if not isinstance(items, list):
        raise ValidationError(f"items must be a list, got {type(items).__name__}.")
    if not items:
        raise ValidationError("items must not be empty.")

    total = 0.0
    for index, item in enumerate(items):
        if not isinstance(item, dict):
            raise ValidationError(f"items[{index}] must be a dict.")

        name = item.get("name")
        quantity = item.get("quantity")
        unit_price = item.get("unit_price")

        if not isinstance(name, str) or not name.strip():
            raise ValidationError(f"items[{index}].name must be a non-empty str.")
        if not isinstance(quantity, int) or isinstance(quantity, bool):
            raise ValidationError(f"items[{index}].quantity must be an int.")
        if quantity < 0:
            raise ValidationError(
                f"items[{index}].quantity must not be negative (got {quantity})."
            )
        if not isinstance(unit_price, (int, float)) or isinstance(unit_price, bool):
            raise ValidationError(f"items[{index}].unit_price must be a number.")
        if unit_price < 0:
            raise ValidationError(
                f"items[{index}].unit_price must not be negative (got {unit_price})."
            )

        total += quantity * float(unit_price)

    return total


def _check_permission(user_id: str, operation: str) -> None:
    """Verify caller permission for a privileged operation (R-020).

    Args:
        user_id: Identifier of the calling user.
        operation: Name of the privileged operation.

    Returns:
        None.

    Raises:
        PermissionDeniedError: If the caller is not permitted.
    """
    if not _PERMISSION_CHECKER(user_id, operation):
        raise PermissionDeniedError(
            f"User '{user_id}' is not permitted to perform '{operation}'."
        )


# ---------------------------------------------------------------------------
# Public API
# ---------------------------------------------------------------------------
def create_order(
    user_id: str, items: List[Dict[str, Any]], payment_token: str
) -> Dict[str, Any]:
    """Create a new order, charge payment, and return a confirmation.

    Args:
        user_id: Identifier of the user placing the order.
        items: List of item dicts, each containing ``name`` (str),
            ``quantity`` (int) and ``unit_price`` (number).
        payment_token: Tokenized payment instrument used to charge the total.

    Returns:
        A confirmation dict with keys ``order_id``, ``user_id``, ``status``,
        ``total``, ``currency``, ``transaction_id`` and ``items``.

    Raises:
        ValidationError: If any parameter is missing or malformed.
        PermissionDeniedError: If the caller may not create orders.
        PaymentError: If the payment is declined.
    """
    logger.debug(
        "Entering create_order(user_id=%r, items=<%d items>, payment_token=%r)",
        user_id,
        len(items) if isinstance(items, list) else 0,
        "***" if payment_token else payment_token,
    )

    _require_non_empty_str(user_id, "user_id")
    _require_non_empty_str(payment_token, "payment_token")
    total = _validate_items(items)
    _check_permission(user_id, "create_order")

    try:
        transaction_id = _GATEWAY.charge(payment_token, total)
    except PaymentError as exc:
        # R-008: include the original error context when re-raising.
        raise PaymentError(
            f"Failed to charge order for user '{user_id}': {exc}"
        ) from exc

    order_id = f"ord_{uuid.uuid4().hex}"
    order = {
        "order_id": order_id,
        "user_id": user_id,
        "status": "confirmed",
        "total": total,
        "currency": _CONFIG.currency,
        "transaction_id": transaction_id,
        "items": [dict(item) for item in items],
    }
    _ORDERS[order_id] = order

    confirmation = dict(order)
    logger.debug("Exiting create_order -> %s", type(confirmation).__name__)
    return confirmation


def cancel_order(user_id: str, order_id: str, reason: str) -> Dict[str, Any]:
    """Cancel an existing order and issue a refund.

    Args:
        user_id: Identifier of the user cancelling the order.
        order_id: Identifier of the order to cancel.
        reason: Human-readable reason for the cancellation.

    Returns:
        A cancellation confirmation dict with keys ``order_id``, ``user_id``,
        ``status``, ``reason``, ``refund_id`` and ``refunded_amount``.

    Raises:
        ValidationError: If any parameter is missing or malformed.
        PermissionDeniedError: If the caller may not cancel this order.
        OrderNotFoundError: If the order does not exist.
        OrderStateError: If the order is already cancelled.
        PaymentError: If the refund fails.
    """
    logger.debug(
        "Entering cancel_order(user_id=%r, order_id=%r, reason=%r)",
        user_id,
        order_id,
        reason,
    )

    _require_non_empty_str(user_id, "user_id")
    _require_non_empty_str(order_id, "order_id")
    _require_non_empty_str(reason, "reason")
    _check_permission(user_id, "cancel_order")

    order = _ORDERS.get(order_id)
    if order is None:
        raise OrderNotFoundError(f"Order '{order_id}' does not exist.")
    if order["user_id"] != user_id:
        raise PermissionDeniedError(
            f"User '{user_id}' may not cancel order '{order_id}'."
        )
    if order["status"] == "cancelled":
        raise OrderStateError(f"Order '{order_id}' is already cancelled.")

    try:
        refund_id = _GATEWAY.refund(order["transaction_id"], order["total"])
    except PaymentError as exc:
        # R-008: preserve original error context.
        raise PaymentError(f"Failed to refund order '{order_id}': {exc}") from exc

    order["status"] = "cancelled"
    order["cancel_reason"] = reason
    order["refund_id"] = refund_id

    confirmation = {
        "order_id": order_id,
        "user_id": user_id,
        "status": "cancelled",
        "reason": reason,
        "refund_id": refund_id,
        "refunded_amount": order["total"],
    }
    logger.debug("Exiting cancel_order -> %s", type(confirmation).__name__)
    return confirmation


def get_order_status(user_id: str, order_id: str) -> Dict[str, Any]:
    """Retrieve the status and summary of an order.

    Args:
        user_id: Identifier of the user requesting the status.
        order_id: Identifier of the order to inspect.

    Returns:
        A status dict with keys ``order_id``, ``user_id``, ``status``,
        ``total`` and ``currency``.

    Raises:
        ValidationError: If any parameter is missing or malformed.
        PermissionDeniedError: If the caller may not view this order.
        OrderNotFoundError: If the order does not exist.
    """
    logger.debug(
        "Entering get_order_status(user_id=%r, order_id=%r)", user_id, order_id
    )

    _require_non_empty_str(user_id, "user_id")
    _require_non_empty_str(order_id, "order_id")
    _check_permission(user_id, "get_order_status")

    order = _ORDERS.get(order_id)
    if order is None:
        raise OrderNotFoundError(f"Order '{order_id}' does not exist.")
    if order["user_id"] != user_id:
        raise PermissionDeniedError(
            f"User '{user_id}' may not view order '{order_id}'."
        )

    status = {
        "order_id": order["order_id"],
        "user_id": order["user_id"],
        "status": order["status"],
        "total": order["total"],
        "currency": order["currency"],
    }
    logger.debug("Exiting get_order_status -> %s", type(status).__name__)
    return status

import unittest


class OrderServiceTests(unittest.TestCase):
    """Unit tests for the public order-service API."""

    def setUp(self) -> None:
        """Reset module state and configuration before each test."""
        _ORDERS.clear()
        configure(
            config=ServiceConfig(currency="USD"),
            permission_checker=_default_permission_checker,
        )
        self.items = [
            {"name": "Widget", "quantity": 2, "unit_price": 9.5},
            {"name": "Gadget", "quantity": 1, "unit_price": 5.0},
        ]

    # -- create_order --------------------------------------------------------
    def test_create_order_success(self) -> None:
        confirmation = create_order("u1", self.items, "tok_valid")
        self.assertEqual(confirmation["status"], "confirmed")
        self.assertEqual(confirmation["total"], 24.0)
        self.assertEqual(confirmation["currency"], "USD")
        self.assertTrue(confirmation["order_id"].startswith("ord_"))
        self.assertTrue(confirmation["transaction_id"].startswith("txn_"))

    def test_create_order_rejects_negative_quantity(self) -> None:
        bad = [{"name": "Widget", "quantity": -1, "unit_price": 9.5}]
        with self.assertRaises(ValidationError):
            create_order("u1", bad, "tok_valid")

    def test_create_order_rejects_negative_price(self) -> None:
        bad = [{"name": "Widget", "quantity": 1, "unit_price": -1.0}]
        with self.assertRaises(ValidationError):
            create_order("u1", bad, "tok_valid")

    def test_create_order_validates_user_id(self) -> None:
        with self.assertRaises(ValidationError):
            create_order("", self.items, "tok_valid")

    def test_create_order_validates_items_not_empty(self) -> None:
        with self.assertRaises(ValidationError):
            create_order("u1", [], "tok_valid")

    def test_create_order_rejects_wrong_types(self) -> None:
        with self.assertRaises(ValidationError):
            create_order("u1", "not-a-list", "tok_valid")  # type: ignore[arg-type]

    def test_create_order_payment_declined(self) -> None:
        with self.assertRaises(PaymentError):
            create_order("u1", self.items, "invalid_token")

    def test_create_order_permission_denied(self) -> None:
        with self.assertRaises(PermissionDeniedError):
            create_order("blocked_user", self.items, "tok_valid")

    # -- cancel_order --------------------------------------------------------
    def test_cancel_order_success(self) -> None:
        order = create_order("u1", self.items, "tok_valid")
        result = cancel_order("u1", order["order_id"], "changed mind")
        self.assertEqual(result["status"], "cancelled")
        self.assertEqual(result["refunded_amount"], 24.0)
        self.assertTrue(result["refund_id"].startswith("rfnd_"))

    def test_cancel_order_not_found(self) -> None:
        with self.assertRaises(OrderNotFoundError):
            cancel_order("u1", "ord_missing", "reason")

    def test_cancel_order_already_cancelled(self) -> None:
        order = create_order("u1", self.items, "tok_valid")
        cancel_order("u1", order["order_id"], "first")
        with self.assertRaises(OrderStateError):
            cancel_order("u1", order["order_id"], "second")

    def test_cancel_order_wrong_user(self) -> None:
        order = create_order("u1", self.items, "tok_valid")
        with self.assertRaises(PermissionDeniedError):
            cancel_order("u2", order["order_id"], "not mine")

    def test_cancel_order_validates_reason(self) -> None:
        order = create_order("u1", self.items, "tok_valid")
        with self.assertRaises(ValidationError):
            cancel_order("u1", order["order_id"], "")

    # -- get_order_status ----------------------------------------------------
    def test_get_order_status_success(self) -> None:
        order = create_order("u1", self.items, "tok_valid")
        status = get_order_status("u1", order["order_id"])
        self.assertEqual(status["status"], "confirmed")
        self.assertEqual(status["total"], 24.0)

    def test_get_order_status_not_found(self) -> None:
        with self.assertRaises(OrderNotFoundError):
            get_order_status("u1", "ord_missing")

    def test_get_order_status_wrong_user(self) -> None:
        order = create_order("u1", self.items, "tok_valid")
        with self.assertRaises(PermissionDeniedError):
            get_order_status("u2", order["order_id"])

    def test_get_order_status_reflects_cancellation(self) -> None:
        order = create_order("u1", self.items, "tok_valid")
        cancel_order("u1", order["order_id"], "done")
        status = get_order_status("u1", order["order_id"])
        self.assertEqual(status["status"], "cancelled")


if __name__ == "__main__":
    unittest.main()
```
