"""Order management service.

This module provides a small order management service with functions to
create, cancel, and inspect orders. Configuration values are externalized
into a :class:`ServiceConfig` object, and all public functions validate
their inputs, log entry/exit, and raise domain-specific exceptions.
"""

from __future__ import annotations

import logging
import uuid
from dataclasses import dataclass
from typing import Any, Dict, List, Optional

logger = logging.getLogger(__name__)


# ---------------------------------------------------------------------------
# Configuration (R-019: all configuration externalized into a config object)
# ---------------------------------------------------------------------------
@dataclass
class ServiceConfig:
    """Configuration for the order service.

    Attributes:
        database_url: Connection string for the order datastore.
        payment_api_key: API key used to authenticate with the payment gateway.
        max_orders_per_minute: Rate limit for order creation per user.
        max_items_per_order: Maximum number of distinct line items per order.
        currency: ISO 4217 currency code used for totals.
    """

    database_url: str = "sqlite:///orders.db"
    payment_api_key: str = "test-payment-key"
    max_orders_per_minute: int = 60
    max_items_per_order: int = 100
    currency: str = "USD"


# Default, module-level configuration instance. Callers may override by
# passing their own ``ServiceConfig`` to the public functions.
DEFAULT_CONFIG = ServiceConfig()


# ---------------------------------------------------------------------------
# Domain-specific exceptions (R-007)
# ---------------------------------------------------------------------------
class OrderServiceError(Exception):
    """Base class for all order service errors."""


class ValidationError(OrderServiceError):
    """Raised when input validation fails."""


class PermissionDeniedError(OrderServiceError):
    """Raised when a caller lacks permission for a privileged operation."""


class PaymentError(OrderServiceError):
    """Raised when payment processing or refunding fails."""


class OrderNotFoundError(OrderServiceError):
    """Raised when a requested order does not exist."""

_ORDERS: Dict[str, Dict[str, Any]] = {}

def _check_permission(user_id: str, order: Optional[Dict[str, Any]] = None) -> None:
    """Verify the caller is permitted to act (R-020).

    Args:
        user_id: Identifier of the caller requesting the operation.
        order: Optional existing order whose owner must match ``user_id``.

    Returns:
        None.

    Raises:
        PermissionDeniedError: If the caller is not authorized.
    """
    if not user_id or not user_id.strip():
        raise PermissionDeniedError("A non-empty user_id is required.")
    if order is not None and order.get("user_id") != user_id:
        raise PermissionDeniedError(
            f"User '{user_id}' is not permitted to access order "
            f"'{order.get('order_id')}'."
        )


def _validate_items(items: List[Dict[str, Any]], config: ServiceConfig) -> float:
    """Validate line items and compute the order total.

    Args:
        items: List of line items, each a mapping with ``name``,
            ``quantity``, and ``unit_price`` keys.
        config: Service configuration providing item limits.

    Returns:
        The computed total price as a float.

    Raises:
        ValidationError: If items are malformed or contain negative values.
    """
    # R-001: validate parameter types before processing.
    if not isinstance(items, list):
        raise ValidationError(f"items must be a list, got {type(items).__name__}.")
    if not items:
        raise ValidationError("items must contain at least one line item.")
    if len(items) > config.max_items_per_order:
        raise ValidationError(
            f"items exceeds maximum of {config.max_items_per_order} per order."
        )

    total = 0.0
    for index, item in enumerate(items):
        if not isinstance(item, dict):
            raise ValidationError(f"item at index {index} must be a dict.")

        name = item.get("name")
        quantity = item.get("quantity")
        unit_price = item.get("unit_price")

        if not isinstance(name, str) or not name.strip():
            raise ValidationError(f"item at index {index} requires a non-empty name.")
        if not isinstance(quantity, int) or isinstance(quantity, bool):
            raise ValidationError(
                f"item '{name}' quantity must be an int, "
                f"got {type(quantity).__name__}."
            )
        if not isinstance(unit_price, (int, float)) or isinstance(unit_price, bool):
            raise ValidationError(
                f"item '{name}' unit_price must be a number, "
                f"got {type(unit_price).__name__}."
            )
        # R-002: reject negative quantity values.
        if quantity < 0:
            raise ValidationError(f"item '{name}' has a negative quantity.")
        if unit_price < 0:
            raise ValidationError(f"item '{name}' has a negative unit_price.")

        total += quantity * unit_price

    return round(total, 2)


def _process_payment(
    payment_token: str, amount: float, config: ServiceConfig
) -> str:
    """Process a payment through the payment gateway (placeholder).

    Args:
        payment_token: Tokenized payment instrument supplied by the caller.
        amount: Amount to charge.
        config: Service configuration providing the payment API key.

    Returns:
        A payment transaction identifier.

    Raises:
        PaymentError: If the payment cannot be processed.
    """
    try:
        if not config.payment_api_key:
            raise ValueError("payment gateway is not configured.")
        # Real implementation would call an external gateway here.
        return f"pay_{uuid.uuid4().hex}"
    except Exception as exc:  # noqa: BLE001 - re-raised as domain error.
        # R-008: include the original error context when re-raising.
        raise PaymentError(f"Payment failed for amount {amount}: {exc}") from exc


def _process_refund(
    payment_id: str, amount: float, config: ServiceConfig
) -> str:
    """Issue a refund through the payment gateway (placeholder).

    Args:
        payment_id: The original payment transaction identifier.
        amount: Amount to refund.
        config: Service configuration providing the payment API key.

    Returns:
        A refund transaction identifier.

    Raises:
        PaymentError: If the refund cannot be processed.
    """
    try:
        if not config.payment_api_key:
            raise ValueError("payment gateway is not configured.")
        return f"ref_{uuid.uuid4().hex}"
    except Exception as exc:  # noqa: BLE001 - re-raised as domain error.
        # R-008: include the original error context when re-raising.
        raise PaymentError(
            f"Refund failed for payment '{payment_id}': {exc}"
        ) from exc

def create_order(
    user_id: str,
    items: List[Dict[str, Any]],
    payment_token: str,
    config: ServiceConfig = DEFAULT_CONFIG,
) -> Dict[str, Any]:
    """Create a new order, charge payment, and return a confirmation.

    Args:
        user_id: Identifier of the user placing the order.
        items: List of line items, each a mapping with ``name`` (str),
            ``quantity`` (int), and ``unit_price`` (number) keys.
        payment_token: Tokenized payment instrument used to charge the order.
        config: Service configuration. Defaults to ``DEFAULT_CONFIG``.

    Returns:
        A confirmation dict with keys ``order_id``, ``user_id``, ``status``,
        ``total``, ``currency``, ``payment_id``, and ``items``.

    Raises:
        ValidationError: If any parameter is missing or malformed.
        PermissionDeniedError: If the caller is not authorized.
        PaymentError: If payment processing fails.
    """
    # R-013: log entry at DEBUG with parameter summary.
    logger.debug(
        "Entering create_order(user_id=%r, item_count=%s, payment_token=***)",
        user_id,
        len(items) if isinstance(items, list) else "n/a",
    )

    # R-001: validate parameter types before processing.
    if not isinstance(user_id, str):
        raise ValidationError(f"user_id must be a str, got {type(user_id).__name__}.")
    if not isinstance(payment_token, str) or not payment_token.strip():
        raise ValidationError("payment_token must be a non-empty str.")

    # R-020: verify caller permissions before the privileged operation.
    _check_permission(user_id)

    total = _validate_items(items, config)
    payment_id = _process_payment(payment_token, total, config)

    order_id = f"ord_{uuid.uuid4().hex}"
    order: Dict[str, Any] = {
        "order_id": order_id,
        "user_id": user_id,
        "status": "confirmed",
        "total": total,
        "currency": config.currency,
        "payment_id": payment_id,
        "items": list(items),
    }
    _ORDERS[order_id] = order

    confirmation = dict(order)
    # R-014: log exit at DEBUG with return value type.
    logger.debug("Exiting create_order -> %s", type(confirmation).__name__)
    return confirmation


def cancel_order(
    user_id: str,
    order_id: str,
    reason: str,
    config: ServiceConfig = DEFAULT_CONFIG,
) -> Dict[str, Any]:
    """Cancel an existing order and issue a refund.

    Args:
        user_id: Identifier of the user cancelling the order.
        order_id: Identifier of the order to cancel.
        reason: Human-readable reason for the cancellation.
        config: Service configuration. Defaults to ``DEFAULT_CONFIG``.

    Returns:
        A cancellation confirmation dict with keys ``order_id``,
        ``user_id``, ``status``, ``reason``, and ``refund_id``.

    Raises:
        ValidationError: If any parameter is missing or malformed.
        PermissionDeniedError: If the caller does not own the order.
        OrderNotFoundError: If the order does not exist.
        PaymentError: If the refund fails.
    """
    # R-013: log entry at DEBUG with parameter summary.
    logger.debug(
        "Entering cancel_order(user_id=%r, order_id=%r, reason=%r)",
        user_id,
        order_id,
        reason,
    )

    # R-001: validate parameter types before processing.
    if not isinstance(user_id, str):
        raise ValidationError(f"user_id must be a str, got {type(user_id).__name__}.")
    if not isinstance(order_id, str) or not order_id.strip():
        raise ValidationError("order_id must be a non-empty str.")
    if not isinstance(reason, str) or not reason.strip():
        raise ValidationError("reason must be a non-empty str.")

    order = _ORDERS.get(order_id)
    if order is None:
        raise OrderNotFoundError(f"Order '{order_id}' was not found.")

    # R-020: verify caller permissions before the privileged operation.
    _check_permission(user_id, order)

    if order["status"] == "cancelled":
        raise ValidationError(f"Order '{order_id}' is already cancelled.")

    refund_id = _process_refund(order["payment_id"], order["total"], config)
    order["status"] = "cancelled"
    order["cancellation_reason"] = reason
    order["refund_id"] = refund_id

    confirmation: Dict[str, Any] = {
        "order_id": order_id,
        "user_id": user_id,
        "status": order["status"],
        "reason": reason,
        "refund_id": refund_id,
    }
    # R-014: log exit at DEBUG with return value type.
    logger.debug("Exiting cancel_order -> %s", type(confirmation).__name__)
    return confirmation


def get_order_status(
    user_id: str,
    order_id: str,
    config: ServiceConfig = DEFAULT_CONFIG,
) -> Dict[str, Any]:
    """Retrieve the status and details of an order.

    Args:
        user_id: Identifier of the user requesting the order.
        order_id: Identifier of the order to inspect.
        config: Service configuration. Defaults to ``DEFAULT_CONFIG``.

    Returns:
        A status dict with keys ``order_id``, ``user_id``, ``status``,
        ``total``, ``currency``, and ``items``.

    Raises:
        ValidationError: If any parameter is missing or malformed.
        PermissionDeniedError: If the caller does not own the order.
        OrderNotFoundError: If the order does not exist.
    """
    # R-013: log entry at DEBUG with parameter summary.
    logger.debug(
        "Entering get_order_status(user_id=%r, order_id=%r)", user_id, order_id
    )

    # R-001: validate parameter types before processing.
    if not isinstance(user_id, str):
        raise ValidationError(f"user_id must be a str, got {type(user_id).__name__}.")
    if not isinstance(order_id, str) or not order_id.strip():
        raise ValidationError("order_id must be a non-empty str.")

    order = _ORDERS.get(order_id)
    if order is None:
        raise OrderNotFoundError(f"Order '{order_id}' was not found.")

    # R-020: verify caller permissions before returning order data.
    _check_permission(user_id, order)

    status: Dict[str, Any] = {
        "order_id": order["order_id"],
        "user_id": order["user_id"],
        "status": order["status"],
        "total": order["total"],
        "currency": order["currency"],
        "items": list(order["items"]),
    }
    # R-014: log exit at DEBUG with return value type.
    logger.debug("Exiting get_order_status -> %s", type(status).__name__)
    return status

import unittest


class OrderServiceTests(unittest.TestCase):
    """Unit tests for the public order service functions."""

    def setUp(self) -> None:
        """Reset the in-memory datastore before each test."""
        _ORDERS.clear()
        self.config = ServiceConfig()
        self.items: List[Dict[str, Any]] = [
            {"name": "Widget", "quantity": 2, "unit_price": 9.99},
            {"name": "Gadget", "quantity": 1, "unit_price": 19.50},
        ]

    def test_create_order_success(self) -> None:
        result = create_order("user-1", self.items, "tok_abc", self.config)
        self.assertEqual(result["status"], "confirmed")
        self.assertEqual(result["total"], round(2 * 9.99 + 19.50, 2))
        self.assertTrue(result["order_id"].startswith("ord_"))
        self.assertTrue(result["payment_id"].startswith("pay_"))

    def test_create_order_rejects_negative_quantity(self) -> None:
        bad_items = [{"name": "Widget", "quantity": -1, "unit_price": 5.0}]
        with self.assertRaises(ValidationError):
            create_order("user-1", bad_items, "tok_abc", self.config)

    def test_create_order_rejects_bad_types(self) -> None:
        with self.assertRaises(ValidationError):
            create_order("user-1", "not-a-list", "tok_abc", self.config)
        with self.assertRaises(ValidationError):
            create_order("user-1", self.items, "", self.config)

    def test_create_order_requires_user(self) -> None:
        with self.assertRaises(PermissionDeniedError):
            create_order("", self.items, "tok_abc", self.config)

    def test_get_order_status_success(self) -> None:
        created = create_order("user-1", self.items, "tok_abc", self.config)
        status = get_order_status("user-1", created["order_id"], self.config)
        self.assertEqual(status["status"], "confirmed")
        self.assertEqual(status["order_id"], created["order_id"])

    def test_get_order_status_permission_enforced(self) -> None:
        created = create_order("user-1", self.items, "tok_abc", self.config)
        with self.assertRaises(PermissionDeniedError):
            get_order_status("user-2", created["order_id"], self.config)

    def test_get_order_status_not_found(self) -> None:
        with self.assertRaises(OrderNotFoundError):
            get_order_status("user-1", "ord_missing", self.config)

    def test_cancel_order_success(self) -> None:
        created = create_order("user-1", self.items, "tok_abc", self.config)
        result = cancel_order(
            "user-1", created["order_id"], "Changed my mind", self.config
        )
        self.assertEqual(result["status"], "cancelled")
        self.assertTrue(result["refund_id"].startswith("ref_"))

    def test_cancel_order_twice_fails(self) -> None:
        created = create_order("user-1", self.items, "tok_abc", self.config)
        cancel_order("user-1", created["order_id"], "First", self.config)
        with self.assertRaises(ValidationError):
            cancel_order("user-1", created["order_id"], "Second", self.config)

    def test_cancel_order_permission_enforced(self) -> None:
        created = create_order("user-1", self.items, "tok_abc", self.config)
        with self.assertRaises(PermissionDeniedError):
            cancel_order("user-2", created["order_id"], "Nope", self.config)

    def test_cancel_order_not_found(self) -> None:
        with self.assertRaises(OrderNotFoundError):
            cancel_order("user-1", "ord_missing", "reason", self.config)

    def test_cancel_order_requires_reason(self) -> None:
        created = create_order("user-1", self.items, "tok_abc", self.config)
        with self.assertRaises(ValidationError):
            cancel_order("user-1", created["order_id"], "", self.config)


if __name__ == "__main__":
    unittest.main()
