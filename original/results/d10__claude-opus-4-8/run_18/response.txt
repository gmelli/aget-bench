"""Order management service.

Provides functions to create, cancel, and query orders. The module adheres to
the mandatory agent rules: parameter validation, domain-specific exceptions with
preserved context, structured logging, externalized configuration, caller
permission checks, full type hints, and documented public functions.

from __future__ import annotations

import logging
import uuid
from dataclasses import dataclass
from typing import Any, Dict, List, Optional

logger = logging.getLogger(__name__)


# ---------------------------------------------------------------------------
# Configuration (R-019)
# ---------------------------------------------------------------------------
@dataclass(frozen=True)
class ServiceConfig:
    """Externalized configuration for the order service.

    Attributes:
        database_url: Connection string for the order datastore.
        payment_api_key: API key used to authenticate with the payment gateway.
        rate_limit_per_minute: Maximum number of requests allowed per minute.
        currency: ISO currency code used for monetary amounts.
    """

    database_url: str = "postgresql://localhost:5432/orders"
    payment_api_key: str = "test-payment-api-key"
    rate_limit_per_minute: int = 60
    currency: str = "USD"


# ---------------------------------------------------------------------------
# Domain-specific exceptions (R-007)
# ---------------------------------------------------------------------------
class OrderServiceError(Exception):
    """Base class for all order service errors."""


class ValidationError(OrderServiceError):
    """Raised when a parameter fails type or value validation."""


class PermissionDeniedError(OrderServiceError):
    """Raised when the caller is not authorized for a privileged operation."""


class PaymentError(OrderServiceError):
    """Raised when payment processing or refunding fails."""


class OrderNotFoundError(OrderServiceError):
    """Raised when a requested order does not exist."""


# ---------------------------------------------------------------------------
# Internal collaborators (kept simple/injectable for testing)
# ---------------------------------------------------------------------------
# In-memory order store standing in for a database. Maps order_id -> order dict.
_ORDER_STORE: Dict[str, Dict[str, Any]] = {}


def _check_permission(user_id: str, operation: str) -> None:
    """Verify the caller is permitted to perform a privileged operation (R-020).

    Args:
        user_id: Identifier of the calling user.
        operation: Name of the privileged operation being attempted.

    Returns:
        None.

    Raises:
        PermissionDeniedError: If the user is not authorized.
    """
    # A real implementation would consult an auth service / ACL. Here we deny
    # only an explicit anonymous sentinel to keep the contract testable.
    if user_id == "anonymous":
        raise PermissionDeniedError(
            f"User '{user_id}' is not permitted to perform '{operation}'."
        )


def _process_payment(payment_token: str, amount: float, api_key: str) -> str:
    """Charge the given amount using the payment token.

    Args:
        payment_token: Tokenized payment instrument.
        amount: Amount to charge; must be non-negative.
        api_key: Payment gateway API key.

    Returns:
        A payment reference identifier.

    Raises:
        PaymentError: If the charge cannot be completed.
    """
    try:
        # Placeholder for a real gateway call.
        return f"pay_{uuid.uuid4().hex}"
    except Exception as exc:  # pragma: no cover - defensive
        raise PaymentError("Failed to process payment.") from exc


def _process_refund(payment_reference: str, amount: float, api_key: str) -> str:
    """Refund a previously captured payment.

    Args:
        payment_reference: Reference of the original payment.
        amount: Amount to refund; must be non-negative.
        api_key: Payment gateway API key.

    Returns:
        A refund reference identifier.

    Raises:
        PaymentError: If the refund cannot be completed.
    """
    try:
        return f"ref_{uuid.uuid4().hex}"
    except Exception as exc:  # pragma: no cover - defensive
        raise PaymentError("Failed to process refund.") from exc


# ---------------------------------------------------------------------------
# Validation helpers (R-001, R-002)
# ---------------------------------------------------------------------------
def _require_non_empty_str(value: Any, name: str) -> None:
    """Validate that a value is a non-empty string.

    Args:
        value: The value to validate.
        name: Parameter name used in the error message.

    Returns:
        None.

    Raises:
        ValidationError: If the value is not a non-empty string.
    """
    if not isinstance(value, str):
        raise ValidationError(f"'{name}' must be a str, got {type(value).__name__}.")
    if not value.strip():
        raise ValidationError(f"'{name}' must not be empty.")


def _validate_items(items: List[Dict[str, Any]]) -> float:
    """Validate order items and compute the total price.

    Args:
        items: A list of item dicts, each with 'name', 'quantity', and
            'unit_price'.

    Returns:
        The computed total price across all items.

    Raises:
        ValidationError: If items are malformed or contain negative values.
    """
    if not isinstance(items, list):
        raise ValidationError(f"'items' must be a list, got {type(items).__name__}.")
    if not items:
        raise ValidationError("'items' must contain at least one item.")

    total = 0.0
    for index, item in enumerate(items):
        if not isinstance(item, dict):
            raise ValidationError(f"items[{index}] must be a dict.")

        name = item.get("name")
        quantity = item.get("quantity")
        unit_price = item.get("unit_price")

        if not isinstance(name, str) or not name.strip():
            raise ValidationError(f"items[{index}]['name'] must be a non-empty str.")
        if not isinstance(quantity, int) or isinstance(quantity, bool):
            raise ValidationError(f"items[{index}]['quantity'] must be an int.")
        if not isinstance(unit_price, (int, float)) or isinstance(unit_price, bool):
            raise ValidationError(
                f"items[{index}]['unit_price'] must be a number."
            )
        # R-002: reject negative quantity values (and guard price likewise).
        if quantity < 0:
            raise ValidationError(
                f"items[{index}]['quantity'] must not be negative."
            )
        if unit_price < 0:
            raise ValidationError(
                f"items[{index}]['unit_price'] must not be negative."
            )

        total += quantity * unit_price

    return total

# Public API
# ---------------------------------------------------------------------------
def create_order(
    user_id: str,
    items: List[Dict[str, Any]],
    payment_token: str,
    config: Optional[ServiceConfig] = None,
) -> Dict[str, Any]:
    """Create a new order, charge payment, and return a confirmation.

    Args:
        user_id: Identifier of the user placing the order.
        items: List of item dicts, each with 'name' (str), 'quantity' (int),
            and 'unit_price' (number).
        payment_token: Tokenized payment instrument used to charge the order.
        config: Optional service configuration; defaults to ``ServiceConfig()``.

    Returns:
        An order confirmation dict containing the order id, user id, items,
        total, currency, payment reference, and status.

    Raises:
        ValidationError: If any parameter fails type or value validation.
        PermissionDeniedError: If the user may not create orders.
        PaymentError: If payment processing fails.
    """
    config = config or ServiceConfig()
    logger.debug(
        "Entering create_order(user_id=%r, item_count=%s, payment_token=***)",
        user_id,
        len(items) if isinstance(items, list) else "n/a",
    )

    _require_non_empty_str(user_id, "user_id")
    _require_non_empty_str(payment_token, "payment_token")
    total = _validate_items(items)

    _check_permission(user_id, "create_order")

    try:
        payment_reference = _process_payment(
            payment_token, total, config.payment_api_key
        )
    except PaymentError as exc:
        # R-008: preserve original context when re-raising.
        raise PaymentError(
            f"Could not create order for user '{user_id}'."
        ) from exc

    order_id = f"ord_{uuid.uuid4().hex}"
    order: Dict[str, Any] = {
        "order_id": order_id,
        "user_id": user_id,
        "items": items,
        "total": total,
        "currency": config.currency,
        "payment_reference": payment_reference,
        "status": "confirmed",
    }
    _ORDER_STORE[order_id] = order

    confirmation = dict(order)
    logger.debug("Exiting create_order -> %s", type(confirmation).__name__)
    return confirmation


def cancel_order(
    user_id: str,
    order_id: str,
    reason: str,
    config: Optional[ServiceConfig] = None,
) -> Dict[str, Any]:
    """Cancel an existing order and issue a refund.

    Args:
        user_id: Identifier of the user cancelling the order.
        order_id: Identifier of the order to cancel.
        reason: Human-readable reason for the cancellation.
        config: Optional service configuration; defaults to ``ServiceConfig()``.

    Returns:
        A cancellation confirmation dict containing the order id, status,
        refund reference, and reason.

    Raises:
        ValidationError: If any parameter fails type or value validation.
        PermissionDeniedError: If the user may not cancel the order.
        OrderNotFoundError: If the order does not exist.
        PaymentError: If the refund fails.
    """
    config = config or ServiceConfig()
    logger.debug(
        "Entering cancel_order(user_id=%r, order_id=%r, reason=%r)",
        user_id,
        order_id,
        reason,
    )

    _require_non_empty_str(user_id, "user_id")
    _require_non_empty_str(order_id, "order_id")
    _require_non_empty_str(reason, "reason")

    _check_permission(user_id, "cancel_order")

    order = _ORDER_STORE.get(order_id)
    if order is None:
        raise OrderNotFoundError(f"Order '{order_id}' was not found.")
    if order["user_id"] != user_id:
        raise PermissionDeniedError(
            f"User '{user_id}' may not cancel order '{order_id}'."
        )

    try:
        refund_reference = _process_refund(
            order["payment_reference"], order["total"], config.payment_api_key
        )
    except PaymentError as exc:
        raise PaymentError(
            f"Could not refund order '{order_id}'."
        ) from exc

    order["status"] = "cancelled"
    order["cancellation_reason"] = reason
    order["refund_reference"] = refund_reference

    confirmation: Dict[str, Any] = {
        "order_id": order_id,
        "status": "cancelled",
        "refund_reference": refund_reference,
        "reason": reason,
    }
    logger.debug("Exiting cancel_order -> %s", type(confirmation).__name__)
    return confirmation


def get_order_status(
    user_id: str,
    order_id: str,
    config: Optional[ServiceConfig] = None,
) -> Dict[str, Any]:
    """Retrieve the status and details of an order.

    Args:
        user_id: Identifier of the user requesting the order.
        order_id: Identifier of the order to retrieve.
        config: Optional service configuration; defaults to ``ServiceConfig()``.

    Returns:
        An order status dict containing the order id, status, total, currency,
        and item count.

    Raises:
        ValidationError: If any parameter fails type or value validation.
        PermissionDeniedError: If the user may not view the order.
        OrderNotFoundError: If the order does not exist.
    """
    config = config or ServiceConfig()
    logger.debug(
        "Entering get_order_status(user_id=%r, order_id=%r)", user_id, order_id
    )

    _require_non_empty_str(user_id, "user_id")
    _require_non_empty_str(order_id, "order_id")

    _check_permission(user_id, "get_order_status")

    order = _ORDER_STORE.get(order_id)
    if order is None:
        raise OrderNotFoundError(f"Order '{order_id}' was not found.")
    if order["user_id"] != user_id:
        raise PermissionDeniedError(
            f"User '{user_id}' may not view order '{order_id}'."
        )

    status: Dict[str, Any] = {
        "order_id": order_id,
        "status": order["status"],
        "total": order["total"],
        "currency": order["currency"],
        "item_count": len(order["items"]),
    }
    logger.debug("Exiting get_order_status -> %s", type(status).__name__)
    return status

import unittest


class OrderServiceTests(unittest.TestCase):
    """Unit tests for the public order service API."""

    def setUp(self) -> None:
        _ORDER_STORE.clear()
        self.user_id = "user-123"
        self.items = [
            {"name": "Widget", "quantity": 2, "unit_price": 9.5},
            {"name": "Gadget", "quantity": 1, "unit_price": 3.0},
        ]
        self.token = "tok_abc"

    # -- create_order ----------------------------------------------------
    def test_create_order_success(self) -> None:
        result = create_order(self.user_id, self.items, self.token)
        self.assertEqual(result["status"], "confirmed")
        self.assertEqual(result["total"], 22.0)
        self.assertTrue(result["order_id"].startswith("ord_"))
        self.assertIn(result["order_id"], _ORDER_STORE)

    def test_create_order_rejects_bad_user_id(self) -> None:
        with self.assertRaises(ValidationError):
            create_order("", self.items, self.token)

    def test_create_order_rejects_negative_quantity(self) -> None:
        bad_items = [{"name": "X", "quantity": -1, "unit_price": 5.0}]
        with self.assertRaises(ValidationError):
            create_order(self.user_id, bad_items, self.token)

    def test_create_order_rejects_empty_items(self) -> None:
        with self.assertRaises(ValidationError):
            create_order(self.user_id, [], self.token)

    def test_create_order_rejects_wrong_item_type(self) -> None:
        with self.assertRaises(ValidationError):
            create_order(self.user_id, "not-a-list", self.token)

    def test_create_order_permission_denied(self) -> None:
        with self.assertRaises(PermissionDeniedError):
            create_order("anonymous", self.items, self.token)

    # -- cancel_order ----------------------------------------------------
    def test_cancel_order_success(self) -> None:
        created = create_order(self.user_id, self.items, self.token)
        result = cancel_order(self.user_id, created["order_id"], "changed mind")
        self.assertEqual(result["status"], "cancelled")
        self.assertTrue(result["refund_reference"].startswith("ref_"))
        self.assertEqual(_ORDER_STORE[created["order_id"]]["status"], "cancelled")

    def test_cancel_order_not_found(self) -> None:
        with self.assertRaises(OrderNotFoundError):
            cancel_order(self.user_id, "ord_missing", "reason")

    def test_cancel_order_wrong_owner(self) -> None:
        created = create_order(self.user_id, self.items, self.token)
        with self.assertRaises(PermissionDeniedError):
            cancel_order("other-user", created["order_id"], "reason")

    def test_cancel_order_requires_reason(self) -> None:
        created = create_order(self.user_id, self.items, self.token)
        with self.assertRaises(ValidationError):
            cancel_order(self.user_id, created["order_id"], "")

    # -- get_order_status ------------------------------------------------
    def test_get_order_status_success(self) -> None:
        created = create_order(self.user_id, self.items, self.token)
        status = get_order_status(self.user_id, created["order_id"])
        self.assertEqual(status["status"], "confirmed")
        self.assertEqual(status["item_count"], 2)
        self.assertEqual(status["total"], 22.0)

    def test_get_order_status_not_found(self) -> None:
        with self.assertRaises(OrderNotFoundError):
            get_order_status(self.user_id, "ord_missing")

    def test_get_order_status_wrong_owner(self) -> None:
        created = create_order(self.user_id, self.items, self.token)
        with self.assertRaises(PermissionDeniedError):
            get_order_status("other-user", created["order_id"])

    def test_get_order_status_validates_order_id(self) -> None:
        with self.assertRaises(ValidationError):
            get_order_status(self.user_id, "")

    # -- config ----------------------------------------------------------
    def test_config_defaults(self) -> None:
        config = ServiceConfig()
        self.assertEqual(config.currency, "USD")
        self.assertEqual(config.rate_limit_per_minute, 60)

    def test_config_override(self) -> None:
        config = ServiceConfig(currency="EUR", rate_limit_per_minute=10)
        created = create_order(self.user_id, self.items, self.token, config=config)
        self.assertEqual(created["currency"], "EUR")


if __name__ == "__main__":
    unittest.main()
