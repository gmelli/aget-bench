"""Order management service.

Provides creation, cancellation, and status retrieval for orders, including
price calculation, payment processing, and refunds.

This module adheres to the mandatory agent rules: all public functions
validate their inputs (R-001/R-002), raise domain-specific exceptions with
preserved context (R-007/R-008), log entry/exit at DEBUG level
(R-013/R-014), source configuration from a config object (R-019), verify
caller permissions before privileged operations (R-020), and carry full
type hints and docstrings (R-025/R-026).
"""

from __future__ import annotations

import logging
import uuid
from dataclasses import dataclass, field
from typing import Any, Dict, List, Optional

logger = logging.getLogger(__name__)


# ---------------------------------------------------------------------------
# Configuration (R-019)
# ---------------------------------------------------------------------------


@dataclass(frozen=True)
class ServiceConfig:
    """Configuration values for the order service.

    Externalizes all tunable/secret values so no magic constants live in the
    business logic (R-019).

    Attributes:
        database_url: Connection string for the order datastore.
        payment_api_key: API key used to authenticate with the payment gateway.
        max_orders_per_minute: Rate limit for order creation per user.
        max_items_per_order: Upper bound on distinct line items per order.
        currency: ISO currency code used for monetary amounts.
    """

    database_url: str = "sqlite:///orders.db"
    payment_api_key: str = "test-key"
    max_orders_per_minute: int = 60
    max_items_per_order: int = 100
    currency: str = "USD"


# ---------------------------------------------------------------------------
# Domain-specific exceptions (R-007)
# ---------------------------------------------------------------------------


class OrderServiceError(Exception):
    """Base class for all order service errors."""


class ValidationError(OrderServiceError):
    """Raised when input parameters fail validation."""


class PermissionDeniedError(OrderServiceError):
    """Raised when a caller lacks permission for a privileged operation."""


class OrderNotFoundError(OrderServiceError):
    """Raised when a requested order does not exist."""


class PaymentError(OrderServiceError):
    """Raised when a payment or refund operation fails."""


# ---------------------------------------------------------------------------
# In-memory datastore (stand-in for a real database)
# ---------------------------------------------------------------------------


@dataclass
class _OrderRecord:
    """Internal representation of a stored order."""

    order_id: str
    user_id: str
    items: List[Dict[str, Any]]
    total: float
    status: str
    payment_reference: str
    cancellation_reason: Optional[str] = None
    refund_reference: Optional[str] = None


@dataclass
class OrderStore:
    """Simple in-memory order store.

    Stands in for a real database (configured via ``ServiceConfig.database_url``)
    so the service logic can be exercised and tested in isolation.
    """

    _orders: Dict[str, _OrderRecord] = field(default_factory=dict)

    def save(self, record: _OrderRecord) -> None:
        """Persist or overwrite an order record.

        Args:
            record: The order record to store.

        Returns:
            None.
        """
        self._orders[record.order_id] = record

    def get(self, order_id: str) -> _OrderRecord:
        """Fetch an order record by ID.

        Args:
            order_id: Identifier of the order to retrieve.

        Returns:
            The matching order record.

        Raises:
            OrderNotFoundError: If no order with the given ID exists.
        """
        try:
            return self._orders[order_id]
        except KeyError as exc:
            raise OrderNotFoundError(f"Order {order_id!r} not found") from exc


# ---------------------------------------------------------------------------
# Payment gateway (stand-in for a real provider)
# ---------------------------------------------------------------------------


class PaymentGateway:
    """Thin payment gateway abstraction.

    A real implementation would call an external payment provider using
    ``config.payment_api_key``. This stand-in succeeds for any truthy token.
    """

    def __init__(self, config: ServiceConfig) -> None:
        """Initialize the gateway.

        Args:
            config: Service configuration supplying the payment API key.

        Returns:
            None.
        """
        self._config = config

    def charge(self, payment_token: str, amount: float) -> str:
        """Charge an amount against a payment token.

        Args:
            payment_token: Tokenized payment instrument.
            amount: Amount to charge in the configured currency.

        Returns:
            A payment reference identifier.

        Raises:
            PaymentError: If the charge cannot be completed.
        """
        if not payment_token:
            raise PaymentError("Missing payment token")
        if amount <= 0:
            raise PaymentError(f"Cannot charge non-positive amount: {amount}")
        try:
            # A real gateway call would happen here using self._config.
            return f"pay_{uuid.uuid4().hex}"
        except Exception as exc:  # pragma: no cover - defensive
            raise PaymentError("Payment charge failed") from exc

    def refund(self, payment_reference: str, amount: float) -> str:
        """Refund a previously charged payment.

        Args:
            payment_reference: Reference returned by a prior ``charge`` call.
            amount: Amount to refund in the configured currency.

        Returns:
            A refund reference identifier.

        Raises:
            PaymentError: If the refund cannot be completed.
        """
        if not payment_reference:
            raise PaymentError("Missing payment reference for refund")
        try:
            return f"ref_{uuid.uuid4().hex}"
        except Exception as exc:  # pragma: no cover - defensive
            raise PaymentError("Payment refund failed") from exc


# ---------------------------------------------------------------------------
# Permission checks (R-020)
# ---------------------------------------------------------------------------


def _check_permission(user_id: str, order: Optional[_OrderRecord] = None) -> None:
    """Verify the caller may operate on the given order.

    Args:
        user_id: Identifier of the calling user.
        order: The order being acted upon, or None for creation.

    Returns:
        None.

    Raises:
        PermissionDeniedError: If the user does not own the order.
    """
    if order is not None and order.user_id != user_id:
        raise PermissionDeniedError(
            f"User {user_id!r} may not access order {order.order_id!r}"
        )


# ---------------------------------------------------------------------------
# Input validation helpers (R-001, R-002)
# ---------------------------------------------------------------------------


def _require_non_empty_str(value: Any, name: str) -> str:
    """Validate that a value is a non-empty string.

    Args:
        value: The value to validate.
        name: Parameter name used in error messages.

    Returns:
        The validated string.

    Raises:
        ValidationError: If the value is not a non-empty string.
    """
    if not isinstance(value, str):
        raise ValidationError(f"{name} must be a str, got {type(value).__name__}")
    if not value.strip():
        raise ValidationError(f"{name} must not be empty")
    return value


def _validate_items(items: Any, config: ServiceConfig) -> List[Dict[str, Any]]:
    """Validate the items list and each line item.

    Enforces type checks (R-001) and rejects negative quantities (R-002).

    Args:
        items: Candidate list of line items.
        config: Service configuration supplying limits.

    Returns:
        The validated list of items.

    Raises:
        ValidationError: If items or any line item is invalid.
    """
    if not isinstance(items, list):
        raise ValidationError(f"items must be a list, got {type(items).__name__}")
    if not items:
        raise ValidationError("items must not be empty")
    if len(items) > config.max_items_per_order:
        raise ValidationError(
            f"items exceeds max of {config.max_items_per_order}"
        )

    for index, item in enumerate(items):
        if not isinstance(item, dict):
            raise ValidationError(f"item[{index}] must be a dict")

        name = item.get("name")
        if not isinstance(name, str) or not name.strip():
            raise ValidationError(f"item[{index}].name must be a non-empty str")

        quantity = item.get("quantity")
        if not isinstance(quantity, int) or isinstance(quantity, bool):
            raise ValidationError(f"item[{index}].quantity must be an int")
        if quantity < 0:  # R-002
            raise ValidationError(
                f"item[{index}].quantity must not be negative: {quantity}"
            )
        if quantity == 0:
            raise ValidationError(f"item[{index}].quantity must be positive")

        unit_price = item.get("unit_price")
        if isinstance(unit_price, bool) or not isinstance(unit_price, (int, float)):
            raise ValidationError(f"item[{index}].unit_price must be a number")
        if unit_price < 0:  # R-002
            raise ValidationError(
                f"item[{index}].unit_price must not be negative: {unit_price}"
            )

    return items


def _calculate_total(items: List[Dict[str, Any]]) -> float:
    """Compute the order total from validated items.

    Args:
        items: Validated line items, each with quantity and unit_price.

    Returns:
        The rounded total price.
    """
    total = sum(item["quantity"] * item["unit_price"] for item in items)
    return round(float(total), 2)

# Public service functions
# ---------------------------------------------------------------------------


def create_order(
    user_id: str,
    items: List[Dict[str, Any]],
    payment_token: str,
    config: Optional[ServiceConfig] = None,
    store: Optional[OrderStore] = None,
    gateway: Optional[PaymentGateway] = None,
) -> Dict[str, Any]:
    """Create a new order, charge payment, and return a confirmation.

    Args:
        user_id: Identifier of the user placing the order.
        items: Line items, each a dict with ``name``, ``quantity`` (positive
            int), and ``unit_price`` (non-negative number).
        payment_token: Tokenized payment instrument used to charge the total.
        config: Optional service configuration; defaults to ``ServiceConfig()``.
        store: Optional order store; defaults to a fresh ``OrderStore``.
        gateway: Optional payment gateway; defaults to one built from config.

    Returns:
        A confirmation dict with keys ``order_id``, ``user_id``, ``status``,
        ``total``, ``currency``, ``items``, and ``payment_reference``.

    Raises:
        ValidationError: If any parameter fails validation.
        PaymentError: If the payment charge fails.
    """
    logger.debug(
        "create_order(user_id=%r, items=%d items, payment_token=***)",
        user_id,
        len(items) if isinstance(items, list) else 0,
    )  # R-013

    config = config or ServiceConfig()
    store = store if store is not None else OrderStore()
    gateway = gateway if gateway is not None else PaymentGateway(config)

    _require_non_empty_str(user_id, "user_id")
    _require_non_empty_str(payment_token, "payment_token")
    validated_items = _validate_items(items, config)

    _check_permission(user_id)  # R-020

    total = _calculate_total(validated_items)

    try:
        payment_reference = gateway.charge(payment_token, total)
    except PaymentError as exc:
        # R-008: preserve original context when re-raising.
        raise PaymentError(f"Failed to charge order for user {user_id!r}") from exc

    order_id = f"ord_{uuid.uuid4().hex}"
    record = _OrderRecord(
        order_id=order_id,
        user_id=user_id,
        items=validated_items,
        total=total,
        status="confirmed",
        payment_reference=payment_reference,
    )
    store.save(record)

    confirmation: Dict[str, Any] = {
        "order_id": order_id,
        "user_id": user_id,
        "status": record.status,
        "total": total,
        "currency": config.currency,
        "items": validated_items,
        "payment_reference": payment_reference,
    }

    logger.debug("create_order -> %s", type(confirmation).__name__)  # R-014
    return confirmation


def cancel_order(
    user_id: str,
    order_id: str,
    reason: str,
    config: Optional[ServiceConfig] = None,
    store: Optional[OrderStore] = None,
    gateway: Optional[PaymentGateway] = None,
) -> Dict[str, Any]:
    """Cancel an existing order and issue a refund.

    Args:
        user_id: Identifier of the user requesting cancellation.
        order_id: Identifier of the order to cancel.
        reason: Human-readable cancellation reason.
        config: Optional service configuration; defaults to ``ServiceConfig()``.
        store: Optional order store; defaults to a fresh ``OrderStore``.
        gateway: Optional payment gateway; defaults to one built from config.

    Returns:
        A cancellation confirmation dict with keys ``order_id``, ``user_id``,
        ``status``, ``reason``, and ``refund_reference``.

    Raises:
        ValidationError: If any parameter fails validation.
        OrderNotFoundError: If the order does not exist.
        PermissionDeniedError: If the user does not own the order.
        PaymentError: If the refund fails.
    """
    logger.debug(
        "cancel_order(user_id=%r, order_id=%r, reason=%r)",
        user_id,
        order_id,
        reason,
    )  # R-013

    config = config or ServiceConfig()
    store = store if store is not None else OrderStore()
    gateway = gateway if gateway is not None else PaymentGateway(config)

    _require_non_empty_str(user_id, "user_id")
    _require_non_empty_str(order_id, "order_id")
    _require_non_empty_str(reason, "reason")

    record = store.get(order_id)
    _check_permission(user_id, record)  # R-020

    if record.status == "cancelled":
        raise ValidationError(f"Order {order_id!r} is already cancelled")

    try:
        refund_reference = gateway.refund(record.payment_reference, record.total)
    except PaymentError as exc:
        raise PaymentError(f"Failed to refund order {order_id!r}") from exc  # R-008

    record.status = "cancelled"
    record.cancellation_reason = reason
    record.refund_reference = refund_reference
    store.save(record)

    confirmation: Dict[str, Any] = {
        "order_id": order_id,
        "user_id": user_id,
        "status": record.status,
        "reason": reason,
        "refund_reference": refund_reference,
    }

    logger.debug("cancel_order -> %s", type(confirmation).__name__)  # R-014
    return confirmation


def get_order_status(
    user_id: str,
    order_id: str,
    config: Optional[ServiceConfig] = None,
    store: Optional[OrderStore] = None,
) -> Dict[str, Any]:
    """Retrieve the status and details of an order.

    Args:
        user_id: Identifier of the user requesting the status.
        order_id: Identifier of the order to look up.
        config: Optional service configuration; defaults to ``ServiceConfig()``.
        store: Optional order store; defaults to a fresh ``OrderStore``.

    Returns:
        A status dict with keys ``order_id``, ``user_id``, ``status``,
        ``total``, ``currency``, and ``items``.

    Raises:
        ValidationError: If any parameter fails validation.
        OrderNotFoundError: If the order does not exist.
        PermissionDeniedError: If the user does not own the order.
    """
    logger.debug(
        "get_order_status(user_id=%r, order_id=%r)", user_id, order_id
    )  # R-013

    config = config or ServiceConfig()
    store = store if store is not None else OrderStore()

    _require_non_empty_str(user_id, "user_id")
    _require_non_empty_str(order_id, "order_id")

    record = store.get(order_id)
    _check_permission(user_id, record)  # R-020

    status: Dict[str, Any] = {
        "order_id": record.order_id,
        "user_id": record.user_id,
        "status": record.status,
        "total": record.total,
        "currency": config.currency,
        "items": record.items,
    }

    logger.debug("get_order_status -> %s", type(status).__name__)  # R-014
    return status

"""Unit tests for order_service."""

import unittest

from order_service import (
    OrderNotFoundError,
    OrderStore,
    PaymentError,
    PaymentGateway,
    PermissionDeniedError,
    ServiceConfig,
    ValidationError,
    cancel_order,
    create_order,
    get_order_status,

class _FailingGateway(PaymentGateway):
    """Gateway whose charge always fails, for error-path tests."""

    def charge(self, payment_token: str, amount: float) -> str:
        raise PaymentError("simulated decline")


def _items():
    return [
        {"name": "Widget", "quantity": 2, "unit_price": 9.99},
        {"name": "Gadget", "quantity": 1, "unit_price": 4.50},
    ]


class CreateOrderTests(unittest.TestCase):
    def setUp(self):
        self.config = ServiceConfig()
        self.store = OrderStore()

    def test_create_order_success(self):
        result = create_order(
            "user-1", _items(), "tok_abc",
            config=self.config, store=self.store,
        )
        self.assertEqual(result["status"], "confirmed")
        self.assertEqual(result["total"], round(2 * 9.99 + 4.50, 2))
        self.assertEqual(result["currency"], "USD")
        self.assertTrue(result["order_id"].startswith("ord_"))
        self.assertTrue(result["payment_reference"].startswith("pay_"))

    def test_invalid_user_id_type(self):
        with self.assertRaises(ValidationError):
            create_order(123, _items(), "tok", store=self.store)

    def test_empty_payment_token(self):
        with self.assertRaises(ValidationError):
            create_order("user-1", _items(), "", store=self.store)

    def test_negative_quantity_rejected(self):
        bad = [{"name": "X", "quantity": -1, "unit_price": 1.0}]
        with self.assertRaises(ValidationError):
            create_order("user-1", bad, "tok", store=self.store)

    def test_negative_unit_price_rejected(self):
        bad = [{"name": "X", "quantity": 1, "unit_price": -5.0}]
        with self.assertRaises(ValidationError):
            create_order("user-1", bad, "tok", store=self.store)

    def test_empty_items_rejected(self):
        with self.assertRaises(ValidationError):
            create_order("user-1", [], "tok", store=self.store)

    def test_items_not_a_list_rejected(self):
        with self.assertRaises(ValidationError):
            create_order("user-1", "nope", "tok", store=self.store)

    def test_payment_failure_wrapped(self):
        gateway = _FailingGateway(self.config)
        with self.assertRaises(PaymentError):
            create_order(
                "user-1", _items(), "tok",
                config=self.config, store=self.store, gateway=gateway,
            )


class CancelOrderTests(unittest.TestCase):
    def setUp(self):
        self.store = OrderStore()
        self.order = create_order("user-1", _items(), "tok", store=self.store)

    def test_cancel_success(self):
        result = cancel_order(
            "user-1", self.order["order_id"], "changed mind", store=self.store,
        )
        self.assertEqual(result["status"], "cancelled")
        self.assertTrue(result["refund_reference"].startswith("ref_"))

    def test_cancel_unknown_order(self):
        with self.assertRaises(OrderNotFoundError):
            cancel_order("user-1", "ord_missing", "reason", store=self.store)

    def test_cancel_wrong_user_denied(self):
        with self.assertRaises(PermissionDeniedError):
            cancel_order(
                "intruder", self.order["order_id"], "reason", store=self.store,
            )

    def test_double_cancel_rejected(self):
        cancel_order("user-1", self.order["order_id"], "reason", store=self.store)
        with self.assertRaises(ValidationError):
            cancel_order(
                "user-1", self.order["order_id"], "again", store=self.store,
            )

    def test_empty_reason_rejected(self):
        with self.assertRaises(ValidationError):
            cancel_order("user-1", self.order["order_id"], "", store=self.store)


class GetOrderStatusTests(unittest.TestCase):

        self.store = OrderStore()

A few notes on how the rules shaped the design:

- **R-001/R-002** — `_validate_items` and `_require_non_empty_str` type-check every parameter and reject negative quantities/prices (booleans are explicitly excluded since `bool` is an `int` subclass).
- **R-007/R-008** — a domain exception hierarchy (`OrderServiceError` and subclasses) replaces generic exceptions, and re-raises use `raise ... from exc` to preserve context.
- **R-013/R-014** — each public function logs entry (with a sanitized parameter summary — note the payment token is masked) and exit (return type).
- **R-019/R-020** — `ServiceConfig` holds DB URL, payment key, and rate limits; `_check_permission` gates the privileged operations.
- **R-025/R-026** — full type hints and Args/Returns/Raises docstrings throughout.
